42 // TMP to check if a type is in the tuple.
43 template <typename T, typename Tuple, std::size_t I = 0u, typename Enable = void>
44 struct type_in_tuple {
45  static_assert(I < std::numeric_limits<std::size_t>::max(), "Overflow error.");
46  static const bool value
47  = std::is_same<T, typename std::tuple_element<I, Tuple>::type>::value || type_in_tuple<T, Tuple, I + 1u>::value;
48 };
49 template <typename T, typename Tuple, std::size_t I>
50 struct type_in_tuple<T, Tuple, I, typename std::enable_if<I == std::tuple_size<Tuple>::value>::type> {
51  static const bool value = false;
52 };
53 }
