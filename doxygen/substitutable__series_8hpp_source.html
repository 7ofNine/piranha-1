78 template <typename Series, typename Derived>
79 class substitutable_series : public Series, detail::substitutable_series_tag
80 {
81  typedef Series base;
82  // Detect subs term.
83  template <typename Term, typename T>
84  struct subs_term_score {
85  static const unsigned value = static_cast<unsigned>(has_subs<typename Term::cf_type, T>::value)
86  + (static_cast<unsigned>(key_has_subs<typename Term::key_type, T>::value) << 1u);
87  };
88  // Case 1: subs only on cf.
89  template <typename T, typename Term>
90  using cf_subs_type = decltype(math::subs(std::declval<typename Term::cf_type const &>(),
91  std::declval<std::string const &>(), std::declval<T const &>()));
92  template <typename T, typename Term>
93  using ret_type_1 = decltype(std::declval<const cf_subs_type<T, Term> &>() * std::declval<Derived const &>());
94  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term, T>::value == 1u, int>::type = 0>
95  static ret_type_1<T, Term> subs_term_impl(const Term &t, const std::string &name, const T &x,
96  const symbol_set &s_set)
97  {
98  Derived tmp;
99  tmp.set_symbol_set(s_set);
100  tmp.insert(Term(typename Term::cf_type(1), t.m_key));
101  // NOTE: use moves here in case the multiplication can take advantage.
102  return math::subs(t.m_cf, name, x) * std::move(tmp);
103  }
104  // Case 2: subs only on key.
105  template <typename T, typename Term>
106  using k_subs_type = typename decltype(std::declval<const typename Term::key_type &>().subs(
107  std::declval<const std::string &>(), std::declval<const T &>(),
108  std::declval<const symbol_set &>()))::value_type::first_type;
109  template <typename T, typename Term>
110  using ret_type_2_ = decltype(std::declval<Derived const &>() * std::declval<const k_subs_type<T, Term> &>());
111  template <typename T, typename Term>
112  using ret_type_2 = typename std::enable_if<is_addable_in_place<ret_type_2_<T, Term>>::value
113  && std::is_constructible<ret_type_2_<T, Term>, int>::value,
114  ret_type_2_<T, Term>>::type;
115  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term, T>::value == 2u, int>::type = 0>
116  static ret_type_2<T, Term> subs_term_impl(const Term &t, const std::string &name, const T &x,
117  const symbol_set &s_set)
118  {
119  ret_type_2<T, Term> retval(0);
120  auto ksubs = t.m_key.subs(name, x, s_set);
121  for (auto &p : ksubs) {
122  Derived tmp;
123  tmp.set_symbol_set(s_set);
124  tmp.insert(Term{t.m_cf, std::move(p.second)});
125  // NOTE: possible use of multadd here in the future.
126  retval += std::move(tmp) * std::move(p.first);
127  }
128  return retval;
129  }
130  // Case 3: subs on cf and key.
131  // NOTE: the checks on type 2 are already present in the alias above.
132  template <typename T, typename Term>
133  using ret_type_3
134  = decltype(std::declval<const cf_subs_type<T, Term> &>() * std::declval<const ret_type_2<T, Term> &>());
135  template <typename T, typename Term, typename std::enable_if<subs_term_score<Term, T>::value == 3u, int>::type = 0>
136  static ret_type_3<T, Term> subs_term_impl(const Term &t, const std::string &name, const T &x,
137  const symbol_set &s_set)
138  {
139  // Accumulator for the sum below. This is the same type resulting from case 2.
140  ret_type_2<T, Term> acc(0);
141  auto ksubs = t.m_key.subs(name, x, s_set);
142  auto cf_subs = math::subs(t.m_cf, name, x);
143  for (auto &p : ksubs) {
144  Derived tmp;
145  tmp.set_symbol_set(s_set);
146  tmp.insert(Term(typename Term::cf_type(1), std::move(p.second)));
147  // NOTE: multadd chance.
148  acc += std::move(tmp) * std::move(p.first);
149  }
150  return std::move(cf_subs) * std::move(acc);
151  }
152  // Initial definition of the subs type.
153  template <typename T>
154  using subs_type_ = decltype(subs_term_impl(std::declval<typename Series::term_type const &>(),
155  std::declval<std::string const &>(), std::declval<const T &>(),
156  std::declval<symbol_set const &>()));
157  // Enable conditionally based on the common requirements in the subs() method, plus on the fact that
158  // the subs type must be returnable.
159  // NOTE: the returnable check here should be enough, as the functions above will not be called if the check
160  // here is not passed.
161  template <typename T>
162  using subs_type =
163  typename std::enable_if<std::is_constructible<subs_type_<T>, int>::value
164  && is_addable_in_place<subs_type_<T>>::value && is_returnable<subs_type_<T>>::value,
165  subs_type_<T>>::type;
166 
167 public:
169  substitutable_series() = default;
171  substitutable_series(const substitutable_series &) = default;
173  substitutable_series(substitutable_series &&) = default;
174  PIRANHA_FORWARDING_CTOR(substitutable_series, base)
176  ~substitutable_series()
177  {
178  PIRANHA_TT_CHECK(is_series, substitutable_series);
179  PIRANHA_TT_CHECK(is_series, Derived);
180  PIRANHA_TT_CHECK(std::is_base_of, substitutable_series, Derived);
181  }
183  substitutable_series &operator=(const substitutable_series &) = default;
185  substitutable_series &operator=(substitutable_series &&) = default;
186  PIRANHA_FORWARDING_ASSIGNMENT(substitutable_series, base)
