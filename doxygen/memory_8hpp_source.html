72 inline void *cpp_aligned_alloc(const std::size_t &alignment, const std::size_t &size)
73 {
74  if (unlikely(alignment == 0u || size > std::numeric_limits<std::size_t>::max() - alignment)) {

77  // This is the actual allocated size.
78  std::size_t a_size = size + alignment;

80  // Allocate enough space for payload and alignment.
81  void *u_ptr = std::malloc(a_size), *orig_u_ptr = u_ptr;
82  if (unlikely(u_ptr == nullptr)) {
83  piranha_throw(std::bad_alloc,);
84  }
85  // Try to align.
86  void *ptr = std::align(alignment,size,u_ptr,a_size);
87  if (unlikely(ptr == nullptr)) {
88  std::free(orig_u_ptr);
89  piranha_throw(std::bad_alloc,);
90  }
91  // Some sanity checks.
92  piranha_assert(orig_a_size >= a_size);
93  // If the size has changed, the pointer must have too.
94  piranha_assert(orig_a_size == a_size || orig_u_ptr != ptr);
95  // Number of bytes taken by the alignment.
96  const std::size_t delta = (orig_a_size == a_size) ? alignment : orig_a_size - a_size;
97  piranha_assert(delta <= alignment);
98  piranha_assert(delta > 0u);
99  // The delta needs to be representable by unsigned char for storage.
100  if (unlikely(delta > std::numeric_limits<unsigned char>::max())) {
101  // NOTE: need to use original pointer, as u_ptr might have been changed by std::align.
102  std::free(orig_u_ptr);
103  piranha_throw(std::bad_alloc,);
104  }
105  // If the original pointer was already aligned, we have to move it forward.
106  if (orig_u_ptr == ptr) {
107  ptr = static_cast<void *>(static_cast<unsigned char *>(ptr) + alignment);
108  }
109  // Record the delta.
110  *(static_cast<unsigned char *>(ptr) - 1) = static_cast<unsigned char>(delta);
111  return ptr;
112 }

135 inline void *aligned_palloc(const std::size_t &alignment, const std::size_t &size)
136 {
137  // Platform-independent part: special values for alignment and size.
138  if (unlikely(size == 0u)) {
139  return nullptr;
140  }
141  if (alignment == 0u) {
142  void *ptr = std::malloc(size);
143  if (unlikely(ptr == nullptr)) {
144  piranha_throw(std::bad_alloc,);
145  }
146  return ptr;
147  }
148 #if defined(PIRANHA_HAVE_POSIX_MEMALIGN)
149  void *ptr;
150  const int retval = ::posix_memalign(&ptr,alignment,size);
151  if (unlikely(retval != 0)) {
152  piranha_throw(std::bad_alloc,);
153  }
154  return ptr;
155 #elif defined(_WIN32)
156  void *ptr = ::_aligned_malloc(size,alignment);
157  if (unlikely(ptr == nullptr)) {
158  piranha_throw(std::bad_alloc,);
159  }
160  return ptr;
161 #else
162  // return detail::cpp_aligned_alloc(alignment,size);
163  piranha_throw(not_implemented_error,"memory alignment primitives are not available");
164 #endif
165 }

222 template <typename T>
223 inline bool alignment_check(const std::size_t &alignment)
224 {
225  // Platform-independent checks.
226  if (alignment == 0u) {
227  return true;
228  }
229  // Alignment must be power of 2.
230  if (unlikely(static_cast<bool>(alignment & (alignment - 1u)))) {
231  return false;
232  }
233  // Alignment must not be less than the natural alignment of T. We just need the '<' check
234  // as we already know that alignment is either a multiple of alignof(T) or a divisor.
235  if (unlikely(alignment < alignof(typename std::decay<T>::type))) {
236  return false;
237  }
238 #if defined(PIRANHA_HAVE_POSIX_MEMALIGN)
239  // Extra check for posix_memalign requirements.
240  if (unlikely(static_cast<bool>(alignment % sizeof(void *)))) {
241  return false;
242  }
243 #endif
244  return true;
245 }

269 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
270 inline void parallel_value_init(T *ptr, const std::size_t &size, const unsigned &n_threads)
271 {
272  using ranges_vector = std::vector<std::pair<T *,T *>>;
273  using rv_size_type = typename ranges_vector::size_type;
274  if (unlikely(ptr == nullptr)) {
275  piranha_assert(!size);
276  return;
277  }
278  // Initing functor.
279  auto init_function = [](T *start, T *end, const unsigned &thread_idx, ranges_vector *rv) {
280  auto orig_start = start;
281  try {
282  for (; start != end; ++start) {
283  ::new (static_cast<void *>(start)) T();
284  }
285  } catch (...) {
286  // Cleanup.
287  for (; orig_start != start; ++orig_start) {
288  orig_start->~T();
289  }
290  // Re-throw.
291  throw;
292  }
293  // If the init was successful and we are in multithreaded mode, record
294  // the range that was inited.
295  if (rv != nullptr) {
296  (*rv)[static_cast<rv_size_type>(thread_idx)].first = orig_start;
297  (*rv)[static_cast<rv_size_type>(thread_idx)].second = end;
298  }
299  };
300  if (n_threads <= 1) {
301  init_function(ptr,ptr + size,0u,nullptr);
302  } else {
303  // Init the ranges vector with (ptr,ptr) pairs, so they are empty ranges.
304  ranges_vector inited_ranges(static_cast<rv_size_type>(n_threads),std::make_pair(ptr,ptr));
305  if (unlikely(inited_ranges.size() != n_threads)) {
306  piranha_throw(std::bad_alloc,);
307  }
308  // Work per thread.
309  const auto wpt = static_cast<std::size_t>(size / n_threads);
310  future_list<decltype(thread_pool::enqueue(0u,init_function,ptr,ptr,0u,&inited_ranges))> f_list;
311  try {
312  for (auto i = 0u; i < n_threads; ++i) {
313  auto start = ptr + i * wpt, end = (i == n_threads - 1u) ? ptr + size : ptr + (i + 1u) * wpt;
314  f_list.push_back(thread_pool::enqueue(i,init_function,start,end,i,&inited_ranges));
315  }
316  f_list.wait_all();
317  f_list.get_all();
318  } catch (...) {
319  f_list.wait_all();
320  // Rollback the ranges that were inited.
321  for (const auto &p: inited_ranges) {
322  for (auto start = p.first; start != p.second; ++start) {
323  start->~T();
324  }
325  }
326  throw;
327  }
328  }
329 }

346 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
347 inline void parallel_destroy(T *ptr, const std::size_t &size, const unsigned &n_threads)
348 {
349  using ranges_vector = std::vector<std::pair<T *,T *>>;
350  using rv_size_type = typename ranges_vector::size_type;
351  // Nothing to be done for null pointers.
352  if (unlikely(ptr == nullptr)) {
353  piranha_assert(!size);
354  return;
355  }
356  // Nothing needs to be done for trivially destructible types:
357  // http://en.cppreference.com/w/cpp/language/lifetime
358  if (std::is_trivially_destructible<T>::value) {
359  return;
360  }
361  // Destroy functor.
362  auto destroy_function = [](T *start, T *end) {
363  for (; start != end; ++start) {
364  start->~T();
365  }
366  };
367  if (n_threads <= 1u) {
368  destroy_function(ptr,ptr + size);
369  } else {
370  // A vector of ranges representing elements yet to be destroyed in case something goes wrong
371  // in the multithreaded part.
372  ranges_vector d_ranges;
373  future_list<decltype(thread_pool::enqueue(0u,destroy_function,ptr,ptr))> f_list;
374  try {
375  d_ranges.resize(static_cast<rv_size_type>(n_threads),std::make_pair(ptr,ptr));
376  if (unlikely(d_ranges.size() != n_threads)) {
377  piranha_throw(std::bad_alloc,);
378  }
379  } catch (...) {
380  // Just perform the single-threaded version, and GTFO.
381  destroy_function(ptr,ptr + size);
382  return;
383  }
384  try {
385  // Work per thread.
386  const auto wpt = static_cast<std::size_t>(size / n_threads);
387  // The ranges to destroy in the cleanup phase are, initially, all the ranges.
388  for (auto i = 0u; i < n_threads; ++i) {
389  auto start = ptr + i * wpt, end = (i == n_threads - 1u) ? ptr + size : ptr + (i + 1u) * wpt;
390  d_ranges[static_cast<rv_size_type>(i)] = std::make_pair(start,end);
391  }
392  // Perform the actual destruction and update the d_ranges vector.
393  for (auto i = 0u; i < n_threads; ++i) {
394  auto start = d_ranges[static_cast<rv_size_type>(i)].first, end = d_ranges[static_cast<rv_size_type>(i)].second;
395  f_list.push_back(thread_pool::enqueue(i,destroy_function,start,end));
396  // The range needs not to be destroyed anymore, as the enqueue/push_back was successful. Replace
397  // with an empty range.
398  d_ranges[static_cast<rv_size_type>(i)].first = ptr;
399  d_ranges[static_cast<rv_size_type>(i)].second = ptr;
400  }
401  f_list.wait_all();
402  // NOTE: T is a container_element, no need to get exceptions here.
403  } catch (...) {
404  f_list.wait_all();
405  // If anything failed in the multithreaded part, just destroy in single-thread the ranges
406  // that were not destroyed.
407  for (const auto &p: d_ranges) {
408  destroy_function(p.first,p.second);
409  }
410  }
411  }
412 }

417 // Deleter functor to be used in std::unique_ptr.
418 template <typename T>
419 class parallel_deleter
420 {
421  public:
422  explicit parallel_deleter(const std::size_t &size, const unsigned &n_threads):
423  m_size(size),m_n_threads(n_threads)

463 template <typename T, typename = typename std::enable_if<is_container_element<T>::value>::type>
464 inline std::unique_ptr<T[],detail::parallel_deleter<T>> make_parallel_array(const std::size_t &size, const unsigned &n_threads)
465 {
466  if (unlikely(size > std::numeric_limits<std::size_t>::max() / sizeof(T))) {
467  piranha_throw(std::bad_alloc,);
468  }
469  // Allocate space. This could be nullptr if size is zero.
470  auto ptr = static_cast<T *>(aligned_palloc(0u,static_cast<std::size_t>(size * sizeof(T))));
471  try {
472  // No problems here with nullptr, will be a no-op.
473  parallel_value_init(ptr,size,n_threads);
474  } catch (...) {
475  piranha_assert(ptr != nullptr);
476  // Free the allocated memory. This is noexcept.
477  aligned_pfree(0u,static_cast<void *>(ptr));
478  throw;
479  }
480  return std::unique_ptr<T[],detail::parallel_deleter<T>>(ptr,detail::parallel_deleter<T>{size,n_threads});
481 }
