68 template <typename Series, typename Derived, typename = void>
69 struct base_series_multiplier_impl
70 {
71  using term_type = typename Series::term_type;
72  using container_type = typename std::decay<decltype(std::declval<Series>()._container())>::type;
73  using c_size_type = typename container_type::size_type;
74  using v_size_type = typename std::vector<term_type const *>::size_type;
75  template <typename Term, typename std::enable_if<!is_less_than_comparable<typename Term::key_type>::value,int>::type = 0>
76  void fill_term_pointers(const container_type &c1, const container_type &c2,
77  std::vector<Term const *> &v1, std::vector<Term const *> &v2)
78  {
79  // If the key is not less-than comparable, we can only copy over the pointers as they are.
80  std::transform(c1.begin(),c1.end(),std::back_inserter(v1),[](const term_type &t) {return &t;});
81  std::transform(c2.begin(),c2.end(),std::back_inserter(v2),[](const term_type &t) {return &t;});
82  }
83  template <typename Term, typename std::enable_if<is_less_than_comparable<typename Term::key_type>::value,int>::type = 0>
84  void fill_term_pointers(const container_type &c1, const container_type &c2,
85  std::vector<Term const *> &v1, std::vector<Term const *> &v2)
86  {
87  // Fetch the number of threads from the derived class.
88  const unsigned n_threads = static_cast<Derived *>(this)->m_n_threads;
89  piranha_assert(n_threads > 0u);
90  // Threading functor.
91  auto thread_func = [n_threads](unsigned thread_idx, const container_type *c, std::vector<term_type const *> *v) {
92  piranha_assert(thread_idx < n_threads);
93  // Total bucket count.
94  const auto b_count = c->bucket_count();
95  // Buckets per thread.
96  const auto bpt = b_count / n_threads;
97  // End index.
98  const auto end = static_cast<c_size_type>((thread_idx == n_threads - 1u) ? b_count :
99  (bpt * (thread_idx + 1u)));
100  // Sorter.
101  auto sorter = [](term_type const *p1, term_type const *p2) {
102  return p1->m_key < p2->m_key;
103  };
104  v_size_type j = 0u;
105  for (auto start = static_cast<c_size_type>(bpt * thread_idx); start < end; ++start) {
106  const auto &b = c->_get_bucket_list(start);
107  v_size_type tmp = 0u;
108  for (const auto &t: b) {
109  v->push_back(&t);
110  ++tmp;
111  }
112  std::stable_sort(v->data() + j, v->data() + j + tmp, sorter);
113  j += tmp;
114  }
115  };
116  if (n_threads == 1u) {
117  thread_func(0u,&c1,&v1);
118  thread_func(0u,&c2,&v2);
119  return;
120  }
121  auto thread_wrapper = [&thread_func,n_threads](const container_type *c, std::vector<term_type const *> *v) {
122  // In the multi-threaded case, each thread needs to work on a separate vector.
123  // We will merge the vectors later.
124  using vv_t = std::vector<std::vector<Term const *>>;
125  using vv_size_t = typename vv_t::size_type;
126  vv_t vv(safe_cast<vv_size_t>(n_threads));
127  // Go with the threads.
128  future_list<std::future<void>> ff_list;
129  try {
130  for (unsigned i = 0u; i < n_threads; ++i) {
131  ff_list.push_back(thread_pool::enqueue(i,thread_func,i,c,&(vv[static_cast<vv_size_t>(i)])));
132  }
133  // First let's wait for everything to finish.
134  ff_list.wait_all();
135  // Then, let's handle the exceptions.
136  ff_list.get_all();
137  } catch (...) {
138  ff_list.wait_all();
139  throw;
140  }
141  // Last, we need to merge everything into v.
142  for (const auto &vi: vv) {
143  v->insert(v->end(),vi.begin(),vi.end());
144  }
145  };
146  thread_wrapper(&c1,&v1);
147  thread_wrapper(&c2,&v2);
148  }
149 };

151 template <typename Series, typename Derived>
152 struct base_series_multiplier_impl<Series,Derived,typename std::enable_if<is_mp_rational<typename Series::term_type::cf_type>::value>::type>
153 {
154  // Useful shortcuts.
155  using term_type = typename Series::term_type;
156  using rat_type = typename term_type::cf_type;
157  using int_type = typename std::decay<decltype(std::declval<rat_type>().num())>::type;
158  using container_type = typename std::decay<decltype(std::declval<Series>()._container())>::type;
159  void fill_term_pointers(const container_type &c1, const container_type &c2,
160  std::vector<term_type const *> &v1, std::vector<term_type const *> &v2)
161  {
162  // Compute the least common multiplier.
163  m_lcm = 1;
164  auto it_f = c1.end();
165  for (auto it = c1.begin(); it != it_f; ++it) {
166  m_lcm = (m_lcm * it->m_cf.den()) / gcd(m_lcm,it->m_cf.den());
167  }
168  it_f = c2.end();
169  for (auto it = c2.begin(); it != it_f; ++it) {
170  m_lcm = (m_lcm * it->m_cf.den()) / gcd(m_lcm,it->m_cf.den());
171  }
172  // All these computations involve only positive numbers,
173  // the GCD must always be positive.
174  piranha_assert(m_lcm.sign() == 1);
175  // Copy over the terms and renormalise to lcm.
176  it_f = c1.end();
177  for (auto it = c1.begin(); it != it_f; ++it) {
178  m_terms1.push_back(term_type(rat_type(m_lcm / it->m_cf.den() * it->m_cf.num(),int_type(1)),it->m_key));
179  }
180  it_f = c2.end();
181  for (auto it = c2.begin(); it != it_f; ++it) {
182  m_terms2.push_back(term_type(rat_type(m_lcm / it->m_cf.den() * it->m_cf.num(),int_type(1)),it->m_key));
183  }
184  // Copy over the pointers.
185  std::transform(m_terms1.begin(),m_terms1.end(),std::back_inserter(v1),[](const term_type &t) {return &t;});
186  std::transform(m_terms2.begin(),m_terms2.end(),std::back_inserter(v2),[](const term_type &t) {return &t;});
187  piranha_assert(v1.size() == c1.size());
188  piranha_assert(v2.size() == c2.size());
189  }
190  std::vector<term_type> m_terms1;
191  std::vector<term_type> m_terms2;
192  int_type m_lcm;
193 };

215 // Some performance ideas:
216 // - optimisation in case one series has 1 term with unitary key and both series same type: multiply directly coefficients;
217 // - optimisation for coefficient series that merges all args, similar to the rational optimisation;

257  // Implementation of finalise().
258  template <typename T, typename std::enable_if<detail::is_mp_rational<typename T::term_type::cf_type>::value,int>::type = 0>
259  void finalise_impl(T &s) const
260  {
261  // Nothing to do if the lcm is unitary.
262  if (math::is_unitary(this->m_lcm)) {
263  return;
264  }
265  // NOTE: this has to be the square of the lcm, as in addition to uniformising
266  // the denominators in each series we are also multiplying the two series.
267  const auto l2 = this->m_lcm * this->m_lcm;
268  auto &container = s._container();
269  // Single thread implementation.
270  if (m_n_threads == 1u) {
271  for (const auto &t: container) {
272  t.m_cf._set_den(l2);
273  t.m_cf.canonicalise();
274  }
275  return;
276  }
277  // Multi-thread implementation.
278  // Buckets per thread.
279  const bucket_size_type bpt = static_cast<bucket_size_type>(container.bucket_count() / m_n_threads);
280  auto thread_func = [l2,&container,this,bpt](unsigned t_idx) {
281  bucket_size_type start_idx = static_cast<bucket_size_type>(t_idx * bpt);
282  // Special handling for the last thread.
283  const bucket_size_type end_idx = t_idx == (this->m_n_threads - 1u) ? container.bucket_count() :
284  static_cast<bucket_size_type>((t_idx + 1u) * bpt);
285  for (; start_idx != end_idx; ++start_idx) {
286  auto &list = container._get_bucket_list(start_idx);
287  for (const auto &t: list) {
288  t.m_cf._set_den(l2);
289  t.m_cf.canonicalise();
290  }
291  }
292  };
293  // Go with the threads.
294  future_list<decltype(thread_pool::enqueue(0u,thread_func,0u))> ff_list;
295  try {
296  for (unsigned i = 0u; i < m_n_threads; ++i) {
297  ff_list.push_back(thread_pool::enqueue(i,thread_func,i));
298  }
299  // First let's wait for everything to finish.
300  ff_list.wait_all();
301  // Then, let's handle the exceptions.
302  ff_list.get_all();
303  } catch (...) {
304  ff_list.wait_all();
305  throw;
306  }
307  }

401  template <typename MultFunctor, typename LimitFunctor>
402  void blocked_multiplication(const MultFunctor &mf, const size_type &start1, const size_type &end1,
403  const LimitFunctor &lf) const
404  {
405  PIRANHA_TT_CHECK(is_function_object,MultFunctor,void,const size_type &, const size_type &);
406  if (unlikely(start1 > end1 || start1 > m_v1.size() || end1 > m_v1.size())) {
407  piranha_throw(std::invalid_argument,"invalid bounds in blocked_multiplication");
408  }
409  // Block size and number of regular blocks.
410  const size_type bsize = safe_cast<size_type>(tuning::get_multiplication_block_size()),
411  nblocks1 = static_cast<size_type>((end1 - start1) / bsize),
412  nblocks2 = static_cast<size_type>(m_v2.size() / bsize);
413  // Start and end of last (possibly irregular) blocks.
414  const size_type i_ir_start = static_cast<size_type>(nblocks1 * bsize + start1), i_ir_end = end1;
415  const size_type j_ir_start = static_cast<size_type>(nblocks2 * bsize), j_ir_end = m_v2.size();
416  for (size_type n1 = 0u; n1 < nblocks1; ++n1) {
417  const size_type i_start = static_cast<size_type>(n1 * bsize + start1),
418  i_end = static_cast<size_type>(i_start + bsize);
419  // regulars1 * regulars2
420  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
421  const size_type j_start = static_cast<size_type>(n2 * bsize),
422  j_end = static_cast<size_type>(j_start + bsize);
423  for (size_type i = i_start; i < i_end; ++i) {
424  const size_type limit = std::min<size_type>(lf(i),j_end);
425  for (size_type j = j_start; j < limit; ++j) {
426  mf(i,j);
427  }
428  }
429  }
430  // regulars1 * rem2
431  for (size_type i = i_start; i < i_end; ++i) {
432  const size_type limit = std::min<size_type>(lf(i),j_ir_end);
433  for (size_type j = j_ir_start; j < limit; ++j) {
434  mf(i,j);
435  }
436  }
437  }
438  // rem1 * regulars2
439  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
440  const size_type j_start = static_cast<size_type>(n2 * bsize),
441  j_end = static_cast<size_type>(j_start + bsize);
442  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
443  const size_type limit = std::min<size_type>(lf(i),j_end);
444  for (size_type j = j_start; j < limit; ++j) {
445  mf(i,j);
446  }
447  }
448  }
449  // rem1 * rem2.
450  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
451  const size_type limit = std::min<size_type>(lf(i),j_ir_end);
452  for (size_type j = j_ir_start; j < limit; ++j) {
453  mf(i,j);
454  }
455  }
456  }

513  template <std::size_t MultArity, typename MultFunctor, typename LimitFunctor>
514  bucket_size_type estimate_final_series_size(const LimitFunctor &lf) const
515  {
516  PIRANHA_TT_CHECK(is_function_object,MultFunctor,void,const size_type &, const size_type &);
517  PIRANHA_TT_CHECK(std::is_constructible,MultFunctor,const base_series_multiplier &, Series &);
518  PIRANHA_TT_CHECK(is_function_object,LimitFunctor,size_type,const size_type &);
519  // Cache these.
520  const size_type size1 = m_v1.size(), size2 = m_v2.size();
521  constexpr std::size_t result_size = MultArity;
522  // If one of the two series is empty, just return 0.
523  if (unlikely(!size1 || !size2)) {
524  return 1u;
525  }
526  // If either series has a size of 1, just return size1 * size2 * result_size.
527  if (size1 == 1u || size2 == 1u) {
528  return static_cast<bucket_size_type>(integer(size1) * size2 * result_size);
529  }
530  // NOTE: Hard-coded number of trials.
531  // NOTE: here consider that in case of extremely sparse series with few terms this will incur in noticeable
532  // overhead, since we will need many term-by-term before encountering the first duplicate.
533  const unsigned n_trials = 15u;
534  // NOTE: Hard-coded value for the estimation multiplier.
535  // NOTE: This value should be tuned for performance/memory usage tradeoffs.
536  const unsigned multiplier = 2u;
537  // Number of threads to use. If there are more threads than trials, then reduce
538  // the number of actual threads to use.
539  // NOTE: this is a bit different from usual, where we do not care if the workload per thread is zero.
540  // We do like this because n_trials is a small number and there still seems to be benefit in running
541  // just 1 trial per thread.
542  const unsigned n_threads = (n_trials >= m_n_threads) ? m_n_threads : n_trials;
543  piranha_assert(n_threads > 0u);
544  // Trials per thread. This will always be at least 1.
545  const unsigned tpt = n_trials / n_threads;
546  piranha_assert(tpt >= 1u);
547  // The cumulative estimate.
548  integer c_estimate(0);
549  // Sync mutex - actually used only in multithreading.
550  std::mutex mut;
551  // The estimation functor.
552  auto estimator = [&lf,size1,n_threads,multiplier,tpt,n_trials,this,&c_estimate,&mut,result_size](unsigned thread_idx) {
553  piranha_assert(thread_idx < n_threads);
554  // Vectors of indices into m_v1.
555  std::vector<size_type> v_idx1(safe_cast<typename std::vector<size_type>::size_type>(size1));
556  std::iota(v_idx1.begin(),v_idx1.end(),size_type(0));
557  // Copy in order to reset to initial state later.
558  const auto v_idx1_copy = v_idx1;
559  // Random number engine.
560  std::mt19937 engine;
561  // Uniform int distribution.
562  using dist_type = std::uniform_int_distribution<size_type>;
563  dist_type dist;
564  // Init the accumulated estimation for averaging later.
565  integer acc(0);
566  // Number of trials for this thread - usual special casing for the last thread.
567  const unsigned cur_trials = (thread_idx == n_threads - 1u) ? (n_trials - thread_idx * tpt) : tpt;
568  // This should always be guaranteed because tpt is never 0.
569  piranha_assert(cur_trials > 0u);
570  // Create and setup the temp series.
571  Series tmp;
572  tmp.set_symbol_set(m_ss);
573  // Create the multiplier.
574  MultFunctor mf(*this,tmp);
575  // Go with the trials.
576  for (auto n = 0u; n < cur_trials; ++n) {
577  // Seed the engine. The seed should be the global trial number, accounting for multiple
578  // threads. This way the estimation will not depend on the number of threads.
579  engine.seed(static_cast<std::mt19937::result_type>(tpt * thread_idx + n));
580  // Reset the indices vector and re-randomise it.
581  // NOTE: we need to do this as every run inside this for loop must be completely independent
582  // of any previous run, we cannot keep any state.
583  v_idx1 = v_idx1_copy;
584  std::shuffle(v_idx1.begin(),v_idx1.end(),engine);
585  // The counter. This will be increased each time a term-by-term multiplication
586  // does not generate a duplicate term.
587  size_type count = 0u;
588  // This will be used to determine the average number of terms in s2
589  // that participate in the multiplication.
590  integer acc_s2(0);
591  auto it1 = v_idx1.begin();
592  for (; it1 != v_idx1.end(); ++it1) {
593  // Get the limit idx in s2.
594  const size_type limit = lf(*it1);
595  // This is the upper limit of an open ended interval, so it needs
596  // to be decreased by one in order to be used in dist. If zero, it means
597  // there are no terms in v2 that can be multiplied by the current term in t1.
598  if (limit == 0u) {
599  continue;
600  }
601  acc_s2 += limit;
602  // Pick a random index in m_v2 within the limit.
603  const size_type idx2 = dist(engine,typename dist_type::param_type(static_cast<size_type>(0u),
604  static_cast<size_type>(limit - 1u)));
605  // Perform term multiplication.
606  mf(*it1,idx2);
607  // Check for unlikely overflows when increasing count.
608  if (unlikely(result_size > std::numeric_limits<size_type>::max() ||
609  count > std::numeric_limits<size_type>::max() - result_size))
610  {
611  piranha_throw(std::overflow_error,"overflow error");
612  }
613  if (tmp.size() != count + result_size) {
614  break;
615  }
616  // Increase cycle variables.
617  count = static_cast<size_type>(count + result_size);
618  }
619  integer add;
620  if (it1 == v_idx1.end()) {
621  // We never found a duplicate. count is now the number of terms in s1
622  // which actually participate in the multiplication, while acc_s2 / count
623  // is the average number of terms in s2 that participate in the multiplication.
624  // The result will be then count * acc_s2 / count = acc_s2.
625  add = acc_s2;
626  } else {
627  // If we found a duplicate, we use the heuristic.
628  add = integer(multiplier) * count * count;
629  }
630  // Fix if zero, so that the average later never results in zero.
631  if (add.sign() == 0) {
632  add = 1;
633  }
634  acc += add;
635  // Reset tmp.
636  tmp._container().clear();
637  }
638  // Accumulate in the shared variable.
639  if (n_threads == 1u) {
640  // No locking needed.
641  c_estimate += acc;
642  } else {
643  std::lock_guard<std::mutex> lock(mut);
644  c_estimate += acc;
645  }
646  };
647  // Run the estimation functor.
648  if (n_threads == 1u) {
649  estimator(0u);
650  } else {
651  future_list<std::future<void>> f_list;
652  try {
653  for (unsigned i = 0u; i < n_threads; ++i) {
654  f_list.push_back(thread_pool::enqueue(i,estimator,i));
655  }
656  // First let's wait for everything to finish.
657  f_list.wait_all();
658  // Then, let's handle the exceptions.
659  f_list.get_all();
660  } catch (...) {
661  f_list.wait_all();
662  throw;
663  }
664  }
665  piranha_assert(c_estimate >= n_trials);
666  // Return the mean.
667  return static_cast<bucket_size_type>(c_estimate / n_trials);
668  }

796  static void sanitise_series(Series &retval, unsigned n_threads)
797  {
798  using term_type = typename Series::term_type;
799  if (unlikely(n_threads == 0u)) {
800  piranha_throw(std::invalid_argument,"invalid number of threads");
801  }
802  auto &container = retval._container();
803  const auto &args = retval.get_symbol_set();
804  // Reset the size to zero before doing anything.
805  container._update_size(static_cast<bucket_size_type>(0u));
806  // Single-thread implementation.
807  if (n_threads == 1u) {
808  const auto it_end = container.end();
809  for (auto it = container.begin(); it != it_end;) {
810  if (unlikely(!it->is_compatible(args))) {
811  piranha_throw(std::invalid_argument,"incompatible term");
812  }
813  if (unlikely(container.size() == std::numeric_limits<bucket_size_type>::max())) {
814  piranha_throw(std::overflow_error,"overflow error in the number of terms of a series");
815  }
816  // First update the size, it will be scaled back in the erase() method if necessary.
817  container._update_size(static_cast<bucket_size_type>(container.size() + 1u));
818  if (unlikely(it->is_ignorable(args))) {
819  it = container.erase(it);
820  } else {
821  ++it;
822  }
823  }
824  return;
825  }
826  // Multi-thread implementation.
827  const auto b_count = container.bucket_count();
828  std::mutex m;
829  integer global_count(0);
830  auto eraser = [b_count,&container,&m,&args,&global_count](const bucket_size_type &start, const bucket_size_type &end) {
831  piranha_assert(start <= end && end <= b_count);
832  (void)b_count;
833  bucket_size_type count = 0u;
834  std::vector<term_type> term_list;
835  // Examine and count the terms bucket-by-bucket.
836  for (bucket_size_type i = start; i != end; ++i) {
837  term_list.clear();
838  const auto &bl = container._get_bucket_list(i);
839  const auto it_f = bl.end();
840  for (auto it = bl.begin(); it != it_f; ++it) {
841  // Check first for compatibility.
842  if (unlikely(!it->is_compatible(args))) {
843  piranha_throw(std::invalid_argument,"incompatible term");
844  }
845  // Check for ignorability.
846  if (unlikely(it->is_ignorable(args))) {
847  term_list.push_back(*it);
848  }
849  // Update the count of terms.
850  if (unlikely(count == std::numeric_limits<bucket_size_type>::max())) {
851  piranha_throw(std::overflow_error,"overflow error in the number of terms of a series");
852  }
853  count = static_cast<bucket_size_type>(count + 1u);
854  }
855  for (auto it = term_list.begin(); it != term_list.end(); ++it) {
856  // NOTE: must use _erase to avoid concurrent modifications
857  // to the number of elements in the table.
858  container._erase(container._find(*it,i));
859  // Account for the erased term.
860  piranha_assert(count > 0u);
861  count = static_cast<bucket_size_type>(count - 1u);
862  }
863  }
864  // Update the global count.
865  std::lock_guard<std::mutex> lock(m);
866  global_count += count;
867  };
868  future_list<decltype(thread_pool::enqueue(0u,eraser,bucket_size_type(),bucket_size_type()))> f_list;
869  try {
870  for (unsigned i = 0u; i < n_threads; ++i) {
871  const auto start = static_cast<bucket_size_type>((b_count / n_threads) * i),
872  end = static_cast<bucket_size_type>((i == n_threads - 1u) ? b_count : (b_count / n_threads) * (i + 1u));
873  f_list.push_back(thread_pool::enqueue(i,eraser,start,end));
874  }
875  // First let's wait for everything to finish.
876  f_list.wait_all();
877  // Then, let's handle the exceptions.
878  f_list.get_all();
879  } catch (...) {
880  f_list.wait_all();
881  // NOTE: there's not need to clear retval here - it was already in an inconsistent
882  // state coming into this method. We rather need to make sure sanitise_series() is always
883  // called in a try/catch block that clears retval in case of errors.
884  throw;
885  }
886  // Final update of the total count.
887  container._update_size(static_cast<bucket_size_type>(global_count));
888  }

921  template <typename LimitFunctor>
922  Series plain_multiplication(const LimitFunctor &lf) const
923  {
924  // Shortcuts.
925  using term_type = typename Series::term_type;
926  using cf_type = typename term_type::cf_type;
927  using key_type = typename term_type::key_type;
928  PIRANHA_TT_CHECK(key_is_multipliable,cf_type,key_type);
929  constexpr std::size_t m_arity = key_type::multiply_arity;
930  // Setup the return value with the merged symbol set.
931  Series retval;
932  retval.set_symbol_set(m_ss);
933  // Do not do anything if one of the two series is empty.
934  if (unlikely(m_v1.empty() || m_v2.empty())) {
935  return retval;
936  }
937  const size_type size1 = m_v1.size(), size2 = m_v2.size();
938  piranha_assert(size1 && size2);
939  // Convert n_threads to size_type for convenience.
940  const size_type n_threads = safe_cast<size_type>(m_n_threads);
941  piranha_assert(n_threads);
942  // Determine if we should estimate the size. We check the threshold, but we always
943  // need to estimate in multithreaded mode.
944  bool estimate = true;
945  const auto e_thr = tuning::get_estimate_threshold();
946  if (integer(m_v1.size()) * m_v2.size() < integer(e_thr) * e_thr && n_threads == 1u) {
947  estimate = false;
948  }
949  if (estimate) {
950  // Estimate and rehash.
951  const auto est = estimate_final_series_size<m_arity,plain_multiplier<false>>(lf);
952  // NOTE: use numeric cast here as safe_cast is expensive, going through an integer-double conversion,
953  // and in this case the behaviour of numeric_cast is appropriate.
954  const auto n_buckets = boost::numeric_cast<bucket_size_type>(std::ceil(static_cast<double>(est)
955  / retval._container().max_load_factor()));
956  piranha_assert(n_buckets > 0u);
957  // Check if we want to use the parallel memory set.
958  // NOTE: it is important here that we use the same n_threads for multiplication and memset as
959  // we tie together pinned threads with potentially different NUMA regions.
960  const unsigned n_threads_rehash = tuning::get_parallel_memory_set() ?
961  static_cast<unsigned>(n_threads) : 1u;
962  retval._container().rehash(n_buckets,n_threads_rehash);
963  }
964  if (n_threads == 1u) {
965  try {
966  // Single-thread case.
967  if (estimate) {
968  blocked_multiplication(plain_multiplier<true>(*this,retval),0u,size1,lf);
969  // If we estimated beforehand, we need to sanitise the series.
970  sanitise_series(retval,static_cast<unsigned>(n_threads));
971  } else {
972  blocked_multiplication(plain_multiplier<false>(*this,retval),0u,size1,lf);
973  }
974  finalise_series(retval);
975  return retval;
976  } catch (...) {
977  retval._container().clear();
978  throw;
979  }
980  }
981  // Multi-threaded case.
982  piranha_assert(estimate);
983  // Init the vector of spinlocks.
984  detail::atomic_flag_array sl_array(safe_cast<std::size_t>(retval._container().bucket_count()));
985  // Init the future list.
986  future_list<std::future<void>> f_list;
987  // Thread block size.
988  const auto block_size = size1 / n_threads;
989  try {
990  for (size_type idx = 0u; idx < n_threads; ++idx) {
991  // Thread functor.
992  auto tf = [idx,this,block_size,n_threads,&sl_array,&retval,&lf]()
993  {
994  // Used to store the result of term multiplication.
995  std::array<term_type,key_type::multiply_arity> tmp_t;
996  // End of retval container (thread-safe).
997  const auto c_end = retval._container().end();
998  // Block functor.
999  // NOTE: this is very similar to the plain functor, but it does the bucket locking
1000  // additionally.
1001  auto f = [&c_end,&tmp_t,this,&retval,&sl_array] (const size_type &i, const size_type &j)
1002  {
1003  // Run the term multiplication.
1004  key_type::multiply(tmp_t,*(this->m_v1[i]),*(this->m_v2[j]),retval.get_symbol_set());
1005  for (std::size_t n = 0u; n < key_type::multiply_arity; ++n) {
1006  auto &container = retval._container();
1007  auto &tmp_term = tmp_t[n];
1008  // Try to locate the term into retval.
1009  auto bucket_idx = container._bucket(tmp_term);
1010  // Lock the bucket.
1011  detail::atomic_lock_guard alg(sl_array[static_cast<std::size_t>(bucket_idx)]);
1012  const auto it = container._find(tmp_term,bucket_idx);
1013  if (it == c_end) {
1014  container._unique_insert(term_insertion(tmp_term),bucket_idx);
1015  } else {
1016  it->m_cf += tmp_term.m_cf;
1017  }
1018  }
1019  };
1020  // Thread block limit.
1021  const auto e1 = (idx == n_threads - 1u) ? this->m_v1.size() :
1022  static_cast<size_type>((idx + 1u) * block_size);
1023  this->blocked_multiplication(f,static_cast<size_type>(idx * block_size),e1,lf);
1024  };
1025  f_list.push_back(thread_pool::enqueue(static_cast<unsigned>(idx),tf));
1026  }
1027  f_list.wait_all();
1028  f_list.get_all();
1029  sanitise_series(retval,static_cast<unsigned>(n_threads));
1030  finalise_series(retval);
1031  } catch (...) {
1032  f_list.wait_all();
1033  // Clean up retval as it might be in an inconsistent state.
1034  retval._container().clear();
1035  throw;
1036  }
1037  return retval;
1038  }
