67 template <typename Series, typename Derived, typename = void>
68 struct base_series_multiplier_impl {
69  using term_type = typename Series::term_type;
70  using container_type = typename std::decay<decltype(std::declval<Series>()._container())>::type;
71  using c_size_type = typename container_type::size_type;
72  using v_size_type = typename std::vector<term_type const *>::size_type;
73  template <typename Term,
74  typename std::enable_if<!is_less_than_comparable<typename Term::key_type>::value, int>::type = 0>
75  void fill_term_pointers(const container_type &c1, const container_type &c2, std::vector<Term const *> &v1,
76  std::vector<Term const *> &v2)
77  {
78  // If the key is not less-than comparable, we can only copy over the pointers as they are.
79  std::transform(c1.begin(), c1.end(), std::back_inserter(v1), [](const term_type &t) { return &t; });
80  std::transform(c2.begin(), c2.end(), std::back_inserter(v2), [](const term_type &t) { return &t; });
81  }
82  template <typename Term,
83  typename std::enable_if<is_less_than_comparable<typename Term::key_type>::value, int>::type = 0>
84  void fill_term_pointers(const container_type &c1, const container_type &c2, std::vector<Term const *> &v1,
85  std::vector<Term const *> &v2)
86  {
87  // Fetch the number of threads from the derived class.
88  const unsigned n_threads = static_cast<Derived *>(this)->m_n_threads;
89  piranha_assert(n_threads > 0u);
90  // Threading functor.
91  auto thread_func
92  = [n_threads](unsigned thread_idx, const container_type *c, std::vector<term_type const *> *v) {
93  piranha_assert(thread_idx < n_threads);
94  // Total bucket count.
95  const auto b_count = c->bucket_count();
96  // Buckets per thread.
97  const auto bpt = b_count / n_threads;
98  // End index.
99  const auto end
100  = static_cast<c_size_type>((thread_idx == n_threads - 1u) ? b_count : (bpt * (thread_idx + 1u)));
101  // Sorter.
102  auto sorter = [](term_type const *p1, term_type const *p2) { return p1->m_key < p2->m_key; };
103  v_size_type j = 0u;
104  for (auto start = static_cast<c_size_type>(bpt * thread_idx); start < end; ++start) {
105  const auto &b = c->_get_bucket_list(start);
106  v_size_type tmp = 0u;
107  for (const auto &t : b) {
108  v->push_back(&t);
109  ++tmp;
110  }
111  std::stable_sort(v->data() + j, v->data() + j + tmp, sorter);
112  j += tmp;
113  }
114  };
115  if (n_threads == 1u) {
116  thread_func(0u, &c1, &v1);
117  thread_func(0u, &c2, &v2);
118  return;
119  }
120  auto thread_wrapper = [&thread_func, n_threads](const container_type *c, std::vector<term_type const *> *v) {
121  // In the multi-threaded case, each thread needs to work on a separate vector.
122  // We will merge the vectors later.
123  using vv_t = std::vector<std::vector<Term const *>>;
124  using vv_size_t = typename vv_t::size_type;
125  vv_t vv(safe_cast<vv_size_t>(n_threads));
126  // Go with the threads.
127  future_list<void> ff_list;
128  try {
129  for (unsigned i = 0u; i < n_threads; ++i) {
130  ff_list.push_back(thread_pool::enqueue(i, thread_func, i, c, &(vv[static_cast<vv_size_t>(i)])));
131  }
132  // First let's wait for everything to finish.
133  ff_list.wait_all();
134  // Then, let's handle the exceptions.
135  ff_list.get_all();
136  } catch (...) {
137  ff_list.wait_all();
138  throw;
139  }
140  // Last, we need to merge everything into v.
141  for (const auto &vi : vv) {
142  v->insert(v->end(), vi.begin(), vi.end());
143  }
144  };
145  thread_wrapper(&c1, &v1);
146  thread_wrapper(&c2, &v2);
147  }
148 };

150 template <typename Series, typename Derived>
151 struct base_series_multiplier_impl<Series, Derived, typename std::enable_if<is_mp_rational<
152  typename Series::term_type::cf_type>::value>::type> {
153  // Useful shortcuts.
154  using term_type = typename Series::term_type;
155  using rat_type = typename term_type::cf_type;
156  using int_type = typename std::decay<decltype(std::declval<rat_type>().num())>::type;
157  using container_type = typename std::decay<decltype(std::declval<Series>()._container())>::type;
158  void fill_term_pointers(const container_type &c1, const container_type &c2, std::vector<term_type const *> &v1,
159  std::vector<term_type const *> &v2)
160  {
161  // Compute the least common multiplier.
162  m_lcm = 1;
163  auto it_f = c1.end();
164  int_type g;
165  for (auto it = c1.begin(); it != it_f; ++it) {
166  math::gcd3(g, m_lcm, it->m_cf.den());
167  math::mul3(m_lcm, m_lcm, it->m_cf.den());
168  int_type::_divexact(m_lcm, m_lcm, g);
169  }
170  it_f = c2.end();
171  for (auto it = c2.begin(); it != it_f; ++it) {
172  math::gcd3(g, m_lcm, it->m_cf.den());
173  math::mul3(m_lcm, m_lcm, it->m_cf.den());
174  int_type::_divexact(m_lcm, m_lcm, g);
175  }
176  // All these computations involve only positive numbers,
177  // the GCD must always be positive.
178  piranha_assert(m_lcm.sign() == 1);
179  // Copy over the terms and renormalise to lcm.
180  it_f = c1.end();
181  for (auto it = c1.begin(); it != it_f; ++it) {
182  // NOTE: these divisions are exact, we could take advantage of that.
183  m_terms1.push_back(term_type(rat_type(m_lcm / it->m_cf.den() * it->m_cf.num(), int_type(1)), it->m_key));
184  }
185  it_f = c2.end();
186  for (auto it = c2.begin(); it != it_f; ++it) {
187  m_terms2.push_back(term_type(rat_type(m_lcm / it->m_cf.den() * it->m_cf.num(), int_type(1)), it->m_key));
188  }
189  // Copy over the pointers.
190  std::transform(m_terms1.begin(), m_terms1.end(), std::back_inserter(v1), [](const term_type &t) { return &t; });
191  std::transform(m_terms2.begin(), m_terms2.end(), std::back_inserter(v2), [](const term_type &t) { return &t; });
192  piranha_assert(v1.size() == c1.size());
193  piranha_assert(v2.size() == c2.size());
194  }
195  std::vector<term_type> m_terms1;
196  std::vector<term_type> m_terms2;
197  int_type m_lcm;
198 };
199 }

221 // Some performance ideas:
222 // - optimisation in case one series has 1 term with unitary key and both series same type: multiply directly
223 // coefficients;
224 // - optimisation for coefficient series that merges all args, similar to the rational optimisation;
225 // - optimisation for load balancing similar to the poly multiplier.
226 template <typename Series>
227 class base_series_multiplier : private detail::base_series_multiplier_impl<Series, base_series_multiplier<Series>>

255  // The purpose of this helper is to move in a coefficient series during insertion. For series,
256  // we know that moves leave the series in a valid state, and series multiplications do not benefit
257  // from an already-constructed destination - hence it is convenient to move them rather than copy.
258  template <typename Term, typename std::enable_if<!is_series<typename Term::cf_type>::value, int>::type = 0>
259  static Term &term_insertion(Term &t)
260  {
261  return t;

264  static Term term_insertion(Term &t)
265  {

268  // Implementation of finalise().
269  template <typename T,
270  typename std::enable_if<detail::is_mp_rational<typename T::term_type::cf_type>::value, int>::type = 0>
271  void finalise_impl(T &s) const
272  {
273  // Nothing to do if the lcm is unitary.
274  if (math::is_unitary(this->m_lcm)) {
275  return;
276  }
277  // NOTE: this has to be the square of the lcm, as in addition to uniformising
278  // the denominators in each series we are also multiplying the two series.
279  const auto l2 = this->m_lcm * this->m_lcm;
280  auto &container = s._container();
281  // Single thread implementation.
282  if (m_n_threads == 1u) {
283  for (const auto &t : container) {
284  t.m_cf._set_den(l2);
285  t.m_cf.canonicalise();
286  }
287  return;
288  }
289  // Multi-thread implementation.
290  // Buckets per thread.
291  const bucket_size_type bpt = static_cast<bucket_size_type>(container.bucket_count() / m_n_threads);
292  auto thread_func = [l2, &container, this, bpt](unsigned t_idx) {
293  bucket_size_type start_idx = static_cast<bucket_size_type>(t_idx * bpt);
294  // Special handling for the last thread.
295  const bucket_size_type end_idx = t_idx == (this->m_n_threads - 1u)
296  ? container.bucket_count()
297  : static_cast<bucket_size_type>((t_idx + 1u) * bpt);
298  for (; start_idx != end_idx; ++start_idx) {
299  auto &list = container._get_bucket_list(start_idx);
300  for (const auto &t : list) {
301  t.m_cf._set_den(l2);
302  t.m_cf.canonicalise();
303  }
304  }
305  };
306  // Go with the threads.
307  future_list<decltype(thread_func(0u))> ff_list;
308  try {
309  for (unsigned i = 0u; i < m_n_threads; ++i) {
310  ff_list.push_back(thread_pool::enqueue(i, thread_func, i));
311  }
312  // First let's wait for everything to finish.
313  ff_list.wait_all();
314  // Then, let's handle the exceptions.
315  ff_list.get_all();
316  } catch (...) {
317  ff_list.wait_all();
318  throw;
319  }
320  }
321  template <typename T,
322  typename std::enable_if<!detail::is_mp_rational<typename T::term_type::cf_type>::value, int>::type = 0>
323  void finalise_impl(T &) const
324  {
325  }

357  explicit base_series_multiplier(const Series &s1, const Series &s2) : m_ss(s1.get_symbol_set())
358  {
359  if (unlikely(s1.get_symbol_set() != s2.get_symbol_set())) {
360  piranha_throw(std::invalid_argument, "incompatible arguments sets");
361  }
362  // The largest series goes first.
363  const Series *p1 = &s1, *p2 = &s2;
364  if (s1.size() < s2.size()) {
365  std::swap(p1, p2);
366  }
367  // This is just an optimisation, no troubles if there is a truncation due to static_cast.
368  m_v1.reserve(static_cast<size_type>(p1->size()));
369  m_v2.reserve(static_cast<size_type>(p2->size()));
370  container_type const *ctr1 = &p1->_container(), *ctr2 = &p2->_container();
371  // NOTE: if the zero element of Series is not absorbing, we need to create a temporary zero series in place
372  // of any factor that is zero, and then use it in the multiplication. This ensures a correct series
373  // multiplication result for coefficient types (such as IEEE floats) for which 0 times x is not necessarily
374  // always 0. The temporary zero series is stored in the m_zero_f member as a collection of 1 term with
375  // zero coefficient.
376  if (!zero_is_absorbing<Series>::value) {
377  using term_type = typename Series::term_type;
378  using cf_type = typename term_type::cf_type;
379  using key_type = typename term_type::key_type;
380  if (p1->empty()) {
381  m_zero_f1.insert(term_type{cf_type(0), key_type(s1.get_symbol_set())});
382  ctr1 = &m_zero_f1;
383  }
384  if (p2->empty()) {
385  m_zero_f2.insert(term_type{cf_type(0), key_type(s1.get_symbol_set())});
386  ctr2 = &m_zero_f2;
387  }
388  }
389  // Set the number of threads.
390  m_n_threads = (ctr1->size() && ctr2->size())
391  ? thread_pool::use_threads(integer(ctr1->size()) * ctr2->size(),
392  integer(settings::get_min_work_per_thread()))
393  : 1u;
394  this->fill_term_pointers(*ctr1, *ctr2, m_v1, m_v2);
395  }

446  template <typename MultFunctor, typename LimitFunctor>
447  void blocked_multiplication(const MultFunctor &mf, const size_type &start1, const size_type &end1,
448  const LimitFunctor &lf) const
449  {
450  PIRANHA_TT_CHECK(is_function_object, MultFunctor, void, const size_type &, const size_type &);
451  if (unlikely(start1 > end1 || start1 > m_v1.size() || end1 > m_v1.size())) {
452  piranha_throw(std::invalid_argument, "invalid bounds in blocked_multiplication");
453  }
454  // Block size and number of regular blocks.
455  const size_type bsize = safe_cast<size_type>(tuning::get_multiplication_block_size()),
456  nblocks1 = static_cast<size_type>((end1 - start1) / bsize),
457  nblocks2 = static_cast<size_type>(m_v2.size() / bsize);
458  // Start and end of last (possibly irregular) blocks.
459  const size_type i_ir_start = static_cast<size_type>(nblocks1 * bsize + start1), i_ir_end = end1;
460  const size_type j_ir_start = static_cast<size_type>(nblocks2 * bsize), j_ir_end = m_v2.size();
461  for (size_type n1 = 0u; n1 < nblocks1; ++n1) {
462  const size_type i_start = static_cast<size_type>(n1 * bsize + start1),
463  i_end = static_cast<size_type>(i_start + bsize);
464  // regulars1 * regulars2
465  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
466  const size_type j_start = static_cast<size_type>(n2 * bsize),
467  j_end = static_cast<size_type>(j_start + bsize);
468  for (size_type i = i_start; i < i_end; ++i) {
469  const size_type limit = std::min<size_type>(lf(i), j_end);
470  for (size_type j = j_start; j < limit; ++j) {
471  mf(i, j);
472  }
473  }
474  }
475  // regulars1 * rem2
476  for (size_type i = i_start; i < i_end; ++i) {
477  const size_type limit = std::min<size_type>(lf(i), j_ir_end);
478  for (size_type j = j_ir_start; j < limit; ++j) {
479  mf(i, j);
480  }
481  }
482  }
483  // rem1 * regulars2
484  for (size_type n2 = 0u; n2 < nblocks2; ++n2) {
485  const size_type j_start = static_cast<size_type>(n2 * bsize),
486  j_end = static_cast<size_type>(j_start + bsize);
487  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
488  const size_type limit = std::min<size_type>(lf(i), j_end);
489  for (size_type j = j_start; j < limit; ++j) {
490  mf(i, j);
491  }
492  }
493  }
494  // rem1 * rem2.
495  for (size_type i = i_ir_start; i < i_ir_end; ++i) {
496  const size_type limit = std::min<size_type>(lf(i), j_ir_end);
497  for (size_type j = j_ir_start; j < limit; ++j) {
498  mf(i, j);
499  }
500  }
501  }

565  template <std::size_t MultArity, typename MultFunctor, typename LimitFunctor>
566  bucket_size_type estimate_final_series_size(const LimitFunctor &lf) const
567  {
568  PIRANHA_TT_CHECK(is_function_object, MultFunctor, void, const size_type &, const size_type &);
569  PIRANHA_TT_CHECK(std::is_constructible, MultFunctor, const base_series_multiplier &, Series &);
570  PIRANHA_TT_CHECK(is_function_object, LimitFunctor, size_type, const size_type &);
571  // Cache these.
572  const size_type size1 = m_v1.size(), size2 = m_v2.size();
573  constexpr std::size_t result_size = MultArity;
574  // If one of the two series is empty, just return 0.
575  if (unlikely(!size1 || !size2)) {
576  return 1u;
577  }
578  // If either series has a size of 1, just return size1 * size2 * result_size.
579  if (size1 == 1u || size2 == 1u) {
580  return static_cast<bucket_size_type>(integer(size1) * size2 * result_size);
581  }
582  // NOTE: Hard-coded number of trials.
583  // NOTE: here consider that in case of extremely sparse series with few terms this will incur in noticeable
584  // overhead, since we will need many term-by-term before encountering the first duplicate.
585  const unsigned n_trials = 15u;
586  // NOTE: Hard-coded value for the estimation multiplier.
587  // NOTE: This value should be tuned for performance/memory usage tradeoffs.
588  const unsigned multiplier = 2u;
589  // Number of threads to use. If there are more threads than trials, then reduce
590  // the number of actual threads to use.
591  // NOTE: this is a bit different from usual, where we do not care if the workload per thread is zero.
592  // We do like this because n_trials is a small number and there still seems to be benefit in running
593  // just 1 trial per thread.
594  const unsigned n_threads = (n_trials >= m_n_threads) ? m_n_threads : n_trials;
595  piranha_assert(n_threads > 0u);
596  // Trials per thread. This will always be at least 1.
597  const unsigned tpt = n_trials / n_threads;
598  piranha_assert(tpt >= 1u);
599  // The cumulative estimate.
600  integer c_estimate(0);
601  // Sync mutex - actually used only in multithreading.
602  std::mutex mut;
603  // The estimation functor.
604  auto estimator = [&lf, size1, n_threads, multiplier, tpt, n_trials, this, &c_estimate, &mut,
605  result_size](unsigned thread_idx) {
606  piranha_assert(thread_idx < n_threads);
607  // Vectors of indices into m_v1.
608  std::vector<size_type> v_idx1(safe_cast<typename std::vector<size_type>::size_type>(size1));
609  std::iota(v_idx1.begin(), v_idx1.end(), size_type(0));
610  // Copy in order to reset to initial state later.
611  const auto v_idx1_copy = v_idx1;
612  // Random number engine.
613  std::mt19937 engine;
614  // Uniform int distribution.
615  using dist_type = std::uniform_int_distribution<size_type>;
616  dist_type dist;
617  // Init the accumulated estimation for averaging later.
618  integer acc(0);
619  // Number of trials for this thread - usual special casing for the last thread.
620  const unsigned cur_trials = (thread_idx == n_threads - 1u) ? (n_trials - thread_idx * tpt) : tpt;
621  // This should always be guaranteed because tpt is never 0.
622  piranha_assert(cur_trials > 0u);
623  // Create and setup the temp series.
624  Series tmp;
625  tmp.set_symbol_set(m_ss);
626  // Create the multiplier.
627  MultFunctor mf(*this, tmp);
628  // Go with the trials.
629  for (auto n = 0u; n < cur_trials; ++n) {
630  // Seed the engine. The seed should be the global trial number, accounting for multiple
631  // threads. This way the estimation will not depend on the number of threads.
632  engine.seed(static_cast<std::mt19937::result_type>(tpt * thread_idx + n));
633  // Reset the indices vector and re-randomise it.
634  // NOTE: we need to do this as every run inside this for loop must be completely independent
635  // of any previous run, we cannot keep any state.
636  v_idx1 = v_idx1_copy;
637  std::shuffle(v_idx1.begin(), v_idx1.end(), engine);
638  // The counter. This will be increased each time a term-by-term multiplication
639  // does not generate a duplicate term.
640  size_type count = 0u;
641  // This will be used to determine the average number of terms in s2
642  // that participate in the multiplication.
643  integer acc_s2(0);
644  auto it1 = v_idx1.begin();
645  for (; it1 != v_idx1.end(); ++it1) {
646  // Get the limit idx in s2.
647  const size_type limit = lf(*it1);
648  // This is the upper limit of an open ended interval, so it needs
649  // to be decreased by one in order to be used in dist. If zero, it means
650  // there are no terms in v2 that can be multiplied by the current term in t1.
651  if (limit == 0u) {
652  continue;
653  }
654  acc_s2 += limit;
655  // Pick a random index in m_v2 within the limit.
656  const size_type idx2
657  = dist(engine, typename dist_type::param_type(static_cast<size_type>(0u),
658  static_cast<size_type>(limit - 1u)));
659  // Perform term multiplication.
660  mf(*it1, idx2);
661  // Check for unlikely overflows when increasing count.
662  if (unlikely(result_size > std::numeric_limits<size_type>::max()
663  || count > std::numeric_limits<size_type>::max() - result_size)) {
664  piranha_throw(std::overflow_error, "overflow error");
665  }
666  if (tmp.size() != count + result_size) {
667  break;
668  }
669  // Increase cycle variables.
670  count = static_cast<size_type>(count + result_size);
671  }
672  integer add;
673  if (it1 == v_idx1.end()) {
674  // We never found a duplicate. count is now the number of terms in s1
675  // which actually participate in the multiplication, while acc_s2 / count
676  // is the average number of terms in s2 that participate in the multiplication.
677  // The result will be then count * acc_s2 / count = acc_s2.
678  add = acc_s2;
679  } else {
680  // If we found a duplicate, we use the heuristic.
681  add = integer(multiplier) * count * count;
682  }
683  // Fix if zero, so that the average later never results in zero.
684  if (add.sign() == 0) {
685  add = 1;
686  }
687  acc += add;
688  // Reset tmp.
689  tmp._container().clear();
690  }
691  // Accumulate in the shared variable.
692  if (n_threads == 1u) {
693  // No locking needed.
694  c_estimate += acc;
695  } else {
696  std::lock_guard<std::mutex> lock(mut);
697  c_estimate += acc;
698  }
699  };
700  // Run the estimation functor.
701  if (n_threads == 1u) {
702  estimator(0u);
703  } else {
704  future_list<void> f_list;
705  try {
706  for (unsigned i = 0u; i < n_threads; ++i) {
707  f_list.push_back(thread_pool::enqueue(i, estimator, i));
708  }
709  // First let's wait for everything to finish.
710  f_list.wait_all();
711  // Then, let's handle the exceptions.
712  f_list.get_all();
713  } catch (...) {
714  f_list.wait_all();
715  throw;
716  }
717  }
718  piranha_assert(c_estimate >= n_trials);
719  // Return the mean.
720  return static_cast<bucket_size_type>(c_estimate / n_trials);
721  }

859  static void sanitise_series(Series &retval, unsigned n_threads)
860  {
861  using term_type = typename Series::term_type;
862  if (unlikely(n_threads == 0u)) {
863  piranha_throw(std::invalid_argument, "invalid number of threads");
864  }
865  auto &container = retval._container();
866  const auto &args = retval.get_symbol_set();
867  // Reset the size to zero before doing anything.
868  container._update_size(static_cast<bucket_size_type>(0u));
869  // Single-thread implementation.
870  if (n_threads == 1u) {
871  const auto it_end = container.end();
872  for (auto it = container.begin(); it != it_end;) {
873  if (unlikely(!it->is_compatible(args))) {
874  piranha_throw(std::invalid_argument, "incompatible term");
875  }
876  if (unlikely(container.size() == std::numeric_limits<bucket_size_type>::max())) {
877  piranha_throw(std::overflow_error, "overflow error in the number of terms of a series");
878  }
879  // First update the size, it will be scaled back in the erase() method if necessary.
880  container._update_size(static_cast<bucket_size_type>(container.size() + 1u));
881  if (unlikely(it->is_ignorable(args))) {
882  it = container.erase(it);
883  } else {
884  ++it;
885  }
886  }
887  return;
888  }
889  // Multi-thread implementation.
890  const auto b_count = container.bucket_count();
891  std::mutex m;
892  integer global_count(0);
893  auto eraser = [b_count, &container, &m, &args, &global_count](const bucket_size_type &start,
894  const bucket_size_type &end) {
895  piranha_assert(start <= end && end <= b_count);
896  (void)b_count;
897  bucket_size_type count = 0u;
898  std::vector<term_type> term_list;
899  // Examine and count the terms bucket-by-bucket.
900  for (bucket_size_type i = start; i != end; ++i) {
901  term_list.clear();
902  const auto &bl = container._get_bucket_list(i);
903  const auto it_f = bl.end();
904  for (auto it = bl.begin(); it != it_f; ++it) {
905  // Check first for compatibility.
906  if (unlikely(!it->is_compatible(args))) {
907  piranha_throw(std::invalid_argument, "incompatible term");
908  }
909  // Check for ignorability.
910  if (unlikely(it->is_ignorable(args))) {
911  term_list.push_back(*it);
912  }
913  // Update the count of terms.
914  if (unlikely(count == std::numeric_limits<bucket_size_type>::max())) {
915  piranha_throw(std::overflow_error, "overflow error in the number of terms of a series");
916  }
917  count = static_cast<bucket_size_type>(count + 1u);
918  }
919  for (auto it = term_list.begin(); it != term_list.end(); ++it) {
920  // NOTE: must use _erase to avoid concurrent modifications
921  // to the number of elements in the table.
922  container._erase(container._find(*it, i));
923  // Account for the erased term.
924  piranha_assert(count > 0u);
925  count = static_cast<bucket_size_type>(count - 1u);
926  }
927  }
928  // Update the global count.
929  std::lock_guard<std::mutex> lock(m);
930  global_count += count;
931  };
932  future_list<decltype(eraser(bucket_size_type(), bucket_size_type()))> f_list;
933  try {
934  for (unsigned i = 0u; i < n_threads; ++i) {
935  const auto start = static_cast<bucket_size_type>((b_count / n_threads) * i),
936  end = static_cast<bucket_size_type>(
937  (i == n_threads - 1u) ? b_count : (b_count / n_threads) * (i + 1u));
938  f_list.push_back(thread_pool::enqueue(i, eraser, start, end));
939  }
940  // First let's wait for everything to finish.
941  f_list.wait_all();
942  // Then, let's handle the exceptions.
943  f_list.get_all();
944  } catch (...) {
945  f_list.wait_all();
946  // NOTE: there's not need to clear retval here - it was already in an inconsistent
947  // state coming into this method. We rather need to make sure sanitise_series() is always
948  // called in a try/catch block that clears retval in case of errors.
949  throw;
950  }
951  // Final update of the total count.
952  container._update_size(static_cast<bucket_size_type>(global_count));
953  }

991  template <typename LimitFunctor>
992  Series plain_multiplication(const LimitFunctor &lf) const
993  {
994  // Shortcuts.
995  using term_type = typename Series::term_type;
996  using cf_type = typename term_type::cf_type;
997  using key_type = typename term_type::key_type;
998  PIRANHA_TT_CHECK(key_is_multipliable, cf_type, key_type);
999  constexpr std::size_t m_arity = key_type::multiply_arity;
1000  // Setup the return value with the merged symbol set.
1001  Series retval;
1002  retval.set_symbol_set(m_ss);
1003  // Do not do anything if one of the two series is empty.
1004  if (unlikely(m_v1.empty() || m_v2.empty())) {
1005  return retval;
1006  }
1007  const size_type size1 = m_v1.size(), size2 = m_v2.size();
1008  (void)size2;
1009  piranha_assert(size1 && size2);
1010  // Convert n_threads to size_type for convenience.
1011  const size_type n_threads = safe_cast<size_type>(m_n_threads);
1012  piranha_assert(n_threads);
1013  // Determine if we should estimate the size. We check the threshold, but we always
1014  // need to estimate in multithreaded mode.
1015  bool estimate = true;
1016  const auto e_thr = tuning::get_estimate_threshold();
1017  if (integer(m_v1.size()) * m_v2.size() < integer(e_thr) * e_thr && n_threads == 1u) {
1018  estimate = false;
1019  }
1020  if (estimate) {
1021  // Estimate and rehash.
1022  const auto est = estimate_final_series_size<m_arity, plain_multiplier<false>>(lf);
1023  // NOTE: use numeric cast here as safe_cast is expensive, going through an integer-double conversion,
1024  // and in this case the behaviour of numeric_cast is appropriate.
1025  const auto n_buckets = boost::numeric_cast<bucket_size_type>(
1026  std::ceil(static_cast<double>(est) / retval._container().max_load_factor()));
1027  piranha_assert(n_buckets > 0u);
1028  // Check if we want to use the parallel memory set.
1029  // NOTE: it is important here that we use the same n_threads for multiplication and memset as
1030  // we tie together pinned threads with potentially different NUMA regions.
1031  const unsigned n_threads_rehash = tuning::get_parallel_memory_set() ? static_cast<unsigned>(n_threads) : 1u;
1032  retval._container().rehash(n_buckets, n_threads_rehash);
1033  }
1034  if (n_threads == 1u) {
1035  try {
1036  // Single-thread case.
1037  if (estimate) {
1038  blocked_multiplication(plain_multiplier<true>(*this, retval), 0u, size1, lf);
1039  // If we estimated beforehand, we need to sanitise the series.
1040  sanitise_series(retval, static_cast<unsigned>(n_threads));
1041  } else {
1042  blocked_multiplication(plain_multiplier<false>(*this, retval), 0u, size1, lf);
1043  }
1044  finalise_series(retval);
1045  return retval;
1046  } catch (...) {
1047  retval._container().clear();
1048  throw;
1049  }
1050  }
1051  // Multi-threaded case.
1052  piranha_assert(estimate);
1053  // Init the vector of spinlocks.
1054  detail::atomic_flag_array sl_array(safe_cast<std::size_t>(retval._container().bucket_count()));
1055  // Init the future list.
1056  future_list<void> f_list;
1057  // Thread block size.
1058  const auto block_size = size1 / n_threads;
1059  try {
1060  for (size_type idx = 0u; idx < n_threads; ++idx) {
1061  // Thread functor.
1062  auto tf = [idx, this, block_size, n_threads, &sl_array, &retval, &lf]() {
1063  // Used to store the result of term multiplication.
1064  std::array<term_type, key_type::multiply_arity> tmp_t;
1065  // End of retval container (thread-safe).
1066  const auto c_end = retval._container().end();
1067  // Block functor.
1068  // NOTE: this is very similar to the plain functor, but it does the bucket locking
1069  // additionally.
1070  auto f = [&c_end, &tmp_t, this, &retval, &sl_array](const size_type &i, const size_type &j) {
1071  // Run the term multiplication.
1072  key_type::multiply(tmp_t, *(this->m_v1[i]), *(this->m_v2[j]), retval.get_symbol_set());
1073  for (std::size_t n = 0u; n < key_type::multiply_arity; ++n) {
1074  auto &container = retval._container();
1075  auto &tmp_term = tmp_t[n];
1076  // Try to locate the term into retval.
1077  auto bucket_idx = container._bucket(tmp_term);
1078  // Lock the bucket.
1079  detail::atomic_lock_guard alg(sl_array[static_cast<std::size_t>(bucket_idx)]);
1080  const auto it = container._find(tmp_term, bucket_idx);
1081  if (it == c_end) {
1082  container._unique_insert(term_insertion(tmp_term), bucket_idx);
1083  } else {
1084  it->m_cf += tmp_term.m_cf;
1085  }
1086  }
1087  };
1088  // Thread block limit.
1089  const auto e1
1090  = (idx == n_threads - 1u) ? this->m_v1.size() : static_cast<size_type>((idx + 1u) * block_size);
1091  this->blocked_multiplication(f, static_cast<size_type>(idx * block_size), e1, lf);
1092  };
1093  f_list.push_back(thread_pool::enqueue(static_cast<unsigned>(idx), tf));
1094  }
1095  f_list.wait_all();
1096  f_list.get_all();
1097  sanitise_series(retval, static_cast<unsigned>(n_threads));
1098  finalise_series(retval);
1099  } catch (...) {
1100  f_list.wait_all();
1101  // Clean up retval as it might be in an inconsistent state.
1102  retval._container().clear();
1103  throw;
1104  }
1105  return retval;
1106  }
