276  // Integration with integrable coefficient.
277  template <typename T = polynomial>
278  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
279  const std::true_type &) const
280  {
281  typedef typename base::term_type term_type;
282  typedef typename term_type::cf_type cf_type;
283  typedef typename term_type::key_type key_type;
284  // Get the partial degree of the monomial in integral form.
285  integer degree;
286  const symbol_set::positions pos(this->m_symbol_set,symbol_set{s});
287  try {
288  degree = safe_cast<integer>(term.m_key.degree(pos,this->m_symbol_set));
289  } catch (const std::invalid_argument &) {
290  piranha_throw(std::invalid_argument,
291  "unable to perform polynomial integration: cannot extract the integral form of an exponent");
292  }
293  // If the degree is negative, integration by parts won't terminate.
294  if (degree.sign() < 0) {
295  piranha_throw(std::invalid_argument,
296  "unable to perform polynomial integration: negative integral exponent");
297  }
298  polynomial tmp;
299  tmp.set_symbol_set(this->m_symbol_set);
300  key_type tmp_key = term.m_key;
301  tmp.insert(term_type(cf_type(1),tmp_key));
302  i_cf_type_p<T> i_cf(math::integrate(term.m_cf,s.get_name()));
303  integrate_type<T> retval(i_cf * tmp);
304  for (integer i(1); i <= degree; ++i) {
305  // Update coefficient and key. These variables are persistent across loop iterations.
306  auto partial_key = tmp_key.partial(pos,this->m_symbol_set);
307  i_cf = math::integrate(i_cf,s.get_name()) * std::move(partial_key.first);
308  // Account for (-1)**i.
309  math::negate(i_cf);
310  // Build the other factor from the derivative of the monomial.
311  tmp = polynomial{};
312  tmp.set_symbol_set(this->m_symbol_set);
313  tmp_key = std::move(partial_key.second);
314  // NOTE: don't move tmp_key, as it needs to hold a valid value
315  // for the next loop iteration.
316  tmp.insert(term_type(cf_type(1),tmp_key));
317  retval += i_cf * tmp;
318  }
319  return retval;
320  }

521  template <typename T = polynomial>
522  integrate_type<T> integrate(const std::string &name) const
523  {
524  typedef typename base::term_type term_type;
525  typedef typename term_type::cf_type cf_type;
526  // Turn name into symbol.
527  const symbol s(name);
528  integrate_type<T> retval(0);
529  const auto it_f = this->m_container.end();
530  for (auto it = this->m_container.begin(); it != it_f; ++it) {
531  // If the derivative of the coefficient is null, we just need to deal with
532  // the integration of the key.
533  if (math::is_zero(math::partial(it->m_cf,name))) {
534  polynomial tmp;
535  symbol_set sset = this->m_symbol_set;
536  // If the variable does not exist in the arguments set, add it.
537  if (!std::binary_search(sset.begin(),sset.end(),s)) {
538  sset.add(s);
539  }
540  tmp.set_symbol_set(sset);
541  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
542  tmp.insert(term_type(cf_type(1),std::move(key_int.second)));
543  retval += (tmp * it->m_cf) / key_int.first;
544  } else {
545  retval += integrate_impl(s,*it,std::integral_constant<bool,is_integrable<cf_type>::value>());
546  }
547  }
548  return retval;

1193  Series truncated_multiplication(const T &max_degree, const Args & ... args) const
1194  {
1195  // NOTE: a possible optimisation here is the following: if the sum degrees of the arguments is less than
1196  // or equal to the max truncation degree, just do the normal multiplication - which can also then take advantage
1197  // of faster Kronecker multiplication, if the series are suitable.
1198  using term_type = typename Series::term_type;

1200  using degree_type = decltype(detail::ps_get_degree(term_type{},this->m_ss));
1201  using size_type = typename base::size_type;
1202  namespace sph = std::placeholders;
1203  static_assert(std::is_same<T,degree_type>::value,"Invalid degree type");
1204  static_assert(detail::has_get_auto_truncate_degree<Series>::value,"Invalid series type");
1205  // First let's create two vectors with the degrees of the terms in the two series.
1206  using d_size_type = typename std::vector<degree_type>::size_type;
1207  std::vector<degree_type> v_d1(safe_cast<d_size_type>(this->m_v1.size())), v_d2(safe_cast<d_size_type>(this->m_v2.size()));
1208  detail::parallel_vector_transform(this->m_n_threads,this->m_v1,v_d1,std::bind(term_degree_getter{},sph::_1,
1209  std::cref(this->m_ss),std::cref(args)...));
1210  detail::parallel_vector_transform(this->m_n_threads,this->m_v2,v_d2,std::bind(term_degree_getter{},sph::_1,
1211  std::cref(this->m_ss),std::cref(args)...));
1212  // Next we need to order the terms in the second series, and also the corresponding degree vector.
1213  // First we create a vector of indices and we fill it.
1214  std::vector<size_type> idx_vector(safe_cast<typename std::vector<size_type>::size_type>(this->m_v2.size()));
1215  std::iota(idx_vector.begin(),idx_vector.end(),size_type(0u));
1216  // Second, we sort the vector of indices according to the degrees in the second series.
1217  std::stable_sort(idx_vector.begin(),idx_vector.end(),[&v_d2](const size_type &i1, const size_type &i2) {
1218  return v_d2[static_cast<d_size_type>(i1)] < v_d2[static_cast<d_size_type>(i2)];
1219  });
1220  // Finally, we apply the permutation to v_d2 and m_v2.
1221  decltype(this->m_v2) v2_copy(this->m_v2.size());
1222  decltype(v_d2) v_d2_copy(v_d2.size());
1223  std::transform(idx_vector.begin(),idx_vector.end(),v2_copy.begin(),[this](const size_type &i) {
1224  return this->m_v2[i];
1225  });
1226  std::transform(idx_vector.begin(),idx_vector.end(),v_d2_copy.begin(),[&v_d2](const size_type &i) {
1227  return v_d2[static_cast<d_size_type>(i)];
1228  });

1230  v_d2 = std::move(v_d2_copy);

1233  auto lf = [&sl](const size_type &idx1) {
1234  return sl[static_cast<typename std::vector<size_type>::size_type>(idx1)];
1235  };

1263  template <typename T>
1264  std::vector<typename base::size_type> get_skip_limits(const std::vector<T> &v_d1, const std::vector<T> &v_d2, const T &max_degree) const
1265  {
1266  // NOTE: this can be parallelised, but we need to check the heuristic
1267  // for selecting the number of threads as it is pretty fast wrt the multiplication.
1268  // Check that we are allowed to call this method.
1269  PIRANHA_TT_CHECK(detail::has_get_auto_truncate_degree,Series);
1270  PIRANHA_TT_CHECK(std::is_same,T,decltype(math::degree(Series{})));
1271  using size_type = typename base::size_type;
1272  using d_size_type = typename std::vector<T>::size_type;
1273  piranha_assert(std::is_sorted(v_d2.begin(),v_d2.end()));
1274  // A vector of indices into the second series.
1275  std::vector<size_type> idx_vector(safe_cast<typename std::vector<size_type>::size_type>(this->m_v2.size()));
1276  std::iota(idx_vector.begin(),idx_vector.end(),size_type(0u));
1277  // The return value.
1278  std::vector<size_type> retval;
1279  for (const auto &d1: v_d1) {
1280  // Here we will find the index of the first term t2 in the second series such that
1281  // the degree d2 of t2 is > max_degree - d1, that is, d1 + d2 > max_degree.
1282  // NOTE: we need to use upper_bound, instead of lower_bound, because we need to find the first
1283  // element which is *strictly* greater than the max degree, as upper bound of a half closed
1284  // interval.
1285  // NOTE: the functor of lower_bound works inversely wrt upper_bound. See the notes on the type
1286  // requirements for the functor here:
1287  // http://en.cppreference.com/w/cpp/algorithm/upper_bound
1288  const T comp = degree_sub(max_degree,d1);
1289  const auto it = std::upper_bound(idx_vector.begin(),idx_vector.end(),comp,[&v_d2](const T &c, const size_type &idx) {
1290  return c < v_d2[static_cast<d_size_type>(idx)];
1291  });
1292  retval.push_back(it == idx_vector.end() ? static_cast<size_type>(idx_vector.size()) : *it);
1293  }
1294  // Check the consistency of the result in debug mode.
1295  auto retval_checker = [&retval,&v_d1,&v_d2,&max_degree,this]() -> bool {
1296  for (decltype(retval.size()) i = 0u; i < retval.size(); ++i) {
1297  // NOTE: this just means that all terms in s2 are within the limit.
1298  if (retval[i] == v_d2.size()) {
1299  continue;
1300  }
1301  if (retval[i] > v_d2.size()) {
1302  return false;
1303  }
1304  if (!(v_d2[static_cast<d_size_type>(retval[i])] >
1305  this->degree_sub(max_degree,v_d1[static_cast<d_size_type>(i)])))
1306  {
1307  return false;
1308  }
1309  }
1310  return true;
1311  };
1312  (void)retval_checker;
1313  piranha_assert(retval.size() == this->m_v1.size());
1314  piranha_assert(retval_checker());
1315  return retval;
1316  }

1463  auto term_cmp = [&r_bucket](term_type const *p1, term_type const *p2)
1464  {
1465  return r_bucket(p1) < r_bucket(p2);

1470  // the multiplication of the term in the first series by the first term in the block
1471  // of the second series. This is essentially the first bucket index of retval in which the task
1472  // will write.

1474  // and the max value of bucket_size_type is 2**nbits - 1.

1476  return r_bucket(v1[std::get<0u>(t1)]) + r_bucket(v2[std::get<1u>(t1)]) <

1555  // a multiple of the number of threads.

1566  // Given the [first,last[ index range in v2, find the first index idx in the v2 range such that the i-th term in v1

1586  }
