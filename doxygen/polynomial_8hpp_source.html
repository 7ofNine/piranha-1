184 // Univariate polynomial GCD via PRS-SR.
185 // Implementation based on subresultant polynomial remainder sequence:
186 // https://en.wikipedia.org/wiki/Polynomial_greatest_common_divisor
187 // See also Zippel, 8.6. This implementation is actually based on
188 // Winkler, chapter 4.
189 // NOTE: here we do not need the full sequence of remainders, consider
190 // removing the vector of polys once this is debugged and tested.
191 // Preconditions:
192 // - univariate polynomials on the same variable.
193 // Type requirements:
194 // - cf type supports exponentiation to unsigned, yielding the same type,
195 // - cf type * cf type is still cf type, and multipliable in-place,
196 // - cf type has divexact.
197 template <typename PType>
198 inline PType gcd_prs_sr(PType a, PType b)
199 {
200  using term_type = typename PType::term_type;
201  using cf_type = typename term_type::cf_type;
202  using key_type = typename term_type::key_type;
203  // Only univariate polynomials are allowed.
204  piranha_assert(a.get_symbol_set().size() == 1u && a.get_symbol_set() == b.get_symbol_set());
205  // If one of the two is zero, the gcd is the other.
206  if (math::is_zero(a) && !math::is_zero(b)) {
207  return b;
208  }
209  if (!math::is_zero(a) && math::is_zero(b)) {
210  return a;
211  }
212  // If both are zero, return zero.
213  if (math::is_zero(a) && math::is_zero(b)) {
214  PType retval;
215  retval.set_symbol_set(a.get_symbol_set());
216  return retval;
217  }
218  // Check exponents.
219  poly_expo_checker(a);
220  poly_expo_checker(b);
221  // Order so that deg(a) >= deg(b).
222  if (poly_lterm(a)->m_key < poly_lterm(b)->m_key) {
223  std::swap(a, b);
224  }
225  // Cache the arguments set. Make a copy as "a" will be moved below.
226  const auto args = a.get_symbol_set();
227  // NOTE: coefficients are alway ctible from ints.
228  cf_type h(1), g(1);
229  // Store the content of a and b for later use.
230  auto a_cont = a.content(), b_cont = b.content();
231  // NOTE: it seems like removing the content from the inputs
232  // can help performance. Let's revisit this after we implement
233  // the sorted data structure for polys.
234  // poly_exact_cf_div(a,a_cont);
235  // poly_exact_cf_div(b,b_cont);
236  std::vector<PType> F;
237  using size_type = typename std::vector<PType>::size_type;
238  F.push_back(std::move(a));
239  F.push_back(std::move(b));
240  PType fprime(F.back());
241  size_type i = 2u;
242  // A key representing the constant univariate monomial.
243  // NOTE: all monomials support construction from init list.
244  key_type zero_key{typename key_type::value_type(0)};
245  while (fprime.size() != 0u && poly_lterm(fprime)->m_key != zero_key) {
246  auto l2 = poly_lterm(F[static_cast<size_type>(i - 2u)]), l1 = poly_lterm(F[static_cast<size_type>(i - 1u)]);
247  // NOTE: we are using the degree here in order to maintain compatibility with
248  // all monomials. There is some small overhead in this, but it should not
249  // matter too much.
250  // NOTE: this will be used only on monomials with integral exponents, so it is always valid.
251  integer delta(l2->m_key.degree(args)), l1d(l1->m_key.degree(args));
252  piranha_assert(delta >= l1d);
253  // NOTE: this is a pseudo-remainder operation here, we don't use the poly method as we need the delta
254  // information below.
255  delta -= l1d;
256  poly_cf_mult(math::pow(l1->m_cf, static_cast<unsigned>(delta + 1)), F[static_cast<size_type>(i - 2u)]);
257  fprime = PType::udivrem(F[static_cast<size_type>(i - 2u)], F[static_cast<size_type>(i - 1u)]).second;
258  if (fprime.size() != 0u) {
259  const cf_type h_delta = math::pow(h, static_cast<unsigned>(delta));
260  auto tmp(fprime);
261  poly_exact_cf_div(tmp, g * h_delta);
262  g = l1->m_cf;
263  h = (math::pow(g, static_cast<unsigned>(delta)) * h);
264  math::divexact(h, h, h_delta);
265  F.push_back(std::move(tmp));
266  safe_integral_adder(i, size_type(1u));
267  }
268  }
269  auto retval = std::move(F.back());
270  auto content = retval.content();
271  poly_exact_cf_div(retval, content);
272  poly_cf_mult(math::gcd(a_cont, b_cont), retval);
273  return retval;
274 }

276 // Establish the limits of the exponents of two polynomials. Will throw if a negative exponent is encountered.
277 // Preconditions:
278 // - equal symbol sets.
279 // - non-empty polys.
280 template <typename PType>
281 inline std::vector<std::pair<typename PType::term_type::key_type::value_type,
282  typename PType::term_type::key_type::value_type>>
283 poly_establish_limits(const PType &n, const PType &d)
284 {
285  using expo_type = typename PType::term_type::key_type::value_type;
286  using v_type = std::vector<expo_type>;
287  using vp_type = std::vector<std::pair<expo_type, expo_type>>;
288  using vp_size_type = typename vp_type::size_type;
289  piranha_assert(n.get_symbol_set() == d.get_symbol_set());
290  piranha_assert(n.size() != 0u && d.size() != 0u);
291  // Return value, init with the first element from the numerator.
292  vp_type retval;
293  v_type tmp;
294  auto it = n._container().begin(), it_f = n._container().end();
295  it->m_key.extract_exponents(tmp, n.get_symbol_set());
296  std::transform(tmp.begin(), tmp.end(), std::back_inserter(retval),
297  [](const expo_type &e) { return std::make_pair(e, e); });
298  for (; it != it_f; ++it) {
299  it->m_key.extract_exponents(tmp, n.get_symbol_set());
300  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
301  // NOTE: greater-than comparability is ensured for all exponent types on which this
302  // method will be used.
303  if (tmp[i] < retval[static_cast<vp_size_type>(i)].first) {
304  retval[static_cast<vp_size_type>(i)].first = tmp[i];
305  } else if (tmp[i] > retval[static_cast<vp_size_type>(i)].second) {
306  retval[static_cast<vp_size_type>(i)].second = tmp[i];
307  }
308  }
309  }
310  // Denominator.
311  it_f = d._container().end();
312  for (it = d._container().begin(); it != it_f; ++it) {
313  it->m_key.extract_exponents(tmp, n.get_symbol_set());
314  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
315  if (tmp[i] < retval[static_cast<vp_size_type>(i)].first) {
316  retval[static_cast<vp_size_type>(i)].first = tmp[i];
317  } else if (tmp[i] > retval[static_cast<vp_size_type>(i)].second) {
318  retval[static_cast<vp_size_type>(i)].second = tmp[i];
319  }
320  }
321  }
322  // Check for negative expos.
323  if (std::any_of(retval.begin(), retval.end(),
324  [](const std::pair<expo_type, expo_type> &p) { return p.first < expo_type(0); })) {
325  piranha_throw(std::invalid_argument, "negative exponents are not allowed");
326  }
327  return retval;
328 }

338 // Convert multivariate polynomial to univariate. Preconditions:
339 // - at least 1 symbolic argument,
340 // - identical symbol set for n and d,
341 // - non-null input polys.
342 template <typename Cf, typename Key>
343 inline std::tuple<polynomial<Cf, k_monomial>, polynomial<Cf, k_monomial>, std::vector<typename k_monomial::value_type>>
344 poly_to_univariate(const polynomial<Cf, Key> &n, const polynomial<Cf, Key> &d)
345 {
346  piranha_assert(n.get_symbol_set().size() > 0u);
347  piranha_assert(n.get_symbol_set() == d.get_symbol_set());
348  piranha_assert(n.size() != 0u && d.size() != 0u);
349  using k_expo_type = typename k_monomial::value_type;
350  using expo_type = typename Key::value_type;
351  // Cache the args.
352  const auto &args = n.get_symbol_set();
353  // Establish the limits of the two polynomials.
354  auto limits = poly_establish_limits(n, d);
355  // Extract just the vector of max exponents.
356  std::vector<expo_type> max_expos;
357  std::transform(limits.begin(), limits.end(), std::back_inserter(max_expos),
358  [](const std::pair<expo_type, expo_type> &p) { return p.second; });
359  using me_size_type = decltype(max_expos.size());
360  // Next we need to check the limits of the codification. We do everything in multiprecision and
361  // then we will attempt to cast back.
362  std::vector<integer> mp_cv;
363  mp_cv.push_back(integer(1));
364  for (decltype(args.size()) i = 0u; i < args.size(); ++i) {
365  mp_cv.push_back(mp_cv.back() * (integer(1) + max_expos[static_cast<me_size_type>(i)]));
366  }
367  // Determine the max univariate expo.
368  integer mp_kM = poly_dot_product(max_expos.begin(), max_expos.end(), mp_cv.begin());
369  // Attempt casting everything down.
370  auto kM = static_cast<k_expo_type>(mp_kM);
371  (void)kM;
372  std::vector<k_expo_type> cv;
373  std::transform(mp_cv.begin(), mp_cv.end(), std::back_inserter(cv),
374  [](const integer &z) { return static_cast<k_expo_type>(z); });
375  // Proceed to the codification.
376  auto encode_poly = [&args, &cv](const polynomial<Cf, Key> &p) -> polynomial<Cf, k_monomial> {
377  using term_type = typename polynomial<Cf, k_monomial>::term_type;
378  polynomial<Cf, k_monomial> retval;
379  // The only symbol will be the first symbol from the input polys.
380  symbol_set ss;
381  ss.add(*args.begin());
382  retval.set_symbol_set(ss);
383  // Prepare rehashed.
384  retval._container().rehash(p._container().bucket_count());
385  // Go with the codification.
386  std::vector<expo_type> tmp_expos;
387  for (const auto &t : p._container()) {
388  // Extract the exponents.
389  t.m_key.extract_exponents(tmp_expos, args);
390  // NOTE: unique insertion?
391  retval.insert(term_type{t.m_cf, k_monomial(static_cast<k_expo_type>(
392  poly_dot_product(tmp_expos.begin(), tmp_expos.end(), cv.begin())))});
393  }
394  return retval;
395  };
396  return std::make_tuple(encode_poly(n), encode_poly(d), cv);
397 }

399 // From univariate to multivariate. Preconditions:
400 // - n has only 1 symbol,
401 // - coding vector and args are consistent,
402 // - non-empty poly.
403 template <typename Key, typename Cf>
404 inline polynomial<Cf, Key> poly_from_univariate(const polynomial<Cf, k_monomial> &n,
405  const std::vector<typename k_monomial::value_type> &cv,
406  const symbol_set &args)
407 {
408  piranha_assert(n.get_symbol_set().size() == 1u);
409  piranha_assert(cv.size() > 1u);
410  piranha_assert(args.size() == cv.size() - 1u);
411  piranha_assert(*n.get_symbol_set().begin() == *args.begin());
412  piranha_assert(n.size() != 0u);
413  using expo_type = typename Key::value_type;
414  using cv_size_type = decltype(cv.size());
415  auto decode_poly = [&args, &cv](const polynomial<Cf, k_monomial> &p) -> polynomial<Cf, Key> {
416  using term_type = typename polynomial<Cf, Key>::term_type;
417  // Init the return value.
418  polynomial<Cf, Key> retval;
419  retval.set_symbol_set(args);
420  retval._container().rehash(p._container().bucket_count());
421  // Go with the decodification.
422  std::vector<expo_type> tmp_expos;
423  using v_size_type = typename std::vector<expo_type>::size_type;
424  tmp_expos.resize(safe_cast<v_size_type>(args.size()));
425  for (const auto &t : p._container()) {
426  for (v_size_type i = 0u; i < tmp_expos.size(); ++i) {
427  // NOTE: use construction here rather than static_cast as expo_type could be integer.
428  tmp_expos[i] = expo_type((t.m_key.get_int() % cv[static_cast<cv_size_type>(i + 1u)])
429  / cv[static_cast<cv_size_type>(i)]);
430  }
431  retval.insert(term_type{t.m_cf, Key(tmp_expos.begin(), tmp_expos.end())});
432  }
433  return retval;
434  };
435  return decode_poly(n);
436 }

445 // This implementation is taken straight from the Geddes book. We used to have an implementation based on the paper
446 // from Liao & Fateman, which supposedly is (or was) implemented in Maple, but that implementation had a bug which
447 // showed
448 // up very rarely. Strangely enough, the sympy implementation of heugcd (also based on the Liao paper) has exactly the
449 // same
450 // issue. Thus the motivation to use the implementation from Geddes.
451 // This function returns:
452 // - a status flag, to signal if the heuristic failed (true if failure occurred),
453 // - the GCD and the 2 cofactors (a / GCD and b / GCD).
454 // The function can also fail via throwing gcdheu_failure, which is a different type of failure from the status flag
455 // (the
456 // gcdheu_failure signals that too large integers are being generated, the status flag signals that too many iterations
457 // have been run). This difference is of no consequence for the user, as the exception is caught in try_gcdheu.
458 template <typename Poly>
459 std::pair<bool, std::tuple<Poly, Poly, Poly>> gcdheu_geddes(const Poly &a, const Poly &b,
460  symbol_set::size_type s_index = 0u)
461 {
462  static_assert(is_mp_integer<typename Poly::term_type::cf_type>::value, "Invalid type.");
463  // Require identical symbol sets.
464  piranha_assert(a.get_symbol_set() == b.get_symbol_set());
465  // Helper function to compute the symmetric version of the modulo operation.
466  // See Geddes 4.2 and:
467  // http://www.maplesoft.com/support/help/maple/view.aspx?path=mod
468  auto mod_s = [](const integer &n, const integer &m) -> integer {
469  // NOTE: require that the denominator is strictly positive.
470  piranha_assert(m > 0);
471  auto retval = n % m;
472  if (retval < -((m - 1) / 2)) {
473  retval += m;
474  } else if (retval > m / 2) {
475  retval -= m;
476  }
477  return retval;
478  };
479  // Apply mod_s to all coefficients in a poly.
480  auto mod_s_poly = [mod_s](const Poly &p, const integer &m) -> Poly {
481  Poly retval(p);
482  auto it = retval._container().begin();
483  for (; it != retval._container().end();) {
484  it->m_cf = mod_s(it->m_cf, m);
485  if (math::is_zero(it->m_cf)) {
486  it = retval._container().erase(it);
487  } else {
488  ++it;
489  }
490  }
491  return retval;
492  };
493  // Cache it.
494  const auto &args = a.get_symbol_set();
495  // The contents of input polys.
496  const integer a_cont = a.content(), b_cont = b.content();
497  // Handle empty polys.
498  if (a.size() == 0u && b.size() == 0u) {
499  Poly retval;
500  retval.set_symbol_set(args);
501  return std::make_pair(false, std::make_tuple(retval, retval, retval));
502  }
503  if (a.size() == 0u) {
504  Poly retval;
505  retval.set_symbol_set(args);
506  // NOTE: 'retval + 1' could be improved performance-wise.
507  return std::make_pair(false, std::make_tuple(b, retval, retval + 1));
508  }
509  if (b.size() == 0u) {
510  Poly retval;
511  retval.set_symbol_set(args);
512  // NOTE: 'retval + 1' could be improved performance-wise.
513  return std::make_pair(false, std::make_tuple(a, retval + 1, retval));
514  }
515  // If we are at the first recursion, check the exponents.
516  if (s_index == 0u) {
517  poly_expo_checker(a);
518  poly_expo_checker(b);
519  }
520  // This is the case in which the polynomials are reduced to two integers (essentially, the zerovariate case).
521  if (a.is_single_coefficient() && b.is_single_coefficient()) {
522  piranha_assert(a.size() == 1u && b.size() == 1u);
523  Poly retval;
524  retval.set_symbol_set(args);
525  auto gab = math::gcd(a_cont, b_cont);
526  // NOTE: the creation of the gcd retval and of the cofactors can be improved performance-wise
527  // (e.g., exact division of coefficients).
528  return std::make_pair(false, std::make_tuple(retval + gab, a / gab, b / gab));
529  }
530  // s_index is the recursion index for the gcdheu() call.
531  piranha_assert(s_index < args.size());
532  // We need to work on primitive polys.
533  // NOTE: performance improvements.
534  const auto ap = a / a_cont;
535  const auto bp = b / b_cont;
536  // The current variable.
537  const std::string var = args[s_index].get_name();
538  integer xi{2 * std::min(ap.height(), bp.height()) + 2};
539  // Functor to update xi at each iteration.
540  auto update_xi = [&xi]() {
541  // NOTE: this is just a way of making xi bigger without too much correlation
542  // to its previous value.
543  xi = (xi * 73794) / 27011;
544  };
545  // NOTE: the values of 6 iterations is taken straight from the original implementation of the algorithm.
546  // The bits limit is larger than the original implementation, on the grounds that GMP should be rather efficient
547  // at bignum. It might as well be that other tuning params are better, but let's keep them hardcoded for now.
548  for (int i = 0; i < 6; ++i) {
549  if (integer(xi.bits_size()) * std::max(ap.degree({var}), bp.degree({var})) > 100000) {
550  piranha_throw(gcdheu_failure, );
551  }
552  auto res = gcdheu_geddes(ap.subs(var, xi), bp.subs(var, xi), static_cast<symbol_set::size_type>(s_index + 1u));
553  if (!res.first) {
554  Poly &gamma = std::get<0u>(res.second);
555  Poly G;
556  G.set_symbol_set(args);
557  unsigned j = 0;
558  while (gamma.size() != 0u) {
559  Poly g(mod_s_poly(gamma, xi));
560  // NOTE: term mult here could be useful, but probably it makes more
561  // sense to just improve the series multiplication routine.
562  G += g * math::pow(Poly{var}, j);
563  gamma -= g;
564  poly_exact_cf_div(gamma, xi);
565  safe_integral_adder(j, 1u);
566  }
567  // NOTE: I don't know if this is possible at all, but it's better to be safe than
568  // sorry. This prevents divisions by zero below.
569  if (G.size() == 0u) {
570  update_xi();
571  continue;
572  }
573  try {
574  // For the division test, we need to operate on the primitive
575  // part of the candidate GCD.
576  poly_exact_cf_div(G, G.content());
577  // The division test.
578  auto cf_a = ap / G;
579  auto cf_b = bp / G;
580  // Need to multiply by the GCD of the contents of a and b in order to produce
581  // the GCD with the largest coefficients.
582  const auto F = math::gcd(a_cont, b_cont);
583  poly_cf_mult(F, G);
584  // The cofactors above have been computed with the primitive parts of G and a,b.
585  // We need to rescale them in order to find the true cofactors, that is, a / G
586  // and b / G.
587  poly_cf_mult(a_cont, cf_a);
588  poly_exact_cf_div(cf_a, F);
589  poly_cf_mult(b_cont, cf_b);
590  poly_exact_cf_div(cf_b, F);
591  return std::make_pair(false, std::make_tuple(std::move(G), std::move(cf_a), std::move(cf_b)));
592  } catch (const math::inexact_division &) {
593  // Continue in case the division check fails.
594  }
595  }
596  update_xi();
597  }
598  return std::make_pair(true, std::make_tuple(Poly{}, Poly{}, Poly{}));
599 }

848  if (degree.sign() < 0) {

869  // for the next loop iteration.
870  tmp.insert(term_type(cf_type(1), tmp_key));
871  retval += i_cf * tmp;
872  }
873  return retval;
874  }

1000  // Univariate euclidean division.
1001  template <bool CheckExpos, typename T>
1002  static std::pair<polynomial, polynomial> udivrem_impl(const T &n, const T &d)
1003  {
1004  using term_type = typename base::term_type;
1005  using cf_type = typename term_type::cf_type;
1006  using key_type = typename term_type::key_type;
1007  // Only univariate polynomials are allowed.
1008  if (unlikely(n.get_symbol_set().size() != 1u || n.get_symbol_set() != d.get_symbol_set())) {
1009  piranha_throw(std::invalid_argument, "only univariate polynomials in the same variable "
1010  "are allowed in the polynomial division with remainder routine");
1011  }
1012  if (unlikely(d.size() == 0u)) {
1013  piranha_throw(zero_division_error, "univariate polynomial division by zero");
1014  }
1015  // Cache the symbol set for brevity.
1016  const auto &args = n.get_symbol_set();
1017  // If the numerator is zero, just return two empty polys.
1018  if (n.size() == 0u) {
1019  polynomial q, r;
1020  q.set_symbol_set(args);
1021  r.set_symbol_set(args);
1022  return std::make_pair(std::move(q), std::move(r));
1023  }
1024  // Check negative exponents, if requested.
1025  if (CheckExpos) {
1026  // Check if there are negative exponents.
1027  detail::poly_expo_checker(n);
1028  detail::poly_expo_checker(d);
1029  }
1030  // Initialisation: quotient is empty, remainder is the numerator.
1031  polynomial q, r(n);
1032  q.set_symbol_set(args);
1033  // Leading term of the denominator, always the same.
1034  const auto lden = detail::poly_lterm(d);
1035  piranha_assert(!math::is_zero(lden->m_cf));
1036  // Temp cf and key used for computations in the loop.
1037  cf_type tmp_cf;
1038  key_type tmp_key;
1039  while (true) {
1040  if (r.size() == 0u) {
1041  break;
1042  }
1043  // Leading term of the remainder.
1044  const auto lr = poly_lterm(r);
1045  if (lr->m_key < lden->m_key) {
1046  break;
1047  }
1048  // NOTE: we want to check that the division is exact here,
1049  // and throw if this is not the case.
1050  // NOTE: this should be made configurable to optimise the case in which we
1051  // know the division will be exact (e.g., in GCD).
1052  math::divexact(tmp_cf, lr->m_cf, lden->m_cf);
1053  key_type::divide(tmp_key, lr->m_key, lden->m_key, args);
1054  term_type t{tmp_cf, tmp_key};
1055  // NOTE: here we are basically progressively removing terms from r until
1056  // it gets to zero. This sounds exactly like the kind of situation in which
1057  // the iteration over the hash table could become really slow. This might
1058  // matter when we need to re-determine the leading term of r above.
1059  // We should consider adding a re-hashing every time we do a +/- operation on

1070  template <typename T>
1071  static T division_impl(const T &n, const T &d)
1072  {

1075  using expo_type = typename term_type::key_type::value_type;
1076  // Cache it.
1077  const auto &args = n.get_symbol_set();

1079  if (args.size() == 1u) {
1080  // NOTE: udivrem contains the check for negative exponents, zero n/d, etc.
1081  auto res = polynomial::udivrem(n, d);
1082  if (res.second.size() != 0u) {
1083  piranha_throw(math::inexact_division, );
1084  }
1085  return std::move(res.first);
1086  }
1087  // Multivariate or zerovariate case.
1088  // NOTE: here we need to check a few things as the mapping to univariate has certain prereqs.
1089  // First we check if n or d are zero.
1090  if (d.size() == 0u) {
1091  piranha_throw(zero_division_error, "polynomial division by zero");
1092  }
1093  if (n.size() == 0u) {

1095  retval.set_symbol_set(args);
1096  return retval;
1097  }

1099  // requires at least 1 var.
1100  if (args.size() == 0u) {

1107  retval.insert(term_type{std::move(tmp_cf), Key(args)});

1114  // NOTE: the check for negative exponents is done in the mapping.
1115  auto ures = std::get<0u>(umap).template udivrem_impl<false>(std::get<0u>(umap), std::get<1u>(umap));
1116  // Check if the division was exact.

1119  }

1126  piranha_assert(p.size() != 0u);

1128  it->m_key.extract_exponents(ret, args);
1129  ++it;
1130  for (; it != p._container().end(); ++it) {
1131  it->m_key.extract_exponents(tmp, args);
1132  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
1133  if (tmp[i] > ret[i]) {
1134  ret[i] = tmp[i];
1135  }
1136  }
1137  }
1138  return ret;
1139  };

1141  r_dv = degree_vector_extractor(retval);

2004  // requires a nonzero number of arguments.

2008  }
2009  if (real_a->size() == 0u) {
2010  return wrap_gcd_cofactors(*real_b, *real_a, *real_b, with_cofactors);
2011  }
2012  if (real_b->size() == 0u) {
2013  return wrap_gcd_cofactors(*real_a, *real_a, *real_b, with_cofactors);

2874  template <typename T>
2875  std::vector<typename base::size_type> _get_skip_limits(const std::vector<T> &v_d1, const std::vector<T> &v_d2,
2876  const T &max_degree) const
2877  {
2878  // NOTE: this can be parallelised, but we need to check the heuristic
2879  // for selecting the number of threads as it is pretty fast wrt the multiplication.
2880  // Check that we are allowed to call this method.
2881  PIRANHA_TT_CHECK(detail::has_get_auto_truncate_degree, Series);
2882  PIRANHA_TT_CHECK(std::is_same, T, decltype(math::degree(Series{})));
2883  using size_type = typename base::size_type;
2884  using d_size_type = typename std::vector<T>::size_type;
2885  piranha_assert(std::is_sorted(v_d2.begin(), v_d2.end()));
2886  // A vector of indices into the second series.
2887  std::vector<size_type> idx_vector(safe_cast<typename std::vector<size_type>::size_type>(this->m_v2.size()));
2888  std::iota(idx_vector.begin(), idx_vector.end(), size_type(0u));
2889  // The return value.
2890  std::vector<size_type> retval;
2891  for (const auto &d1 : v_d1) {

2893  // the degree d2 of t2 is > max_degree - d1, that is, d1 + d2 > max_degree.
2894  // NOTE: we need to use upper_bound, instead of lower_bound, because we need to find the first
2895  // element which is *strictly* greater than the max degree, as upper bound of a half closed
2896  // interval.

2898  // requirements for the functor here:

2900  const T comp = degree_sub(max_degree, d1);
2901  const auto it = std::upper_bound(
2902  idx_vector.begin(), idx_vector.end(), comp,
2903  [&v_d2](const T &c, const size_type &idx) { return c < v_d2[static_cast<d_size_type>(idx)]; });
2904  retval.push_back(it == idx_vector.end() ? static_cast<size_type>(idx_vector.size()) : *it);
2905  }

2910  if (retval[i] == v_d2.size()) {
2911  continue;
2912  }
2913  if (retval[i] > v_d2.size()) {
2914  return false;
2915  }
2916  if (!(v_d2[static_cast<d_size_type>(retval[i])]
2917  > this->degree_sub(max_degree, v_d1[static_cast<d_size_type>(i)]))) {
2918  return false;
2919  }
2920  }
2921  return true;
2922  };

2924  piranha_assert(retval.size() == this->m_v1.size());
2925  piranha_assert(retval_checker());

3075  // - the current term index from s1,
3076  // - the first term index in s2,

3089  auto term_cmp = [&r_bucket](term_type const *p1, term_type const *p2) { return r_bucket(p1) < r_bucket(p2); };

3093  // the multiplication of the term in the first series by the first term in the block
3094  // of the second series. This is essentially the first bucket index of retval in which the task
3095  // will write.
