304  // Integration with integrable coefficient.
305  template <typename T = polynomial>
306  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
307  const std::true_type &) const
308  {
309  typedef typename base::term_type term_type;
310  typedef typename term_type::cf_type cf_type;
311  typedef typename term_type::key_type key_type;
312  // Get the partial degree of the monomial in integral form.
313  integer degree;
314  const symbol_set::positions pos(this->m_symbol_set, symbol_set{s});
315  try {
316  degree = safe_cast<integer>(term.m_key.degree(pos, this->m_symbol_set));
317  } catch (const safe_cast_failure &) {
318  piranha_throw(std::invalid_argument,
319  "unable to perform polynomial integration: cannot extract the integral form of an exponent");
320  }
321  // If the degree is negative, integration by parts won't terminate.
322  if (degree.sign() < 0) {
323  piranha_throw(std::invalid_argument,
324  "unable to perform polynomial integration: negative integral exponent");
325  }
326  polynomial tmp;
327  tmp.set_symbol_set(this->m_symbol_set);
328  key_type tmp_key = term.m_key;
329  tmp.insert(term_type(cf_type(1), tmp_key));
330  i_cf_type_p<T> i_cf(math::integrate(term.m_cf, s.get_name()));
331  integrate_type<T> retval(i_cf * tmp);
332  for (integer i(1); i <= degree; ++i) {
333  // Update coefficient and key. These variables are persistent across loop iterations.
334  auto partial_key = tmp_key.partial(pos, this->m_symbol_set);
335  i_cf = math::integrate(i_cf, s.get_name()) * std::move(partial_key.first);
336  // Account for (-1)**i.
337  math::negate(i_cf);
338  // Build the other factor from the derivative of the monomial.
339  tmp = polynomial{};
340  tmp.set_symbol_set(this->m_symbol_set);
341  tmp_key = std::move(partial_key.second);
342  // NOTE: don't move tmp_key, as it needs to hold a valid value
343  // for the next loop iteration.
344  tmp.insert(term_type(cf_type(1), tmp_key));
345  retval += i_cf * tmp;
346  }
347  return retval;
348  }

447  // before calling the runner functor, which performs the actual multiplication.

449  static polynomial um_tm_implementation(const polynomial &p1, const polynomial &p2, const Functor &runner)
450  {
451  const auto &ss1 = p1.get_symbol_set(), &ss2 = p2.get_symbol_set();
452  if (ss1 == ss2) {
453  return runner(p1, p2);
454  }
455  // If the symbol sets are not the same, we need to merge them and make
456  // copies of the original operands as needed.
457  auto merge = ss1.merge(ss2);
458  const bool need_copy_1 = (merge != ss1), need_copy_2 = (merge != ss2);
459  if (need_copy_1) {
460  polynomial copy_1(p1.extend_symbol_set(merge));
461  if (need_copy_2) {
462  polynomial copy_2(p2.extend_symbol_set(merge));
463  return runner(copy_1, copy_2);
464  }
465  return runner(copy_1, p2);
466  } else {
467  polynomial copy_2(p2.extend_symbol_set(merge));
468  return runner(p1, copy_2);
469  }

625  template <typename T = polynomial>
626  integrate_type<T> integrate(const std::string &name) const
627  {
628  typedef typename base::term_type term_type;
629  typedef typename term_type::cf_type cf_type;
630  // Turn name into symbol.
631  const symbol s(name);
632  integrate_type<T> retval(0);
633  const auto it_f = this->m_container.end();
634  for (auto it = this->m_container.begin(); it != it_f; ++it) {
635  // If the derivative of the coefficient is null, we just need to deal with
636  // the integration of the key.
637  if (math::is_zero(math::partial(it->m_cf, name))) {
638  polynomial tmp;
639  symbol_set sset = this->m_symbol_set;
640  // If the variable does not exist in the arguments set, add it.
641  if (!std::binary_search(sset.begin(), sset.end(), s)) {
642  sset.add(s);
643  }
644  tmp.set_symbol_set(sset);
645  auto key_int = it->m_key.integrate(s, this->m_symbol_set);
646  tmp.insert(term_type(cf_type(1), std::move(key_int.second)));
647  retval += (tmp * it->m_cf) / key_int.first;
648  } else {
649  retval += integrate_impl(s, *it, std::integral_constant<bool, is_integrable<cf_type>::value>());
650  }
651  }
652  return retval;

1093  = vp->data() + ((t_idx == this->m_n_threads - 1u) ? vp->size() : ((t_idx + 1u) * block_size));
1094  // We need to make sure we have at least 1 element to process. This is guaranteed
1095  // in the single-threaded implementation but not in multithreading.

1102  mm_vec minmax_values;

1104  // NOTE: we can use this as we are sure the series has at least one element (start != end).
1105  std::transform((*start)->m_key.begin(), (*start)->m_key.end(), std::back_inserter(minmax_values),
1106  [](const expo_type &v) { return std::make_pair(v, v); });
1107  // Move to the next element and go with the loop.
1108  ++start;
1109  for (; start != end; ++start) {
1110  piranha_assert(monomial_checker(**start));

1112  // the same as the first or second input range:
1113  // http://stackoverflow.com/questions/19200528/is-it-safe-for-the-input-iterator-and-output-iterator-in-stdtransform-to-be-fr

1116  minmax_values.begin(), update_minmax{});
1117  }

1120  piranha_assert(mmv->empty());
1121  // Just move in the local minmax values in single-threaded mode.
1122  *mmv = std::move(minmax_values);
1123  } else {
1124  std::lock_guard<std::mutex> lock(mut);
1125  if (mmv->empty()) {

1127  *mmv = std::move(minmax_values);
1128  } else {
1129  piranha_assert(minmax_values.size() == mmv->size());

1132  = [](const std::pair<expo_type, expo_type> &p1, const std::pair<expo_type, expo_type> &p2) {

1451  template <typename T, typename... Args>
1452  Series _truncated_multiplication(const T &max_degree, const Args &... args) const
1453  {
1454  // NOTE: a possible optimisation here is the following: if the sum degrees of the arguments is less than
1455  // or equal to the max truncation degree, just do the normal multiplication - which can also then take
1456  // advantage of faster Kronecker multiplication, if the series are suitable.
1457  using term_type = typename Series::term_type;
1458  // NOTE: degree type is the same in total and partial.
1459  using degree_type = decltype(ps_get_degree(term_type{}, this->m_ss));
1460  using size_type = typename base::size_type;
1461  namespace sph = std::placeholders;
1462  static_assert(std::is_same<T, degree_type>::value, "Invalid degree type");
1463  static_assert(detail::has_get_auto_truncate_degree<Series>::value, "Invalid series type");
1464  // First let's create two vectors with the degrees of the terms in the two series.
1465  using d_size_type = typename std::vector<degree_type>::size_type;
1466  std::vector<degree_type> v_d1(safe_cast<d_size_type>(this->m_v1.size())),
1467  v_d2(safe_cast<d_size_type>(this->m_v2.size()));
1468  detail::parallel_vector_transform(
1469  this->m_n_threads, this->m_v1, v_d1,
1470  std::bind(term_degree_getter{}, sph::_1, std::cref(this->m_ss), std::cref(args)...));
1471  detail::parallel_vector_transform(
1472  this->m_n_threads, this->m_v2, v_d2,
1473  std::bind(term_degree_getter{}, sph::_1, std::cref(this->m_ss), std::cref(args)...));
1474  // Next we need to order the terms in the second series, and also the corresponding degree vector.
1475  // First we create a vector of indices and we fill it.
1476  std::vector<size_type> idx_vector(safe_cast<typename std::vector<size_type>::size_type>(this->m_v2.size()));
1477  std::iota(idx_vector.begin(), idx_vector.end(), size_type(0u));
1478  // Second, we sort the vector of indices according to the degrees in the second series.
1479  std::stable_sort(idx_vector.begin(), idx_vector.end(), [&v_d2](const size_type &i1, const size_type &i2) {
1480  return v_d2[static_cast<d_size_type>(i1)] < v_d2[static_cast<d_size_type>(i2)];
1481  });
1482  // Finally, we apply the permutation to v_d2 and m_v2.
1483  decltype(this->m_v2) v2_copy(this->m_v2.size());
1484  decltype(v_d2) v_d2_copy(v_d2.size());
1485  std::transform(idx_vector.begin(), idx_vector.end(), v2_copy.begin(),
1486  [this](const size_type &i) { return this->m_v2[i]; });
1487  std::transform(idx_vector.begin(), idx_vector.end(), v_d2_copy.begin(),
1488  [&v_d2](const size_type &i) { return v_d2[static_cast<d_size_type>(i)]; });
1489  this->m_v2 = std::move(v2_copy);
1490  v_d2 = std::move(v_d2_copy);
1491  // Now get the skip limits and we build the limits functor.
1492  const auto sl = _get_skip_limits(v_d1, v_d2, max_degree);
1493  auto lf = [&sl](const size_type &idx1) {
1494  return sl[static_cast<typename std::vector<size_type>::size_type>(idx1)];
1495  };
1496  return this->plain_multiplication(lf);
1497  }

1525  template <typename T>
1526  std::vector<typename base::size_type> _get_skip_limits(const std::vector<T> &v_d1, const std::vector<T> &v_d2,
1527  const T &max_degree) const
1528  {
1529  // NOTE: this can be parallelised, but we need to check the heuristic
1530  // for selecting the number of threads as it is pretty fast wrt the multiplication.
1531  // Check that we are allowed to call this method.
1532  PIRANHA_TT_CHECK(detail::has_get_auto_truncate_degree, Series);
1533  PIRANHA_TT_CHECK(std::is_same, T, decltype(math::degree(Series{})));
1534  using size_type = typename base::size_type;
1535  using d_size_type = typename std::vector<T>::size_type;
1536  piranha_assert(std::is_sorted(v_d2.begin(), v_d2.end()));
1537  // A vector of indices into the second series.
1538  std::vector<size_type> idx_vector(safe_cast<typename std::vector<size_type>::size_type>(this->m_v2.size()));
1539  std::iota(idx_vector.begin(), idx_vector.end(), size_type(0u));
1540  // The return value.
1541  std::vector<size_type> retval;
1542  for (const auto &d1 : v_d1) {
1543  // Here we will find the index of the first term t2 in the second series such that
1544  // the degree d2 of t2 is > max_degree - d1, that is, d1 + d2 > max_degree.
1545  // NOTE: we need to use upper_bound, instead of lower_bound, because we need to find the first
1546  // element which is *strictly* greater than the max degree, as upper bound of a half closed
1547  // interval.
1548  // NOTE: the functor of lower_bound works inversely wrt upper_bound. See the notes on the type
1549  // requirements for the functor here:
1550  // http://en.cppreference.com/w/cpp/algorithm/upper_bound
1551  const T comp = degree_sub(max_degree, d1);
1552  const auto it = std::upper_bound(
1553  idx_vector.begin(), idx_vector.end(), comp,
1554  [&v_d2](const T &c, const size_type &idx) { return c < v_d2[static_cast<d_size_type>(idx)]; });
1555  retval.push_back(it == idx_vector.end() ? static_cast<size_type>(idx_vector.size()) : *it);
1556  }
1557  // Check the consistency of the result in debug mode.
1558  auto retval_checker = [&retval, &v_d1, &v_d2, &max_degree, this]() -> bool {
1559  for (decltype(retval.size()) i = 0u; i < retval.size(); ++i) {
1560  // NOTE: this just means that all terms in s2 are within the limit.
1561  if (retval[i] == v_d2.size()) {
1562  continue;
1563  }
1564  if (retval[i] > v_d2.size()) {
1565  return false;
1566  }
1567  if (!(v_d2[static_cast<d_size_type>(retval[i])]
1568  > this->degree_sub(max_degree, v_d1[static_cast<d_size_type>(i)]))) {
1569  return false;
1570  }
1571  }
1572  return true;
1573  };
1574  (void)retval_checker;
1575  piranha_assert(retval.size() == this->m_v1.size());
1576  piranha_assert(retval_checker());
1577  return retval;
1578  }

1720  void sparse_kronecker_multiplication(Series &retval) const
1721  {
1722  using bucket_size_type = typename base::bucket_size_type;
1723  using size_type = typename base::size_type;
1724  using term_type = typename Series::term_type;
1725  // Type representing multiplication tasks:
1726  // - the current term index from s1,
1727  // - the first term index in s2,
1728  // - the last term index in s2.
1729  using task_type = std::tuple<size_type, size_type, size_type>;
1730  // Cache a few quantities.
1731  auto &v1 = this->m_v1;
1732  auto &v2 = this->m_v2;
1733  const auto size1 = v1.size();
1734  const auto size2 = v2.size();
1735  auto &container = retval._container();
1736  // A convenience functor to compute the destination bucket
1737  // of a term into retval.
1738  auto r_bucket = [&container](term_type const *p) { return container._bucket_from_hash(p->hash()); };
1739  // Sort input terms according to bucket positions in retval.
1740  auto term_cmp = [&r_bucket](term_type const *p1, term_type const *p2) { return r_bucket(p1) < r_bucket(p2); };
1741  std::stable_sort(v1.begin(), v1.end(), term_cmp);
1742  std::stable_sort(v2.begin(), v2.end(), term_cmp);
1743  // Task comparator. It will compare the bucket index of the terms resulting from
1744  // the multiplication of the term in the first series by the first term in the block
1745  // of the second series. This is essentially the first bucket index of retval in which the task
1746  // will write.
1747  // NOTE: this is guaranteed not to overflow as the max bucket size in the hash set is 2**(nbits-1),
1748  // and the max value of bucket_size_type is 2**nbits - 1.
1749  auto task_cmp = [&r_bucket, &v1, &v2](const task_type &t1, const task_type &t2) {
1750  return r_bucket(v1[std::get<0u>(t1)]) + r_bucket(v2[std::get<1u>(t1)])
1751  < r_bucket(v1[std::get<0u>(t2)]) + r_bucket(v2[std::get<1u>(t2)]);
1752  };
1753  // Task block size.
1754  const size_type block_size = safe_cast<size_type>(tuning::get_multiplication_block_size());
1755  // Task splitter: split a task in block_size sized tasks and append them to out.
1756  auto task_split = [block_size](const task_type &t, std::vector<task_type> &out) {
1757  size_type start = std::get<1u>(t), end = std::get<2u>(t);
1758  while (static_cast<size_type>(end - start) > block_size) {
1759  out.emplace_back(std::get<0u>(t), start, static_cast<size_type>(start + block_size));
1760  start = static_cast<size_type>(start + block_size);
1761  }
1762  if (end != start) {
1763  out.emplace_back(std::get<0u>(t), start, end);
1764  }
1765  };
1766  // End of the container, always the same value.
1767  const auto it_end = container.end();
1768  // Function to perform all the term-by-term multiplications in a task, using tmp_term
1769  // as a temporary value for the computation of the result.
1770  auto task_consume = [&v1, &v2, &container, it_end, this](const task_type &task, term_type &tmp_term) {
1771  // Get the term in the first series.
1772  term_type const *t1 = v1[std::get<0u>(task)];
1773  // Get pointers to the second series.
1774  term_type const **start2 = &(v2[std::get<1u>(task)]), **end2 = &(v2[std::get<2u>(task)]);
1775  // NOTE: these will have to be adapted for kd_monomial.
1776  using int_type = decltype(t1->m_key.get_int());
1777  // Get shortcuts to cf and key in t1.
1778  const auto &cf1 = t1->m_cf;
1779  const int_type key1 = t1->m_key.get_int();
1780  // Iterate over the task.
1781  for (; start2 != end2; ++start2) {
1782  // Const ref to the current term in the second series.
1783  const auto &cur = **start2;
1784  // Add the keys.
1785  // NOTE: this will have to be adapted for kd_monomial.
1786  tmp_term.m_key.set_int(static_cast<int_type>(key1 + cur.m_key.get_int()));
1787  // Try to locate the term into retval.
1788  auto bucket_idx = container._bucket(tmp_term);
1789  const auto it = container._find(tmp_term, bucket_idx);
1790  if (it == it_end) {
1791  // NOTE: for coefficient series, we might want to insert with move() below,
1792  // as we are not going to re-use the allocated resources in tmp.m_cf.
1793  // Take care of multiplying the coefficient.
1794  detail::cf_mult_impl(tmp_term.m_cf, cf1, cur.m_cf);
1795  container._unique_insert(tmp_term, bucket_idx);
1796  } else {
1797  // NOTE: here we need to decide if we want to give the same treatment to fmp as we did with
1798  // cf_mult_impl.
1799  // For the moment it is an implementation detail of this class.
1800  this->fma_wrap(it->m_cf, cf1, cur.m_cf);
1801  }
1802  }
1803  };
1804  if (this->m_n_threads == 1u) {
1805  try {
1806  // Single threaded case.
1807  // Create the vector of tasks.
1808  std::vector<task_type> tasks;
1809  for (decltype(v1.size()) i = 0u; i < size1; ++i) {
1810  task_split(std::make_tuple(i, size_type(0u), size2), tasks);
1811  }
1812  // Sort the tasks.
1813  std::stable_sort(tasks.begin(), tasks.end(), task_cmp);
1814  // Iterate over the tasks and run the multiplication.
1815  term_type tmp_term;
1816  for (const auto &t : tasks) {
1817  task_consume(t, tmp_term);
1818  }
1819  this->sanitise_series(retval, this->m_n_threads);
1820  this->finalise_series(retval);
1821  } catch (...) {
1822  retval._container().clear();
1823  throw;
1824  }
1825  return;
1826  }
1827  // Number of buckets in retval.
1828  const bucket_size_type bucket_count = container.bucket_count();
1829  // Compute the number of zones in which the output container will be subdivided,
1830  // a multiple of the number of threads.
1831  // NOTE: zm is a tuning parameter.
1832  const unsigned zm = 10u;
1833  const bucket_size_type n_zones = static_cast<bucket_size_type>(integer(this->m_n_threads) * zm);
1834  // Number of buckets per zone (can be zero).
1835  const bucket_size_type bpz = static_cast<bucket_size_type>(bucket_count / n_zones);
1836  // For each zone, we need to define a vector of tasks that will write only into that zone.
1837  std::vector<std::vector<task_type>> task_table;
1838  task_table.resize(safe_cast<decltype(task_table.size())>(n_zones));
1839  // Lower bound implementation. Adapted from:
1840  // http://en.cppreference.com/w/cpp/algorithm/lower_bound
1841  // Given the [first,last[ index range in v2, find the first index idx in the v2 range such that the i-th
1842  // term in v1 multiplied by the idx-th term in v2 will be written into retval at a bucket index not less than
1843  // zb.
1844  auto l_bound = [&v1, &v2, &r_bucket, &task_split](size_type first, size_type last, bucket_size_type zb,
1845  size_type i) -> size_type {
1846  piranha_assert(first <= last);
1847  bucket_size_type ib = r_bucket(v1[i]);
1848  // Avoid zb - ib below wrapping around.
1849  if (zb < ib) {
1850  return 0u;
1851  }
1852  const auto cmp = static_cast<bucket_size_type>(zb - ib);
1853  size_type idx, step, count = static_cast<size_type>(last - first);
1854  while (count > 0u) {
1855  idx = first;
1856  step = static_cast<size_type>(count / 2u);
1857  idx = static_cast<size_type>(idx + step);
1858  if (r_bucket(v2[idx]) < cmp) {
1859  idx = static_cast<size_type>(idx + 1u);
1860  first = idx;
1861  if (count <= step + 1u) {
1862  break;
1863  }
1864  count = static_cast<size_type>(count - (step + 1u));
1865  } else {
1866  count = step;
1867  }
1868  }
1869  return first;
1870  };
1871  // Fill the task table.
1872  auto table_filler = [&task_table, bpz, zm, this, bucket_count, size1, size2, &l_bound, &task_split,
1873  &task_cmp](const unsigned &thread_idx) {
1874  for (unsigned n = 0u; n < zm; ++n) {
1875  std::vector<task_type> cur_tasks;
1876  // [a,b[ is the container zone.
1877  bucket_size_type a = static_cast<bucket_size_type>(thread_idx * bpz * zm + n * bpz);
1878  bucket_size_type b;
1879  if (n == zm - 1u && thread_idx == this->m_n_threads - 1u) {
1880  // Special casing if this is the last zone in the container.
1881  b = bucket_count;
1882  } else {
1883  b = static_cast<bucket_size_type>(a + bpz);
1884  }
1885  // First batch of tasks.
1886  for (size_type i = 0u; i < size1; ++i) {
1887  auto t = std::make_tuple(i, l_bound(0u, size2, a, i), l_bound(0u, size2, b, i));
1888  if (std::get<1u>(t) == 0u && std::get<2u>(t) == 0u) {
1889  // This means that all the next tasks we will compute will be empty,
1890  // no sense in calculating them.
1891  break;
1892  }
1893  task_split(t, cur_tasks);
1894  }
1895  // Second batch of tasks.
1896  // Note: we can always compute a,b + bucket_count because of the limits on the maximum value of
1897  // bucket_count.
1898  for (size_type i = 0u; i < size1; ++i) {
1899  auto t = std::make_tuple(i, l_bound(0u, size2, static_cast<bucket_size_type>(a + bucket_count), i),
1900  l_bound(0u, size2, static_cast<bucket_size_type>(b + bucket_count), i));
1901  if (std::get<1u>(t) == 0u && std::get<2u>(t) == 0u) {
1902  break;
1903  }
1904  task_split(t, cur_tasks);
1905  }
1906  // Sort the task vector.
1907  std::stable_sort(cur_tasks.begin(), cur_tasks.end(), task_cmp);
1908  // Move the vector of tasks in the table.
1909  task_table[static_cast<decltype(task_table.size())>(thread_idx * zm + n)] = std::move(cur_tasks);
1910  }
1911  };
1912  // Go with the threads to fill the task table.
