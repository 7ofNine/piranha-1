185  template <typename T = trigonometric_series>
186  t_degree_type<T> t_degree() const
187  {
188  using term_type = typename T::term_type;
189  auto it = std::max_element(this->m_container.begin(),this->m_container.end(),[this](const term_type &t1, const term_type &t2) {
190  return this->get_t_degree(t1,this->m_symbol_set) < this->get_t_degree(t2,this->m_symbol_set);
191  });
192  return (it == this->m_container.end()) ? t_degree_type<T>(0) : get_t_degree(*it,this->m_symbol_set);
193  }
