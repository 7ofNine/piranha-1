192  t_degree_type<T> t_degree() const
193  {
194  using term_type = typename T::term_type;
195  auto it = std::max_element(this->m_container.begin(),this->m_container.end(),[this](const term_type &t1, const term_type &t2) {
196  return this->get_t_degree(t1,this->m_symbol_set) < this->get_t_degree(t2,this->m_symbol_set);
