48 // A simple RAII holder for an array of atomic flags. Bare minimum functionality.
49 struct atomic_flag_array
50 {
51  using value_type = std::atomic_flag;
52  // This constructor will init all the flags in the array to false.
53  explicit atomic_flag_array(const std::size_t &size):m_ptr(nullptr),m_size(size)
54  {
55  if (unlikely(size > std::numeric_limits<std::size_t>::max() / sizeof(value_type))) {
56  piranha_throw(std::bad_alloc,);
57  }
58  // NOTE: this throws if allocation fails, after this line everything is noexcept.
59  m_ptr = static_cast<value_type *>(aligned_palloc(0u,size * sizeof(value_type)));
60  for (std::size_t i = 0u; i < m_size; ++i) {
61  // NOTE: atomic_flag should support aggregate init syntax:
62  // http://en.cppreference.com/w/cpp/atomic/atomic
63  // But it results in warnings, let's avoid initialisation
64  // via ctor and just set the flag to false later.
65  ::new (static_cast<void *>(m_ptr + i)) value_type;
66  (m_ptr + i)->clear();
67  }
68  }
69  ~atomic_flag_array()
70  {
71  // atomic_flag is guaranteed to have a trivial dtor,
72  // so we can just free the storage:
73  // http://en.cppreference.com/w/cpp/atomic/atomic
74  aligned_pfree(0u,static_cast<void *>(m_ptr));
75  }
76  // Delete explicitly all other ctors/assignment operators.
77  atomic_flag_array() = delete;
78  atomic_flag_array(const atomic_flag_array &) = delete;
79  atomic_flag_array(atomic_flag_array &&) = delete;
80  atomic_flag_array &operator=(const atomic_flag_array &) = delete;
81  atomic_flag_array &operator=(atomic_flag_array &&) = delete;
82  value_type &operator[](const std::size_t &i)
83  {
84  return *(m_ptr + i);
85  }
86  const value_type &operator[](const std::size_t &i) const
87  {
88  return *(m_ptr + i);
89  }
90  // Data members.
91  value_type *m_ptr;
92  const std::size_t m_size;
93 };

95 // A simple spinlock built on top of std::atomic_flag. See for reference:
96 // http://en.cppreference.com/w/cpp/atomic/atomic_flag
97 // http://stackoverflow.com/questions/26583433/c11-implementation-of-spinlock-using-atomic
98 // The memory order specification is to squeeze out some extra performance with respect to the
99 // default behaviour of atomic types.
100 struct atomic_lock_guard
101 {
102  explicit atomic_lock_guard(std::atomic_flag &af):m_af(af)
103  {
104  while (m_af.test_and_set(std::memory_order_acquire)) {}
105  }
106  ~atomic_lock_guard()
107  {
108  m_af.clear(std::memory_order_release);
109  }
110  // Delete explicitly all other ctors/assignment operators.
111  atomic_lock_guard() = delete;
112  atomic_lock_guard(const atomic_lock_guard &) = delete;
113  atomic_lock_guard(atomic_lock_guard &&) = delete;
114  atomic_lock_guard &operator=(const atomic_lock_guard &) = delete;
115  atomic_lock_guard &operator=(atomic_lock_guard &&) = delete;
116  // Data members.
117  std::atomic_flag &m_af;
118 };
