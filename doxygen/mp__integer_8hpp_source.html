84 // Greatest common divisor using the euclidean algorithm.
85 // NOTE: this can yield negative values, depending on the signs
86 // of a and b. Supports C++ integrals and mp_integer.
87 // NOTE: using this with C++ integrals unchecked on ranges can result in undefined
88 // behaviour.
89 template <typename T>
90 inline T gcd_euclidean(T a, T b)
91 {
92  while (true) {
93  if (math::is_zero(a)) {
94  return b;
95  }
96  // NOTE: the difference in implementation here is because
97  // we want to prevent compiler warnings when T is a short int,
98  // hence the static cast. For mp_integer, the in-place version
99  // might be faster.
100  gcd_mod(b,a);
101  if (math::is_zero(b)) {
102  return a;
103  }
104  gcd_mod(a,b);
105  }
106 }

128 // Considering an array of UIn (unsigned integrals) of a certain size as a continuous sequence of bits,
129 // read the index-th URet (unsigned integral) that can be extracted from the sequence of bits.
130 // The parameter IBits is the number of upper bits of UIn that should be discarded in the computation
131 // (i.e., not considered as part of the continuous sequence of bits). RBits has the same meaning,
132 // but for the output value: it's the number of upper bits that are not considered as part of the
133 // return type.
134 template <typename URet, unsigned IBits = 0u, unsigned RBits = 0u, typename UIn,
135  typename std::enable_if<!read_uint_opt<URet,IBits,RBits,UIn>::value,int>::type = 0>
136 inline URet read_uint(const UIn *ptr, std::size_t size, std::size_t index)
137 {
138  // We can work only with unsigned integer types.
139  static_assert(std::is_integral<UIn>::value && std::is_unsigned<UIn>::value,"Invalid type.");
140  static_assert(std::is_integral<URet>::value && std::is_unsigned<URet>::value,"Invalid type.");
141  // Bit size of the return type.
142  constexpr unsigned r_bits = static_cast<unsigned>(std::numeric_limits<URet>::digits);
143  // Bit size of the input type.
144  constexpr unsigned i_bits = static_cast<unsigned>(std::numeric_limits<UIn>::digits);
145  // The ignored bits in the input type cannot be larger than or equal to its bit size.
146  static_assert(IBits < i_bits,"Invalid ignored input bits size");
147  // Same for for RBits.
148  static_assert(RBits < r_bits,"Invalid ignored output bits size");
149  // Bits effectively considered in the input type.
150  constexpr unsigned ei_bits = i_bits - IBits;
151  // Bits effectively considered in the output type.
152  constexpr unsigned er_bits = r_bits - RBits;
153  // Index in input array from where we will start reading, and bit index within
154  // that element from which the actual reading will start.
155  // NOTE: we need to protect against multiplication overflows in the upper level routine.
156  std::size_t s_index = static_cast<std::size_t>((er_bits * index) / ei_bits),
157  r_index = static_cast<std::size_t>((er_bits * index) % ei_bits);
158  // Check that we are not going to read past the end.
159  piranha_assert(s_index < size);
160  // Check for null.
161  piranha_assert(ptr != nullptr);
162  // Init the return value as 0.
163  URet retval = 0u;
164  // Bits read so far.
165  unsigned read_bits = 0u;
166  while (s_index < size && read_bits < er_bits) {
167  const unsigned unread_bits = er_bits - read_bits,
168  // This can be different from ei_bits only on the first iteration.
169  available_bits = static_cast<unsigned>(ei_bits - r_index),
170  bits_to_read = (available_bits > unread_bits) ? unread_bits : available_bits;
171  // On the first iteration, we might need to drop the first r_index bits.
172  UIn tmp = static_cast<UIn>(ptr[s_index] >> r_index);
173  // Zero out the bits we don't want to read.
174  tmp = clear_top_bits(tmp,i_bits - bits_to_read);
175  // Accumulate into return value.
176  retval = static_cast<URet>(retval + (static_cast<URet>(tmp) << read_bits));
177  // Update loop variables.
178  read_bits += bits_to_read;
179  ++s_index;
180  r_index = 0u;
181  }
182  piranha_assert(s_index == size || read_bits == er_bits);
183  return retval;
184 }

342  template <typename T, typename std::enable_if<std::is_signed<T>::value && std::is_integral<T>::value,int>::type = 0>
343  bool attempt_direct_construction(T n)
344  {
345  try {
346  if (n >= T(0)) {
347  // We can attempt conversion of a non-negative signed directly into limb_t.
348  m_limbs[0u] = boost::numeric_cast<limb_t>(n);
349  } else if (n >= -safe_abs_sint<T>::value) {
350  // Conversion of a negative signed whose value can be safely negated.
351  m_limbs[0u] = boost::numeric_cast<limb_t>(-n);
352  } else {
353  // We cannot convert safely n to a limb_t in this case.
354  return false;
355  }
356  if (n > T(0)) {
357  // Strictly positive value, size is 1.
358  _mp_size = mpz_size_t(1);
359  }
360  if (n < T(0)) {
361  // Strictly negative value, size is -1.
362  _mp_size = mpz_size_t(-1);
363  }
364  // NOTE: if n is zero, size has been inited to zero already.
365  return true;
366  } catch (...) {
367  return false;
368  }
369  }

385  template <typename Integer>
386  void construct_from_integral(Integer n)
387  {
388  // Don't do anything if n is zero or is the direct construction
389  // into the first limb succeeds.
390  if (!n || likely(attempt_direct_construction(n))) {
391  return;
392  }
393  const auto orig_n = n;
394  limb_t bit_idx = 0;
395  while (n != Integer(0)) {
396  if (bit_idx == limb_bits * 2u) {
397  // Clear out before throwing, as this is used in mp_integer as well.
398  _mp_size = 0;
399  m_limbs[0u] = 0u;
400  m_limbs[1u] = 0u;
401  piranha_throw(std::overflow_error,"insufficient bit width");
402  }
403  // NOTE: in C++11 division will round to zero always (for negative numbers as well).
404  // The bit shift operator >> has implementation defined behaviour if n is signed and negative,
405  // so we do not use it.
406  const Integer quot = static_cast<Integer>(n / Integer(2)), rem = static_cast<Integer>(n % Integer(2));
407  if (rem) {
408  set_bit(bit_idx);
409  }
410  n = quot;
411  ++bit_idx;
412  }
413  fix_sign_ctor(orig_n);
414  }

448  m_limbs[static_cast<size_type>(quot)] = static_cast<limb_t>(m_limbs[static_cast<size_type>(quot)] | static_cast<limb_t>(limb_t(1) << rem));
449  // Update the size if needed. The new size must be at least quot + 1, as we set a bit
450  // in the limb with index quot.

452  if (_mp_size < 0) {

455  }
456  } else {

458  _mp_size = new_size;
459  }
460  }

484  // Read-only mpz view class. After creation, this class can be used
485  // as const mpz_t argument in GMP functions, thanks to the implicit conversion
486  // operator.
487  template <typename T = static_integer, typename = void>
488  class static_mpz_view
489  {
490  public:
491  // Safe, checked above.
492  static const auto max_tot_nbits = 2u * T::limb_bits;
493  // Check the conversion below.
494  static_assert(max_tot_nbits / unsigned(GMP_NUMB_BITS) + 1u <= std::numeric_limits<std::size_t>::max(),
495  "Overflow error.");
496  // Number of GMP limbs to use.
497  static const std::size_t max_n_gmp_limbs = static_cast<std::size_t>(
498  max_tot_nbits % unsigned(GMP_NUMB_BITS) == 0u ?
499  max_tot_nbits / unsigned(GMP_NUMB_BITS) :
500  max_tot_nbits / unsigned(GMP_NUMB_BITS) + 1u);
501  static_assert(max_n_gmp_limbs >= 1u,"Invalid number of GMP limbs.");
502  static_mpz_view(): m_mpz(),m_limbs() {}
503  explicit static_mpz_view(const static_integer &n)
504  {
505  std::size_t asize;
506  bool sign;
507  if (n._mp_size < 0) {
508  sign = false;
509  asize = static_cast<std::size_t>(-n._mp_size);
510  } else {
511  sign = true;
512  asize = static_cast<std::size_t>(n._mp_size);
513  }
514  piranha_assert(asize <= 2u);
515  const auto tot_nbits = asize * T::limb_bits;
516  const std::size_t n_gmp_limbs = static_cast<std::size_t>(
517  tot_nbits % unsigned(GMP_NUMB_BITS) == 0u ?
518  tot_nbits / unsigned(GMP_NUMB_BITS) :
519  tot_nbits / unsigned(GMP_NUMB_BITS) + 1u);
520  // Overflow check when using read_uint.
521  static_assert(unsigned(GMP_LIMB_BITS) <= std::numeric_limits<std::size_t>::max() / max_n_gmp_limbs,
522  "Overflow error.");
523  for (std::size_t i = 0u; i < n_gmp_limbs; ++i) {
524  m_limbs[i] = read_uint< ::mp_limb_t,T::total_bits - T::limb_bits,
525  unsigned(GMP_LIMB_BITS - GMP_NUMB_BITS)>
526  (n.m_limbs.data(),asize,i);
527  }
528  // Fill in the missing limbs, otherwise we have indeterminate values.
529  for (std::size_t i = n_gmp_limbs; i < max_n_gmp_limbs; ++i) {
530  m_limbs[i] = 0u;
531  }
532  // Final assignment.
533  static_assert(max_n_gmp_limbs <= static_cast<std::make_unsigned<mpz_alloc_t>::type>
534  (std::numeric_limits<mpz_alloc_t>::max()),
535  "Overflow error.");

674  static int raw_add(static_integer &res, const static_integer &x, const static_integer &y)
675  {
676  piranha_assert(x.abs_size() <= 2 && y.abs_size() <= 2);
677  const dlimb_t lo = static_cast<dlimb_t>(static_cast<dlimb_t>(x.m_limbs[0u]) + y.m_limbs[0u]);
678  const dlimb_t hi = static_cast<dlimb_t>((static_cast<dlimb_t>(x.m_limbs[1u]) + y.m_limbs[1u]) + (lo >> limb_bits));
679  // NOTE: exit before modifying anything here, so that res is not modified.
680  if (unlikely(static_cast<limb_t>(hi >> limb_bits) != 0u)) {
681  return 1;
682  }

684  res.m_limbs[1u] = static_cast<limb_t>(hi);
685  res._mp_size = res.calculate_n_limbs();
686  res.clear_extra_bits();
687  return 0;

689  static void raw_sub(static_integer &res, const static_integer &x, const static_integer &y)
690  {
691  piranha_assert(x.abs_size() <= 2 && y.abs_size() <= 2);
692  piranha_assert(x.abs_size() >= y.abs_size());
693  piranha_assert(x.m_limbs[1u] >= y.m_limbs[1u]);
694  const bool has_borrow = x.m_limbs[0u] < y.m_limbs[0u];
695  piranha_assert(x.m_limbs[1u] > y.m_limbs[1u] || !has_borrow);
696  res.m_limbs[0u] = static_cast<limb_t>(x.m_limbs[0u] - y.m_limbs[0u]);
697  res.m_limbs[1u] = static_cast<limb_t>((x.m_limbs[1u] - y.m_limbs[1u]) - limb_t(has_borrow));
698  res._mp_size = res.calculate_n_limbs();
699  res.clear_extra_bits();
700  }

877  // lshift by n bits.
878  int lshift(::mp_bitcnt_t n)
879  {
880  // Shift by zero or on a zero value has no effect
881  // and it is always successful.
882  if (n == 0u || _mp_size == 0) {
883  return 0;
884  }
885  // Too much shift, this can never work on a nonzero value.
886  if (n >= 2u * limb_bits) {
887  return 1;
888  }
889  limb_t hi;
890  limb_t lo;
891  if (n >= limb_bits) {
892  if (m_limbs[1u] != 0u) {
893  // Too much shift.
894  return 1;
895  }
896  // Move lo to hi and set lo to zero.
897  hi = m_limbs[0u];
898  lo = 0u;
899  if (n == limb_bits) {
900  // If n is exactly limb_bits,
901  // we can assign the new hi/lo and stop here.
902  m_limbs[0u] = lo;
903  m_limbs[1u] = hi;
904  // The size has to be 2.
905  _mp_size = (_mp_size > 0) ? static_cast<mpz_size_t>(2) :
906  static_cast<mpz_size_t>(-2);
907  return 0;
908  }
909  // Moving lo to hi and setting lo to zero
910  // means we already shifted by limb_bits.
911  n = static_cast<::mp_bitcnt_t>(n - limb_bits);
912  } else {
913  // hi and lo are the original ones.
914  lo = m_limbs[0u];
915  hi = m_limbs[1u];
916  }
917  // Check that hi will not be shifted too much. Note that
918  // here and below n can never be zero, so we never shift too much.
919  piranha_assert(n > 0u);
920  if (hi >= (limb_t(1) << (limb_bits - n))) {
921  return 1;
922  }
923  // Shift hi and lo. hi gets the carry over from lo.
924  hi = static_cast<limb_t>(ulshift(hi, n) + (lo >> (limb_bits - n)));
925  // NOTE: ulshift makes sure the operation is well defined also for
926  // short integral types. This will be a no-op if lo has been set to zero
927  // above.
928  lo = ulshift(lo, n);
929  // Assign back hi and lo.
930  m_limbs[0u] = lo;
931  m_limbs[1u] = hi;
932  // Now update the size.
933  bool sign = true;
934  if (_mp_size < 0) {
935  sign = false;
936  }
937  // We know the size is at least 1 as the case of size 0 was already
938  // handled in the beginning.
939  piranha_assert(abs_size() >= 1);
940  mpz_size_t asize = 1;
941  asize = static_cast<mpz_size_t>(asize + static_cast<mpz_size_t>(m_limbs[1u] != 0u));
942  _mp_size = static_cast<mpz_size_t>(sign ? asize : -asize);
943  // Extra bits could in principle be remaining in lo after the left shift.
944  clear_extra_bits();
945  return 0;
946  }
947  // rshift by n bits.
948  void rshift(::mp_bitcnt_t n)
949  {
950  // Shift by zero or on a zero value has no effect.
951  if (n == 0u || _mp_size == 0) {
952  return;
953  }
954  // Shift by 2 * limb_bits or greater will produce zero.
955  if (n >= 2u * limb_bits) {
956  _mp_size = 0;
957  m_limbs[0u] = 0u;
958  m_limbs[1u] = 0u;
959  return;
960  }
961  if (n >= limb_bits) {
962  // NOTE: this is ok if n == limb_bits, no special casing needed.
963  m_limbs[0u] = static_cast<limb_t>(m_limbs[1u] >> (n - limb_bits));
964  m_limbs[1u] = 0u;
965  // The size could be zero or +-1, depending
966  // on the new content of m_limbs[0] and the previous
967  // sign of _mp_size.
968  _mp_size = (m_limbs[0u] == 0u) ? mpz_size_t(0) :
969  static_cast<mpz_size_t>((_mp_size > 0) ? 1 : -1);
970  return;
971  }
972  // Here we know that 0 < n < limb_bits.
973  piranha_assert(n > 0u && n < limb_bits);
974  // This represents the bits in hi that will be shifted down into lo.
975  // We move them up so we can add tmp to the new lo to account for them.
976  // NOTE: here we could have excess bits if limb_bits != total_bits, we will
977  // clean up below with clear_extra_bits().
978  const limb_t tmp(ulshift(m_limbs[1u],limb_bits - n));
979  m_limbs[0u] = static_cast<limb_t>((m_limbs[0u] >> n) + tmp);
980  m_limbs[1u] = static_cast<limb_t>(m_limbs[1u] >> n);
981  clear_extra_bits();
982  const auto new_asize = calculate_n_limbs();
983  _mp_size = static_cast<mpz_size_t>((_mp_size > 0) ? new_asize : -new_asize);
984  }
985  // Division.
986  static void div(static_integer &q, static_integer &r, const static_integer &a, const static_integer &b)
987  {
988  piranha_assert(!b.is_zero());
989  // NOTE: here in principle q/r could overlap with a or b (e.g., in in-place division).
990  // We need to first read everything we need from a and b, and only then write into q/r.
991  // Store the signs.
992  const bool signa = a._mp_size >= 0, signb = b._mp_size >= 0;
993  // Compute the result in dlimb_t.
994  const dlimb_t ad = static_cast<dlimb_t>(a.m_limbs[0u] + (static_cast<dlimb_t>(a.m_limbs[1u]) << limb_bits)),
995  bd = static_cast<dlimb_t>(b.m_limbs[0u] + (static_cast<dlimb_t>(b.m_limbs[1u]) << limb_bits));
996  const dlimb_t qd = static_cast<dlimb_t>(ad / bd), rd = static_cast<dlimb_t>(ad % bd);
997  // Convert back to array of limb_t.
998  q.m_limbs[0u] = static_cast<limb_t>(qd);
999  q.m_limbs[1u] = static_cast<limb_t>(qd >> limb_bits);
1000  q.clear_extra_bits();
1001  q._mp_size = q.calculate_n_limbs();
1002  r.m_limbs[0u] = static_cast<limb_t>(rd);
1003  r.m_limbs[1u] = static_cast<limb_t>(rd >> limb_bits);
1004  r.clear_extra_bits();
1005  r._mp_size = r.calculate_n_limbs();
1006  // The sign of the remainder is the same as the numerator.
1007  if (!signa) {
1008  r.negate();
1009  }
1010  // The sign of the quotient is the sign of a/b.
1011  if (signa != signb) {
1012  q.negate();
1013  }
1014  }
1015  // Compute the number of bits used in the representation of the integer.
1016  limb_t bits_size() const
1017  {
1018  using size_type = typename limbs_type::size_type;
1019  const auto asize = abs_size();
1020  if (asize == 0) {
1021  return 0u;
1022  }
1023  const auto idx = static_cast<size_type>(asize - 1);
1024  limb_t size = static_cast<limb_t>(limb_bits * idx), limb = m_limbs[idx];
1025  while (limb != 0u) {
1026  ++size;
1027  limb = static_cast<limb_t>(limb >> 1u);
1028  }
1029  return size;
1030  }
1031  limb_t test_bit(const limb_t &idx) const
1032  {
1033  using size_type = typename limbs_type::size_type;
1034  piranha_assert(idx < limb_bits * 2u);
1035  const auto quot = static_cast<limb_t>(idx / limb_bits), rem = static_cast<limb_t>(idx % limb_bits);
1036  return (static_cast<limb_t>(m_limbs[static_cast<size_type>(quot)] & static_cast<limb_t>(limb_t(1u) << rem)) != 0u);
1037  }
1038  // Some metaprogramming to make sure we can compute the hash safely. A bit of paranoid defensive programming.
1039  struct hash_checks
1040  {
1041  // Total number of bits that can be stored. We know already this operation is safe.
1042  static const limb_t tot_bits = static_cast<limb_t>(limb_bits * 2u);
1043  static const unsigned nbits_size_t = static_cast<unsigned>(std::numeric_limits<std::size_t>::digits);
1044  static const limb_t q = static_cast<limb_t>(tot_bits / nbits_size_t);
1045  static const limb_t r = static_cast<limb_t>(tot_bits % nbits_size_t);
1046  // Max number of size_t elements that can be extracted.
1047  static const limb_t max_n_size_t = q + limb_t(r != 0u);
1048  static const bool value =
1049  // Compute tot_nbits as unsigned.
1050  (tot_bits < std::numeric_limits<unsigned>::max()) &&
1051  // Convert n_size_t to std::size_t for use as function param in read_uint.
1052  (max_n_size_t < std::numeric_limits<std::size_t>::max()) &&

1617  }

1641  }

1661  template <typename T>
1662  T convert_to_impl(typename std::enable_if<std::is_floating_point<T>::value>::type * = nullptr) const
1663  {
1664  const int s = sign();
1665  // Special case for zero.
1666  if (s == 0) {
1667  return T(0);
1668  }
1669  // Extract a GMP mpz to work with.
1670  detail::mpz_raii tmp;
1671  if (m_int.is_static()) {
1672  auto v = m_int.g_st().get_mpz_view();
1673  ::mpz_set(&tmp.m_mpz,v);
1674  } else {
1675  ::mpz_set(&tmp.m_mpz,&m_int.g_dy());
1676  }
1677  // Work on absolute value.
1678  if (s < 0) {
1679  ::mpz_neg(&tmp.m_mpz,&tmp.m_mpz);
1680  }
1681  const unsigned radix = static_cast<unsigned>(std::numeric_limits<T>::radix);
1682  // NOTE: radix must be between 2 and 62 for GMP functions to work.
1683  if (unlikely(radix < 2u || radix > 62u)) {
1684  piranha_throw(std::overflow_error,"overflow in conversion to floating-point type");
1685  }
1686  unsigned long r_size;
1687  try {
1688  r_size = boost::numeric_cast<unsigned long>(::mpz_sizeinbase(&tmp.m_mpz,static_cast<int>(radix)));
1689  } catch (...) {
1690  piranha_throw(std::overflow_error,"overflow in conversion to floating-point type");
1691  }
1692  // NOTE: sizeinbase might return the correct value, or increased by one. Check
1693  // which one is which.
1694  // https://gmplib.org/manual/Miscellaneous-Integer-Functions.html#Miscellaneous-Integer-Functions
1695  piranha_assert(r_size >= 1u);
1696  detail::mpz_raii tmp2, tmp3;
1697  ::mpz_ui_pow_ui(&tmp2.m_mpz,static_cast<unsigned long>(radix),r_size - 1ul);
1698  ::mpz_sub_ui(&tmp2.m_mpz,&tmp2.m_mpz,1ul);
1699  if (::mpz_cmp(&tmp2.m_mpz,&tmp.m_mpz) > 0) {
1700  --r_size;
1701  }
1702  // Init return value.
1703  T retval(0);
1704  int exp = 0;
1705  for (unsigned long i = 0u; i < r_size; ++i) {
1706  const auto rem = ::mpz_fdiv_q_ui(&tmp.m_mpz,&tmp.m_mpz,static_cast<unsigned long>(radix));
1707  const auto exp_val = std::scalbn(static_cast<T>(rem),exp);
1708  if (unlikely(exp_val == HUGE_VAL)) {

1710  if (std::numeric_limits<T>::has_infinity) {
1711  retval = std::numeric_limits<T>::infinity();
1712  break;
1713  } else {
1714  piranha_throw(std::overflow_error,"overflow in conversion to floating-point type");
1715  }
1716  }
1717  retval += exp_val;
1718  if (unlikely(exp == std::numeric_limits<int>::max())) {
1719  piranha_throw(std::overflow_error,"overflow in conversion to floating-point type");
1720  }
1721  ++exp;
1722  }
1723  // Adjust sign.
1724  if (s < 0) {
1725  retval = std::copysign(retval,std::numeric_limits<T>::lowest());
1726  }
1727  return retval;
1728  }

2417  explicit mp_integer(const ::mpz_t z)
2418  {
2419  // Get the number of limbs in use.
2420  const std::size_t size = ::mpz_size(z);
2421  // Nothing to do if z is zero.
2422  if (size == 0u) {
2423  return;
2424  }
2425  // Get a read-only pointer to the limbs of z.
2426  // NOTE: there is a specific mpz_limbs_read() function
2427  // in later GMP versions for this.
2428  const ::mp_limb_t *l_ptr = z->_mp_d;
2429  // Effective number of bits used per limb in static storage.
2430  const auto limb_bits = detail::integer_union<NBits>::s_storage::limb_bits;
2431  // Here we are checking roughly if we need static or dynamic
2432  // storage, based on the number of limbs used in z and the available
2433  // bits in static storage. It is a conservative check, meaning there
2434  // could be cases in which z fits in static but the check below fails
2435  // and forces dynamic storage.
2436  // Example: the mpz has a value of 1 and 64 bit limb, mp_integer
2437  // has 16 bit limb.
2438  // We could replace with mpz sizeinbase() but performance would be worse
2439  // probably. Need to invesitgate.
2440  if (unsigned(GMP_NUMB_BITS) > (limb_bits * 2u) / size) {
2441  promote();
2442  ::mpz_set(&m_int.g_dy(),z);
2443  } else {
2444  // Limb type in static storage.
2445  using limb_t = typename detail::integer_union<NBits>::s_storage::limb_t;
2446  // Number of total bits per limb in static storage (>= limb_bits).
2447  const auto total_bits = detail::integer_union<NBits>::s_storage::total_bits;
2448  // The total number of bits we will need to extract from z. We know we can compute this
2449  // because we know z fits static, and we can always represent the total number of bits
2450  // in static.
2451  const limb_t tot_nbits = static_cast<limb_t>(unsigned(GMP_NUMB_BITS) * size),
2452  q = static_cast<limb_t>(tot_nbits / limb_bits),
2453  r = static_cast<limb_t>(tot_nbits % limb_bits),
2454  n_limbs = static_cast<limb_t>(q + static_cast<limb_t>(r != 0u));
2455  piranha_assert(n_limbs <= 2u && n_limbs > 0u);
2456  // NOTE: if here the number of static limbs used will be only 1, the second
2457  // limb has already been zeroed out by the intial construction.
2458  for (std::size_t i = 0u; i < n_limbs; ++i) {
2459  m_int.g_st().m_limbs[i] = detail::read_uint<limb_t,unsigned(GMP_LIMB_BITS - GMP_NUMB_BITS),

2461  }

2463  m_int.g_st()._mp_size = m_int.g_st().calculate_n_limbs();

2465  if (mpz_sgn(z) == -1) {
