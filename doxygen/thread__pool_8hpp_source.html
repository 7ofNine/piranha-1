61 // Task queue class. Inspired by:
62 // https://github.com/progschj/ThreadPool
63 class task_queue
64 {
65  struct runner
66  {
67  runner(task_queue *ptr, unsigned n):m_ptr(ptr),m_n(n) {}
68  void operator()() const
69  {
70  // Don't stop if we cannot bind.
71  try {
72  thread_management::bind_to_proc(m_n);
73  } catch (...) {
74  // NOTE: logging candidate.
75  }
76  try {
77  while (true) {
78  std::unique_lock<std::mutex> lock(m_ptr->m_mutex);
79  while (!m_ptr->m_stop && m_ptr->m_tasks.empty()) {
80  // Need to wait for something to happen only if the task
81  // list is empty and we are not stopping.
82  // NOTE: wait will be noexcept in C++14.
83  m_ptr->m_cond.wait(lock);
84  }
85  if (m_ptr->m_stop && m_ptr->m_tasks.empty()) {
86  // If the stop flag was set, and we do not have more tasks,
87  // just exit.
88  break;
89  }
90  // NOTE: move constructor of std::function could throw, unfortunately.
91  std::function<void()> task(std::move(m_ptr->m_tasks.front()));
92  m_ptr->m_tasks.pop();
93  lock.unlock();
94  task();
95  }
96  } catch (...) {
97  // The errors we could get here are:
98  // - threading primitives,
99  // - move-construction of std::function,
100  // - queue popping (I guess unlikely, as the destructor of std::function
101  // is noexcept).
102  // In any case, not much that can be done to recover from this, better to abort.
103  // NOTE: logging candidate.
104  std::abort();
105  }
106  // Free the MPFR caches.
107  ::mpfr_free_cache();
108  }
109  task_queue *m_ptr;
110  const unsigned m_n;
111  };
112  public:
113  task_queue(unsigned n):m_stop(false)
114  {
115  m_thread.reset(::new std::thread(runner{this,n}));
116  }
117  ~task_queue()
118  {
119  // NOTE: logging candidate (catch any exception,
120  // log it and abort as there is not much we can do).
121  try {
122  stop();
123  } catch (...) {
124  std::abort();
125  }
126  }
127  template <typename F, typename ... Args>
128  auto enqueue(F &&f, Args && ... args) -> std::future<decltype(f(args...))>
129  {
130  using f_ret_type = decltype(f(args...));
131  using p_task_type = std::packaged_task<f_ret_type()>;
132  auto task = std::make_shared<p_task_type>(std::bind(std::forward<F>(f),std::forward<Args>(args)...));
133  std::future<f_ret_type> res = task->get_future();
134  {
135  std::unique_lock<std::mutex> lock(m_mutex);
136  if (m_stop) {
137  // Enqueueing is not allowed if the queue is stopped.
138  piranha_throw(std::runtime_error,"cannot enqueue task while task queue is stopping");
139  }
140  m_tasks.push([task](){(*task)();});
141  }
142  // NOTE: notify_one is noexcept.
143  m_cond.notify_one();
144  return res;
145  }
146  // NOTE: we call this only from dtor, it is here in order to be able to test it.
147  // So the exception handling in dtor will suffice, keep it in mind if things change.
148  void stop()
149  {
150  {
151  std::unique_lock<std::mutex> lock(m_mutex);
152  if (m_stop) {
153  // Already stopped.
154  return;
155  }
156  m_stop = true;
157  }
158  // Notify the thread that queue has been stopped, wait for it
159  // to consume the remaining tasks and exit.
160  m_cond.notify_one();
161  m_thread->join();
162  }
163  private:
164  bool m_stop;
165  std::condition_variable m_cond;
166  std::mutex m_mutex;
167  std::queue<std::function<void()>> m_tasks;
168  std::unique_ptr<std::thread> m_thread;
169 };

171 inline std::vector<std::unique_ptr<task_queue>> get_initial_thread_queues()
172 {
173  std::vector<std::unique_ptr<task_queue>> retval;
174  const unsigned candidate = runtime_info::get_hardware_concurrency(), hc = (candidate > 0u) ? candidate : 1u;
175  retval.reserve(static_cast<decltype(retval.size())>(hc));
176  for (unsigned i = 0u; i < hc; ++i) {

178  }

284  static void resize(unsigned new_size)
285  {
286  if (unlikely(new_size == 0u)) {
287  piranha_throw(std::invalid_argument,"cannot resize the thread pool to zero");
288  }
289  std::lock_guard<std::mutex> lock(s_mutex);
290  using size_type = decltype(base::s_queues.size());
291  const size_type new_s = static_cast<size_type>(new_size);
292  if (unlikely(new_s != new_size)) {
293  piranha_throw(std::invalid_argument,"invalid size");
294  }
295  std::vector<std::unique_ptr<detail::task_queue>> new_queues;
296  new_queues.reserve(new_s);
297  for (size_type i = 0u; i < new_s; ++i) {
298  new_queues.emplace_back(::new detail::task_queue(static_cast<unsigned>(i)));
299  }
300  // NOTE: here the allocator is not swapped, as std::allocator won't propagate on swap.
301  // Besides, all instances of std::allocator are equal, so the operation is well-defined.
302  // http://en.cppreference.com/w/cpp/container/vector/swap
303  // If an exception gets actually thrown, no big deal.
304  new_queues.swap(base::s_queues);
305  }

326  template <typename Int, use_threads_enabler<Int> = 0>
327  static unsigned use_threads(const Int &work_size, const Int &min_work_per_thread)
328  {
329  // Check input params.
330  if (unlikely(work_size <= Int(0))) {
331  piranha_throw(std::invalid_argument,"invalid value for work size");
332  }
333  if (unlikely(min_work_per_thread <= Int(0))) {
334  piranha_throw(std::invalid_argument,"invalid value for minimum work per thread");
335  }
336  // Don't use threads if we are not in the main thread.
337  if (std::this_thread::get_id() != runtime_info::get_main_thread_id()) {
338  return 1u;
339  }
340  std::lock_guard<std::mutex> lock(s_mutex);
341  const auto n_threads = static_cast<unsigned>(base::s_queues.size());
342  piranha_assert(n_threads);
343  if (work_size / n_threads >= min_work_per_thread) {
344  // Enough work per thread, use them all.
345  return n_threads;
346  }
347  // Return a number of threads such that each thread consumes at least min_work_per_thread.
348  // Never return 0.
349  return static_cast<unsigned>(std::max(Int(1),static_cast<Int>(work_size / min_work_per_thread)));
350  }
