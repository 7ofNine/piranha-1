66 // Task queue class. Inspired by:
67 // https://github.com/progschj/ThreadPool
68 struct task_queue {
69  struct runner {
70  runner(task_queue *ptr, unsigned n, bool bind) : m_ptr(ptr), m_n(n), m_bind(bind)
71  {
72  }
73  void operator()() const
74  {
75  if (m_bind) {
76  try {
77  bind_to_proc(m_n);
78  } catch (...) {
79  // Don't stop if we cannot bind.
80  // NOTE: logging candidate.
81  }
82  }
83  try {
84  while (true) {
85  std::unique_lock<std::mutex> lock(m_ptr->m_mutex);
86  while (!m_ptr->m_stop && m_ptr->m_tasks.empty()) {
87  // Need to wait for something to happen only if the task
88  // list is empty and we are not stopping.
89  // NOTE: wait will be noexcept in C++14.
90  m_ptr->m_cond.wait(lock);
91  }
92  if (m_ptr->m_stop && m_ptr->m_tasks.empty()) {
93  // If the stop flag was set, and we do not have more tasks,
94  // just exit.
95  break;
96  }
97  // NOTE: move constructor of std::function could throw, unfortunately.
98  std::function<void()> task(std::move(m_ptr->m_tasks.front()));
99  m_ptr->m_tasks.pop();
100  lock.unlock();
101  task();
102  }
103  } catch (...) {
104  // The errors we could get here are:
105  // - threading primitives,
106  // - move-construction of std::function,
107  // - queue popping (I guess unlikely, as the destructor of std::function
108  // is noexcept).
109  // In any case, not much that can be done to recover from this, better to abort.
110  // NOTE: logging candidate.
111  std::abort();
112  }
113  // Free the MPFR caches.
114  ::mpfr_free_cache();
115  }
116  task_queue *m_ptr;
117  const unsigned m_n;
118  const bool m_bind;
119  };

121  task_queue(unsigned n, bool bind) : m_stop(false)
122  {
123  // NOTE: this seems to be ok wrt order of evaluation, since the ctor of runner cannot throw.
124  // In general, it could happen that:
125  // - new allocates,
126  // - runner is constructed and throws,
127  // - the memory allocated by new is not freed.
128  // See the classic: http://gotw.ca/gotw/056.htm
129  m_thread.reset(new std::thread(runner{this, n, bind}));
130  }
131  ~task_queue()
132  {
133  // NOTE: logging candidate (catch any exception,
134  // log it and abort as there is not much we can do).
135  try {
136  stop();
137  } catch (...) {
138  std::abort();
139  }

141  // Small utility to remove reference_wrapper.
142  template <typename T>
143  struct unwrap_ref {
144  using type = T;
145  };
146  template <typename T>
147  struct unwrap_ref<std::reference_wrapper<T>> {
148  using type = T;
149  };
150  template <typename T>
151  using unwrap_ref_t = typename unwrap_ref<T>::type;
152  // NOTE: the functor F will be forwarded to std::bind in order to create a nullary wrapper. The nullary wrapper
153  // will create copies of the input arguments, and it will then pass these copies as lvalue refs to a copy of the
154  // original functor when the call operator is invoked (with special handling of reference wrappers). Thus, the
155  // real invocation of F is not simply F(args), but this more complicated type below.
156  // NOTE: this is one place where it seems we really want decay instead of uncvref, as decay is applied
157  // also by std::bind() to F.
158  template <typename F, typename... Args>
159  using f_ret_type = decltype(std::declval<decay_t<F> &>()(std::declval<unwrap_ref_t<uncvref_t<Args>> &>()...));
160  // enqueue() will be enabled if:
161  // - f_ret_type is a valid type (checked in the return type),
162  // - we can construct the nullary wrapper via std::bind() (this requires F and Args to be ctible from the input
163  // arguments),
164  // - we can build a packaged_task from the nullary wrapper (requires F and Args to be move/copy ctible),
165  // - the return type of F is returnable.
166  template <typename F, typename... Args>
167  using enabler
168  = enable_if_t<conjunction<std::is_constructible<decay_t<F>, F>, std::is_constructible<uncvref_t<Args>, Args>...,
169  disjunction<std::is_copy_constructible<decay_t<F>>,
170  std::is_move_constructible<decay_t<F>>>,
171  conjunction<disjunction<std::is_copy_constructible<uncvref_t<Args>>,
172  std::is_move_constructible<uncvref_t<Args>>>>...,
173  is_returnable<f_ret_type<F, Args...>>>::value,
174  int>;
175  // Main enqueue function.
176  template <typename F, typename... Args, enabler<F &&, Args &&...> = 0>
177  std::future<f_ret_type<F &&, Args &&...>> enqueue(F &&f, Args &&... args)
178  {
179  using ret_type = f_ret_type<F &&, Args &&...>;
180  using p_task_type = std::packaged_task<ret_type()>;
181  // NOTE: here we have a multi-stage construction of the task:
182  // - std::bind() turns F into a nullary functor,
183  // - std::packaged_task gives us the std::future machinery,
184  // - std::function (in m_tasks) gives the uniform type interface via type erasure.
185  auto task = std::make_shared<p_task_type>(std::bind(std::forward<F>(f), std::forward<Args>(args)...));
186  std::future<ret_type> res = task->get_future();
187  {
188  std::unique_lock<std::mutex> lock(m_mutex);
189  if (unlikely(m_stop)) {
190  // Enqueueing is not allowed if the queue is stopped.
191  piranha_throw(std::runtime_error, "cannot enqueue task while the task queue is stopping");
192  }
193  m_tasks.push([task]() { (*task)(); });
194  }
195  // NOTE: notify_one is noexcept.
196  m_cond.notify_one();
197  return res;
198  }
199  // NOTE: we call this only from dtor, it is here in order to be able to test it.
200  // So the exception handling in dtor will suffice, keep it in mind if things change.
201  void stop()
202  {
203  {
204  std::unique_lock<std::mutex> lock(m_mutex);
205  if (m_stop) {
206  // Already stopped.
207  return;
208  }
209  m_stop = true;
210  }
211  // Notify the thread that queue has been stopped, wait for it
212  // to consume the remaining tasks and exit.
213  m_cond.notify_one();
214  m_thread->join();
215  }
216 
217  bool m_stop;
218  std::condition_variable m_cond;
219  std::mutex m_mutex;
220  std::queue<std::function<void()>> m_tasks;
221  std::unique_ptr<std::thread> m_thread;
222 };

224 // Type to represent thread queues: a vector of task queues paired with a set of thread ids.
225 using thread_queues_t = std::pair<std::vector<std::unique_ptr<task_queue>>, std::unordered_set<std::thread::id>>;

227 inline thread_queues_t get_initial_thread_queues()
228 {
229  thread_queues_t retval;
230  // Create the vector of queues.
231  const unsigned candidate = runtime_info::get_hardware_concurrency(), hc = (candidate > 0u) ? candidate : 1u;
232  retval.first.reserve(static_cast<decltype(retval.first.size())>(hc));
233  for (unsigned i = 0u; i < hc; ++i) {
234  // NOTE: thread binding is disabled on startup.
235  retval.first.emplace_back(::new task_queue(i, false));
236  }
237  // Generate the set of thread IDs.
238  for (const auto &ptr : retval.first) {
239  auto p = retval.second.insert(ptr->m_thread->get_id());
240  (void)p;
241  piranha_assert(p.second);
242  }
243  return retval;
244 }

379  static void resize(unsigned new_size)
380  {
381  if (unlikely(new_size == 0u)) {
382  piranha_throw(std::invalid_argument, "cannot resize the thread pool to zero");
383  }
384  // NOTE: need to lock here as we are reading the s_bind member.
385  detail::atomic_lock_guard lock(s_atf);
386  auto new_queues = create_new_queues(new_size, base::s_bind);
387  // NOTE: here the allocator is not swapped, as std::allocator won't propagate on swap.
388  // Besides, all instances of std::allocator are equal, so the operation is well-defined.
389  // http://en.cppreference.com/w/cpp/container/vector/swap
390  // This holds for both std::vector and std::unordered_set.
391  // If an exception gets actually thrown, no big deal.
392  // NOTE: the dtor of the queues is effectively noexcept, as the program will just abort in case of errors
393  // in the dtor.
394  new_queues.swap(base::s_queues);

455  template <typename Int, use_threads_enabler<Int> = 0>
456  static unsigned use_threads(const Int &work_size, const Int &min_work_per_thread)
457  {
458  // Check input params.
459  if (unlikely(work_size <= Int(0))) {
460  piranha_throw(std::invalid_argument, "invalid value of " + boost::lexical_cast<std::string>(work_size)
461  + " for work size (it must be strictly positive)");
462  }
463  if (unlikely(min_work_per_thread <= Int(0))) {
464  piranha_throw(std::invalid_argument, "invalid value of "
465  + boost::lexical_cast<std::string>(min_work_per_thread)
466  + " for minimum work per thread (it must be strictly positive)");
467  }
468  detail::atomic_lock_guard lock(s_atf);
469  // Don't use threads if the calling thread belongs to the pool.
470  if (base::s_queues.second.find(std::this_thread::get_id()) != base::s_queues.second.end()) {
471  return 1u;
472  }
473  const auto n_threads = static_cast<unsigned>(base::s_queues.first.size());
474  piranha_assert(n_threads);
475  if (work_size / n_threads >= min_work_per_thread) {
476  // Enough work per thread, use them all.
477  return n_threads;
478  }
479  // Return a number of threads such that each thread consumes at least min_work_per_thread.
480  // Never return 0.
481  return static_cast<unsigned>(std::max(Int(1), static_cast<Int>(work_size / min_work_per_thread)));
482  }
