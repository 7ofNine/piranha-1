68 // Task queue class. Inspired by:
69 // https://github.com/progschj/ThreadPool
70 struct task_queue {
71  struct runner {
72  runner(task_queue *ptr, unsigned n, bool bind) : m_ptr(ptr), m_n(n), m_bind(bind)
73  {
74  }
75  void operator()() const
76  {
77  if (m_bind) {
78  try {
79  bind_to_proc(m_n);
80  } catch (...) {
81  // Don't stop if we cannot bind.
82  // NOTE: logging candidate.
83  }
84  }
85  try {
86  while (true) {
87  std::unique_lock<std::mutex> lock(m_ptr->m_mutex);
88  while (!m_ptr->m_stop && m_ptr->m_tasks.empty()) {
89  // Need to wait for something to happen only if the task
90  // list is empty and we are not stopping.
91  // NOTE: wait will be noexcept in C++14.
92  m_ptr->m_cond.wait(lock);
93  }
94  if (m_ptr->m_stop && m_ptr->m_tasks.empty()) {
95  // If the stop flag was set, and we do not have more tasks,
96  // just exit.
97  break;
98  }
99  // NOTE: move constructor of std::function could throw, unfortunately.
100  std::function<void()> task(std::move(m_ptr->m_tasks.front()));
101  m_ptr->m_tasks.pop();
102  lock.unlock();
103  task();
104  }
105  } catch (...) {
106  // The errors we could get here are:
107  // - threading primitives,
108  // - move-construction of std::function,
109  // - queue popping (I guess unlikely, as the destructor of std::function
110  // is noexcept).
111  // In any case, not much that can be done to recover from this, better to abort.
112  // NOTE: logging candidate.
113  std::abort();
114  }
115  // Free the MPFR caches.
116  ::mpfr_free_cache();
117  }
118  task_queue *m_ptr;
119  const unsigned m_n;
120  const bool m_bind;
121  };
122 
123  task_queue(unsigned n, bool bind) : m_stop(false)
124  {
125  // NOTE: this seems to be ok wrt order of evaluation, since the ctor of runner cannot throw.
126  // In general, it could happen that:
127  // - new allocates,
128  // - runner is constructed and throws,
129  // - the memory allocated by new is not freed.
130  // See the classic: http://gotw.ca/gotw/056.htm
131  m_thread.reset(new std::thread(runner{this, n, bind}));
132  }
133  ~task_queue()
134  {
135  // NOTE: logging candidate (catch any exception,
136  // log it and abort as there is not much we can do).
137  try {
138  stop();
139  } catch (...) {
140  std::abort();
141  }
142  }
143  // Small utility to remove reference_wrapper.
144  template <typename T>
145  struct unwrap_ref {
146  using type = T;
147  };
148  template <typename T>
149  struct unwrap_ref<std::reference_wrapper<T>> {
150  using type = T;
151  };
152  template <typename T>
153  using unwrap_ref_t = typename unwrap_ref<T>::type;
154  // NOTE: the functor F will be forwarded to std::bind in order to create a nullary wrapper. The nullary wrapper
155  // will create copies of the input arguments, and it will then pass these copies as lvalue refs to a copy of the
156  // original functor when the call operator is invoked (with special handling of reference wrappers). Thus, the
157  // real invocation of F is not simply F(args), but this more complicated type below.
158  // NOTE: this is one place where it seems we really want decay instead of uncvref, as decay is applied
159  // also by std::bind() to F.
160  template <typename F, typename... Args>
161  using f_ret_type = decltype(std::declval<decay_t<F> &>()(std::declval<unwrap_ref_t<uncvref_t<Args>> &>()...));
162  // enqueue() will be enabled if:
163  // - f_ret_type is a valid type (checked in the return type),
164  // - we can construct the nullary wrapper via std::bind() (this requires F and Args to be ctible from the input
165  // arguments),
166  // - we can build a packaged_task from the nullary wrapper (requires F and Args to be move/copy ctible),
167  // - the return type of F is returnable.
168  template <typename F, typename... Args>
169  using enabler
170  = enable_if_t<conjunction<std::is_constructible<decay_t<F>, F>, std::is_constructible<uncvref_t<Args>, Args>...,
171  disjunction<std::is_copy_constructible<decay_t<F>>,
172  std::is_move_constructible<decay_t<F>>>,
173  conjunction<disjunction<std::is_copy_constructible<uncvref_t<Args>>,
174  std::is_move_constructible<uncvref_t<Args>>>>...,
175  is_returnable<f_ret_type<F, Args...>>>::value,
176  int>;
177  // Main enqueue function.
178  template <typename F, typename... Args, enabler<F &&, Args &&...> = 0>
179  std::future<f_ret_type<F &&, Args &&...>> enqueue(F &&f, Args &&... args)
180  {
181  using ret_type = f_ret_type<F &&, Args &&...>;
182  using p_task_type = std::packaged_task<ret_type()>;
183  // NOTE: here we have a multi-stage construction of the task:
184  // - std::bind() turns F into a nullary functor,
185  // - std::packaged_task gives us the std::future machinery,
186  // - std::function (in m_tasks) gives the uniform type interface via type erasure.
187  auto task = std::make_shared<p_task_type>(std::bind(std::forward<F>(f), std::forward<Args>(args)...));
188  std::future<ret_type> res = task->get_future();
189  {
190  std::unique_lock<std::mutex> lock(m_mutex);
191  if (unlikely(m_stop)) {
192  // Enqueueing is not allowed if the queue is stopped.
193  piranha_throw(std::runtime_error, "cannot enqueue task while the task queue is stopping");
194  }
195  m_tasks.push([task]() { (*task)(); });
196  }
197  // NOTE: notify_one is noexcept.
198  m_cond.notify_one();
199  return res;
200  }
201  // NOTE: we call this only from dtor, it is here in order to be able to test it.
202  // So the exception handling in dtor will suffice, keep it in mind if things change.
203  void stop()
204  {
205  {
206  std::unique_lock<std::mutex> lock(m_mutex);
207  if (m_stop) {
208  // Already stopped.
209  return;
210  }
211  m_stop = true;
212  }
213  // Notify the thread that queue has been stopped, wait for it
214  // to consume the remaining tasks and exit.
215  m_cond.notify_one();
216  m_thread->join();
217  }
218 
219  bool m_stop;
220  std::condition_variable m_cond;
221  std::mutex m_mutex;
222  std::queue<std::function<void()>> m_tasks;
223  std::unique_ptr<std::thread> m_thread;
224 };

229 inline thread_queues_t get_initial_thread_queues()
230 {
231  // NOTE: we used to have this print statement here, but it turns out that
232  // in certain setups the cout object is not yet constructed at this point,
233  // and a segfault is generated. I *think* it is possible to enforce the creation
234  // of cout via construction of an init object:
235  // http://en.cppreference.com/w/cpp/io/ios_base/Init
236  // However, this is hardly essential. Let's leave this disabled for the moment.
237  // std::cout << "Initializing the thread pool.\n";
238  thread_queues_t retval;
239  // Create the vector of queues.
240  const unsigned candidate = runtime_info::get_hardware_concurrency(), hc = (candidate > 0u) ? candidate : 1u;
241  retval.first.reserve(static_cast<decltype(retval.first.size())>(hc));
242  for (unsigned i = 0u; i < hc; ++i) {
243  // NOTE: thread binding is disabled on startup.
244  retval.first.emplace_back(::new task_queue(i, false));
245  }
246  // Generate the set of thread IDs.
247  for (const auto &ptr : retval.first) {
248  auto p = retval.second.insert(ptr->m_thread->get_id());
249  (void)p;
250  piranha_assert(p.second);
251  }
252  return retval;
253 }

399  static void resize(unsigned new_size)
400  {
401  if (unlikely(new_size == 0u)) {
402  piranha_throw(std::invalid_argument, "cannot resize the thread pool to zero");
403  }
404  // NOTE: need to lock here as we are reading the s_bind member.

413  // in the dtor.

475  template <typename Int, use_threads_enabler<Int> = 0>
476  static unsigned use_threads(const Int &work_size, const Int &min_work_per_thread)
477  {
478  // Check input params.
479  if (unlikely(work_size <= Int(0))) {
480  piranha_throw(std::invalid_argument, "invalid value of " + boost::lexical_cast<std::string>(work_size)
481  + " for work size (it must be strictly positive)");
482  }
483  if (unlikely(min_work_per_thread <= Int(0))) {
484  piranha_throw(std::invalid_argument, "invalid value of "
485  + boost::lexical_cast<std::string>(min_work_per_thread)
486  + " for minimum work per thread (it must be strictly positive)");
487  }
488  detail::atomic_lock_guard lock(s_atf);
489  // Don't use threads if the calling thread belongs to the pool.
490  if (base::s_queues.second.find(std::this_thread::get_id()) != base::s_queues.second.end()) {
491  return 1u;
492  }
493  const auto n_threads = static_cast<unsigned>(base::s_queues.first.size());
494  piranha_assert(n_threads);
495  if (work_size / n_threads >= min_work_per_thread) {
496  // Enough work per thread, use them all.
497  return n_threads;
498  }
499  // Return a number of threads such that each thread consumes at least min_work_per_thread.
500  // Never return 0.
501  return static_cast<unsigned>(std::max(Int(1), static_cast<Int>(work_size / min_work_per_thread)));
502  }

578  {

580  try {
581  m_list.emplace_back();
582  } catch (...) {
583  // If we get some error here, we want to make sure we wait on the future
584  // before escaping out.
585  // NOTE: calling wait() on an invalid future is UB.
586  if (f.valid()) {

601  if (f.valid()) {
602  wait_or_abort(f);
603  }
604  }
