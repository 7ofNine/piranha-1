54 template <typename T>
55 struct ps_term_score
56 {
57  typedef typename T::cf_type cf_type;
58  typedef typename T::key_type key_type;
59  static const unsigned value = static_cast<unsigned>(has_degree<cf_type>::value && has_ldegree<cf_type>::value) +
60  (static_cast<unsigned>(key_has_degree<key_type>::value && key_has_ldegree<key_type>::value) << 1u);
61 };

197  // Total degree truncation.
198  // Case 1: coefficient can truncate, no degree or ldegree in key.
199  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,T>::value &&
200  (detail::ps_term_score<Term>::value >> 1u) == 0u,int>::type = 0>
201  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &)
202  {
203  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree),t.m_key));
204  }
205  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
206  // NOTE: here we do not have support for key truncation (yet), so we decide based on the low degree of the key:
207  // if it is larger than the max degree, remove the term, otherwise keep it - it is an all-or-nothing scenario.
208  template <typename Term, typename T, typename std::enable_if<!has_truncate_degree<typename Term::cf_type,T>::value &&
209  (detail::ps_term_score<Term>::value >> 1u) == 1u &&
210  is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>())),T>::value,int>::type = 0>
211  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
212  {
213  if (t.m_key.ldegree(s) > max_degree) {
214  // Term must be discarded.
215  return std::make_pair(false,Term());
216  } else {
217  // Keep the term as it is.
218  return std::make_pair(true,Term(t.m_cf,t.m_key));
219  }
220  }
221  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the coefficient truncation.
222  // NOTE: again, no key truncation, thus we decrement the real degree of the coefficient by the low degree of the key. This way we will keep
223  // all the important parts, plus some garbage.
224  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,
225  decltype(std::declval<const T &>() - std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>()))>::value &&
226  (detail::ps_term_score<Term>::value >> 1u) == 1u,int>::type = 0>
227  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
228  {
229  // The truncation level for the coefficient must be modified in order to take
230  // into account the degree of the key.
231  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree - t.m_key.ldegree(s)),t.m_key));
232  }
233  // Enabler for total degree truncation.
234  template <typename T, typename U>
235  using truncate_degree_enabler = typename std::enable_if<detail::true_tt<
236  decltype(truncate_term(std::declval<const typename U::term_type &>(),std::declval<const T &>(),std::declval<const symbol_set &>()))
237  >::value,int>::type;
238  // Partial degree truncation.
239  // Case 1: coefficient can truncate, no degree or ldegree in key.
240  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,T>::value &&
241  (detail::ps_term_score<Term>::value >> 1u) == 0u,int>::type = 0>
242  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &names, const symbol_set::positions &, const symbol_set &)
243  {
244  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree,names),t.m_key));
245  }
246  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
247  template <typename Term, typename T, typename std::enable_if<!has_truncate_degree<typename Term::cf_type,T>::value &&
248  (detail::ps_term_score<Term>::value >> 1u) == 1u &&
249  is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set::positions &>(),
250  std::declval<const symbol_set &>())),T>::value,int>::type = 0>
251  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &, const symbol_set::positions &p, const symbol_set &s)
252  {
253  if (t.m_key.ldegree(p,s) > max_degree) {
254  return std::make_pair(false,Term());
255  } else {
256  return std::make_pair(true,Term(t.m_cf,t.m_key));
257  }
258  }
259  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the coefficient truncation.
260  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,
261  decltype(std::declval<const T &>() - std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set::positions &>(),
262  std::declval<const symbol_set &>()))>::value &&
263  (detail::ps_term_score<Term>::value >> 1u) == 1u,int>::type = 0>
264  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &names, const symbol_set::positions &p, const symbol_set &s)
265  {
266  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree - t.m_key.ldegree(p,s),names),t.m_key));
267  }
