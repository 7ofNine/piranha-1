46 struct ps_term_score
47 {
48  typedef typename T::cf_type cf_type;
49  typedef typename T::key_type key_type;
50  static const unsigned value = static_cast<unsigned>(has_degree<cf_type>::value && has_ldegree<cf_type>::value) +
51  (static_cast<unsigned>(key_has_degree<key_type>::value && key_has_ldegree<key_type>::value) << 1u);
52 };

162  // Total degree truncation.
163  // Case 1: coefficient can truncate, no degree or ldegree in key.
164  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,T>::value &&
165  (detail::ps_term_score<Term>::value >> 1u) == 0u,int>::type = 0>
166  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &)
167  {
168  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree),t.m_key));
169  }
170  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
171  // NOTE: here we do not have support for key truncation (yet), so we decide based on the low degree of the key:
172  // if it is larger than the max degree, remove the term, otherwise keep it - it is an all-or-nothing scenario.
173  template <typename Term, typename T, typename std::enable_if<!has_truncate_degree<typename Term::cf_type,T>::value &&
174  (detail::ps_term_score<Term>::value >> 1u) == 1u &&
175  is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>())),T>::value,int>::type = 0>
176  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
177  {
178  if (t.m_key.ldegree(s) > max_degree) {
179  // Term must be discarded.
180  return std::make_pair(false,Term());
181  } else {
182  // Keep the term as it is.
183  return std::make_pair(true,Term(t.m_cf,t.m_key));
184  }
185  }
186  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the coefficient truncation.
187  // NOTE: again, no key truncation, thus we decrement the real degree of the coefficient by the low degree of the key. This way we will keep
188  // all the important parts, plus some garbage.
189  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,
190  decltype(std::declval<const T &>() - std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set &>()))>::value &&
191  (detail::ps_term_score<Term>::value >> 1u) == 1u,int>::type = 0>
192  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
193  {
194  // The truncation level for the coefficient must be modified in order to take
195  // into account the degree of the key.
196  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree - t.m_key.ldegree(s)),t.m_key));
197  }
198  // Enabler for total degree truncation.
199  template <typename T, typename U>
200  using truncate_degree_enabler = typename std::enable_if<detail::true_tt<
201  decltype(truncate_term(std::declval<const typename U::term_type &>(),std::declval<const T &>(),std::declval<const symbol_set &>()))
202  >::value,int>::type;
203  // Partial degree truncation.
204  // Case 1: coefficient can truncate, no degree or ldegree in key.
205  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,T>::value &&
206  (detail::ps_term_score<Term>::value >> 1u) == 0u,int>::type = 0>
207  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &names, const symbol_set::positions &, const symbol_set &)
208  {
209  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree,names),t.m_key));
210  }
211  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
212  template <typename Term, typename T, typename std::enable_if<!has_truncate_degree<typename Term::cf_type,T>::value &&
213  (detail::ps_term_score<Term>::value >> 1u) == 1u &&
214  is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set::positions &>(),
215  std::declval<const symbol_set &>())),T>::value,int>::type = 0>
216  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &, const symbol_set::positions &p, const symbol_set &s)
217  {
218  if (t.m_key.ldegree(p,s) > max_degree) {
219  return std::make_pair(false,Term());
220  } else {
221  return std::make_pair(true,Term(t.m_cf,t.m_key));
222  }
223  }
224  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the coefficient truncation.
225  template <typename Term, typename T, typename std::enable_if<has_truncate_degree<typename Term::cf_type,
226  decltype(std::declval<const T &>() - std::declval<const typename Term::key_type &>().ldegree(std::declval<const symbol_set::positions &>(),
227  std::declval<const symbol_set &>()))>::value &&
228  (detail::ps_term_score<Term>::value >> 1u) == 1u,int>::type = 0>
229  static std::pair<bool,Term> truncate_term(const Term &t, const T &max_degree, const std::vector<std::string> &names, const symbol_set::positions &p, const symbol_set &s)
230  {
231  return std::make_pair(true,Term(math::truncate_degree(t.m_cf,max_degree - t.m_key.ldegree(p,s),names),t.m_key));
232  }
