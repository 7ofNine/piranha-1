56 // Detect power series terms.
57 template <typename T>
58 struct ps_term_score {
59  typedef typename T::cf_type cf_type;
60  typedef typename T::key_type key_type;
61  static const unsigned value
62  = static_cast<unsigned>(has_degree<cf_type>::value && has_ldegree<cf_type>::value)
63  + (static_cast<unsigned>(key_has_degree<key_type>::value && key_has_ldegree<key_type>::value) << 1u);
64 };

230  // Total degree truncation.
231  // Case 1: coefficient can truncate, no degree or ldegree in key.
232  template <typename Term, typename T,
233  typename std::enable_if<has_truncate_degree<typename Term::cf_type, T>::value
234  && (detail::ps_term_score<Term>::value >> 1u) == 0u,
235  int>::type
236  = 0>
237  static std::pair<bool, Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &)
238  {
239  return std::make_pair(true, Term(math::truncate_degree(t.m_cf, max_degree), t.m_key));
240  }
241  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.
242  // NOTE: here we do not have support for key truncation (yet), so we decide based on the low degree of the key:
243  // if it is larger than the max degree, remove the term, otherwise keep it - it is an all-or-nothing scenario.
244  template <
245  typename Term, typename T,
246  typename std::
247  enable_if<!has_truncate_degree<typename Term::cf_type, T>::value
248  && (detail::ps_term_score<Term>::value >> 1u) == 1u
249  && is_greater_than_comparable<decltype(std::declval<const typename Term::key_type &>()
250  .ldegree(std::declval<const symbol_set &>())),
251  T>::value,
252  int>::type
253  = 0>
254  static std::pair<bool, Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
255  {
256  if (t.m_key.ldegree(s) > max_degree) {
257  // Term must be discarded.
258  return std::make_pair(false, Term());
259  } else {
260  // Keep the term as it is.
261  return std::make_pair(true, Term(t.m_cf, t.m_key));
262  }
263  }
264  // Case 3: coefficient can truncate, degree and ldegree in key, the new max degree type can be used in the
265  // coefficient truncation.
266  // NOTE: again, no key truncation, thus we decrement the real degree of the coefficient by the low degree of the
267  // key. This way we will keep
268  // all the important parts, plus some garbage.
269  template <
270  typename Term, typename T,
271  typename std::enable_if<has_truncate_degree<typename Term::cf_type,
272  decltype(std::declval<const T &>()
273  - std::declval<const typename Term::key_type &>().ldegree(
274  std::declval<const symbol_set &>()))>::value
275  && (detail::ps_term_score<Term>::value >> 1u) == 1u,
276  int>::type
277  = 0>
278  static std::pair<bool, Term> truncate_term(const Term &t, const T &max_degree, const symbol_set &s)
279  {
280  // The truncation level for the coefficient must be modified in order to take
281  // into account the degree of the key.
282  return std::make_pair(true, Term(math::truncate_degree(t.m_cf, max_degree - t.m_key.ldegree(s)), t.m_key));
283  }
284  // Enabler for total degree truncation.
285  template <typename T, typename U>
286  using truncate_degree_enabler =
287  typename std::enable_if<detail::true_tt<decltype(truncate_term(std::declval<const typename U::term_type &>(),
288  std::declval<const T &>(),
289  std::declval<const symbol_set &>()))>::value,
290  int>::type;

304  // Case 2: coefficient cannot truncate, degree and ldegree in key, degrees are greater_than comparable.

519  Derived truncate_degree(const T &max_degree) const
520  {
521  Derived retval;
522  retval.m_symbol_set = this->m_symbol_set;
523  const auto it_f = this->m_container.end();
524  for (auto it = this->m_container.begin(); it != it_f; ++it) {
525  auto tmp = this->truncate_term(*it, max_degree, retval.m_symbol_set);

527  retval.insert(std::move(tmp.second));
528  }
