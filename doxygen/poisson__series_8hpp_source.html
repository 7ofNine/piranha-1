151  // Special sin/cos implementation when we have reached the first polynomial coefficient in the hierarchy.
152  template <bool IsCos, typename RetT, typename T, typename std::enable_if<std::is_base_of<detail::polynomial_tag,
153  typename T::term_type::cf_type>::value,int>::type = 0>
154  RetT special_sin_cos(const T &s) const
155  {
156  // Do something only if the series is equivalent to a polynomial.
157  if (s.is_single_coefficient() && !s.empty()) {
158  try {
159  // Shortcuts.
160  typedef typename RetT::term_type term_type;
161  typedef typename term_type::cf_type cf_type;
162  typedef typename term_type::key_type key_type;
163  typedef typename key_type::value_type value_type;
164  // Try to get the integral combination from the poly coefficient.
165  auto lc = s._container().begin()->m_cf.integral_combination();
166  // Change sign if needed.
167  bool sign_change = false;
168  if (!lc.empty() && lc.begin()->second.sign() < 0) {
169  std::for_each(lc.begin(),lc.end(),[](std::pair<const std::string,integer> &p) {p.second.negate();});
170  sign_change = true;
171  }
172  // Return value.
173  RetT retval;
174  // Build vector of integral multipliers.
175  std::vector<value_type> v;
176  for (auto it = lc.begin(); it != lc.end(); ++it) {
177  retval.m_symbol_set.add(it->first);
178  // NOTE: this should probably be a safe_cast.
179  // The value type here could be anything, and not guaranteed to be castable,
180  // even if in the current implementation this is guaranteed to be a signed
181  // int of some kind.
182  v.push_back(static_cast<value_type>(it->second));
183  }
184  // Build term, fix signs and flavour and move-insert it.
185  term_type term(cf_type(1),key_type(v.begin(),v.end()));
186  if (!IsCos) {
187  term.m_key.set_flavour(false);
188  if (sign_change) {
189  // NOTE: negate is supported by any coefficient type.
190  math::negate(term.m_cf);
191  }
192  }
193  retval.insert(std::move(term));
194  return retval;
195  } catch (const std::invalid_argument &) {
196  // Interpret invalid_argument as a failure in extracting integral combination,
197  // and move on.
198  }
199  }
200  return special_sin_cos_failure(*this,std::integral_constant<bool,IsCos>{});
201  }
202  // The coefficient is not a polynomial: recurse to the inner coefficient type, if the current coefficient type
203  // is suitable.
204  template <bool IsCos, typename RetT, typename T, typename std::enable_if<!std::is_base_of<detail::polynomial_tag,
205  typename T::term_type::cf_type>::value,int>::type = 0>
206  RetT special_sin_cos(const T &s) const
207  {
208  if (s.is_single_coefficient() && !s.empty()) {
209  return special_sin_cos<IsCos,RetT>(s._container().begin()->m_cf);
210  }
211  return special_sin_cos_failure(*this,std::integral_constant<bool,IsCos>{});
212  }

317  template <typename T = poisson_series>
318  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
319  const std::true_type &) const
320  {
321  typedef typename base::term_type term_type;
322  typedef typename term_type::cf_type cf_type;
323  integer degree;
324  try {
325  degree = safe_cast<integer>(math::degree(term.m_cf,{s.get_name()}));
326  } catch (const std::invalid_argument &) {
327  piranha_throw(std::invalid_argument,
328  "unable to perform Poisson series integration: cannot convert polynomial degree to an integer");
329  }
330  // If the variable is in both cf and key, and the cf degree is negative, we cannot integrate.
331  if (degree.sign() < 0) {
332  piranha_throw(std::invalid_argument,
333  "unable to perform Poisson series integration: polynomial coefficient has negative integral degree");
334  }
335  // Init retval and auxiliary quantities for the iteration.
336  auto key_int = term.m_key.integrate(s,this->m_symbol_set);
337  // NOTE: here we are sure that the variable is contained in the monomial.
338  piranha_assert(key_int.first != 0);
339  poisson_series tmp;
340  tmp.set_symbol_set(this->m_symbol_set);
341  // NOTE: not move for .second, as it is needed in the loop below.
342  tmp.insert(term_type(cf_type(1),key_int.second));
343  i_cf_type_p<T> p_cf(term.m_cf / std::move(key_int.first));
344  integrate_type<T> retval(p_cf * tmp);
345  for (integer i(1); i <= degree; ++i) {
346  key_int = key_int.second.integrate(s,this->m_symbol_set);
347  piranha_assert(key_int.first != 0);
348  p_cf = math::partial(p_cf / std::move(key_int.first),s.get_name());
349  // Sign change due to the second portion of integration by part.
350  math::negate(p_cf);
351  tmp = poisson_series{};
352  tmp.set_symbol_set(this->m_symbol_set);
353  // NOTE: don't move second.
354  tmp.insert(term_type(cf_type(1),key_int.second));
355  retval += p_cf * tmp;
356  }
357  return retval;
358  }

484  template <typename T = poisson_series>
485  integrate_type<T> integrate(const std::string &name) const
486  {
487  typedef typename base::term_type term_type;
488  typedef typename term_type::cf_type cf_type;
489  // Turn name into symbol.
490  const symbol s(name);
491  // Init the return value.
492  integrate_type<T> retval(0);
493  const auto it_f = this->m_container.end();
494  for (auto it = this->m_container.begin(); it != it_f; ++it) {
495  // Integrate the key first.
496  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
497  // If the variable does not appear in the monomial, try deferring the integration
498  // to the coefficient.
499  if (key_int.first == 0) {
500  integrate_coefficient_only(retval,it,name);
501  continue;
502  }
503  // The variable is in the monomial, let's check if the variable is also in the coefficient.
504  if (math::is_zero(math::partial(it->m_cf,name))) {
505  // No variable in the coefficient, proceed with the integrated key and divide by multiplier.
506  poisson_series tmp;
507  tmp.set_symbol_set(this->m_symbol_set);
508  tmp.insert(term_type(cf_type(1),std::move(key_int.second)));
509  retval += (std::move(tmp) * it->m_cf) / key_int.first;
510  } else {
511  // With the variable both in the coefficient and the key, we only know how to proceed with polynomials.
512  retval += integrate_impl(s,*it,std::integral_constant<bool,std::is_base_of<detail::polynomial_tag,Cf>::value>());
513  }
514  }
515  return retval;
516  }

555  template <typename T = poisson_series>
556  ti_type<T> t_integrate() const
557  {
558  using return_type = ti_type<T>;
559  // Calling series types.
560  using term_type = typename base::term_type;
561  // The value type of the trigonometric key.
562  using k_value_type = typename base::term_type::key_type::value_type;
563  // Divisor series types.
564  using d_series_type = typename base::term_type::cf_type;
565  using d_term_type = typename d_series_type::term_type;
566  using d_cf_type = typename d_term_type::cf_type;
567  using d_key_type = typename d_term_type::key_type;
568  // Initialise the return value.
569  return_type retval(0);
570  // The symbol set for the divisor series coming out of the trigonometric arguments.
571  symbol_set div_symbols;
572  for (auto it = this->m_symbol_set.begin(); it != this->m_symbol_set.end(); ++it) {
573  div_symbols.add(std::string("\\nu_{") + it->get_name() + "}");
574  }
575  // A temp vector of integers used to normalise the divisors coming
576  // out of the integration operation from the trig keys.
577  std::vector<integer> tmp_int;
578  // Build the return value.
579  const auto it_f = this->m_container.end();
580  for (auto it = this->m_container.begin(); it != it_f; ++it) {
581  // Clear the tmp integer vector.
582  tmp_int.clear();
583  // Get the vector of trigonometric multipliers.
584  const auto trig_vector = it->m_key.unpack(this->m_symbol_set);
585  // Copy it over to the tmp_int as integer values.
586  std::transform(trig_vector.begin(),trig_vector.end(),std::back_inserter(tmp_int),
587  [](const k_value_type &n) {return integer(n);});
588  // Determine the common divisor.
589  // NOTE: both the divisor and the trigonometric key share the canonical form in which the
590  // first nonzero multiplier is positive, so we don't need to account for sign flips when
591  // constructing a divisor from the trigonometric part. We just need to take care
592  // of the common divisor.
593  integer cd(0);
594  bool first_nonzero_found = false;
595  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
596  // NOTE: gcd is safe, operating on integers.
597  cd = detail::gcd(cd,*it2);
598  if (!first_nonzero_found && !math::is_zero(*it2)) {
599  piranha_assert(*it2 > 0);
600  first_nonzero_found = true;
601  }
602  }
603  if (unlikely(math::is_zero(cd))) {
604  piranha_throw(std::invalid_argument,"an invalid trigonometric term was encountered while "
605  "attempting a time integration");
606  }
607  // Take the abs of the cd.
608  cd = cd.abs();
609  // Divide the vector by the common divisor.
610  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
611  *it2 /= cd;
612  }
613  // Build the temporary divisor series from the trigonometric arguments.
614  d_series_type div_series;
615  div_series.set_symbol_set(div_symbols);
616  // Build the divisor key.
617  typename d_key_type::value_type exponent(1);
618  d_key_type div_key;
619  div_key.insert(tmp_int.begin(),tmp_int.end(),exponent);
620  // Finish building the temporary divisor series.
621  div_series.insert(d_term_type(d_cf_type(1),std::move(div_key)));
622  // Temporary Poisson series from the current term, with the trig flavour flipped.
623  poisson_series tmp_ps;
624  tmp_ps.set_symbol_set(this->m_symbol_set);
625  auto tmp_key = it->m_key;
626  tmp_key.set_flavour(!tmp_key.get_flavour());
627  tmp_ps.insert(term_type(it->m_cf,std::move(tmp_key)));
628  // Update the return value.
629  auto tmp = (std::move(tmp_ps) * std::move(div_series)) / cd;
630  // It also needs a negation, if the original trig key is a sine.
631  if (!it->m_key.get_flavour()) {
632  math::negate(tmp);
633  }
634  retval += std::move(tmp);
635  }
636  return retval;
637  }
