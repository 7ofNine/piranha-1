198  // Convert an input polynomial to a Poisson series of type RetT. The conversion
199  // will be successful if the polynomial can be reduced to an integral linear combination of
200  // symbols.
201  template <bool IsCos, typename RetT, typename Poly>
202  static RetT poly_to_ps(const Poly &poly)
203  {
204  // Shortcuts.
205  typedef typename RetT::term_type term_type;
206  typedef typename term_type::cf_type cf_type;
207  typedef typename term_type::key_type key_type;
208  typedef typename key_type::value_type value_type;
209  // Try to get the integral combination from the poly coefficient.
210  auto lc = poly.integral_combination();
211  // Change sign if needed.
212  bool sign_change = false;
213  if (!lc.empty() && lc.begin()->second.sign() < 0) {
214  std::for_each(lc.begin(), lc.end(), [](std::pair<const std::string, integer> &p) { p.second.negate(); });
215  sign_change = true;
216  }
217  // Return value.
218  RetT retval;
219  // Build vector of integral multipliers.
220  std::vector<value_type> v;
221  for (auto it = lc.begin(); it != lc.end(); ++it) {
222  retval.m_symbol_set.add(it->first);
223  // NOTE: this should probably be a safe_cast.
224  // The value type here could be anything, and not guaranteed to be castable,
225  // even if in the current implementation this is guaranteed to be a signed
226  // int of some kind.
227  v.push_back(static_cast<value_type>(it->second));
228  }
229  // Build term, fix signs and flavour and move-insert it.
230  term_type term(cf_type(1), key_type(v.begin(), v.end()));
231  if (!IsCos) {
232  term.m_key.set_flavour(false);
233  if (sign_change) {
234  // NOTE: negate is supported by any coefficient type.
235  math::negate(term.m_cf);
236  }
237  }
238  retval.insert(std::move(term));
239  return retval;
240  }
241  // Special sin/cos implementation when we have reached the first polynomial coefficient in the hierarchy.
242  template <bool IsCos, typename RetT, typename T,
243  typename std::enable_if<std::is_base_of<detail::polynomial_tag, typename T::term_type::cf_type>::value,
244  int>::type
245  = 0>
246  RetT special_sin_cos(const T &s) const
247  {
248  // Do something only if the series is equivalent to a polynomial.
249  if (s.is_single_coefficient() && !s.empty()) {
250  try {
251  return poly_to_ps<IsCos, RetT>(s._container().begin()->m_cf);
252  } catch (const std::invalid_argument &) {
253  // Interpret invalid_argument as a failure in extracting integral combination,
254  // and move on.
255  }
256  }
257  return special_sin_cos_failure(*this, std::integral_constant<bool, IsCos>{});
258  }

376  template <typename T>
377  struct integrate_type_<T,
378  typename std::
379  enable_if<std::is_base_of<detail::polynomial_tag, typename T::term_type::cf_type>::value
380  && basic_integrate_requirements<T, pc_res_type<T>>::value &&
381  // We need to be able to add in the npc type.
382  is_addable_in_place<pc_res_type<T>, npc_res_type<T>>::value &&

388  std::vector<std::string>{}))>::value

391  std::is_constructible<i_cf_type_p<T>, i_cf_type<T>>::value &&
392  // This type needs also to be negated.
393  has_negate<i_cf_type_p<T>>::value>::type> {
394  using type = pc_res_type<T>;
395  };
396  // The final typedef.
397  template <typename T>
398  using integrate_type = typename std::enable_if<is_returnable<typename integrate_type_<T>::type>::value,
399  typename integrate_type_<T>::type>::type;
400  template <typename T = poisson_series>
401  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
402  const std::true_type &) const
403  {
404  typedef typename base::term_type term_type;
405  typedef typename term_type::cf_type cf_type;
406  integer degree;
407  try {
408  degree = safe_cast<integer>(math::degree(term.m_cf, {s.get_name()}));
409  } catch (const std::invalid_argument &) {
410  piranha_throw(
411  std::invalid_argument,
412  "unable to perform Poisson series integration: cannot convert polynomial degree to an integer");
413  }
414  // If the variable is in both cf and key, and the cf degree is negative, we cannot integrate.
415  if (degree.sign() < 0) {
416  piranha_throw(
417  std::invalid_argument,
418  "unable to perform Poisson series integration: polynomial coefficient has negative integral degree");
419  }
420  // Init retval and auxiliary quantities for the iteration.
421  auto key_int = term.m_key.integrate(s, this->m_symbol_set);
422  // NOTE: here we are sure that the variable is contained in the monomial.
423  piranha_assert(key_int.first != 0);
424  poisson_series tmp;
425  tmp.set_symbol_set(this->m_symbol_set);
426  // NOTE: not move for .second, as it is needed in the loop below.
427  tmp.insert(term_type(cf_type(1), key_int.second));
428  i_cf_type_p<T> p_cf(term.m_cf / std::move(key_int.first));
429  integrate_type<T> retval(p_cf * tmp);
430  for (integer i(1); i <= degree; ++i) {
431  key_int = key_int.second.integrate(s, this->m_symbol_set);
432  piranha_assert(key_int.first != 0);
433  p_cf = math::partial(p_cf / std::move(key_int.first), s.get_name());
434  // Sign change due to the second portion of integration by part.
435  math::negate(p_cf);
436  tmp = poisson_series{};
437  tmp.set_symbol_set(this->m_symbol_set);
438  // NOTE: don't move second.
439  tmp.insert(term_type(cf_type(1), key_int.second));
440  retval += p_cf * tmp;
441  }
442  return retval;
443  }

452  // Time integration.
453  // First the implementation for divisor series.
454  template <typename T>
455  using ti_type_divisor_
456  = decltype((std::declval<const T &>() * std::declval<const typename T::term_type::cf_type &>())
457  / std::declval<const integer &>());
458  template <typename T>
459  using ti_type_divisor = typename std::enable_if<std::is_constructible<ti_type_divisor_<T>, int>::value
460  && is_addable_in_place<ti_type_divisor_<T>>::value
461  && std::is_base_of<detail::divisor_series_tag,
462  typename T::term_type::cf_type>::value,
463  ti_type_divisor_<T>>::type;
464  template <typename T = poisson_series>
465  ti_type_divisor<T> t_integrate_impl(const std::vector<std::string> &names) const
466  {
467  using return_type = ti_type<T>;
468  // Calling series types.
469  using term_type = typename base::term_type;
470  // The value type of the trigonometric key.
471  using k_value_type = typename term_type::key_type::value_type;
472  // Divisor series types.
473  using d_series_type = typename base::term_type::cf_type;
474  using d_term_type = typename d_series_type::term_type;
475  using d_cf_type = typename d_term_type::cf_type;
476  using d_key_type = typename d_term_type::key_type;
477  // Initialise the return value.
478  return_type retval(0);
479  // Setup of the symbol set.
480  piranha_assert(names.size() == this->m_symbol_set.size());
481  symbol_set div_symbols;
482  for (const auto &name : names) {
483  div_symbols.add(name);
484  }
485  // A temp vector of integers used to normalise the divisors coming
486  // out of the integration operation from the trig keys.
487  std::vector<integer> tmp_int;
488  // Build the return value.
489  const auto it_f = this->m_container.end();
490  for (auto it = this->m_container.begin(); it != it_f; ++it) {
491  // Clear the tmp integer vector.
492  tmp_int.clear();
493  // Get the vector of trigonometric multipliers.
494  const auto trig_vector = it->m_key.unpack(this->m_symbol_set);
495  // Copy it over to the tmp_int as integer values.
496  std::transform(trig_vector.begin(), trig_vector.end(), std::back_inserter(tmp_int),
497  [](const k_value_type &n) { return integer(n); });
498  // Determine the common divisor.
499  // NOTE: both the divisor and the trigonometric key share the canonical form in which the
500  // first nonzero multiplier is positive, so we don't need to account for sign flips when
501  // constructing a divisor from the trigonometric part. We just need to take care
502  // of the common divisor.
503  integer cd(0);
504  bool first_nonzero_found = false;
505  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
506  // NOTE: gcd is safe, operating on integers.
507  cd = math::gcd(cd, *it2);
508  if (!first_nonzero_found && !math::is_zero(*it2)) {
509  piranha_assert(*it2 > 0);
510  first_nonzero_found = true;
511  }
512  }
513  if (unlikely(math::is_zero(cd))) {
514  piranha_throw(std::invalid_argument, "an invalid trigonometric term was encountered while "
515  "attempting a time integration");
516  }
517  // Take the abs of the cd.
518  cd = cd.abs();
519  // Divide the vector by the common divisor.
520  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
521  *it2 /= cd;
522  }
523  // Build the temporary divisor series from the trigonometric arguments.
524  d_series_type div_series;
525  div_series.set_symbol_set(div_symbols);
526  // Build the divisor key.
527  typename d_key_type::value_type exponent(1);
528  d_key_type div_key;
529  div_key.insert(tmp_int.begin(), tmp_int.end(), exponent);
530  // Finish building the temporary divisor series.
531  div_series.insert(d_term_type(d_cf_type(1), std::move(div_key)));
532  // Temporary Poisson series from the current term, with the trig flavour flipped.
533  poisson_series tmp_ps;
534  tmp_ps.set_symbol_set(this->m_symbol_set);
535  auto tmp_key = it->m_key;
536  tmp_key.set_flavour(!tmp_key.get_flavour());
537  tmp_ps.insert(term_type(it->m_cf, std::move(tmp_key)));
538  // Update the return value.
539  auto tmp = (std::move(tmp_ps) * std::move(div_series)) / cd;
540  // It also needs a negation, if the original trig key is a sine.
541  if (!it->m_key.get_flavour()) {
542  math::negate(tmp);
543  }
544  retval += std::move(tmp);
545  }
546  return retval;
547  }
548  // Implementation when the coefficient is a rational function.
549  template <
550  typename T = poisson_series,
551  typename std::enable_if<std::is_base_of<detail::rational_function_tag, typename T::term_type::cf_type>::value,
552  int>::type
553  = 0>
554  T t_integrate_impl(const std::vector<std::string> &names) const
555  {
556  // Types from base.
557  using term_type = typename base::term_type;
558  using r_type = typename term_type::cf_type;
559  using p_type = typename r_type::p_type;
560  using key_type = typename term_type::key_type;
561  piranha_assert(names.size() == this->get_symbol_set().size());
562  T retval;
563  retval.set_symbol_set(this->get_symbol_set());
564  for (const auto &t : this->_container()) {
565  // Get the flavour of the current trig monomial.
566  const bool flavour = t.m_key.get_flavour();
567  // Get the vector of trigonometric multipliers.
568  const auto trig_vector = t.m_key.unpack(this->m_symbol_set);
569  // Construct a polynomial from the extracted multipliers, using
570  // the names in linear combination.
571  p_type tmp;
572  for (decltype(trig_vector.size()) i = 0u; i < trig_vector.size(); ++i) {
573  tmp += p_type{names[static_cast<decltype(names.size())>(i)]} * trig_vector[i];
574  }
575  if (math::is_zero(tmp)) {
576  piranha_throw(std::invalid_argument, "an invalid trigonometric term was encountered while "
577  "attempting a time integration");
578  }
579  // Construct the new coefficient from the current coefficient,
580  // divided by the newly constructed poly.
581  r_type r{t.m_cf / std::move(tmp)};
582  // Need to negate the coefficient if the current trig monomial is a sine.
583  if (!flavour) {
584  math::negate(r);
585  }
586  // Create a copy of the current trig monomial and flip its flavour.
587  key_type tmp_key{t.m_key};
588  tmp_key.set_flavour(!flavour);
589  // Create new term from new coefficient and key, and insert it.
590  retval.insert(term_type{std::move(r), std::move(tmp_key)});
591  }
592  return retval;
593  }

716  template <typename T = poisson_series>
717  integrate_type<T> integrate(const std::string &name) const
718  {
719  typedef typename base::term_type term_type;
720  typedef typename term_type::cf_type cf_type;
721  // Turn name into symbol.
722  const symbol s(name);
723  // Init the return value.
724  integrate_type<T> retval(0);
725  const auto it_f = this->m_container.end();
726  for (auto it = this->m_container.begin(); it != it_f; ++it) {
727  // Integrate the key first.
728  auto key_int = it->m_key.integrate(s, this->m_symbol_set);
729  // If the variable does not appear in the monomial, try deferring the integration
730  // to the coefficient.
731  if (key_int.first == 0) {
732  integrate_coefficient_only(retval, it, name);
733  continue;
734  }
735  // The variable is in the monomial, let's check if the variable is also in the coefficient.
736  if (math::is_zero(math::partial(it->m_cf, name))) {
737  // No variable in the coefficient, proceed with the integrated key and divide by multiplier.
738  poisson_series tmp;
739  tmp.set_symbol_set(this->m_symbol_set);
740  tmp.insert(term_type(cf_type(1), std::move(key_int.second)));
741  retval += (std::move(tmp) * it->m_cf) / key_int.first;
742  } else {
743  // With the variable both in the coefficient and the key, we only know how to proceed with polynomials.
744  retval += integrate_impl(
745  s, *it, std::integral_constant<bool, std::is_base_of<detail::polynomial_tag, Cf>::value>());
746  }
747  }
748  return retval;

843  template <typename T>
844  using call_enabler = typename std::enable_if<key_is_multipliable<typename T::term_type::cf_type,
845  typename T::term_type::key_type>::value,
846  int>::type;
847  void divide_by_two(Series &s) const
848  {
849  // NOTE: if we ever implement multi-threaded series division we most likely need
850  // to revisit this.
851  piranha_assert(this->m_n_threads > 0u);
852  if (this->m_n_threads == 1u) {
853  // This is possible, as the requirements of series divisibility and trig key
854  // multipliability overlap.
855  s /= 2;
856  } else {
857  using bucket_size_type = typename base::bucket_size_type;
858  using term_type = typename Series::term_type;
859  auto &container = s._container();
860  std::atomic<bucket_size_type> total_erase_count(0u);
861  auto divider
862  = [&container, &total_erase_count, this](bucket_size_type start_idx, bucket_size_type end_idx) {
863  // A vector of terms to be erased at each bucket iteration.
864  std::vector<term_type> term_list;
865  // Total number of terms erased by this thread.
866  bucket_size_type erase_count = 0u;
867  for (; start_idx != end_idx; ++start_idx) {
868  // Reset the list of terms to be erased.
869  term_list.clear();
870  const auto &list = container._get_bucket_list(start_idx);
871  for (const auto &t : list) {
872  t.m_cf /= 2;
873  if (unlikely(t.is_ignorable(this->m_ss))) {
874  term_list.push_back(t);
875  }
876  }
877  for (const auto &t : term_list) {
878  container._erase(container._find(t, start_idx));
879  erase_count = static_cast<bucket_size_type>(erase_count + 1u);
880  }
881  }
882  // Update the global counter of erased terms.
883  total_erase_count += erase_count;
884  };
885  // Buckets per thread.
886  const auto bpt = static_cast<bucket_size_type>(container.bucket_count() / this->m_n_threads);
887  // Go with the threads.
888  future_list<decltype(divider(0u, 0u))> ff_list;
889  try {
890  for (unsigned i = 0u; i < this->m_n_threads; ++i) {
891  const auto start_idx = static_cast<bucket_size_type>(bpt * i);
892  // Special casing for the last thread.
893  const auto end_idx = (i == this->m_n_threads - 1u) ? container.bucket_count()
894  : static_cast<bucket_size_type>(bpt * (i + 1u));
895  ff_list.push_back(thread_pool::enqueue(i, divider, start_idx, end_idx));
896  }
897  // First let's wait for everything to finish.
898  ff_list.wait_all();
899  // Then, let's handle the exceptions.
900  ff_list.get_all();
901  } catch (...) {
902  ff_list.wait_all();
903  // Clear out the container as it might be in an inconsistent state.
904  container.clear();
905  throw;
906  }
907  // Final size update - all of this is noexcept.
908  const auto tot = total_erase_count.load();
909  piranha_assert(tot <= s.size());
910  container._update_size(static_cast<bucket_size_type>(s.size() - tot));
911  }
912  }
