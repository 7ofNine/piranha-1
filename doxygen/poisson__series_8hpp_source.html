164  // Special sin/cos implementation when we have reached the first polynomial coefficient in the hierarchy.
165  template <bool IsCos, typename RetT, typename T, typename std::enable_if<std::is_base_of<detail::polynomial_tag,
166  typename T::term_type::cf_type>::value,int>::type = 0>
167  RetT special_sin_cos(const T &s) const
168  {
169  // Do something only if the series is equivalent to a polynomial.
170  if (s.is_single_coefficient() && !s.empty()) {
171  try {
172  // Shortcuts.
173  typedef typename RetT::term_type term_type;
174  typedef typename term_type::cf_type cf_type;
175  typedef typename term_type::key_type key_type;
176  typedef typename key_type::value_type value_type;
177  // Try to get the integral combination from the poly coefficient.
178  auto lc = s._container().begin()->m_cf.integral_combination();
179  // Change sign if needed.
180  bool sign_change = false;
181  if (!lc.empty() && lc.begin()->second.sign() < 0) {
182  std::for_each(lc.begin(),lc.end(),[](std::pair<const std::string,integer> &p) {p.second.negate();});
183  sign_change = true;
184  }
185  // Return value.
186  RetT retval;
187  // Build vector of integral multipliers.
188  std::vector<value_type> v;
189  for (auto it = lc.begin(); it != lc.end(); ++it) {
190  retval.m_symbol_set.add(it->first);
191  // NOTE: this should probably be a safe_cast.
192  // The value type here could be anything, and not guaranteed to be castable,
193  // even if in the current implementation this is guaranteed to be a signed
194  // int of some kind.
195  v.push_back(static_cast<value_type>(it->second));
196  }
197  // Build term, fix signs and flavour and move-insert it.
198  term_type term(cf_type(1),key_type(v.begin(),v.end()));
199  if (!IsCos) {
200  term.m_key.set_flavour(false);
201  if (sign_change) {
202  // NOTE: negate is supported by any coefficient type.
203  math::negate(term.m_cf);
204  }
205  }
206  retval.insert(std::move(term));
207  return retval;
208  } catch (const std::invalid_argument &) {
209  // Interpret invalid_argument as a failure in extracting integral combination,
210  // and move on.
211  }
212  }
213  return special_sin_cos_failure(*this,std::integral_constant<bool,IsCos>{});
214  }
215  // The coefficient is not a polynomial: recurse to the inner coefficient type, if the current coefficient type
216  // is suitable.
217  template <bool IsCos, typename RetT, typename T, typename std::enable_if<!std::is_base_of<detail::polynomial_tag,
218  typename T::term_type::cf_type>::value,int>::type = 0>
219  RetT special_sin_cos(const T &s) const

222  return special_sin_cos<IsCos,RetT>(s._container().begin()->m_cf);
223  }
224  return special_sin_cos_failure(*this,std::integral_constant<bool,IsCos>{});
225  }

330  template <typename T = poisson_series>
331  integrate_type<T> integrate_impl(const symbol &s, const typename base::term_type &term,
332  const std::true_type &) const
333  {
334  typedef typename base::term_type term_type;
335  typedef typename term_type::cf_type cf_type;
336  integer degree;
337  try {
338  degree = safe_cast<integer>(math::degree(term.m_cf,{s.get_name()}));
339  } catch (const std::invalid_argument &) {
340  piranha_throw(std::invalid_argument,
341  "unable to perform Poisson series integration: cannot convert polynomial degree to an integer");
342  }
343  // If the variable is in both cf and key, and the cf degree is negative, we cannot integrate.
344  if (degree.sign() < 0) {
345  piranha_throw(std::invalid_argument,
346  "unable to perform Poisson series integration: polynomial coefficient has negative integral degree");
347  }
348  // Init retval and auxiliary quantities for the iteration.
349  auto key_int = term.m_key.integrate(s,this->m_symbol_set);
350  // NOTE: here we are sure that the variable is contained in the monomial.
351  piranha_assert(key_int.first != 0);
352  poisson_series tmp;
353  tmp.set_symbol_set(this->m_symbol_set);
354  // NOTE: not move for .second, as it is needed in the loop below.
355  tmp.insert(term_type(cf_type(1),key_int.second));
356  i_cf_type_p<T> p_cf(term.m_cf / std::move(key_int.first));
357  integrate_type<T> retval(p_cf * tmp);
358  for (integer i(1); i <= degree; ++i) {
359  key_int = key_int.second.integrate(s,this->m_symbol_set);
360  piranha_assert(key_int.first != 0);
361  p_cf = math::partial(p_cf / std::move(key_int.first),s.get_name());
362  // Sign change due to the second portion of integration by part.
363  math::negate(p_cf);
364  tmp = poisson_series{};
365  tmp.set_symbol_set(this->m_symbol_set);
366  // NOTE: don't move second.
367  tmp.insert(term_type(cf_type(1),key_int.second));
368  retval += p_cf * tmp;
369  }
370  return retval;
371  }

388  // Implementation of time integration.
389  template <typename T = poisson_series>
390  ti_type<T> t_integrate_impl(const std::vector<std::string> &names) const
391  {
392  using return_type = ti_type<T>;
393  // Calling series types.
394  using term_type = typename base::term_type;
395  // The value type of the trigonometric key.
396  using k_value_type = typename base::term_type::key_type::value_type;
397  // Divisor series types.
398  using d_series_type = typename base::term_type::cf_type;
399  using d_term_type = typename d_series_type::term_type;
400  using d_cf_type = typename d_term_type::cf_type;
401  using d_key_type = typename d_term_type::key_type;
402  // Initialise the return value.
403  return_type retval(0);
404  // Setup of the symbol set.
405  piranha_assert(names.size() == this->m_symbol_set.size());
406  symbol_set div_symbols;
407  for (const auto &name: names) {
408  div_symbols.add(name);
409  }
410  // A temp vector of integers used to normalise the divisors coming
411  // out of the integration operation from the trig keys.
412  std::vector<integer> tmp_int;
413  // Build the return value.
414  const auto it_f = this->m_container.end();
415  for (auto it = this->m_container.begin(); it != it_f; ++it) {
416  // Clear the tmp integer vector.
417  tmp_int.clear();
418  // Get the vector of trigonometric multipliers.
419  const auto trig_vector = it->m_key.unpack(this->m_symbol_set);
420  // Copy it over to the tmp_int as integer values.
421  std::transform(trig_vector.begin(),trig_vector.end(),std::back_inserter(tmp_int),
422  [](const k_value_type &n) {return integer(n);});
423  // Determine the common divisor.
424  // NOTE: both the divisor and the trigonometric key share the canonical form in which the
425  // first nonzero multiplier is positive, so we don't need to account for sign flips when
426  // constructing a divisor from the trigonometric part. We just need to take care
427  // of the common divisor.
428  integer cd(0);
429  bool first_nonzero_found = false;
430  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
431  // NOTE: gcd is safe, operating on integers.
432  cd = detail::gcd(cd,*it2);
433  if (!first_nonzero_found && !math::is_zero(*it2)) {
434  piranha_assert(*it2 > 0);
435  first_nonzero_found = true;
436  }
437  }
438  if (unlikely(math::is_zero(cd))) {
439  piranha_throw(std::invalid_argument,"an invalid trigonometric term was encountered while "
440  "attempting a time integration");
441  }
442  // Take the abs of the cd.
443  cd = cd.abs();
444  // Divide the vector by the common divisor.
445  for (auto it2 = tmp_int.begin(); it2 != tmp_int.end(); ++it2) {
446  *it2 /= cd;
447  }
448  // Build the temporary divisor series from the trigonometric arguments.
449  d_series_type div_series;
450  div_series.set_symbol_set(div_symbols);
451  // Build the divisor key.
452  typename d_key_type::value_type exponent(1);
453  d_key_type div_key;
454  div_key.insert(tmp_int.begin(),tmp_int.end(),exponent);
455  // Finish building the temporary divisor series.
456  div_series.insert(d_term_type(d_cf_type(1),std::move(div_key)));
457  // Temporary Poisson series from the current term, with the trig flavour flipped.
458  poisson_series tmp_ps;
459  tmp_ps.set_symbol_set(this->m_symbol_set);
460  auto tmp_key = it->m_key;
461  tmp_key.set_flavour(!tmp_key.get_flavour());
462  tmp_ps.insert(term_type(it->m_cf,std::move(tmp_key)));
463  // Update the return value.
464  auto tmp = (std::move(tmp_ps) * std::move(div_series)) / cd;
465  // It also needs a negation, if the original trig key is a sine.
466  if (!it->m_key.get_flavour()) {
467  math::negate(tmp);
468  }
469  retval += std::move(tmp);
470  }
471  return retval;

582  template <typename T = poisson_series>
583  integrate_type<T> integrate(const std::string &name) const
584  {
585  typedef typename base::term_type term_type;
586  typedef typename term_type::cf_type cf_type;
587  // Turn name into symbol.
588  const symbol s(name);
589  // Init the return value.
590  integrate_type<T> retval(0);
591  const auto it_f = this->m_container.end();
592  for (auto it = this->m_container.begin(); it != it_f; ++it) {
593  // Integrate the key first.
594  auto key_int = it->m_key.integrate(s,this->m_symbol_set);
595  // If the variable does not appear in the monomial, try deferring the integration
596  // to the coefficient.
597  if (key_int.first == 0) {
598  integrate_coefficient_only(retval,it,name);
599  continue;
600  }
601  // The variable is in the monomial, let's check if the variable is also in the coefficient.
602  if (math::is_zero(math::partial(it->m_cf,name))) {
603  // No variable in the coefficient, proceed with the integrated key and divide by multiplier.
604  poisson_series tmp;
605  tmp.set_symbol_set(this->m_symbol_set);
606  tmp.insert(term_type(cf_type(1),std::move(key_int.second)));
607  retval += (std::move(tmp) * it->m_cf) / key_int.first;
608  } else {
609  // With the variable both in the coefficient and the key, we only know how to proceed with polynomials.

704 class series_multiplier<Series,detail::ps_series_multiplier_enabler<Series>> : public base_series_multiplier<Series>
705 {
706  using base = base_series_multiplier<Series>;
707  template <typename T>
708  using call_enabler = typename std::enable_if<key_is_multipliable<typename T::term_type::cf_type,
709  typename T::term_type::key_type>::value,int>::type;
710  void divide_by_two(Series &s) const
711  {
712  // NOTE: if we ever implement multi-threaded series division we most likely need
713  // to revisit this.
714  piranha_assert(this->m_n_threads > 0u);
715  if (this->m_n_threads == 1u) {
716  // This is possible, as the requirements of series divisibility and trig key
717  // multipliability overlap.
718  s /= 2;
719  } else {
720  using bucket_size_type = typename base::bucket_size_type;
721  using term_type = typename Series::term_type;
722  auto &container = s._container();
723  std::atomic<bucket_size_type> total_erase_count(0u);
724  auto divider = [&container,&total_erase_count,this](bucket_size_type start_idx, bucket_size_type end_idx) {
725  // A vector of terms to be erased at each bucket iteration.
726  std::vector<term_type> term_list;
727  // Total number of terms erased by this thread.
728  bucket_size_type erase_count = 0u;
729  for (; start_idx != end_idx; ++start_idx) {
730  // Reset the list of terms to be erased.
731  term_list.clear();
732  const auto &list = container._get_bucket_list(start_idx);
733  for (const auto &t: list) {
734  t.m_cf /= 2;
735  if (unlikely(t.is_ignorable(this->m_ss))) {
736  term_list.push_back(t);
737  }
738  }
739  for (const auto &t: term_list) {
740  container._erase(container._find(t,start_idx));
741  erase_count = static_cast<bucket_size_type>(erase_count + 1u);
742  }
743  }
744  // Update the global counter of erased terms.
745  total_erase_count += erase_count;
746  };
747  // Buckets per thread.
748  const auto bpt = static_cast<bucket_size_type>(container.bucket_count() / this->m_n_threads);
749  // Go with the threads.
750  future_list<decltype(thread_pool::enqueue(0u,divider,0u,0u))> ff_list;
751  try {
752  for (unsigned i = 0u; i < this->m_n_threads; ++i) {
753  const auto start_idx = static_cast<bucket_size_type>(bpt * i);
754  // Special casing for the last thread.
755  const auto end_idx = (i == this->m_n_threads - 1u) ? container.bucket_count() :
756  static_cast<bucket_size_type>(bpt * (i + 1u));
757  ff_list.push_back(thread_pool::enqueue(i,divider,start_idx,end_idx));
758  }
759  // First let's wait for everything to finish.
760  ff_list.wait_all();
761  // Then, let's handle the exceptions.
762  ff_list.get_all();
763  } catch (...) {
764  ff_list.wait_all();
765  // Clear out the container as it might be in an inconsistent state.
766  container.clear();
767  throw;
768  }
769  // Final size update - all of this is noexcept.
770  const auto tot = total_erase_count.load();
771  piranha_assert(tot <= s.size());
772  container._update_size(static_cast<bucket_size_type>(s.size() - tot));
773  }
774  }
775  public:
777  using base::base;
779 
790  template <typename T = Series, call_enabler<T> = 0>
791  Series operator()() const
792  {
793  auto retval(this->plain_multiplication());
794  divide_by_two(retval);
795  return retval;
796  }
