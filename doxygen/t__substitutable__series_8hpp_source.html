127  is_addable_in_place<return_type_<T1,U1,Term1>>::value,return_type_<T1,U1,Term1>>::type;
128  template <typename T1, typename U1, typename Term1>
129  static return_type<T1,U1,Term1> subs(const Term1 &t, const std::string &name, const T1 &c, const U1 &s, const symbol_set &s_set)
130  {
131  return_type<T1,U1,Term1> retval(0);
132  const auto key_subs = t.m_key.t_subs(name,c,s,s_set);
133  for (const auto &x: key_subs) {
134  Derived tmp;
135  tmp.m_symbol_set = s_set;

137  retval += x.first * tmp;
138  }
139  return retval;
