523  template <typename U = T, partial_enabler<U> = 0>
524  std::pair<U,monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
525  {
526  if (!is_compatible(args)) {
527  piranha_throw(std::invalid_argument,"invalid size of arguments set");
528  }
529  // Cannot take derivative wrt more than one variable, and the position of that variable
530  // must be compatible with the monomial.
531  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
532  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
533  }
534  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
535  // variable with zero exponent.
536  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
537  // there is a valid position and hence the size must be not zero.
538  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
539  return std::make_pair(U(0),monomial(args));
540  }
541  // Copy of the original monomial.
542  monomial m(*this);
543  auto m_b = m.begin();
544  // Original exponent.
545  U v(m_b[*p.begin()]);
546  // Decrement the exponent in the monomial.
547  ip_dec(m_b[*p.begin()]);
548  return std::make_pair(std::move(v),std::move(m));
549  }

580  template <typename U = T, integrate_enabler<U> = 0>
581  std::pair<U,monomial> integrate(const symbol &s, const symbol_set &args) const
582  {
583  typedef typename base::size_type size_type;
584  if (!is_compatible(args)) {
585  piranha_throw(std::invalid_argument,"invalid size of arguments set");
586  }
587  monomial retval;
588  T expo(0), one(1);
589  for (size_type i = 0u; i < args.size(); ++i) {
590  if (math::is_zero(expo) && s < args[i]) {
591  // If we went past the position of s in args and still we
592  // have not performed the integration, it means that we need to add
593  // a new exponent.
594  retval.push_back(one);
595  expo = one;
596  }
597  retval.push_back((*this)[i]);
598  if (args[i] == s) {
599  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
600  // we are never going to land here as args[i] is at this point never going to be s.
601  ip_inc(retval[i]);
602  if (math::is_zero(retval[i])) {
603  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
604  }
605  expo = retval[i];
606  }
607  }
608  // If expo is still zero, it means we need to add a new exponent at the end.
609  if (math::is_zero(expo)) {
610  retval.push_back(one);
611  expo = one;
612  }
613  return std::make_pair(std::move(expo),std::move(retval));
614  }

717  {
