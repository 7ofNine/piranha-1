472  template <typename U = T, linarg_enabler<U> = 0>
473  std::string linear_argument(const symbol_set &args) const
474  {
475  if (!is_compatible(args)) {
476  piranha_throw(std::invalid_argument,"invalid size of arguments set");
477  }
478  typedef typename base::size_type size_type;
479  const size_type size = this->size();
480  size_type n_linear = 0u, candidate = 0u;
481  for (size_type i = 0u; i < size; ++i) {
482  integer tmp;
483  try {
484  tmp = safe_cast<integer>((*this)[i]);
485  } catch (const std::invalid_argument &) {
486  piranha_throw(std::invalid_argument,"exponent is not an integer");
487  }
488  if (tmp == 0) {
489  continue;
490  }
491  if (tmp != 1) {
492  piranha_throw(std::invalid_argument,"exponent is not unitary");
493  }
494  candidate = i;
495  ++n_linear;
496  }
497  if (n_linear != 1u) {
498  piranha_throw(std::invalid_argument,"monomial is not linear");
499  }

565  template <typename U = T, partial_enabler<U> = 0>
566  std::pair<U,monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
567  {
568  if (!is_compatible(args)) {
569  piranha_throw(std::invalid_argument,"invalid size of arguments set");
570  }
571  // Cannot take derivative wrt more than one variable, and the position of that variable
572  // must be compatible with the monomial.
573  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
574  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
575  }
576  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
577  // variable with zero exponent.
578  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
579  // there is a valid position and hence the size must be not zero.
580  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
581  return std::make_pair(U(0),monomial(args));
582  }
583  // Copy of the original monomial.
584  monomial m(*this);
585  auto m_b = m.begin();
586  // Original exponent.
587  U v(m_b[*p.begin()]);
588  // Decrement the exponent in the monomial.
589  ip_dec(m_b[*p.begin()]);
590  return std::make_pair(std::move(v),std::move(m));
591  }

622  template <typename U = T, integrate_enabler<U> = 0>
623  std::pair<U,monomial> integrate(const symbol &s, const symbol_set &args) const
624  {
625  typedef typename base::size_type size_type;
626  if (!is_compatible(args)) {
627  piranha_throw(std::invalid_argument,"invalid size of arguments set");
628  }
629  monomial retval;
630  T expo(0), one(1);
631  for (size_type i = 0u; i < args.size(); ++i) {
632  if (math::is_zero(expo) && s < args[i]) {
633  // If we went past the position of s in args and still we
634  // have not performed the integration, it means that we need to add
635  // a new exponent.
636  retval.push_back(one);
637  expo = one;
638  }
639  retval.push_back((*this)[i]);
640  if (args[i] == s) {
641  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
642  // we are never going to land here as args[i] is at this point never going to be s.
643  ip_inc(retval[i]);
644  if (math::is_zero(retval[i])) {
645  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
646  }
647  expo = retval[i];
648  }
649  }
650  // If expo is still zero, it means we need to add a new exponent at the end.
651  if (math::is_zero(expo)) {
652  retval.push_back(one);
653  expo = one;
654  }
655  return std::make_pair(std::move(expo),std::move(retval));
656  }

706  {

724  if (!num_str.empty() && !den_str.empty()) {
725  os << "\\frac{" << num_str << "}{" << den_str << "}";

727  os << num_str;

757  template <typename U>
758  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
759  {
760  using return_type = eval_type<U>;
761  using size_type = typename base::size_type;
762  // NOTE: the positions map must have the same number of elements as this, and it must
763  // be made of consecutive positions [0,size - 1].
764  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
765  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
766  }
767  // Args must be compatible with this.
768  if (unlikely(args.size() != this->size())) {
769  piranha_throw(std::invalid_argument,"invalid size of arguments set");
770  }
771  return_type retval(1);
772  auto it = pmap.begin();
773  for (size_type i = 0u; i < this->size(); ++i, ++it) {
774  piranha_assert(it != pmap.end() && it->first == i);
775  retval *= math::pow(it->second,(*this)[i]);
776  }
777  piranha_assert(it == pmap.end());
778  return retval;
779  }
