539  template <typename U = T, linarg_enabler<U> = 0>
540  std::string linear_argument(const symbol_set &args) const
541  {
542  if (!is_compatible(args)) {
543  piranha_throw(std::invalid_argument, "invalid size of arguments set");
544  }
545  typedef typename base::size_type size_type;
546  const size_type size = this->size();
547  size_type n_linear = 0u, candidate = 0u;
548  for (size_type i = 0u; i < size; ++i) {
549  integer tmp;
550  try {
551  tmp = safe_cast<integer>((*this)[i]);
552  } catch (const safe_cast_failure &) {
553  piranha_throw(std::invalid_argument, "exponent is not an integer");
554  }
555  if (math::is_zero(tmp)) {
556  continue;
557  }
558  if (!math::is_unitary(tmp)) {
559  piranha_throw(std::invalid_argument, "exponent is not unitary");
560  }
561  candidate = i;
562  ++n_linear;
563  }
564  if (n_linear != 1u) {
565  piranha_throw(std::invalid_argument, "monomial is not linear");
566  }
567  return args[static_cast<decltype(args.size())>(candidate)].get_name();
568  }

591  monomial pow(const U &x, const symbol_set &args) const

633  template <typename U = T, partial_enabler<U> = 0>
634  std::pair<U, monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
635  {
636  if (!is_compatible(args)) {
637  piranha_throw(std::invalid_argument, "invalid size of arguments set");
638  }
639  // Cannot take derivative wrt more than one variable, and the position of that variable
640  // must be compatible with the monomial.
641  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
642  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
643  }
644  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
645  // variable with zero exponent.
646  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
647  // there is a valid position and hence the size must be not zero.
648  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
649  return std::make_pair(U(0), monomial(args));
650  }
651  // Copy of the original monomial.
652  monomial m(*this);
653  auto m_b = m.begin();
654  // Original exponent.
655  U v(m_b[*p.begin()]);
656  // Decrement the exponent in the monomial.
657  ip_dec(m_b[*p.begin()]);
658  return std::make_pair(std::move(v), std::move(m));
659  }

690  template <typename U = T, integrate_enabler<U> = 0>
691  std::pair<U, monomial> integrate(const symbol &s, const symbol_set &args) const
692  {
693  typedef typename base::size_type size_type;
694  if (!is_compatible(args)) {
695  piranha_throw(std::invalid_argument, "invalid size of arguments set");
696  }
697  monomial retval;
698  T expo(0), one(1);
699  for (size_type i = 0u; i < args.size(); ++i) {
700  if (math::is_zero(expo) && s < args[i]) {
701  // If we went past the position of s in args and still we
702  // have not performed the integration, it means that we need to add
703  // a new exponent.
704  retval.push_back(one);
705  expo = one;
706  }
707  retval.push_back((*this)[i]);
708  if (args[i] == s) {
709  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
710  // we are never going to land here as args[i] is at this point never going to be s.
711  ip_inc(retval[i]);
712  if (unlikely(math::is_zero(retval[i]))) {
713  piranha_throw(std::invalid_argument,
714  "unable to perform monomial integration: negative unitary exponent");
715  }
716  expo = retval[i];
717  }
718  }
719  // If expo is still zero, it means we need to add a new exponent at the end.
720  if (math::is_zero(expo)) {
721  retval.push_back(one);
722  expo = one;
723  }
724  return std::make_pair(std::move(expo), std::move(retval));
725  }

738  void print(std::ostream &os, const symbol_set &args) const
739  {
740  if (unlikely(args.size() != this->size())) {
741  piranha_throw(std::invalid_argument, "invalid size of arguments set");
742  }
743  bool empty_output = true;
744  for (typename base::size_type i = 0u; i < this->size(); ++i) {
745  if (!math::is_zero((*this)[i])) {
746  // If we are going to print a symbol, and something has been printed before,
747  // then we are going to place the multiplication sign.
748  if (!empty_output) {
749  os << '*';
750  }

752  empty_output = false;
753  if (!math::is_unitary((*this)[i])) {
754  os << "**" << detail::prepare_for_print((*this)[i]);
755  }

772  void print_tex(std::ostream &os, const symbol_set &args) const
773  {
774  if (unlikely(args.size() != this->size())) {
775  piranha_throw(std::invalid_argument, "invalid size of arguments set");
776  }
777  std::ostringstream oss_num, oss_den, *cur_oss;
778  const T zero(0);
779  T cur_value;
780  for (typename base::size_type i = 0u; i < this->size(); ++i) {
781  cur_value = (*this)[i];
782  if (!math::is_zero(cur_value)) {
783  // NOTE: use this weird form for the test because the presence of operator<()
784  // is already guaranteed and thus we don't need additional requirements on T.
785  if (zero < cur_value) {
786  cur_oss = std::addressof(oss_num);
787  } else {
788  math::negate(cur_value);
789  cur_oss = std::addressof(oss_den);
790  }
791  (*cur_oss) << "{" << args[i].get_name() << "}";
792  if (!math::is_unitary(cur_value)) {
793  (*cur_oss) << "^{" << detail::prepare_for_print(cur_value) << "}";
794  }
795  }
796  }
797  const std::string num_str = oss_num.str(), den_str = oss_den.str();
798  if (!num_str.empty() && !den_str.empty()) {
799  os << "\\frac{" << num_str << "}{" << den_str << "}";
800  } else if (!num_str.empty() && den_str.empty()) {
801  os << num_str;
802  } else if (num_str.empty() && !den_str.empty()) {
803  os << "\\frac{1}{" << den_str << "}";
804  }
805  }

842  template <typename U>
843  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
844  {
845  using return_type = eval_type<U>;
846  using size_type = typename base::size_type;
847  // NOTE: the positions map must have the same number of elements as this, and it must
848  // be made of consecutive positions [0,size - 1].
849  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
850  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
851  }
852  // Args must be compatible with this.
853  if (unlikely(args.size() != this->size())) {
854  piranha_throw(std::invalid_argument, "invalid size of arguments set");
855  }
856  return_type retval(1);
857  auto it = pmap.begin();
858  for (size_type i = 0u; i < this->size(); ++i, ++it) {
859  piranha_assert(it != pmap.end() && it->first == i);
860  retval *= math::pow(it->second, (*this)[i]);
861  }
862  piranha_assert(it == pmap.end());
863  return retval;
864  }
