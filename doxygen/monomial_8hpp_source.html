540  template <typename U = T, linarg_enabler<U> = 0>
541  std::string linear_argument(const symbol_set &args) const
542  {
543  if (!is_compatible(args)) {
544  piranha_throw(std::invalid_argument, "invalid size of arguments set");
545  }
546  typedef typename base::size_type size_type;
547  const size_type size = this->size();
548  size_type n_linear = 0u, candidate = 0u;
549  for (size_type i = 0u; i < size; ++i) {
550  integer tmp;
551  try {
552  tmp = safe_cast<integer>((*this)[i]);
553  } catch (const std::invalid_argument &) {
554  piranha_throw(std::invalid_argument, "exponent is not an integer");
555  }
556  if (tmp == 0) {
557  continue;
558  }
559  if (tmp != 1) {
560  piranha_throw(std::invalid_argument, "exponent is not unitary");
561  }
562  candidate = i;
563  ++n_linear;
564  }
565  if (n_linear != 1u) {
566  piranha_throw(std::invalid_argument, "monomial is not linear");
567  }
568  return args[static_cast<decltype(args.size())>(candidate)].get_name();
569  }

634  template <typename U = T, partial_enabler<U> = 0>
635  std::pair<U, monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
636  {
637  if (!is_compatible(args)) {
638  piranha_throw(std::invalid_argument, "invalid size of arguments set");
639  }
640  // Cannot take derivative wrt more than one variable, and the position of that variable
641  // must be compatible with the monomial.
642  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
643  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
644  }
645  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
646  // variable with zero exponent.
647  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
648  // there is a valid position and hence the size must be not zero.
649  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
650  return std::make_pair(U(0), monomial(args));
651  }
652  // Copy of the original monomial.
653  monomial m(*this);
654  auto m_b = m.begin();
655  // Original exponent.
656  U v(m_b[*p.begin()]);
657  // Decrement the exponent in the monomial.
658  ip_dec(m_b[*p.begin()]);
659  return std::make_pair(std::move(v), std::move(m));
660  }

691  template <typename U = T, integrate_enabler<U> = 0>
692  std::pair<U, monomial> integrate(const symbol &s, const symbol_set &args) const
693  {
694  typedef typename base::size_type size_type;
695  if (!is_compatible(args)) {
696  piranha_throw(std::invalid_argument, "invalid size of arguments set");
697  }
698  monomial retval;
699  T expo(0), one(1);
700  for (size_type i = 0u; i < args.size(); ++i) {
701  if (math::is_zero(expo) && s < args[i]) {
702  // If we went past the position of s in args and still we
703  // have not performed the integration, it means that we need to add
704  // a new exponent.
705  retval.push_back(one);
706  expo = one;
707  }
708  retval.push_back((*this)[i]);
709  if (args[i] == s) {
710  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
711  // we are never going to land here as args[i] is at this point never going to be s.
712  ip_inc(retval[i]);
713  if (math::is_zero(retval[i])) {
714  piranha_throw(std::invalid_argument,
715  "unable to perform monomial integration: negative unitary exponent");
716  }
717  expo = retval[i];
718  }
719  }
720  // If expo is still zero, it means we need to add a new exponent at the end.
721  if (math::is_zero(expo)) {
722  retval.push_back(one);
723  expo = one;
724  }
725  return std::make_pair(std::move(expo), std::move(retval));
726  }

739  void print(std::ostream &os, const symbol_set &args) const
740  {
741  if (unlikely(args.size() != this->size())) {
742  piranha_throw(std::invalid_argument, "invalid size of arguments set");
743  }
744  bool empty_output = true;
745  for (typename base::size_type i = 0u; i < this->size(); ++i) {
746  if (!math::is_zero((*this)[i])) {
747  // If we are going to print a symbol, and something has been printed before,
748  // then we are going to place the multiplication sign.
749  if (!empty_output) {
750  os << '*';
751  }
752  os << args[i].get_name();
753  empty_output = false;
754  if (!math::is_unitary((*this)[i])) {
755  os << "**" << detail::prepare_for_print((*this)[i]);
756  }
757  }

773  void print_tex(std::ostream &os, const symbol_set &args) const
774  {
775  if (unlikely(args.size() != this->size())) {
776  piranha_throw(std::invalid_argument, "invalid size of arguments set");
777  }
778  std::ostringstream oss_num, oss_den, *cur_oss;
779  const T zero(0);
780  T cur_value;
781  for (typename base::size_type i = 0u; i < this->size(); ++i) {
782  cur_value = (*this)[i];
783  if (!math::is_zero(cur_value)) {
784  // NOTE: the sequencing rules of the ternary operator ensure that math::negate(cur_value) is
785  // evaluated after zero < cur_value.
786  cur_oss
787  = (zero < cur_value) ? std::addressof(oss_num) : (math::negate(cur_value), std::addressof(oss_den));
788  (*cur_oss) << "{" << args[i].get_name() << "}";
789  if (!math::is_unitary(cur_value)) {
790  (*cur_oss) << "^{" << detail::prepare_for_print(cur_value) << "}";
791  }
792  }
793  }
794  const std::string num_str = oss_num.str(), den_str = oss_den.str();
795  if (!num_str.empty() && !den_str.empty()) {
796  os << "\\frac{" << num_str << "}{" << den_str << "}";
797  } else if (!num_str.empty() && den_str.empty()) {
798  os << num_str;
799  } else if (num_str.empty() && !den_str.empty()) {
800  os << "\\frac{1}{" << den_str << "}";
801  }
802  }

828  template <typename U>
829  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
830  {
831  using return_type = eval_type<U>;
832  using size_type = typename base::size_type;
833  // NOTE: the positions map must have the same number of elements as this, and it must
834  // be made of consecutive positions [0,size - 1].
835  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
836  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
837  }
838  // Args must be compatible with this.
839  if (unlikely(args.size() != this->size())) {
840  piranha_throw(std::invalid_argument, "invalid size of arguments set");
841  }
842  return_type retval(1);
843  auto it = pmap.begin();
844  for (size_type i = 0u; i < this->size(); ++i, ++it) {
845  piranha_assert(it != pmap.end() && it->first == i);
846  retval *= math::pow(it->second, (*this)[i]);
847  }
848  piranha_assert(it == pmap.end());
849  return retval;
850  }
