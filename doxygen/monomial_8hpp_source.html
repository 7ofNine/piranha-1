492  template <typename U = T, linarg_enabler<U> = 0>
493  std::string linear_argument(const symbol_set &args) const
494  {
495  if (!is_compatible(args)) {
496  piranha_throw(std::invalid_argument, "invalid size of arguments set");
497  }
498  typedef typename base::size_type size_type;
499  const size_type size = this->size();
500  size_type n_linear = 0u, candidate = 0u;
501  for (size_type i = 0u; i < size; ++i) {
502  integer tmp;
503  try {
504  tmp = safe_cast<integer>((*this)[i]);
505  } catch (const std::invalid_argument &) {
506  piranha_throw(std::invalid_argument, "exponent is not an integer");
507  }
508  if (tmp == 0) {
509  continue;
510  }
511  if (tmp != 1) {
512  piranha_throw(std::invalid_argument, "exponent is not unitary");
513  }
514  candidate = i;
515  ++n_linear;
516  }
517  if (n_linear != 1u) {
518  piranha_throw(std::invalid_argument, "monomial is not linear");
519  }
520  return args[static_cast<decltype(args.size())>(candidate)].get_name();
521  }

586  template <typename U = T, partial_enabler<U> = 0>
587  std::pair<U, monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
588  {
589  if (!is_compatible(args)) {
590  piranha_throw(std::invalid_argument, "invalid size of arguments set");
591  }
592  // Cannot take derivative wrt more than one variable, and the position of that variable
593  // must be compatible with the monomial.
594  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
595  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
596  }
597  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
598  // variable with zero exponent.
599  // NOTE: safe to take this->begin() here, as the checks on the positions above ensure
600  // there is a valid position and hence the size must be not zero.
601  if (!p.size() || math::is_zero(this->begin()[*p.begin()])) {
602  return std::make_pair(U(0), monomial(args));
603  }
604  // Copy of the original monomial.
605  monomial m(*this);
606  auto m_b = m.begin();
607  // Original exponent.
608  U v(m_b[*p.begin()]);
609  // Decrement the exponent in the monomial.
610  ip_dec(m_b[*p.begin()]);
611  return std::make_pair(std::move(v), std::move(m));
612  }

643  template <typename U = T, integrate_enabler<U> = 0>
644  std::pair<U, monomial> integrate(const symbol &s, const symbol_set &args) const
645  {
646  typedef typename base::size_type size_type;
647  if (!is_compatible(args)) {
648  piranha_throw(std::invalid_argument, "invalid size of arguments set");
649  }
650  monomial retval;
651  T expo(0), one(1);
652  for (size_type i = 0u; i < args.size(); ++i) {
653  if (math::is_zero(expo) && s < args[i]) {
654  // If we went past the position of s in args and still we
655  // have not performed the integration, it means that we need to add
656  // a new exponent.
657  retval.push_back(one);
658  expo = one;
659  }
660  retval.push_back((*this)[i]);
661  if (args[i] == s) {
662  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
663  // we are never going to land here as args[i] is at this point never going to be s.
664  ip_inc(retval[i]);
665  if (math::is_zero(retval[i])) {
666  piranha_throw(std::invalid_argument,
667  "unable to perform monomial integration: negative unitary exponent");
668  }
669  expo = retval[i];
670  }
671  }
672  // If expo is still zero, it means we need to add a new exponent at the end.
673  if (math::is_zero(expo)) {
674  retval.push_back(one);
675  expo = one;
676  }
677  return std::make_pair(std::move(expo), std::move(retval));
678  }

728  {

780  template <typename U>
781  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
782  {
783  using return_type = eval_type<U>;
784  using size_type = typename base::size_type;
785  // NOTE: the positions map must have the same number of elements as this, and it must
786  // be made of consecutive positions [0,size - 1].
787  if (unlikely(pmap.size() != this->size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
788  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
789  }
790  // Args must be compatible with this.
791  if (unlikely(args.size() != this->size())) {
792  piranha_throw(std::invalid_argument, "invalid size of arguments set");
793  }
794  return_type retval(1);
795  auto it = pmap.begin();
796  for (size_type i = 0u; i < this->size(); ++i, ++it) {
797  piranha_assert(it != pmap.end() && it->first == i);
798  retval *= math::pow(it->second, (*this)[i]);
799  }
800  piranha_assert(it == pmap.end());
801  return retval;
802  }

891  std::vector<std::pair<ipow_subs_type<U>, monomial>> ipow_subs(const std::string &s, const integer &n, const U &x,
892  const symbol_set &args) const
893  {

896  piranha_throw(std::invalid_argument, "invalid size of arguments set");
897  }
898  s_type retval_s(1);
899  monomial retval_key;
900  for (typename base::size_type i = 0u; i < this->size(); ++i) {
