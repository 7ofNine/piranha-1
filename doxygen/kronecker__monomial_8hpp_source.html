96 template <typename Archive, typename T>
97 inline void load(Archive &ar, piranha::boost_s11n_key_wrapper<piranha::kronecker_monomial<T>> &k, unsigned)
98 {
99  if (std::is_same<Archive, boost::archive::binary_iarchive>::value) {
100  T value;
101  piranha::boost_load(ar, value);
102  k.key().set_int(value);
103  } else {
104  typename piranha::kronecker_monomial<T>::v_type tmp;
105  piranha::boost_load(ar, tmp);
106  if (unlikely(tmp.size() != k.ss().size())) {
107  piranha_throw(std::invalid_argument, "invalid size detected in the deserialization of a Kronercker "
108  "monomial: the deserialized size is "
109  + std::to_string(tmp.size())
110  + " but the reference symbol set has a size of "
111  + std::to_string(k.ss().size()));
112  }
113  k.key() = piranha::kronecker_monomial<T>(tmp);
114  }
115 }

463  bool is_compatible(const symbol_set &args) const noexcept
464  {
465  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
466  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
467  // in other methods.
468  const auto s = args.size();
469  // No args means the value must also be zero.
470  if (s == 0u) {
471  return !m_value;
472  }
473  const auto &limits = ka::get_limits();
474  // If we overflow the maximum size available, we cannot use this object as key in series.
475  if (s >= limits.size()) {
476  return false;
477  }
478  const auto &l = limits[static_cast<decltype(limits.size())>(s)];
479  // Value is compatible if it is within the bounds for the given size.
480  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
481  }

710  std::string linear_argument(const symbol_set &args) const
711  {
712  const auto v = unpack(args);
713  const auto size = args.size();
714  decltype(args.size()) n_linear = 0u, candidate = 0u;
715  for (typename v_type::size_type i = 0u; i < size; ++i) {
716  integer tmp = safe_cast<integer>(v[i]);
717  if (tmp.sign() == 0) {
718  continue;
719  }
720  if (tmp != 1) {
721  piranha_throw(std::invalid_argument, "exponent is not unitary");
722  }
723  candidate = i;
724  ++n_linear;
725  }
726  if (n_linear != 1u) {
727  piranha_throw(std::invalid_argument, "monomial is not linear");
728  }
729  return args[static_cast<decltype(args.size())>(candidate)].get_name();
730  }

816  void print_tex(std::ostream &os, const symbol_set &args) const
817  {
818  const auto tmp = unpack(args);
819  std::ostringstream oss_num, oss_den, *cur_oss;
820  const value_type zero(0), one(1);
821  value_type cur_value;
822  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
823  cur_value = tmp[i];

829  if (cur_value != one) {

835  if (!num_str.empty() && !den_str.empty()) {
836  os << "\\frac{" << num_str << "}{" << den_str << "}";

863  std::pair<T, kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
864  {
865  auto v = unpack(args);
866  // Cannot take derivative wrt more than one variable, and the position of that variable
867  // must be compatible with the monomial.
868  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
869  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
870  }
871  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
872  // variable with zero exponent.
873  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
874  // there is a valid position and hence the size must be not zero.
875  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
876  return std::make_pair(T(0), kronecker_monomial(args));
877  }
878  auto v_b = v.begin();
879  // Original exponent.
880  T n(v_b[*p.begin()]);
881  // Decrement the exponent in the monomial.
882  if (unlikely(n == std::numeric_limits<T>::min())) {
883  piranha_throw(std::invalid_argument, "negative overflow error in the calculation of the "
884  "partial derivative of a monomial");
885  }
886  v_b[*p.begin()] = static_cast<T>(n - T(1));
887  kronecker_monomial tmp_km;
888  tmp_km.m_value = ka::encode(v);
889  return std::make_pair(n, std::move(tmp_km));
890  }

912  std::pair<T, kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
913  {
914  v_type v = unpack(args), retval;
915  value_type expo(0), one(1);
916  for (min_int<typename v_type::size_type, decltype(args.size())> i = 0u; i < args.size(); ++i) {
917  if (math::is_zero(expo) && s < args[i]) {
918  // If we went past the position of s in args and still we
919  // have not performed the integration, it means that we need to add
920  // a new exponent.
921  retval.push_back(one);
922  expo = one;
923  }
924  retval.push_back(v[i]);
925  if (args[i] == s) {
926  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
927  // we are never going to land here as args[i] is at this point never going to be s.
928  if (unlikely(retval[i] == std::numeric_limits<value_type>::max())) {
929  piranha_throw(std::invalid_argument,
930  "positive overflow error in the calculation of the integral of a monomial");
931  }
932  retval[i] = static_cast<value_type>(retval[i] + value_type(1));
933  if (math::is_zero(retval[i])) {
934  piranha_throw(std::invalid_argument,
935  "unable to perform monomial integration: negative unitary exponent");
936  }
937  expo = retval[i];
938  }
939  }
940  // If expo is still zero, it means we need to add a new exponent at the end.
941  if (math::is_zero(expo)) {
942  retval.push_back(one);
943  expo = one;
944  }
945  return std::make_pair(expo, kronecker_monomial(ka::encode(retval)));
946  }

972  template <typename U>
973  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
974  {
975  using return_type = eval_type<U>;
976  using size_type = typename v_type::size_type;
977  // NOTE: here we can check the pmap size only against args.
978  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
979  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
980  }
981  auto v = unpack(args);
982  return_type retval(1);
983  auto it = pmap.begin();
984  for (min_int<size_type, decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
985  piranha_assert(it != pmap.end() && it->first == i);
