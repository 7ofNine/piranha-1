380  bool is_compatible(const symbol_set &args) const noexcept
381  {
382  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
383  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
384  // in other methods.
385  const auto s = args.size();
386  // No args means the value must also be zero.
387  if (s == 0u) {
388  return !m_value;
389  }
390  const auto &limits = ka::get_limits();
391  // If we overflow the maximum size available, we cannot use this object as key in series.
392  if (s >= limits.size()) {
393  return false;
394  }
395  const auto &l = limits[static_cast<decltype(limits.size())>(s)];
396  // Value is compatible if it is within the bounds for the given size.
397  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
398  }

680  void print_tex(std::ostream &os, const symbol_set &args) const
681  {
682  const auto tmp = unpack(args);
683  std::ostringstream oss_num, oss_den, *cur_oss;
684  const value_type zero(0), one(1);
685  value_type cur_value;
686  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
687  cur_value = tmp[i];
688  if (cur_value != zero) {
689  // NOTE: here negate() is safe because of the symmetry in kronecker_array.
690  cur_oss = (cur_value > zero) ? std::addressof(oss_num) : (math::negate(cur_value),std::addressof(oss_den));
691  (*cur_oss) << "{" << args[i].get_name() << "}";
692  if (cur_value != one) {
693  (*cur_oss) << "^{" << static_cast<long long>(cur_value) << "}";
694  }
695  }
696  }
697  const std::string num_str = oss_num.str(), den_str = oss_den.str();
698  if (!num_str.empty() && !den_str.empty()) {
699  os << "\\frac{" << num_str << "}{" << den_str << "}";
700  } else if (!num_str.empty() && den_str.empty()) {
701  os << num_str;
702  } else if (num_str.empty() && !den_str.empty()) {
703  os << "\\frac{1}{" << den_str << "}";
704  }

725  std::pair<T,kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
726  {
727  auto v = unpack(args);
728  // Cannot take derivative wrt more than one variable, and the position of that variable
729  // must be compatible with the monomial.
730  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
731  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
732  }
733  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
734  // variable with zero exponent.
735  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
736  // there is a valid position and hence the size must be not zero.
737  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
738  return std::make_pair(T(0),kronecker_monomial(args));
739  }
740  auto v_b = v.begin();
741  // Original exponent.
742  T n(v_b[*p.begin()]);
743  // Decrement the exponent in the monomial.
744  if (unlikely(n == std::numeric_limits<T>::min())) {
745  piranha_throw(std::invalid_argument,"negative overflow error in the calculation of the "
746  "partial derivative of a monomial");
747  }
748  v_b[*p.begin()] = static_cast<T>(n - T(1));
749  kronecker_monomial tmp_km;
750  tmp_km.m_value = ka::encode(v);
751  return std::make_pair(n,std::move(tmp_km));
752  }

774  std::pair<T,kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
775  {
776  v_type v = unpack(args), retval;
777  value_type expo(0), one(1);
778  for (min_int<typename v_type::size_type,decltype(args.size())> i = 0u; i < args.size(); ++i) {
779  if (math::is_zero(expo) && s < args[i]) {
780  // If we went past the position of s in args and still we
781  // have not performed the integration, it means that we need to add
782  // a new exponent.
783  retval.push_back(one);
784  expo = one;
785  }
786  retval.push_back(v[i]);
787  if (args[i] == s) {
788  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
789  // we are never going to land here as args[i] is at this point never going to be s.
790  if (unlikely(retval[i] == std::numeric_limits<value_type>::max())) {
791  piranha_throw(std::invalid_argument,"positive overflow error in the calculation of the integral of a monomial");
792  }
793  retval[i] = static_cast<value_type>(retval[i] + value_type(1));
794  if (math::is_zero(retval[i])) {
795  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
796  }
797  expo = retval[i];
798  }
799  }
800  // If expo is still zero, it means we need to add a new exponent at the end.
801  if (math::is_zero(expo)) {
802  retval.push_back(one);
803  expo = one;
804  }
805  return std::make_pair(expo,kronecker_monomial(ka::encode(retval)));
806  }

832  template <typename U>
833  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
834  {
835  using return_type = eval_type<U>;
836  using size_type = typename v_type::size_type;

838  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
839  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
840  }
841  auto v = unpack(args);
842  return_type retval(1);

844  for (min_int<size_type,decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
845  piranha_assert(it != pmap.end() && it->first == i);
846  retval *= math::pow(it->second,v[i]);
847  }
848  piranha_assert(it == pmap.end());
