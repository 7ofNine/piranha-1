96 template <typename Archive, typename T>
97 inline void load(Archive &ar, piranha::boost_s11n_key_wrapper<piranha::kronecker_monomial<T>> &k, unsigned)
98 {
99  if (std::is_same<Archive, boost::archive::binary_iarchive>::value) {
100  T value;
101  piranha::boost_load(ar, value);
102  k.key().set_int(value);
103  } else {
104  typename piranha::kronecker_monomial<T>::v_type tmp;
105  piranha::boost_load(ar, tmp);
106  if (unlikely(tmp.size() != k.ss().size())) {
107  piranha_throw(std::invalid_argument, "invalid size detected in the deserialization of a Kronercker "
108  "monomial: the deserialized size is "
109  + std::to_string(tmp.size())
110  + " but the reference symbol set has a size of "
111  + std::to_string(k.ss().size()));
112  }
113  k.key() = piranha::kronecker_monomial<T>(tmp);
114  }
115 }

451  bool is_compatible(const symbol_set &args) const noexcept
452  {
453  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
454  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
455  // in other methods.
456  const auto s = args.size();
457  // No args means the value must also be zero.
458  if (s == 0u) {
459  return !m_value;
460  }
461  const auto &limits = ka::get_limits();
462  // If we overflow the maximum size available, we cannot use this object as key in series.
463  if (s >= limits.size()) {
464  return false;
465  }
466  const auto &l = limits[static_cast<decltype(limits.size())>(s)];
467  // Value is compatible if it is within the bounds for the given size.
468  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
469  }

683  std::string linear_argument(const symbol_set &args) const
684  {
685  const auto v = unpack(args);
686  const auto size = args.size();
687  decltype(args.size()) n_linear = 0u, candidate = 0u;
688  for (typename v_type::size_type i = 0u; i < size; ++i) {
689  integer tmp = safe_cast<integer>(v[i]);
690  if (tmp.sign() == 0) {
691  continue;
692  }
693  if (tmp != 1) {
694  piranha_throw(std::invalid_argument, "exponent is not unitary");
695  }
696  candidate = i;
697  ++n_linear;
698  }
699  if (n_linear != 1u) {
700  piranha_throw(std::invalid_argument, "monomial is not linear");
701  }
702  return args[static_cast<decltype(args.size())>(candidate)].get_name();
703  }

790  {
791  const auto tmp = unpack(args);
792  std::ostringstream oss_num, oss_den, *cur_oss;
793  const value_type zero(0), one(1);

836  std::pair<T, kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
837  {
838  auto v = unpack(args);
839  // Cannot take derivative wrt more than one variable, and the position of that variable
840  // must be compatible with the monomial.
841  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
842  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
843  }
844  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
845  // variable with zero exponent.
846  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
847  // there is a valid position and hence the size must be not zero.
848  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
849  return std::make_pair(T(0), kronecker_monomial(args));
850  }
851  auto v_b = v.begin();
852  // Original exponent.
853  T n(v_b[*p.begin()]);
854  // Decrement the exponent in the monomial.
855  if (unlikely(n == std::numeric_limits<T>::min())) {
856  piranha_throw(std::invalid_argument, "negative overflow error in the calculation of the "
857  "partial derivative of a monomial");
858  }
859  v_b[*p.begin()] = static_cast<T>(n - T(1));
860  kronecker_monomial tmp_km;
861  tmp_km.m_value = ka::encode(v);
862  return std::make_pair(n, std::move(tmp_km));
863  }

885  std::pair<T, kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
886  {
887  v_type v = unpack(args), retval;
888  value_type expo(0), one(1);
889  for (min_int<typename v_type::size_type, decltype(args.size())> i = 0u; i < args.size(); ++i) {
890  if (math::is_zero(expo) && s < args[i]) {
891  // If we went past the position of s in args and still we
892  // have not performed the integration, it means that we need to add
893  // a new exponent.
894  retval.push_back(one);
895  expo = one;
896  }
897  retval.push_back(v[i]);
898  if (args[i] == s) {
899  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
900  // we are never going to land here as args[i] is at this point never going to be s.
901  if (unlikely(retval[i] == std::numeric_limits<value_type>::max())) {
902  piranha_throw(std::invalid_argument,
903  "positive overflow error in the calculation of the integral of a monomial");
904  }
905  retval[i] = static_cast<value_type>(retval[i] + value_type(1));
906  if (math::is_zero(retval[i])) {
907  piranha_throw(std::invalid_argument,
908  "unable to perform monomial integration: negative unitary exponent");
909  }
910  expo = retval[i];
911  }
912  }
913  // If expo is still zero, it means we need to add a new exponent at the end.
914  if (math::is_zero(expo)) {
915  retval.push_back(one);
916  expo = one;
917  }
918  return std::make_pair(expo, kronecker_monomial(ka::encode(retval)));
919  }

945  template <typename U>
946  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
947  {
948  using return_type = eval_type<U>;
949  using size_type = typename v_type::size_type;
950  // NOTE: here we can check the pmap size only against args.
951  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
952  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
953  }
954  auto v = unpack(args);
955  return_type retval(1);
956  auto it = pmap.begin();
957  for (min_int<size_type, decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
958  piranha_assert(it != pmap.end() && it->first == i);
959  retval *= math::pow(it->second, v[i]);
960  }
961  piranha_assert(it == pmap.end());
962  return retval;
963  }
