392  bool is_compatible(const symbol_set &args) const noexcept
393  {
394  // NOTE: the idea here is to avoid unpack()ing for performance reasons: these checks
395  // are already part of unpack(), and that's why unpack() is used instead of is_compatible()
396  // in other methods.
397  const auto s = args.size();
398  // No args means the value must also be zero.
399  if (s == 0u) {
400  return !m_value;
401  }
402  const auto &limits = ka::get_limits();
403  // If we overflow the maximum size available, we cannot use this object as key in series.
404  if (s >= limits.size()) {
405  return false;
406  }
407  const auto &l = limits[static_cast<decltype(limits.size())>(s)];
408  // Value is compatible if it is within the bounds for the given size.
409  return (m_value >= std::get<1u>(l) && m_value <= std::get<2u>(l));
410  }

618  std::string linear_argument(const symbol_set &args) const
619  {
620  const auto v = unpack(args);
621  const auto size = args.size();
622  decltype(args.size()) n_linear = 0u, candidate = 0u;
623  for (typename v_type::size_type i = 0u; i < size; ++i) {

625  if (tmp.sign() == 0) {
626  continue;

630  }

724  void print_tex(std::ostream &os, const symbol_set &args) const
725  {
726  const auto tmp = unpack(args);
727  std::ostringstream oss_num, oss_den, *cur_oss;
728  const value_type zero(0), one(1);
729  value_type cur_value;
730  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
731  cur_value = tmp[i];
732  if (cur_value != zero) {
733  // NOTE: here negate() is safe because of the symmetry in kronecker_array.
734  cur_oss = (cur_value > zero) ? std::addressof(oss_num) : (math::negate(cur_value),std::addressof(oss_den));

736  if (cur_value != one) {

742  if (!num_str.empty() && !den_str.empty()) {
743  os << "\\frac{" << num_str << "}{" << den_str << "}";

769  std::pair<T,kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
770  {
771  auto v = unpack(args);
772  // Cannot take derivative wrt more than one variable, and the position of that variable
773  // must be compatible with the monomial.
774  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
775  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
776  }
777  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
778  // variable with zero exponent.
779  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
780  // there is a valid position and hence the size must be not zero.
781  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
782  return std::make_pair(T(0),kronecker_monomial(args));
783  }
784  auto v_b = v.begin();
785  // Original exponent.
786  T n(v_b[*p.begin()]);
787  // Decrement the exponent in the monomial.
788  if (unlikely(n == std::numeric_limits<T>::min())) {
789  piranha_throw(std::invalid_argument,"negative overflow error in the calculation of the "
790  "partial derivative of a monomial");
791  }
792  v_b[*p.begin()] = static_cast<T>(n - T(1));
793  kronecker_monomial tmp_km;
794  tmp_km.m_value = ka::encode(v);
795  return std::make_pair(n,std::move(tmp_km));
796  }

818  std::pair<T,kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
819  {
820  v_type v = unpack(args), retval;
821  value_type expo(0), one(1);
822  for (min_int<typename v_type::size_type,decltype(args.size())> i = 0u; i < args.size(); ++i) {
823  if (math::is_zero(expo) && s < args[i]) {
824  // If we went past the position of s in args and still we
825  // have not performed the integration, it means that we need to add
826  // a new exponent.
827  retval.push_back(one);
828  expo = one;
829  }
830  retval.push_back(v[i]);
831  if (args[i] == s) {
832  // NOTE: here using i is safe: if retval gained an extra exponent in the condition above,
833  // we are never going to land here as args[i] is at this point never going to be s.
834  if (unlikely(retval[i] == std::numeric_limits<value_type>::max())) {
835  piranha_throw(std::invalid_argument,"positive overflow error in the calculation of the integral of a monomial");
836  }
837  retval[i] = static_cast<value_type>(retval[i] + value_type(1));
838  if (math::is_zero(retval[i])) {
839  piranha_throw(std::invalid_argument,"unable to perform monomial integration: negative unitary exponent");
840  }
841  expo = retval[i];
842  }
843  }
844  // If expo is still zero, it means we need to add a new exponent at the end.
845  if (math::is_zero(expo)) {
846  retval.push_back(one);
847  expo = one;
848  }
849  return std::make_pair(expo,kronecker_monomial(ka::encode(retval)));
850  }

876  template <typename U>
877  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
878  {
879  using return_type = eval_type<U>;
880  using size_type = typename v_type::size_type;
881  // NOTE: here we can check the pmap size only against args.
882  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
883  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
884  }
885  auto v = unpack(args);
886  return_type retval(1);
887  auto it = pmap.begin();
888  for (min_int<size_type,decltype(args.size())> i = 0u; i < args.size(); ++i, ++it) {
889  piranha_assert(it != pmap.end() && it->first == i);
890  retval *= math::pow(it->second,v[i]);
891  }
892  piranha_assert(it == pmap.end());
893  return retval;
894  }
