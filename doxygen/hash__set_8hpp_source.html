126  // Node class for bucket element.
127  struct node
128  {
129  typedef typename std::aligned_storage<sizeof(T),alignof(T)>::type storage_type;
130  node() : m_next(nullptr) {}
131  // Erase all other ctors/assignments, we do not want to
132  // copy around this object as m_storage is not what it is
133  // (and often could be uninitialized, which would lead to UB if used).
134  node(const node &) = delete;
135  node(node &&) = delete;
136  node &operator=(const node &) = delete;
137  node &operator=(node &&) = delete;
138  // NOTE: the idea here is that we use these helpers only *after* the object has been constructed,
139  // for constructing the object we must always access m_storage directly. The chain of two casts
140  // seems to be the only standard-conforming way of getting out a pointer to T.
141  // http://stackoverflow.com/questions/1082378/what-is-the-basic-use-of-aligned-storage
142  // http://stackoverflow.com/questions/13466556/aligned-storage-and-strict-aliasing
143  // http://en.cppreference.com/w/cpp/types/aligned_storage
144  // A couple of further notes:
145  // - pointer casting to/from void * is ok (4.10.2 and 5.2.9.13, via static_cast);
146  // - the placement new operator (which we use to construct the object into the storage)
147  // is guaranteed to construct the object at the beginning of the storage, and taking
148  // a void * out of the storage will return the address of the object stored within;
149  // - the lifetime of the original POD storage_type ends when we reuse its storage via
150  // placement new.
151  // This page contains some more info about storage reuse:
152  // http://en.cppreference.com/w/cpp/language/lifetime#Storage_reuse
153  // See also this link, regarding static_cast vs reinterpret_cast for this very specific
154  // usage:
155  // http://stackoverflow.com/questions/19300142/static-cast-and-reinterpret-cast-for-stdaligned-storage
156  // Apparently, it is equivalent.
157  const T *ptr() const
158  {
159  piranha_assert(m_next);
160  return static_cast<const T *>(static_cast<const void *>(&m_storage));
161  }
162  T *ptr()
163  {
164  piranha_assert(m_next);
165  return static_cast<T *>(static_cast<void *>(&m_storage));
166  }
167  storage_type m_storage;
168  node *m_next;
169  };

224  list(const list &other):m_node()
225  {
226  try {
227  auto cur = &m_node;
228  auto other_cur = &other.m_node;
229  while (other_cur->m_next) {
230  if (cur->m_next) {
231  // This assert means we are operating on the last element
232  // of the list, as we are doing back-insertions.
233  piranha_assert(cur->m_next == &terminator);
234  // Create a new node with content equal to other_cur
235  // and linking forward to the terminator.
236  std::unique_ptr<node> new_node(::new node());
237  ::new (static_cast<void *>(&new_node->m_storage)) T(*other_cur->ptr());
238  new_node->m_next = &terminator;
239  // Link the new node.
240  cur->m_next = new_node.release();
241  cur = cur->m_next;
242  } else {
243  // This means this is the first node.
244  ::new (static_cast<void *>(&cur->m_storage)) T(*other_cur->ptr());
245  cur->m_next = &terminator;
246  }
247  other_cur = other_cur->m_next;
248  }
249  } catch (...) {
250  destroy();
251  throw;
252  }
253  }

275  void steal_from_rvalue(list &&other)
276  {
277  piranha_assert(empty());
278  // Do something only if there is content in the other.
279  if (other.m_node.m_next) {
280  // Move construct current first node with first node of other.
281  ::new (static_cast<void *>(&m_node.m_storage)) T(std::move(*other.m_node.ptr()));
282  // Link remaining content of other into this.
283  m_node.m_next = other.m_node.m_next;
284  // Destroy first node of other.
285  other.m_node.ptr()->~T();
286  other.m_node.m_next = nullptr;
287  }
288  piranha_assert(other.empty());
289  }

291  node *insert(U &&item, typename std::enable_if<std::is_same<T,typename std::decay<U>::type>::value>::type * = nullptr)
292  {
293  // NOTE: optimize with likely/unlikely?
294  if (m_node.m_next) {
295  // Create the new node and forward-link it to the second node.
296  std::unique_ptr<node> new_node(::new node());
297  ::new (static_cast<void *>(&new_node->m_storage)) T(std::forward<U>(item));
298  new_node->m_next = m_node.m_next;
299  // Link first node to the new node.
300  m_node.m_next = new_node.release();

302  } else {
303  ::new (static_cast<void *>(&m_node.m_storage)) T(std::forward<U>(item));
304  m_node.m_next = &terminator;
305  return &m_node;
306  }

328  void destroy()
329  {
330  node *cur = &m_node;
331  while (cur->m_next) {
332  // Store the current value temporarily.
333  auto old = cur;
334  // Assign the next.
335  cur = cur->m_next;
336  // Destroy the old payload and erase connections.
337  old->ptr()->~T();
338  old->m_next = nullptr;
339  // If the old node was not the initial one, delete it.
340  if (old != &m_node) {
341  ::delete old;
342  }
343  }
344  // After destruction, the list should be equivalent to a default-constructed one.
345  piranha_assert(empty());
346  }

454  void init_from_n_buckets(const size_type &n_buckets, unsigned n_threads)
455  {
456  piranha_assert(!ptr() && !m_log2_size && !m_n_elements);
457  if (unlikely(!n_threads)) {
458  piranha_throw(std::invalid_argument,"the number of threads must be strictly positive");
459  }
460  // Proceed to actual construction only if the requested number of buckets is nonzero.
461  if (!n_buckets) {
462  return;
463  }
464  const size_type log2_size = get_log2_from_hint(n_buckets);
465  const size_type size = size_type(1u) << log2_size;
466  auto new_ptr = allocator().allocate(size);
467  if (unlikely(!new_ptr)) {
468  piranha_throw(std::bad_alloc,);
469  }
470  if (n_threads == 1u) {
471  // Default-construct the elements of the array.
472  // NOTE: this is a noexcept operation, no need to account for rolling back.
473  for (size_type i = 0u; i < size; ++i) {
474  allocator().construct(&new_ptr[i]);
475  }
476  } else {
477  // Sync variables.
478  using crs_type = std::vector<std::pair<size_type,size_type>>;
479  crs_type constructed_ranges(static_cast<typename crs_type::size_type>(n_threads),std::make_pair(size_type(0u),size_type(0u)));
480  if (unlikely(constructed_ranges.size() != n_threads)) {
481  piranha_throw(std::bad_alloc,);
482  }
483  // Thread function.
484  auto thread_function = [this,new_ptr,&constructed_ranges](const size_type &start, const size_type &end, const unsigned &thread_idx) {
485  for (size_type i = start; i != end; ++i) {
486  this->allocator().construct(&new_ptr[i]);
487  }
488  constructed_ranges[thread_idx] = std::make_pair(start,end);
489  };
490  // Work per thread.
491  const auto wpt = size / n_threads;
492  future_list<decltype(thread_pool::enqueue(0u,thread_function,0u,0u,0u))> f_list;
493  try {
494  for (unsigned i = 0u; i < n_threads; ++i) {
495  const auto start = static_cast<size_type>(wpt * i),
496  end = static_cast<size_type>((i == n_threads - 1u) ? size : wpt * (i + 1u));
497  f_list.push_back(thread_pool::enqueue(i,thread_function,start,end,i));
498  }
499  f_list.wait_all();
500  // NOTE: no need to get_all() here, as we know no exceptions will be generated inside thread_func.
501  } catch (...) {
502  // NOTE: everything in thread_func is noexcept, if we are here the exception was thrown by enqueue or push_back.
503  // Wait for everything to wind down.
504  f_list.wait_all();
505  // Destroy what was constructed.
506  for (const auto &r: constructed_ranges) {
507  for (size_type i = r.first; i != r.second; ++i) {
508  allocator().destroy(&new_ptr[i]);
509  }
510  }
511  // Deallocate before re-throwing.
512  allocator().deallocate(new_ptr,size);
513  throw;
514  }
515  }
516  // Assign the members.
517  ptr() = new_ptr;
518  m_log2_size = log2_size;
519  }

532  }

572  // Run a consistency check on the set, will return false if something is wrong.
573  bool sanity_check() const
574  {
575  // Ignore sanity checks on shutdown.
576  if (environment::shutdown()) {
577  return true;
578  }
579  size_type count = 0u;
580  for (size_type i = 0u; i < bucket_count(); ++i) {
581  for (auto it = ptr()[i].begin(); it != ptr()[i].end(); ++it) {
582  if (_bucket(*it) != i) {
583  return false;
584  }
585  ++count;
586  }
587  }
588  if (count != m_n_elements) {
589  return false;
590  }
591  // m_log2_size must not be equal to or greater than the number of bits of size_type.
592  if (m_log2_size >= unsigned(std::numeric_limits<size_type>::digits)) {
593  return false;
594  }
595  // The container pointer must be consistent with the other members.
596  if (!ptr() && (m_log2_size || m_n_elements)) {
597  return false;
598  }
599  // Check size is consistent with number of iterator traversals.
600  count = 0u;
601  for (auto it = begin(); it != end(); ++it, ++count) {}
602  if (count != m_n_elements) {
603  return false;
604  }
605  return true;
606  }

616  return i;

983  std::pair<iterator,bool> insert(U &&k)
984  {
985  auto b_count = bucket_count();
986  // Handle the case of a set with no buckets.
987  if (unlikely(!b_count)) {
988  _increase_size();
989  // Update the bucket count.
990  b_count = 1u;

993  auto bucket_idx = _bucket(k);
994  const auto it = _find(k,bucket_idx);

1000  piranha_throw(std::overflow_error,"maximum number of elements reached");

1006  bucket_idx = _bucket(k);

1104  void rehash(const size_type &new_size, unsigned n_threads = 1u)
1105  {
1106  if (unlikely(!n_threads)) {
1107  piranha_throw(std::invalid_argument,"the number of threads must be strictly positive");
1108  }
1109  // If rehash is requested to zero, do something only if there are no items stored in the set.
1110  if (!new_size) {
1111  if (!size()) {
1112  clear();
1113  }
1114  return;
1115  }
1116  // Do nothing if rehashing to the new size would lead to exceeding the max load factor.
1117  if (static_cast<double>(size()) / static_cast<double>(new_size) > max_load_factor()) {
1118  return;
1119  }
1120  // Create a new set with needed amount of buckets.
1121  hash_set new_set(new_size,hash(),k_equal(),n_threads);
1122  try {
1123  const auto it_f = _m_end();
1124  for (auto it = _m_begin(); it != it_f; ++it) {
1125  const auto new_idx = new_set._bucket(*it);
1126  new_set._unique_insert(std::move(*it),new_idx);
1127  }
1128  } catch (...) {
1129  // Clear up both this and the new set upon any kind of error.
1130  clear();
1131  new_set.clear();
1132  throw;
1133  }
1134  // Retain the number of elements.
1135  new_set.m_n_elements = m_n_elements;
1136  // Clear the old set.
1137  clear();
1138  // Assign the new set.
1139  *this = std::move(new_set);
1140  }

1353  local_iterator _erase(const_iterator it)
1354  {
1355  // Verify the iterator is valid.
1356  piranha_assert(it.m_set == this);
1357  piranha_assert(it.m_idx < bucket_count());
1358  piranha_assert(!ptr()[it.m_idx].empty());
1359  piranha_assert(it.m_it != ptr()[it.m_idx].end());
1360  auto &bucket = ptr()[it.m_idx];
1361  // If the pointed-to element is the first one in the bucket, we need special care.
1362  if (&*it == &*bucket.m_node.ptr()) {
1363  // Destroy the payload.
1364  bucket.m_node.ptr()->~T();
1365  if (bucket.m_node.m_next == &bucket.terminator) {
1366  // Special handling if this was the only element.
1367  bucket.m_node.m_next = nullptr;
1368  return bucket.end();
1369  } else {
1370  // Store the link in the second element (this could be the terminator).
1371  auto tmp = bucket.m_node.m_next->m_next;
1372  // Move-construct from the second element, and then destroy it.
1373  ::new (static_cast<void *>(&bucket.m_node.m_storage)) T(std::move(*bucket.m_node.m_next->ptr()));
1374  bucket.m_node.m_next->ptr()->~T();
1375  ::delete bucket.m_node.m_next;
1376  // Establish the new link.
1377  bucket.m_node.m_next = tmp;
1378  return bucket.begin();
1379  }
1380  } else {
1381  auto b_it = bucket.begin();
1382  auto prev_b_it = b_it;
1383  const auto b_it_f = bucket.end();
1384  ++b_it;
1385  for (; b_it != b_it_f; ++b_it, ++prev_b_it) {
1386  if (&*b_it == &*it) {
1387  // Assign to the previous element the next link of the current one.
1388  prev_b_it.m_ptr->m_next = b_it.m_ptr->m_next;
1389  // Delete the current one.
1390  b_it.m_ptr->ptr()->~T();
1391  ::delete b_it.m_ptr;
1392  break;
1393  };
1394  }
1395  // We never want to go through the whole list, it means the element
1396  // to which 'it' refers is not here: assert that the iterator we just
1397  // erased was not end() - i.e., it was pointing to something.
1398  piranha_assert(b_it.m_ptr);
1399  // Move forward the iterator that originally preceded the erased item.
1400  // It will now point to the item past the erased one or to the local end().
1401  return ++prev_b_it;
1402  }
