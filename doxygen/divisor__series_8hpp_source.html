127  // Partial utils.
128  // Handle exponent increase in a safe way.
129  template <typename T, typename std::enable_if<std::is_integral<T>::value, int>::type = 0>
130  static void expo_increase(T &e)
131  {
132  if (unlikely(e == std::numeric_limits<T>::max())) {
133  piranha_throw(std::overflow_error, "overflow in the computation of the partial derivative "
134  "of a divisor series");
135  }
136  e = static_cast<T>(e + T(1));
137  }
138  template <typename T, typename std::enable_if<!std::is_integral<T>::value, int>::type = 0>
139  static void expo_increase(T &e)
140  {
141  ++e;
142  }
143  // Safe computation of the integral multiplier.
144  template <typename T, typename std::enable_if<std::is_integral<T>::value, int>::type = 0>
145  static auto safe_mult(const T &n, const T &m) -> decltype(n * m)
146  {
147  auto ret(integer(n) * m);
148  ret.negate();
149  return static_cast<decltype(n * m)>(ret);
150  }
151  template <typename T, typename std::enable_if<!std::is_integral<T>::value, int>::type = 0>
152  static auto safe_mult(const T &n, const T &m) -> decltype(-(n * m))
153  {
154  return -(n * m);
155  }
156  // This is the first stage - the second part of the chain rule for each term.
157  template <typename T>
158  using d_partial_type_0
159  = decltype(std::declval<const dv_type &>() * std::declval<const dv_type &>() * std::declval<const T &>());
160  template <typename T>
161  using d_partial_type_1 = decltype(std::declval<const T &>() * std::declval<const d_partial_type_0<T> &>());
162  // Requirements on the final type for the first stage.
163  template <typename T>
164  using d_partial_type =
165  typename std::enable_if<std::is_constructible<d_partial_type_1<T>, d_partial_type_0<T>>::value
166  && std::is_constructible<d_partial_type_1<T>, int>::value
167  && is_addable_in_place<d_partial_type_1<T>>::value,
168  d_partial_type_1<T>>::type;
169  template <typename T = divisor_series>
170  d_partial_type<T> d_partial_impl(typename T::term_type::key_type &key, const symbol_set::positions &pos) const
171  {
172  using term_type = typename base::term_type;
173  using cf_type = typename term_type::cf_type;
174  using key_type = typename term_type::key_type;
175  piranha_assert(key.size() != 0u);
176  // Construct the first part of the derivative.
177  key_type tmp_div;
178  // Insert all the terms that depend on the variable, apart from the
179  // first one.
180  const auto it_f = key.m_container.end();
181  auto it = key.m_container.begin(), it_b(it);
182  auto first(*it), first_copy(*it);
183  // Size type of the multipliers' vector.
184  using vs_type = decltype(first.v.size());
185  ++it;
186  for (; it != it_f; ++it) {
187  tmp_div.m_container.insert(*it);
188  }
189  // Remove the first term from the original key.
190  key.m_container.erase(it_b);
191  // Extract from the first dependent term the aij and the exponent, and multiply+negate them.
192  const auto mult = safe_mult(first.e, first.v[static_cast<vs_type>(pos.back())]);
193  // Increase by one the exponent of the first dep. term.
194  expo_increase(first.e);
195  // Insert the modified first term. Don't move, as we need first below.
196  tmp_div.m_container.insert(first);
197  // Now build the first part of the derivative.
198  divisor_series tmp_ds;
199  tmp_ds.set_symbol_set(this->m_symbol_set);
200  tmp_ds.insert(term_type(cf_type(1), std::move(tmp_div)));
201  // Init the retval.
202  d_partial_type<T> retval(mult * tmp_ds);
203  // Now the second part of the derivative, if appropriate.
204  if (!key.m_container.empty()) {
205  // Build a series with only the first dependent term and unitary coefficient.
206  key_type tmp_div_01;
207  tmp_div_01.m_container.insert(std::move(first_copy));
208  divisor_series tmp_ds_01;
209  tmp_ds_01.set_symbol_set(this->m_symbol_set);
210  tmp_ds_01.insert(term_type(cf_type(1), std::move(tmp_div_01)));
211  // Recurse.
212  retval += tmp_ds_01 * d_partial_impl(key, pos);
213  }
214  return retval;
215  }
216  template <typename T = divisor_series>
217  d_partial_type<T> divisor_partial(const typename T::term_type &term, const symbol_set::positions &pos) const
218  {
219  using term_type = typename base::term_type;
220  // Return zero if the variable is not in the series.
221  if (pos.size() == 0u) {
222  return d_partial_type<T>(0);
223  }
224  // Initial split of the divisor.
225  auto sd = term.m_key.split(pos, this->m_symbol_set);
226  // If the variable is not in the divisor, just return zero.
227  if (sd.first.size() == 0u) {
228  return d_partial_type<T>(0);
229  }
230  // Init the constant part of the derivative: the coefficient and the part of the divisor
231  // which does not depend on the variable.
232  divisor_series tmp_ds;
233  tmp_ds.set_symbol_set(this->m_symbol_set);
234  tmp_ds.insert(term_type(term.m_cf, std::move(sd.second)));
235  // Construct and return the result.
236  return tmp_ds * d_partial_impl(sd.first, pos);
237  }

289  // Special invert() implementation when we have reached the first polynomial coefficient in the hierarchy.
290  template <typename RetT, typename T,
291  typename std::enable_if<std::is_base_of<detail::polynomial_tag, typename T::term_type::cf_type>::value,
292  int>::type
293  = 0>
294  RetT special_invert(const T &s) const
295  {
296  if (s.is_single_coefficient() && !s.empty()) {
297  try {
298  using term_type = typename RetT::term_type;
299  using cf_type = typename term_type::cf_type;
300  using key_type = typename term_type::key_type;
301  // Extract the linear combination from the poly coefficient.
302  auto lc = s._container().begin()->m_cf.integral_combination();
303  // NOTE: lc cannot be empty as we are coming in with a non-zero polynomial.
304  piranha_assert(!lc.empty());
305  std::vector<integer> v_int;
306  symbol_set ss;
307  for (auto it = lc.begin(); it != lc.end(); ++it) {
308  ss.add(symbol(it->first));
309  v_int.push_back(it->second);
310  }
311  // We need to canonicalise the term: switch the sign if the first
312  // nonzero element is negative, and divide by the common denom.
313  bool first_nonzero_found = false, need_negate = false;
314  integer cd(0);
315  for (auto &n : v_int) {
316  if (!first_nonzero_found && !math::is_zero(n)) {
317  if (n < 0) {
318  need_negate = true;
319  }
320  first_nonzero_found = true;
321  }
322  if (need_negate) {
323  math::negate(n);
324  }
325  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
326  // NOTE: the gcd computation here is safe as we are operating on integers.
327  math::gcd3(cd, cd, n);
328  }
329  // Common denominator could be negative.
330  if (cd.sign() < 0) {
331  math::negate(cd);
332  }
333  // It should never be zero: if all elements in v_int are zero, we would not have been
334  // able to extract the linear combination.
335  piranha_assert(!math::is_zero(cd));
336  // Divide by the cd.
337  for (auto &n : v_int) {
338  n /= cd;
339  }
340  // Now build the key.
341  key_type tmp_key;
342  integer exponent(1);
343  tmp_key.insert(v_int.begin(), v_int.end(), exponent);
344  // The return value.
345  RetT retval;
346  retval.set_symbol_set(ss);
347  retval.insert(term_type(cf_type(1), std::move(tmp_key)));
348  // If we negated in the canonicalisation, we need to re-negate
349  // the common divisor before the final division.
350  if (need_negate) {
351  math::negate(cd);
352  }
353  return RetT(retval / cd.pow(exponent));
354  } catch (const std::invalid_argument &) {
355  // Interpret invalid_argument as a failure in extracting integral combination,
356  // and move on.
357  }
358  }
359  return math::invert(*static_cast<const base *>(this));
360  }
361  // The coefficient is not a polynomial: recurse to the inner coefficient type, if the current coefficient type
362  // is suitable.
363  template <typename RetT, typename T,
364  typename std::enable_if<!std::is_base_of<detail::polynomial_tag, typename T::term_type::cf_type>::value,
365  int>::type
366  = 0>
367  RetT special_invert(const T &s) const
368  {
369  if (s.is_single_coefficient() && !s.empty()) {
370  return special_invert<RetT>(s._container().begin()->m_cf);
371  }
372  return math::invert(*static_cast<const base *>(this));
373  }

524  template <typename T = divisor_series>
525  integrate_type<T> integrate(const std::string &name) const
526  {
527  using term_type = typename base::term_type;
528  using cf_type = typename term_type::cf_type;
529  integrate_type<T> retval(0);
530  const auto it_f = this->m_container.end();
531  // Turn name into symbol position.
532  const symbol_set::positions pos(this->m_symbol_set, symbol_set{symbol(name)});
533  for (auto it = this->m_container.begin(); it != it_f; ++it) {
534  if (pos.size() == 1u) {
535  // If the variable is in the symbol set, then we need to make sure
536  // that each multiplier associated to it is zero. Otherwise, the divisor
537  // depends on the variable and we cannot perform the integration.
538  const auto it2_f = it->m_key.m_container.end();
539  for (auto it2 = it->m_key.m_container.begin(); it2 != it2_f; ++it2) {
540  using size_type = decltype(it2->v.size());
541  piranha_assert(pos.back() < it2->v.size());
542  if (unlikely(it2->v[static_cast<size_type>(pos.back())] != 0)) {
543  piranha_throw(std::invalid_argument, "unable to integrate with respect to divisor variables");
544  }
545  }
546  }
547  divisor_series tmp;
548  tmp.set_symbol_set(this->m_symbol_set);
549  tmp.insert(term_type(cf_type(1), it->m_key));
550  retval += math::integrate(it->m_cf, name) * tmp;
551  }
