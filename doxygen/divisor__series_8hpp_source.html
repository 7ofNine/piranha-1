131  // Handle exponent increase in a safe way.
132  template <typename T, typename std::enable_if<std::is_integral<T>::value,int>::type = 0>
133  static void expo_increase(T &e)

146  // Safe computation of the integral multiplier.
147  template <typename T, typename std::enable_if<std::is_integral<T>::value,int>::type = 0>
148  static auto safe_mult(const T &n, const T &m) -> decltype(n*m)
149  {
150  auto ret(integer(n) * m);
151  ret.negate();
152  return static_cast<decltype(n * m)>(ret);
153  }

159  // This is the first stage - the second part of the chain rule for each term.
160  template <typename T>
161  using d_partial_type_0 = decltype(std::declval<const dv_type &>() * std::declval<const dv_type &>() * std::declval<const T &>());
162  template <typename T>
163  using d_partial_type_1 = decltype(std::declval<const T &>() * std::declval<const d_partial_type_0<T> &>());
164  // Requirements on the final type for the first stage.
165  template <typename T>
166  using d_partial_type = typename std::enable_if<std::is_constructible<d_partial_type_1<T>,d_partial_type_0<T>>::value &&
167  std::is_constructible<d_partial_type_1<T>,int>::value &&
168  is_addable_in_place<d_partial_type_1<T>>::value,
169  d_partial_type_1<T>>::type;
170  template <typename T = divisor_series>
171  d_partial_type<T> d_partial_impl(typename T::term_type::key_type &key, const symbol_set::positions &pos) const
172  {
173  using term_type = typename base::term_type;
174  using cf_type = typename term_type::cf_type;
175  using key_type = typename term_type::key_type;
176  piranha_assert(key.size() != 0u);
177  // Construct the first part of the derivative.
178  key_type tmp_div;
179  // Insert all the terms that depend on the variable, apart from the
180  // first one.
181  const auto it_f = key.m_container.end();
182  auto it = key.m_container.begin(), it_b(it);
183  auto first(*it), first_copy(*it);
184  // Size type of the multipliers' vector.
185  using vs_type = decltype(first.v.size());
186  ++it;
187  for (; it != it_f; ++it) {
188  tmp_div.m_container.insert(*it);
189  }
190  // Remove the first term from the original key.
191  key.m_container.erase(it_b);
192  // Extract from the first dependent term the aij and the exponent, and multiply+negate them.
193  const auto mult = safe_mult(first.e,first.v[static_cast<vs_type>(pos.back())]);
194  // Increase by one the exponent of the first dep. term.
195  expo_increase(first.e);
196  // Insert the modified first term. Don't move, as we need first below.
197  tmp_div.m_container.insert(first);
198  // Now build the first part of the derivative.
199  divisor_series tmp_ds;
200  tmp_ds.set_symbol_set(this->m_symbol_set);
201  tmp_ds.insert(term_type(cf_type(1),std::move(tmp_div)));
202  // Init the retval.
203  d_partial_type<T> retval(mult * tmp_ds);
204  // Now the second part of the derivative, if appropriate.
205  if (!key.m_container.empty()) {
206  // Build a series with only the first dependent term and unitary coefficient.
207  key_type tmp_div_01;
208  tmp_div_01.m_container.insert(std::move(first_copy));
209  divisor_series tmp_ds_01;
210  tmp_ds_01.set_symbol_set(this->m_symbol_set);
211  tmp_ds_01.insert(term_type(cf_type(1),std::move(tmp_div_01)));
212  // Recurse.
213  retval += tmp_ds_01 * d_partial_impl(key,pos);
214  }
215  return retval;
216  }
217  template <typename T = divisor_series>
218  d_partial_type<T> divisor_partial(const typename T::term_type &term, const symbol_set::positions &pos) const
219  {
220  using term_type = typename base::term_type;
221  // Return zero if the variable is not in the series.
222  if (pos.size() == 0u) {
223  return d_partial_type<T>(0);
224  }
225  // Initial split of the divisor.
226  auto sd = term.m_key.split(pos,this->m_symbol_set);
227  // If the variable is not in the divisor, just return zero.
228  if (sd.first.size() == 0u) {
229  return d_partial_type<T>(0);
230  }
231  // Init the constant part of the derivative: the coefficient and the part of the divisor
232  // which does not depend on the variable.
233  divisor_series tmp_ds;
234  tmp_ds.set_symbol_set(this->m_symbol_set);
235  tmp_ds.insert(term_type(term.m_cf,std::move(sd.second)));
236  // Construct and return the result.
237  return tmp_ds * d_partial_impl(sd.first,pos);
238  }

286  // Special invert() implementation when we have reached the first polynomial coefficient in the hierarchy.
287  template <typename RetT, typename T, typename std::enable_if<std::is_base_of<detail::polynomial_tag,
288  typename T::term_type::cf_type>::value,int>::type = 0>
289  RetT special_invert(const T &s) const
290  {
291  if (s.is_single_coefficient() && !s.empty()) {
292  try {
293  using term_type = typename RetT::term_type;
294  using cf_type = typename term_type::cf_type;
295  using key_type = typename term_type::key_type;
296  // Extract the linear combination from the poly coefficient.
297  auto lc = s._container().begin()->m_cf.integral_combination();
298  // NOTE: lc cannot be empty as we are coming in with a non-zero polynomial.
299  piranha_assert(!lc.empty());
300  std::vector<integer> v_int;
301  symbol_set ss;
302  for (auto it = lc.begin(); it != lc.end(); ++it) {
303  ss.add(symbol(it->first));
304  v_int.push_back(it->second);
305  }
306  // We need to canonicalise the term: switch the sign if the first
307  // nonzero element is negative, and divide by the common denom.
308  bool first_nonzero_found = false, need_negate = false;
309  integer cd(0);
310  for (auto &n: v_int) {
311  if (!first_nonzero_found && !math::is_zero(n)) {
312  if (n < 0) {
313  need_negate = true;
314  }
315  first_nonzero_found = true;
316  }
317  if (need_negate) {
318  math::negate(n);
319  }
320  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
321  // NOTE: the gcd computation here is safe as we are operating on integers.
322  cd = detail::gcd(cd,n);
323  }
324  // Common denominator could be negative.
325  if (cd.sign() < 0) {
326  math::negate(cd);
327  }
328  // It should never be zero: if all elements in v_int are zero, we would not have been
329  // able to extract the linear combination.
330  piranha_assert(!math::is_zero(cd));
331  // Divide by the cd.
332  for (auto &n: v_int) {
333  n /= cd;
334  }
335  // Now build the key.
336  key_type tmp_key;
337  integer exponent(1);
338  tmp_key.insert(v_int.begin(),v_int.end(),exponent);
339  // The return value.
340  RetT retval;
341  retval.set_symbol_set(ss);
342  retval.insert(term_type(cf_type(1),std::move(tmp_key)));
343  // If we negated in the canonicalisation, we need to re-negate
344  // the common divisor before the final division.
345  if (need_negate) {
346  math::negate(cd);
347  }
348  return RetT(retval / cd.pow(exponent));
349  } catch (const std::invalid_argument &) {
350  // Interpret invalid_argument as a failure in extracting integral combination,
351  // and move on.
352  }
353  }
354  return math::invert(*static_cast<const base *>(this));
355  }
356  // The coefficient is not a polynomial: recurse to the inner coefficient type, if the current coefficient type
357  // is suitable.
358  template <typename RetT, typename T, typename std::enable_if<!std::is_base_of<detail::polynomial_tag,
359  typename T::term_type::cf_type>::value,int>::type = 0>
360  RetT special_invert(const T &s) const
361  {
362  if (s.is_single_coefficient() && !s.empty()) {
363  return special_invert<RetT>(s._container().begin()->m_cf);
364  }
365  return math::invert(*static_cast<const base *>(this));
366  }

504  template <typename T = divisor_series>
505  integrate_type<T> integrate(const std::string &name) const
506  {
507  using term_type = typename base::term_type;
508  using cf_type = typename term_type::cf_type;
509  integrate_type<T> retval(0);
510  const auto it_f = this->m_container.end();
511  // Turn name into symbol position.
512  const symbol_set::positions pos(this->m_symbol_set,symbol_set{symbol(name)});
513  for (auto it = this->m_container.begin(); it != it_f; ++it) {
514  if (pos.size() == 1u) {
515  // If the variable is in the symbol set, then we need to make sure
516  // that each multiplier associated to it is zero. Otherwise, the divisor
517  // depends on the variable and we cannot perform the integration.
518  const auto it2_f = it->m_key.m_container.end();
519  for (auto it2 = it->m_key.m_container.begin(); it2 != it2_f; ++it2) {
520  using size_type = decltype(it2->v.size());
521  piranha_assert(pos.back() < it2->v.size());
522  if (unlikely(it2->v[static_cast<size_type>(pos.back())] != 0)) {
523  piranha_throw(std::invalid_argument,"unable to integrate with respect to divisor variables");
524  }
525  }
526  }
527  divisor_series tmp;
528  tmp.set_symbol_set(this->m_symbol_set);
529  tmp.insert(term_type(cf_type(1),it->m_key));
530  retval += math::integrate(it->m_cf,name) * tmp;
531  }
532  return retval;
