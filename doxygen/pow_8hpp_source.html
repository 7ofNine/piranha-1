45 template <typename T, typename U>
46 using integer_pow_enabler =
47  typename std::enable_if<(is_mp_integer<T>::value && is_mp_integer_interoperable_type<U>::value)
48  || (is_mp_integer<U>::value && is_mp_integer_interoperable_type<T>::value) ||
49  // NOTE: here we are catching two arguments with potentially different
50  // bits. BUT this case is not caught in the pow_impl, so we should be ok as long
51  // as we don't allow interoperablity with different bits.
52  (is_mp_integer<T>::value &&is_mp_integer<U>::value)
53  || (std::is_integral<T>::value && std::is_integral<U>::value)>::type;
54 
55 // Enabler for the pow overload for arithmetic and floating-point types.
56 template <typename T, typename U>
57 using pow_fp_arith_enabler =
58  typename std::enable_if<std::is_arithmetic<T>::value && std::is_arithmetic<U>::value
59  && (std::is_floating_point<T>::value || std::is_floating_point<U>::value)>::type;
60 }
