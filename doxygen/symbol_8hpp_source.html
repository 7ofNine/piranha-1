168  static std::string const *get_pointer(const std::string &name)
169  {
170  std::lock_guard<std::mutex> lock(m_mutex);
171  auto it = m_symbol_list.find(name);
172  if (it == m_symbol_list.end()) {
173  // If symbol is not present in the list, add it.
174  auto retval = m_symbol_list.insert(name);
175  piranha_assert(retval.second);
176  it = retval.first;
177  }
178  // Final check, just for peace of mind.
179  piranha_assert(it != m_symbol_list.end());
180  // Extract and return the pointer.
181  // NOTE: this is legal, as std::set is a standard container:
182  // http://www.gotw.ca/gotw/050.htm
183  return &*it;
184  }
