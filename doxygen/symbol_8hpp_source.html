173  {
174  std::lock_guard<std::mutex> lock(m_mutex);
175  auto it = m_symbol_list.find(name);
176  if (it == m_symbol_list.end()) {
177  // If symbol is not present in the list, add it.
178  auto retval = m_symbol_list.insert(name);
179  piranha_assert(retval.second);
180  it = retval.first;
181  }
182  // Final check, just for peace of mind.
183  piranha_assert(it != m_symbol_list.end());
184  // Extract and return the pointer.
185  // NOTE: this is legal, as std::set is a standard container:
186  // http://www.gotw.ca/gotw/050.htm
187  return &*it;
