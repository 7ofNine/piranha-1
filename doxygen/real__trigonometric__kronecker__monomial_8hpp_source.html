191  static bool canonicalise_impl(v_type &unpacked)
192  {
193  const auto size = unpacked.size();
194  bool sign_change = false;
195  for (decltype(unpacked.size()) i = 0u; i < size; ++i) {
196  if (sign_change || unpacked[i] < value_type(0)) {
197  unpacked[i] = static_cast<value_type>(-unpacked[i]);
198  sign_change = true;
199  } else if (unpacked[i] > value_type(0)) {
200  break;
201  }
202  }

446  bool is_compatible(const symbol_set &args) const noexcept
447  {
448  const auto s = args.size();
449  // No args means the value must also be zero.
450  if (s == 0u) {
451  return !m_value;
452  }
453  const auto &limits = ka::get_limits();
454  // If we overflow the maximum size available, we cannot use this object as key in series.
455  if (s >= limits.size()) {
456  return false;
457  }
458  const auto &l = limits[s];
459  // Value is compatible if it is within the bounds for the given size.
460  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
461  return false;
462  }
463  // Now check for the first multiplier.
464  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
465  // we do not need to put @throw specifications in the doc.
466  const auto unpacked = unpack(args);
467  // We know that s != 0.
468  piranha_assert(unpacked.size() > 0u);
469  for (typename v_type::size_type i = 0u; i < s; ++i) {
470  if (unpacked[i] < value_type(0)) {
471  return false;
472  } else if (unpacked[i] > value_type(0)) {
473  break;
474  }
475  }
476  return true;
477  }

688  template <typename Cf, multiply_enabler<Cf> = 0>
689  static void multiply(std::array<term<Cf,real_trigonometric_kronecker_monomial>,multiply_arity> &res,
690  const term<Cf,real_trigonometric_kronecker_monomial> &t1, const term<Cf,real_trigonometric_kronecker_monomial> &t2,
691  const symbol_set &args)
692  {
693  // Coefficients first.
694  detail::cf_mult_impl(res[0u].m_cf,t1.m_cf,t2.m_cf);
695  res[1u].m_cf = res[0u].m_cf;
696  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
697  if (f1 && f2) {
698  // cos, cos: no change.
699  } else if (!f1 && !f2) {
700  // sin, sin: negate the plus.
701  math::negate(res[0u].m_cf);
702  } else if (!f1 && f2) {
703  // sin, cos: no change.
704  } else {
705  // cos, sin: negate the minus.
706  math::negate(res[1u].m_cf);
707  }
708  // Now the keys.
709  auto &retval_plus = res[0u].m_key;
710  auto &retval_minus = res[1u].m_key;
711  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
712  bool sign_plus = false, sign_minus = false;
713  const auto size = args.size();
714  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
715  v_type result_plus, result_minus;
716  for (typename v_type::size_type i = 0u; i < size; ++i) {
717  result_plus.push_back(tmp1[i]);
718  detail::safe_integral_adder(result_plus[i],tmp2[i]);
719  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
720  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
721  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
722  // to int and safe_adder won't work as it expects identical types.
723  result_minus.push_back(tmp1[i]);
724  detail::safe_integral_adder(result_minus[i],static_cast<value_type>(-tmp2[i]));
725  }
726  // Handle sign changes.
727  sign_plus = canonicalise_impl(result_plus);
728  sign_minus = canonicalise_impl(result_minus);
729  // Compute them before assigning, so in case of exceptions we do not touch the return values.
730  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
731  retval_plus.m_value = re_plus;
732  retval_minus.m_value = re_minus;
733  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
734  retval_plus.m_flavour = f;
735  retval_minus.m_flavour = f;
736  // If multiplier sign was changed and the result is a sine, negate the coefficient.
737  if (sign_plus && !res[0u].m_key.get_flavour()) {
738  math::negate(res[0u].m_cf);
739  }
740  if (sign_minus && !res[1u].m_key.get_flavour()) {
741  math::negate(res[1u].m_cf);
742  }
743  }

798  void print(std::ostream &os, const symbol_set &args) const
799  {
800  // Don't print anything in case all multipliers are zero.
801  if (m_value == value_type(0)) {
802  return;
803  }
804  if (m_flavour) {
805  os << "cos(";
806  } else {
807  os << "sin(";
808  }
809  const auto tmp = unpack(args);
810  piranha_assert(tmp.size() == args.size());
811  const value_type zero(0), one(1), m_one(-1);
812  bool empty_output = true;
813  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
814  if (tmp[i] != zero) {
815  // A positive multiplier, in case previous output exists, must be preceded
816  // by a "+" sign.
817  if (tmp[i] > zero && !empty_output) {
818  os << '+';
819  }
820  // Print the multiplier, unless it's "-1": in that case, just print the minus sign.
821  if (tmp[i] == m_one) {
822  os << '-';
823  } else if (tmp[i] != one) {
824  os << detail::prepare_for_print(tmp[i]) << '*';
825  }
826  // Finally, print name of variable.
827  os << args[i].get_name();
828  empty_output = false;
829  }
830  }
831  os << ")";
832  }

896  std::pair<T,real_trigonometric_kronecker_monomial> partial(const symbol_set::positions &p, const symbol_set &args) const
897  {
898  auto v = unpack(args);
899  // Cannot take derivative wrt more than one variable, and the position of that variable
900  // must be compatible with the monomial.
901  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
902  piranha_throw(std::invalid_argument,"invalid size of symbol_set::positions");
903  }
904  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
905  // variable with zero multiplier.
906  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
907  // there is a valid position and hence the size must be not zero.
908  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
909  return std::make_pair(T(0),real_trigonometric_kronecker_monomial(args));
910  }
911  auto v_b = v.begin();
912  // Original multiplier.
913  T n(v_b[*p.begin()]);
914  // Create a copy of this.
915  real_trigonometric_kronecker_monomial tmp_m(*this);
916  // Flip the flavour.
917  tmp_m.set_flavour(!get_flavour());
918  // Flip the sign of the multiplier as needed.
919  if (get_flavour()) {
920  // NOTE: this is safe, as it is coming out of a k codification which
921  // is symmetric.
922  math::negate(n);
923  }
924  return std::make_pair(std::move(n),std::move(tmp_m));
925  }

990  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
991  {
992  using return_type = eval_type<U>;
993  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
994  piranha_throw(std::invalid_argument,"invalid positions map for evaluation");
995  }
996  // Run the unpack before the checks below in order to check the suitability of args.
997  auto v = unpack(args);
998  if (args.size() == 0u) {
999  if (get_flavour()) {
1000  return return_type(1);
1001  } else {
1002  return return_type(0);
1003  }
1004  }
1005  using tmp_type = decltype(std::declval<U const &>() * std::declval<value_type const &>());
1006  tmp_type tmp(0);
1007  auto it = pmap.begin();
1008  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i, ++it) {
1009  piranha_assert(it != pmap.end() && it->first == i);
1010  // NOTE: here it might make sense to use multiply_accumulate. There might be a perf gain
1011  // for things like real. Maybe fast FMA on Haswell too. Remember to adapt the type requirements
1012  // in case we implement this.
1013  tmp += it->second * v[i];
1014  }
1015  piranha_assert(it == pmap.end());
1016  if (get_flavour()) {
1017  return math::cos(tmp);
1018  }
1019  return math::sin(tmp);

1128  template <typename U>
1129  std::vector<std::pair<t_subs_type<U>,
1130  real_trigonometric_kronecker_monomial>> t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1131  {
1132  typedef decltype(this->t_subs(name,c,s,args)) ret_type;
1133  typedef typename ret_type::value_type::first_type res_type;
1134  const auto v = unpack(args);
1135  v_type new_v;
1136  value_type n(0);
1137  // Build the new vector key.
1138  for (min_int<decltype(args.size()),typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1139  if (args[i].get_name() == name) {
1140  new_v.push_back(value_type(0));
1141  n = v[i];
1142  } else {
1143  new_v.push_back(v[i]);
1144  }
1145  }
1146  // Absolute value of the multiplier.
1147  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1148  // Prepare the powers of c and s to be used in the multiple angles formulae.
1149  std::unordered_map<value_type,U> c_map, s_map;
1150  c_map[0] = U(1);
1151  s_map[0] = U(1);
1152  for (value_type k(0); k < abs_n; ++k) {
1153  c_map[k + value_type(1)] = c_map[k] * c;
1154  s_map[k + value_type(1)] = s_map[k] * s;
1155  }
1156  // Init with the first element in the summation.
1157  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n])),
1158  sin_nx((sin_phase(abs_n) * math::binomial(abs_n,value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1159  for (value_type k(0); k < abs_n; ++k) {
1160  const value_type p = abs_n - (k + value_type(1));
1161  piranha_assert(p >= value_type(0));
1162  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as long
1163  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1164  // left-to-right associativity of multiplication.
1165  res_type tmp(math::binomial(abs_n,k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1166  cos_nx += cos_phase(p) * tmp;
1167  sin_nx += sin_phase(p) * tmp;
1168  }
1169  // Change sign as necessary.
1170  if (abs_n != n) {
1171  math::negate(sin_nx);
1172  }
1173  // Buld the new keys and canonicalise as needed.
1174  const bool sign_changed = canonicalise_impl(new_v);
1175  piranha_assert(new_v.size() == v.size());
1176  const auto new_int = ka::encode(new_v);
1177  real_trigonometric_kronecker_monomial cos_key(new_int,true), sin_key(new_int,false);
1178  ret_type retval;
1179  if (get_flavour()) {
1180  retval.push_back(std::make_pair(std::move(cos_nx),std::move(cos_key)));
1181  retval.push_back(std::make_pair(std::move(sin_nx),std::move(sin_key)));
1182  // Need to flip the sign on the sin * sin product if sign was not changed.
1183  if (!sign_changed) {
1184  math::negate(retval[1u].first);
1185  }
1186  } else {
1187  retval.push_back(std::make_pair(std::move(sin_nx),std::move(cos_key)));
1188  retval.push_back(std::make_pair(std::move(cos_nx),std::move(sin_key)));
1189  // Need to flip the sign on the cos * sin product if sign was changed.
1190  if (sign_changed) {
1191  math::negate(retval[1u].first);
1192  }
1193  }
1194  return retval;
