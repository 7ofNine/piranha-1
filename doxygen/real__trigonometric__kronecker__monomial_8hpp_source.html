204  // Implementation of canonicalisation.
205  static bool canonicalise_impl(v_type &unpacked)
206  {
207  const auto size = unpacked.size();
208  bool sign_change = false;
209  for (decltype(unpacked.size()) i = 0u; i < size; ++i) {
210  if (sign_change || unpacked[i] < value_type(0)) {
211  unpacked[i] = static_cast<value_type>(-unpacked[i]);
212  sign_change = true;
213  } else if (unpacked[i] > value_type(0)) {
214  break;
215  }
216  }
217  return sign_change;
218  }

472  bool is_compatible(const symbol_set &args) const noexcept
473  {
474  const auto s = args.size();
475  // No args means the value must also be zero.
476  if (s == 0u) {
477  return !m_value;
478  }
479  const auto &limits = ka::get_limits();
480  // If we overflow the maximum size available, we cannot use this object as key in series.
481  if (s >= limits.size()) {
482  return false;
483  }
484  const auto &l = limits[s];
485  // Value is compatible if it is within the bounds for the given size.
486  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
487  return false;
488  }
489  // Now check for the first multiplier.
490  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
491  // we do not need to put @throw specifications in the doc.
492  const auto unpacked = unpack(args);
493  // We know that s != 0.
494  piranha_assert(unpacked.size() > 0u);
495  for (typename v_type::size_type i = 0u; i < s; ++i) {
496  if (unpacked[i] < value_type(0)) {
497  return false;
498  } else if (unpacked[i] > value_type(0)) {
499  break;
500  }
501  }
502  return true;
503  }

721  template <typename Cf, multiply_enabler<Cf> = 0>
722  static void multiply(std::array<term<Cf, real_trigonometric_kronecker_monomial>, multiply_arity> &res,
723  const term<Cf, real_trigonometric_kronecker_monomial> &t1,
724  const term<Cf, real_trigonometric_kronecker_monomial> &t2, const symbol_set &args)
725  {
726  // Coefficients first.
727  detail::cf_mult_impl(res[0u].m_cf, t1.m_cf, t2.m_cf);
728  res[1u].m_cf = res[0u].m_cf;
729  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
730  if (f1 && f2) {
731  // cos, cos: no change.
732  } else if (!f1 && !f2) {
733  // sin, sin: negate the plus.
734  math::negate(res[0u].m_cf);
735  } else if (!f1 && f2) {
736  // sin, cos: no change.
737  } else {
738  // cos, sin: negate the minus.
739  math::negate(res[1u].m_cf);
740  }
741  // Now the keys.
742  auto &retval_plus = res[0u].m_key;
743  auto &retval_minus = res[1u].m_key;
744  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
745  bool sign_plus = false, sign_minus = false;
746  const auto size = args.size();
747  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
748  v_type result_plus, result_minus;
749  for (typename v_type::size_type i = 0u; i < size; ++i) {
750  result_plus.push_back(tmp1[i]);
751  detail::safe_integral_adder(result_plus[i], tmp2[i]);
752  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
753  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
754  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
755  // to int and safe_adder won't work as it expects identical types.
756  result_minus.push_back(tmp1[i]);
757  detail::safe_integral_adder(result_minus[i], static_cast<value_type>(-tmp2[i]));
758  }
759  // Handle sign changes.
760  sign_plus = canonicalise_impl(result_plus);
761  sign_minus = canonicalise_impl(result_minus);
762  // Compute them before assigning, so in case of exceptions we do not touch the return values.
763  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
764  retval_plus.m_value = re_plus;
765  retval_minus.m_value = re_minus;
766  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
767  retval_plus.m_flavour = f;
768  retval_minus.m_flavour = f;
769  // If multiplier sign was changed and the result is a sine, negate the coefficient.
770  if (sign_plus && !res[0u].m_key.get_flavour()) {
771  math::negate(res[0u].m_cf);
772  }
773  if (sign_minus && !res[1u].m_key.get_flavour()) {
774  math::negate(res[1u].m_cf);
775  }
776  }

831  void print(std::ostream &os, const symbol_set &args) const
832  {
833  // Don't print anything in case all multipliers are zero.
834  if (m_value == value_type(0)) {
835  return;
836  }
837  if (m_flavour) {
838  os << "cos(";
839  } else {
840  os << "sin(";
841  }
842  const auto tmp = unpack(args);
843  piranha_assert(tmp.size() == args.size());
844  const value_type zero(0), one(1), m_one(-1);
845  bool empty_output = true;
846  for (decltype(tmp.size()) i = 0u; i < tmp.size(); ++i) {
847  if (tmp[i] != zero) {
848  // A positive multiplier, in case previous output exists, must be preceded
849  // by a "+" sign.
850  if (tmp[i] > zero && !empty_output) {
851  os << '+';
852  }
853  // Print the multiplier, unless it's "-1": in that case, just print the minus sign.
854  if (tmp[i] == m_one) {
855  os << '-';
856  } else if (tmp[i] != one) {
857  os << detail::prepare_for_print(tmp[i]) << '*';
858  }
859  // Finally, print name of variable.
860  os << args[i].get_name();
861  empty_output = false;
862  }
863  }
864  os << ")";
865  }

893  // by a "+" sign.
894  if (tmp[i] > zero && !empty_output) {
895  os << '+';
896  }
897  // Print the multiplier, unless it's "-1": in that case, just print the minus sign.
898  if (tmp[i] == m_one) {
899  os << '-';
900  } else if (tmp[i] != one) {
901  os << static_cast<long long>(tmp[i]);
902  }
903  // Finally, print name of variable.
904  os << '{' << args[i].get_name() << '}';
905  empty_output = false;
906  }
907  }
908  os << "\\right)}";

930  std::pair<T, real_trigonometric_kronecker_monomial> partial(const symbol_set::positions &p,
931  const symbol_set &args) const
932  {
933  auto v = unpack(args);
934  // Cannot take derivative wrt more than one variable, and the position of that variable
935  // must be compatible with the monomial.
936  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
937  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
938  }
939  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
940  // variable with zero multiplier.
941  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
942  // there is a valid position and hence the size must be not zero.
943  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
944  return std::make_pair(T(0), real_trigonometric_kronecker_monomial(args));
945  }
946  auto v_b = v.begin();
947  // Original multiplier.
948  T n(v_b[*p.begin()]);
949  // Create a copy of this.
950  real_trigonometric_kronecker_monomial tmp_m(*this);
951  // Flip the flavour.
952  tmp_m.set_flavour(!get_flavour());
953  // Flip the sign of the multiplier as needed.
954  if (get_flavour()) {
955  // NOTE: this is safe, as it is coming out of a k codification which
956  // is symmetric.
957  math::negate(n);
958  }
959  return std::make_pair(std::move(n), std::move(tmp_m));
960  }

981  std::pair<T, real_trigonometric_kronecker_monomial> integrate(const symbol &s, const symbol_set &args) const
982  {

984  for (min_int<decltype(args.size()), typename v_type::size_type> i = 0u; i < args.size(); ++i) {
985  if (args[i] == s && !math::is_zero(v[i])) {
986  integer tmp_n(v[i]);
987  real_trigonometric_kronecker_monomial tmp_m(*this);

991  if (!get_flavour()) {
992  tmp_n.negate();
993  }
994  return std::make_pair(static_cast<value_type>(tmp_n), std::move(tmp_m));
995  }
996  }

1024  template <typename U>
1025  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
1026  {
1027  using return_type = eval_type<U>;
1028  if (unlikely(pmap.size() != args.size() || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
1029  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
1030  }
1031  // Run the unpack before the checks below in order to check the suitability of args.
1032  auto v = unpack(args);
1033  if (args.size() == 0u) {
1034  if (get_flavour()) {
1035  return return_type(1);
1036  } else {
1037  return return_type(0);
1038  }
1039  }
1040  using tmp_type = decltype(std::declval<U const &>() * std::declval<value_type const &>());
1041  tmp_type tmp(0);
1042  auto it = pmap.begin();
1043  for (min_int<decltype(args.size()), typename v_type::size_type> i = 0u; i < args.size(); ++i, ++it) {
1044  piranha_assert(it != pmap.end() && it->first == i);
1045  // NOTE: here it might make sense to use multiply_accumulate. There might be a perf gain
1046  // for things like real. Maybe fast FMA on Haswell too. Remember to adapt the type requirements
1047  // in case we implement this.
1048  tmp += it->second * v[i];
1049  }
1050  piranha_assert(it == pmap.end());
1051  if (get_flavour()) {
1052  return math::cos(tmp);
1053  }
1054  return math::sin(tmp);
1055  }

1169  template <typename U>
1170  std::vector<std::pair<t_subs_type<U>, real_trigonometric_kronecker_monomial>>
1171  t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1172  {
1173  typedef decltype(this->t_subs(name, c, s, args)) ret_type;
1174  typedef typename ret_type::value_type::first_type res_type;
1175  const auto v = unpack(args);
1176  v_type new_v;
1177  value_type n(0);
1178  // Build the new vector key.
1179  for (min_int<decltype(args.size()), typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1180  if (args[i].get_name() == name) {
1181  new_v.push_back(value_type(0));
1182  n = v[i];
1183  } else {
1184  new_v.push_back(v[i]);
1185  }
1186  }
1187  // Absolute value of the multiplier.
1188  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1189  // Prepare the powers of c and s to be used in the multiple angles formulae.
1190  std::unordered_map<value_type, U> c_map, s_map;
1191  c_map[0] = U(1);
1192  s_map[0] = U(1);
1193  for (value_type k(0); k < abs_n; ++k) {
1194  c_map[k + value_type(1)] = c_map[k] * c;
1195  s_map[k + value_type(1)] = s_map[k] * s;
1196  }
1197  // Init with the first element in the summation.
1198  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n, value_type(0)))
1199  * (c_map[value_type(0)] * s_map[abs_n])),
1200  sin_nx((sin_phase(abs_n) * math::binomial(abs_n, value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1201  for (value_type k(0); k < abs_n; ++k) {
1202  const value_type p = abs_n - (k + value_type(1));
1203  piranha_assert(p >= value_type(0));
1204  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as
1205  // long
1206  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1207  // left-to-right associativity of multiplication.
1208  res_type tmp(math::binomial(abs_n, k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1209  cos_nx += cos_phase(p) * tmp;
1210  sin_nx += sin_phase(p) * tmp;
1211  }
1212  // Change sign as necessary.
1213  if (abs_n != n) {
1214  math::negate(sin_nx);
1215  }
1216  // Buld the new keys and canonicalise as needed.
1217  const bool sign_changed = canonicalise_impl(new_v);
1218  piranha_assert(new_v.size() == v.size());
1219  const auto new_int = ka::encode(new_v);
1220  real_trigonometric_kronecker_monomial cos_key(new_int, true), sin_key(new_int, false);
1221  ret_type retval;
1222  if (get_flavour()) {
1223  retval.push_back(std::make_pair(std::move(cos_nx), std::move(cos_key)));
1224  retval.push_back(std::make_pair(std::move(sin_nx), std::move(sin_key)));
1225  // Need to flip the sign on the sin * sin product if sign was not changed.
1226  if (!sign_changed) {
1227  math::negate(retval[1u].first);
1228  }
1229  } else {
1230  retval.push_back(std::make_pair(std::move(sin_nx), std::move(cos_key)));
1231  retval.push_back(std::make_pair(std::move(cos_nx), std::move(sin_key)));
1232  // Need to flip the sign on the cos * sin product if sign was changed.
1233  if (sign_changed) {
1234  math::negate(retval[1u].first);
1235  }
1236  }
1237  return retval;
1238  }
