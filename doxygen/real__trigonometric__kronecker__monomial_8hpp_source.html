82 template <typename Archive, typename T>
83 inline void save(Archive &ar,
84  const piranha::boost_s11n_key_wrapper<piranha::real_trigonometric_kronecker_monomial<T>> &k, unsigned)
85 {
86  if (std::is_same<Archive, boost::archive::binary_oarchive>::value) {
87  piranha::boost_save(ar, k.key().get_int());
88  } else {
89  auto tmp = k.key().unpack(k.ss());
90  piranha::boost_save(ar, tmp);
91  }
92  piranha::boost_save(ar, k.key().get_flavour());
93 }

95 template <typename Archive, typename T>
96 inline void load(Archive &ar, piranha::boost_s11n_key_wrapper<piranha::real_trigonometric_kronecker_monomial<T>> &k,
97  unsigned)
98 {
99  if (std::is_same<Archive, boost::archive::binary_iarchive>::value) {
100  T value;
101  piranha::boost_load(ar, value);
102  k.key().set_int(value);
103  } else {
104  typename piranha::real_trigonometric_kronecker_monomial<T>::v_type tmp;
105  piranha::boost_load(ar, tmp);
106  if (unlikely(tmp.size() != k.ss().size())) {
107  piranha_throw(std::invalid_argument, "invalid size detected in the deserialization of a real Kronercker "
108  "trigonometric monomial: the deserialized size is "
109  + std::to_string(tmp.size())
110  + " but the reference symbol set has a size of "
111  + std::to_string(k.ss().size()));
112  }
113  // NOTE: here the exception safety is basic, as the last boost_load() could fail in principle.
114  // It does not really matter much, as there's no real dependency between the multipliers and the flavour,
115  // any combination is valid.
116  k.key() = piranha::real_trigonometric_kronecker_monomial<T>(tmp.begin(), tmp.end());
117  }
118  // The flavour loading is common.
119  bool f;
120  piranha::boost_load(ar, f);
121  k.key().set_flavour(f);
122 }

264  // Implementation of canonicalisation.
265  static bool canonicalise_impl(v_type &unpacked)
266  {
267  const auto size = unpacked.size();
268  bool sign_change = false;
269  for (decltype(unpacked.size()) i = 0u; i < size; ++i) {
270  if (sign_change || unpacked[i] < value_type(0)) {
271  unpacked[i] = static_cast<value_type>(-unpacked[i]);
272  sign_change = true;
273  } else if (unpacked[i] > value_type(0)) {
274  break;
275  }
276  }
277  return sign_change;
278  }

512  bool is_compatible(const symbol_set &args) const noexcept
513  {
514  const auto s = args.size();
515  // No args means the value must also be zero.
516  if (s == 0u) {
517  return !m_value;
518  }
519  const auto &limits = ka::get_limits();
520  // If we overflow the maximum size available, we cannot use this object as key in series.
521  if (s >= limits.size()) {
522  return false;
523  }
524  const auto &l = limits[s];
525  // Value is compatible if it is within the bounds for the given size.
526  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
527  return false;
528  }
529  // Now check for the first multiplier.
530  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
531  // we do not need to put @throw specifications in the doc.
532  const auto unpacked = unpack(args);
533  // We know that s != 0.
534  piranha_assert(unpacked.size() > 0u);
535  for (typename v_type::size_type i = 0u; i < s; ++i) {
536  if (unpacked[i] < value_type(0)) {
537  return false;
538  } else if (unpacked[i] > value_type(0)) {
539  break;
540  }
541  }
542  return true;
543  }

761  template <typename Cf, multiply_enabler<Cf> = 0>
762  static void multiply(std::array<term<Cf, real_trigonometric_kronecker_monomial>, multiply_arity> &res,
763  const term<Cf, real_trigonometric_kronecker_monomial> &t1,
764  const term<Cf, real_trigonometric_kronecker_monomial> &t2, const symbol_set &args)
765  {
766  // Coefficients first.
767  detail::cf_mult_impl(res[0u].m_cf, t1.m_cf, t2.m_cf);
768  res[1u].m_cf = res[0u].m_cf;
769  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
770  if (f1 && f2) {
771  // cos, cos: no change.
772  } else if (!f1 && !f2) {
773  // sin, sin: negate the plus.
774  math::negate(res[0u].m_cf);
775  } else if (!f1 && f2) {
776  // sin, cos: no change.
777  } else {
778  // cos, sin: negate the minus.
779  math::negate(res[1u].m_cf);
780  }
781  // Now the keys.
782  auto &retval_plus = res[0u].m_key;
783  auto &retval_minus = res[1u].m_key;
784  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
785  bool sign_plus = false, sign_minus = false;
786  const auto size = args.size();
787  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
788  v_type result_plus, result_minus;
789  for (typename v_type::size_type i = 0u; i < size; ++i) {
790  result_plus.push_back(tmp1[i]);
791  detail::safe_integral_adder(result_plus[i], tmp2[i]);
792  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
793  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
794  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
795  // to int and safe_adder won't work as it expects identical types.
796  result_minus.push_back(tmp1[i]);
797  detail::safe_integral_adder(result_minus[i], static_cast<value_type>(-tmp2[i]));
798  }
799  // Handle sign changes.
800  sign_plus = canonicalise_impl(result_plus);
801  sign_minus = canonicalise_impl(result_minus);
802  // Compute them before assigning, so in case of exceptions we do not touch the return values.
803  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
804  retval_plus.m_value = re_plus;
805  retval_minus.m_value = re_minus;
806  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
807  retval_plus.m_flavour = f;
808  retval_minus.m_flavour = f;
809  // If multiplier sign was changed and the result is a sine, negate the coefficient.
810  if (sign_plus && !res[0u].m_key.get_flavour()) {
811  math::negate(res[0u].m_cf);
812  }
813  if (sign_minus && !res[1u].m_key.get_flavour()) {
814  math::negate(res[1u].m_cf);
815  }
816  }

871  void print(std::ostream &os, const symbol_set &args) const
872  {

970  std::pair<T, real_trigonometric_kronecker_monomial> partial(const symbol_set::positions &p,
971  const symbol_set &args) const
972  {
973  auto v = unpack(args);
974  // Cannot take derivative wrt more than one variable, and the position of that variable
975  // must be compatible with the monomial.
976  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
977  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
978  }
979  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
980  // variable with zero multiplier.
981  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
982  // there is a valid position and hence the size must be not zero.
983  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
984  return std::make_pair(T(0), real_trigonometric_kronecker_monomial(args));
985  }
986  auto v_b = v.begin();
987  // Original multiplier.
988  T n(v_b[*p.begin()]);
989  // Create a copy of this.
990  real_trigonometric_kronecker_monomial tmp_m(*this);
991  // Flip the flavour.
992  tmp_m.set_flavour(!get_flavour());
993  // Flip the sign of the multiplier as needed.
994  if (get_flavour()) {
995  // NOTE: this is safe, as it is coming out of a k codification which
996  // is symmetric.
997  math::negate(n);
998  }
999  return std::make_pair(std::move(n), std::move(tmp_m));

1066  {

1209  template <typename U>
1210  std::vector<std::pair<t_subs_type<U>, real_trigonometric_kronecker_monomial>>
1211  t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1212  {
1213  typedef decltype(this->t_subs(name, c, s, args)) ret_type;
1214  typedef typename ret_type::value_type::first_type res_type;
1215  const auto v = unpack(args);
1216  v_type new_v;
1217  value_type n(0);
1218  // Build the new vector key.
1219  for (min_int<decltype(args.size()), typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1220  if (args[i].get_name() == name) {
1221  new_v.push_back(value_type(0));
1222  n = v[i];
1223  } else {
1224  new_v.push_back(v[i]);
1225  }
1226  }
1227  // Absolute value of the multiplier.
1228  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1229  // Prepare the powers of c and s to be used in the multiple angles formulae.
1230  std::unordered_map<value_type, U> c_map, s_map;
1231  c_map[0] = U(1);
1232  s_map[0] = U(1);
1233  for (value_type k(0); k < abs_n; ++k) {
1234  c_map[k + value_type(1)] = c_map[k] * c;
1235  s_map[k + value_type(1)] = s_map[k] * s;
1236  }
1237  // Init with the first element in the summation.
1238  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n, value_type(0)))
1239  * (c_map[value_type(0)] * s_map[abs_n])),
1240  sin_nx((sin_phase(abs_n) * math::binomial(abs_n, value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1241  for (value_type k(0); k < abs_n; ++k) {
1242  const value_type p = abs_n - (k + value_type(1));
1243  piranha_assert(p >= value_type(0));
1244  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as
1245  // long
1246  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1247  // left-to-right associativity of multiplication.
1248  res_type tmp(math::binomial(abs_n, k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1249  cos_nx += cos_phase(p) * tmp;
1250  sin_nx += sin_phase(p) * tmp;
1251  }
1252  // Change sign as necessary.
1253  if (abs_n != n) {
1254  math::negate(sin_nx);
1255  }
1256  // Buld the new keys and canonicalise as needed.
1257  const bool sign_changed = canonicalise_impl(new_v);
1258  piranha_assert(new_v.size() == v.size());
1259  const auto new_int = ka::encode(new_v);
1260  real_trigonometric_kronecker_monomial cos_key(new_int, true), sin_key(new_int, false);
1261  ret_type retval;
1262  if (get_flavour()) {
1263  retval.push_back(std::make_pair(std::move(cos_nx), std::move(cos_key)));
1264  retval.push_back(std::make_pair(std::move(sin_nx), std::move(sin_key)));
1265  // Need to flip the sign on the sin * sin product if sign was not changed.
1266  if (!sign_changed) {
1267  math::negate(retval[1u].first);
1268  }
1269  } else {
1270  retval.push_back(std::make_pair(std::move(sin_nx), std::move(cos_key)));
1271  retval.push_back(std::make_pair(std::move(cos_nx), std::move(sin_key)));
1272  // Need to flip the sign on the cos * sin product if sign was changed.
1273  if (sign_changed) {
1274  math::negate(retval[1u].first);
1275  }
1276  }
1277  return retval;
1278  }
