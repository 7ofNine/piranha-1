82 template <typename Archive, typename T>
83 inline void save(Archive &ar,
84  const piranha::boost_s11n_key_wrapper<piranha::real_trigonometric_kronecker_monomial<T>> &k, unsigned)
85 {
86  if (std::is_same<Archive, boost::archive::binary_oarchive>::value) {
87  piranha::boost_save(ar, k.key().get_int());
88  } else {
89  auto tmp = k.key().unpack(k.ss());

91  }
92  piranha::boost_save(ar, k.key().get_flavour());
93 }

95 template <typename Archive, typename T>
96 inline void load(Archive &ar, piranha::boost_s11n_key_wrapper<piranha::real_trigonometric_kronecker_monomial<T>> &k,
97  unsigned)
98 {
99  if (std::is_same<Archive, boost::archive::binary_iarchive>::value) {
100  T value;
101  piranha::boost_load(ar, value);
102  k.key().set_int(value);
103  } else {
104  typename piranha::real_trigonometric_kronecker_monomial<T>::v_type tmp;
105  piranha::boost_load(ar, tmp);
106  if (unlikely(tmp.size() != k.ss().size())) {
107  piranha_throw(std::invalid_argument, "invalid size detected in the deserialization of a real Kronercker "
108  "trigonometric monomial: the deserialized size is "
109  + std::to_string(tmp.size())
110  + " but the reference symbol set has a size of "
111  + std::to_string(k.ss().size()));
112  }
113  // NOTE: here the exception safety is basic, as the last boost_load() could fail in principle.
114  // It does not really matter much, as there's no real dependency between the multipliers and the flavour,
115  // any combination is valid.
116  k.key() = piranha::real_trigonometric_kronecker_monomial<T>(tmp.begin(), tmp.end());
117  }
118  // The flavour loading is common.
119  bool f;
120  piranha::boost_load(ar, f);
121  k.key().set_flavour(f);
122 }

264  // Implementation of canonicalisation.
265  static bool canonicalise_impl(v_type &unpacked)
266  {
267  const auto size = unpacked.size();
268  bool sign_change = false;
269  for (decltype(unpacked.size()) i = 0u; i < size; ++i) {
270  if (sign_change || unpacked[i] < value_type(0)) {
271  unpacked[i] = static_cast<value_type>(-unpacked[i]);
272  sign_change = true;
273  } else if (unpacked[i] > value_type(0)) {
274  break;
275  }
276  }
277  return sign_change;
278  }

522  bool is_compatible(const symbol_set &args) const noexcept
523  {
524  const auto s = args.size();
525  // No args means the value must also be zero.
526  if (s == 0u) {
527  return !m_value;
528  }
529  const auto &limits = ka::get_limits();
530  // If we overflow the maximum size available, we cannot use this object as key in series.
531  if (s >= limits.size()) {
532  return false;
533  }
534  const auto &l = limits[s];
535  // Value is compatible if it is within the bounds for the given size.
536  if (m_value < std::get<1u>(l) || m_value > std::get<2u>(l)) {
537  return false;
538  }
539  // Now check for the first multiplier.
540  // NOTE: here we have already checked all the conditions that could lead to unpack() throwing, so
541  // we do not need to put @throw specifications in the doc.
542  const auto unpacked = unpack(args);
543  // We know that s != 0.
544  piranha_assert(unpacked.size() > 0u);
545  for (typename v_type::size_type i = 0u; i < s; ++i) {
546  if (unpacked[i] < value_type(0)) {
547  return false;
548  } else if (unpacked[i] > value_type(0)) {
549  break;
550  }
551  }
552  return true;
553  }

801  template <typename Cf, multiply_enabler<Cf> = 0>
802  static void multiply(std::array<term<Cf, real_trigonometric_kronecker_monomial>, multiply_arity> &res,
803  const term<Cf, real_trigonometric_kronecker_monomial> &t1,
804  const term<Cf, real_trigonometric_kronecker_monomial> &t2, const symbol_set &args)
805  {
806  // Coefficients first.
807  detail::cf_mult_impl(res[0u].m_cf, t1.m_cf, t2.m_cf);
808  res[1u].m_cf = res[0u].m_cf;
809  const bool f1 = t1.m_key.get_flavour(), f2 = t2.m_key.get_flavour();
810  if (f1 && f2) {
811  // cos, cos: no change.
812  } else if (!f1 && !f2) {
813  // sin, sin: negate the plus.
814  math::negate(res[0u].m_cf);
815  } else if (!f1 && f2) {
816  // sin, cos: no change.
817  } else {
818  // cos, sin: negate the minus.
819  math::negate(res[1u].m_cf);
820  }
821  // Now the keys.
822  auto &retval_plus = res[0u].m_key;
823  auto &retval_minus = res[1u].m_key;
824  // Flags to signal if a sign change in the multipliers was needed as part of the canonicalization.
825  bool sign_plus = false, sign_minus = false;
826  const auto size = args.size();
827  const auto tmp1 = t1.m_key.unpack(args), tmp2 = t2.m_key.unpack(args);
828  v_type result_plus, result_minus;
829  for (typename v_type::size_type i = 0u; i < size; ++i) {
830  result_plus.push_back(tmp1[i]);
831  detail::safe_integral_adder(result_plus[i], tmp2[i]);
832  // NOTE: it is safe here to take the negative because in kronecker_array we are guaranteed
833  // that the range of each element is symmetric, so if tmp2[i] is representable also -tmp2[i] is.
834  // NOTE: the static cast here is because if value_type is narrower than int, the unary minus will promote
835  // to int and safe_adder won't work as it expects identical types.
836  result_minus.push_back(tmp1[i]);
837  detail::safe_integral_adder(result_minus[i], static_cast<value_type>(-tmp2[i]));
838  }
839  // Handle sign changes.
840  sign_plus = canonicalise_impl(result_plus);
841  sign_minus = canonicalise_impl(result_minus);
842  // Compute them before assigning, so in case of exceptions we do not touch the return values.
843  const auto re_plus = ka::encode(result_plus), re_minus = ka::encode(result_minus);
844  retval_plus.m_value = re_plus;
845  retval_minus.m_value = re_minus;
846  const bool f = (t1.m_key.get_flavour() == t2.m_key.get_flavour());
847  retval_plus.m_flavour = f;
848  retval_minus.m_flavour = f;
849  // If multiplier sign was changed and the result is a sine, negate the coefficient.
850  if (sign_plus && !res[0u].m_key.get_flavour()) {
851  math::negate(res[0u].m_cf);
852  }
853  if (sign_minus && !res[1u].m_key.get_flavour()) {
854  math::negate(res[1u].m_cf);
855  }
856  }

1010  std::pair<T, real_trigonometric_kronecker_monomial> partial(const symbol_set::positions &p,
1011  const symbol_set &args) const
1012  {
1013  auto v = unpack(args);
1014  // Cannot take derivative wrt more than one variable, and the position of that variable
1015  // must be compatible with the monomial.
1016  if (p.size() > 1u || (p.size() == 1u && p.back() >= args.size())) {
1017  piranha_throw(std::invalid_argument, "invalid size of symbol_set::positions");
1018  }
1019  // Derivative wrt a variable not in the monomial: position is empty, or refers to a
1020  // variable with zero multiplier.
1021  // NOTE: safe to take v.begin() here, as the checks on the positions above ensure
1022  // there is a valid position and hence the size must be not zero.
1023  if (!p.size() || math::is_zero(v.begin()[*p.begin()])) {
1024  return std::make_pair(T(0), real_trigonometric_kronecker_monomial(args));
1025  }
1026  auto v_b = v.begin();
1027  // Original multiplier.
1028  T n(v_b[*p.begin()]);
1029  // Create a copy of this.
1030  real_trigonometric_kronecker_monomial tmp_m(*this);
1031  // Flip the flavour.
1032  tmp_m.set_flavour(!get_flavour());
1033  // Flip the sign of the multiplier as needed.
1034  if (get_flavour()) {
1035  // NOTE: this is safe, as it is coming out of a k codification which
1036  // is symmetric.
1037  math::negate(n);
1038  }
1039  return std::make_pair(std::move(n), std::move(tmp_m));

1249  template <typename U>
1250  std::vector<std::pair<t_subs_type<U>, real_trigonometric_kronecker_monomial>>
1251  t_subs(const std::string &name, const U &c, const U &s, const symbol_set &args) const
1252  {
1253  typedef decltype(this->t_subs(name, c, s, args)) ret_type;
1254  typedef typename ret_type::value_type::first_type res_type;
1255  const auto v = unpack(args);
1256  v_type new_v;
1257  value_type n(0);
1258  // Build the new vector key.
1259  for (min_int<decltype(args.size()), typename v_type::size_type> i = 0u; i < args.size(); ++i) {
1260  if (args[i].get_name() == name) {
1261  new_v.push_back(value_type(0));
1262  n = v[i];
1263  } else {
1264  new_v.push_back(v[i]);
1265  }
1266  }
1267  // Absolute value of the multiplier.
1268  const value_type abs_n = (n >= 0) ? n : static_cast<value_type>(-n);
1269  // Prepare the powers of c and s to be used in the multiple angles formulae.
1270  std::unordered_map<value_type, U> c_map, s_map;
1271  c_map[0] = U(1);
1272  s_map[0] = U(1);
1273  for (value_type k(0); k < abs_n; ++k) {
1274  c_map[k + value_type(1)] = c_map[k] * c;
1275  s_map[k + value_type(1)] = s_map[k] * s;
1276  }
1277  // Init with the first element in the summation.
1278  res_type cos_nx((cos_phase(abs_n) * math::binomial(abs_n, value_type(0)))
1279  * (c_map[value_type(0)] * s_map[abs_n])),
1280  sin_nx((sin_phase(abs_n) * math::binomial(abs_n, value_type(0))) * (c_map[value_type(0)] * s_map[abs_n]));
1281  for (value_type k(0); k < abs_n; ++k) {
1282  const value_type p = abs_n - (k + value_type(1));
1283  piranha_assert(p >= value_type(0));
1284  // NOTE: here the type is slightly different from the decltype() that determines the return type, but as
1285  // long
1286  // as binomial(value_type,value_type) returns integer there will be no difference because of the
1287  // left-to-right associativity of multiplication.
1288  res_type tmp(math::binomial(abs_n, k + value_type(1)) * (c_map[k + value_type(1)] * s_map[p]));
1289  cos_nx += cos_phase(p) * tmp;
1290  sin_nx += sin_phase(p) * tmp;
1291  }
1292  // Change sign as necessary.
1293  if (abs_n != n) {
1294  math::negate(sin_nx);
1295  }
1296  // Buld the new keys and canonicalise as needed.
1297  const bool sign_changed = canonicalise_impl(new_v);
1298  piranha_assert(new_v.size() == v.size());
1299  const auto new_int = ka::encode(new_v);
1300  real_trigonometric_kronecker_monomial cos_key(new_int, true), sin_key(new_int, false);
1301  ret_type retval;
1302  if (get_flavour()) {
1303  retval.push_back(std::make_pair(std::move(cos_nx), std::move(cos_key)));
1304  retval.push_back(std::make_pair(std::move(sin_nx), std::move(sin_key)));
1305  // Need to flip the sign on the sin * sin product if sign was not changed.
1306  if (!sign_changed) {
1307  math::negate(retval[1u].first);
1308  }
1309  } else {
1310  retval.push_back(std::make_pair(std::move(sin_nx), std::move(cos_key)));
1311  retval.push_back(std::make_pair(std::move(cos_nx), std::move(sin_key)));
1312  // Need to flip the sign on the cos * sin product if sign was changed.
1313  if (sign_changed) {
1314  math::negate(retval[1u].first);
1315  }
1316  }
1317  return retval;
1318  }
