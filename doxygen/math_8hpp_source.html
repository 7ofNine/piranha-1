819 template <typename T>
820 inline detail::pbracket_type<T> pbracket(const T &f, const T &g, const std::vector<std::string> &p_list,
821  const std::vector<std::string> &q_list)
822 {
823  using return_type = detail::pbracket_type<T>;
824  if (p_list.size() != q_list.size()) {
825  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
826  }
827  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
828  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
829  }
830  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
831  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
832  }
833  return_type retval = return_type(0);
834  for (decltype(p_list.size()) i = 0u; i < p_list.size(); ++i) {
835  // NOTE: could use multadd/sub here, if we implement it for series.
836  retval = retval + partial(f,q_list[i]) * partial(g,p_list[i]);
837  retval = retval - partial(f,p_list[i]) * partial(g,q_list[i]);
838  }
839  return retval;
840 }

873 template <typename T>
874 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
875  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
876 {
877  using p_type = decltype(math::pbracket(*new_q[0],*new_p[0],p_list,q_list));
878  if (p_list.size() != q_list.size()) {
879  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
880  }
881  if (new_p.size() != new_q.size()) {
882  piranha_throw(std::invalid_argument,"the number of new coordinates is different from the number of new momenta");
883  }
884  if (p_list.size() != new_p.size()) {
885  piranha_throw(std::invalid_argument,"the number of new momenta is different from the number of momenta");
886  }
887  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
888  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
889  }
890  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
891  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
892  }
893  const auto size = new_p.size();
894  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
895  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
896  // NOTE: no need for actually doing computations when i == j.
897  if (i != j && !math::is_zero(math::pbracket(*new_p[i],*new_p[j],p_list,q_list))) {
898  return false;
899  }
900  if (i != j && !math::is_zero(math::pbracket(*new_q[i],*new_q[j],p_list,q_list))) {
901  return false;
902  }
903  // Poisson bracket needs to be zero for i != j, one for i == j.
904  // NOTE: cast from bool to int is always 0 or 1.
905  if (math::pbracket(*new_q[i],*new_p[j],p_list,q_list) != p_type(static_cast<int>(i == j))) {
906  return false;
907  }
908  }
909  }
910  return true;
911 }
