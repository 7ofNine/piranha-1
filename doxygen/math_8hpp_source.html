1133 template <typename T>
1134 inline detail::pbracket_type<T> pbracket(const T &f, const T &g, const std::vector<std::string> &p_list,
1135  const std::vector<std::string> &q_list)
1136 {
1137  using return_type = detail::pbracket_type<T>;
1138  if (p_list.size() != q_list.size()) {
1139  piranha_throw(std::invalid_argument, "the number of coordinates is different from the number of momenta");
1140  }
1141  if (std::unordered_set<std::string>(p_list.begin(), p_list.end()).size() != p_list.size()) {
1142  piranha_throw(std::invalid_argument, "the list of momenta contains duplicate entries");
1143  }
1144  if (std::unordered_set<std::string>(q_list.begin(), q_list.end()).size() != q_list.size()) {
1145  piranha_throw(std::invalid_argument, "the list of coordinates contains duplicate entries");
1146  }
1147  return_type retval = return_type(0);
1148  for (decltype(p_list.size()) i = 0u; i < p_list.size(); ++i) {
1149  // NOTE: could use multadd/sub here, if we implement it for series.
1150  retval = retval + partial(f, q_list[i]) * partial(g, p_list[i]);
1151  retval = retval - partial(f, p_list[i]) * partial(g, q_list[i]);
1152  }
1153  return retval;
1154 }

1189 template <typename T>
1190 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
1191  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
1192 {
1193  using p_type = decltype(math::pbracket(*new_q[0], *new_p[0], p_list, q_list));
1194  if (p_list.size() != q_list.size()) {
1195  piranha_throw(std::invalid_argument, "the number of coordinates is different from the number of momenta");
1196  }
1197  if (new_p.size() != new_q.size()) {
1198  piranha_throw(std::invalid_argument,
1199  "the number of new coordinates is different from the number of new momenta");
1200  }
1201  if (p_list.size() != new_p.size()) {
1202  piranha_throw(std::invalid_argument, "the number of new momenta is different from the number of momenta");
1203  }
1204  if (std::unordered_set<std::string>(p_list.begin(), p_list.end()).size() != p_list.size()) {
1205  piranha_throw(std::invalid_argument, "the list of momenta contains duplicate entries");
1206  }
1207  if (std::unordered_set<std::string>(q_list.begin(), q_list.end()).size() != q_list.size()) {
1208  piranha_throw(std::invalid_argument, "the list of coordinates contains duplicate entries");
1209  }
1210  const auto size = new_p.size();
1211  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
1212  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
1213  // NOTE: no need for actually doing computations when i == j.
1214  if (i != j && !math::is_zero(math::pbracket(*new_p[i], *new_p[j], p_list, q_list))) {
1215  return false;
1216  }
1217  if (i != j && !math::is_zero(math::pbracket(*new_q[i], *new_q[j], p_list, q_list))) {
1218  return false;
1219  }
1220  // Poisson bracket needs to be zero for i != j, one for i == j.
1221  // NOTE: cast from bool to int is always 0 or 1.
1222  if (math::pbracket(*new_q[i], *new_p[j], p_list, q_list) != p_type(static_cast<int>(i == j))) {
1223  return false;
1224  }
1225  }
1226  }
1227  return true;
1228 }
