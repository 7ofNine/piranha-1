1131 template <typename T>
1132 inline detail::pbracket_type<T> pbracket(const T &f, const T &g, const std::vector<std::string> &p_list,
1133  const std::vector<std::string> &q_list)
1134 {
1135  using return_type = detail::pbracket_type<T>;
1136  if (p_list.size() != q_list.size()) {
1137  piranha_throw(std::invalid_argument, "the number of coordinates is different from the number of momenta");
1138  }
1139  if (std::unordered_set<std::string>(p_list.begin(), p_list.end()).size() != p_list.size()) {
1140  piranha_throw(std::invalid_argument, "the list of momenta contains duplicate entries");
1141  }
1142  if (std::unordered_set<std::string>(q_list.begin(), q_list.end()).size() != q_list.size()) {
1143  piranha_throw(std::invalid_argument, "the list of coordinates contains duplicate entries");
1144  }
1145  return_type retval = return_type(0);
1146  for (decltype(p_list.size()) i = 0u; i < p_list.size(); ++i) {
1147  // NOTE: could use multadd/sub here, if we implement it for series.
1148  retval = retval + partial(f, q_list[i]) * partial(g, p_list[i]);
1149  retval = retval - partial(f, p_list[i]) * partial(g, q_list[i]);
1150  }
1151  return retval;
1152 }

1187 template <typename T>
1188 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
1189  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
1190 {
1191  using p_type = decltype(math::pbracket(*new_q[0], *new_p[0], p_list, q_list));
1192  if (p_list.size() != q_list.size()) {
1193  piranha_throw(std::invalid_argument, "the number of coordinates is different from the number of momenta");
1194  }
1195  if (new_p.size() != new_q.size()) {
1196  piranha_throw(std::invalid_argument,
1197  "the number of new coordinates is different from the number of new momenta");
1198  }
1199  if (p_list.size() != new_p.size()) {
1200  piranha_throw(std::invalid_argument, "the number of new momenta is different from the number of momenta");
1201  }
1202  if (std::unordered_set<std::string>(p_list.begin(), p_list.end()).size() != p_list.size()) {
1203  piranha_throw(std::invalid_argument, "the list of momenta contains duplicate entries");
1204  }
1205  if (std::unordered_set<std::string>(q_list.begin(), q_list.end()).size() != q_list.size()) {
1206  piranha_throw(std::invalid_argument, "the list of coordinates contains duplicate entries");
1207  }
1208  const auto size = new_p.size();
1209  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
1210  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
1211  // NOTE: no need for actually doing computations when i == j.
1212  if (i != j && !math::is_zero(math::pbracket(*new_p[i], *new_p[j], p_list, q_list))) {
1213  return false;
1214  }
1215  if (i != j && !math::is_zero(math::pbracket(*new_q[i], *new_q[j], p_list, q_list))) {
1216  return false;
1217  }
1218  // Poisson bracket needs to be zero for i != j, one for i == j.
1219  // NOTE: cast from bool to int is always 0 or 1.
1220  if (math::pbracket(*new_q[i], *new_p[j], p_list, q_list) != p_type(static_cast<int>(i == j))) {
1221  return false;
1222  }
1223  }
1224  }
1225  return true;
1226 }
1227 }
