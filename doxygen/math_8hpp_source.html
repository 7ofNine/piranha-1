1146  const std::vector<std::string> &q_list)
1147 {

1150  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
1151  }
1152  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
1153  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
1154  }
1155  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
1156  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");

1198 template <typename T>
1199 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
1200  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
1201 {
1202  using p_type = decltype(math::pbracket(*new_q[0],*new_p[0],p_list,q_list));
1203  if (p_list.size() != q_list.size()) {
1204  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
1205  }
1206  if (new_p.size() != new_q.size()) {
1207  piranha_throw(std::invalid_argument,"the number of new coordinates is different from the number of new momenta");
1208  }
1209  if (p_list.size() != new_p.size()) {
1210  piranha_throw(std::invalid_argument,"the number of new momenta is different from the number of momenta");
1211  }
1212  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
1213  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
1214  }
1215  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
1216  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
1217  }
1218  const auto size = new_p.size();
1219  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
1220  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
1221  // NOTE: no need for actually doing computations when i == j.
1222  if (i != j && !math::is_zero(math::pbracket(*new_p[i],*new_p[j],p_list,q_list))) {
1223  return false;
1224  }
1225  if (i != j && !math::is_zero(math::pbracket(*new_q[i],*new_q[j],p_list,q_list))) {
1226  return false;
1227  }
1228  // Poisson bracket needs to be zero for i != j, one for i == j.
1229  // NOTE: cast from bool to int is always 0 or 1.
1230  if (math::pbracket(*new_q[i],*new_p[j],p_list,q_list) != p_type(static_cast<int>(i == j))) {
1231  return false;
1232  }
1233  }
1234  }
1235  return true;
1236 }
