836 template <typename T>
837 inline detail::pbracket_type<T> pbracket(const T &f, const T &g, const std::vector<std::string> &p_list,
838  const std::vector<std::string> &q_list)
839 {
840  using return_type = detail::pbracket_type<T>;
841  if (p_list.size() != q_list.size()) {
842  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
843  }
844  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
845  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
846  }
847  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
848  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
849  }
850  return_type retval = return_type(0);
851  for (decltype(p_list.size()) i = 0u; i < p_list.size(); ++i) {
852  // NOTE: could use multadd/sub here, if we implement it for series.
853  retval = retval + partial(f,q_list[i]) * partial(g,p_list[i]);
854  retval = retval - partial(f,p_list[i]) * partial(g,q_list[i]);
855  }
856  return retval;
857 }

890 template <typename T>
891 inline bool is_canonical_impl(const std::vector<T const *> &new_p, const std::vector<T const *> &new_q,
892  const std::vector<std::string> &p_list, const std::vector<std::string> &q_list)
893 {
894  using p_type = decltype(math::pbracket(*new_q[0],*new_p[0],p_list,q_list));
895  if (p_list.size() != q_list.size()) {
896  piranha_throw(std::invalid_argument,"the number of coordinates is different from the number of momenta");
897  }
898  if (new_p.size() != new_q.size()) {
899  piranha_throw(std::invalid_argument,"the number of new coordinates is different from the number of new momenta");
900  }
901  if (p_list.size() != new_p.size()) {
902  piranha_throw(std::invalid_argument,"the number of new momenta is different from the number of momenta");
903  }
904  if (std::unordered_set<std::string>(p_list.begin(),p_list.end()).size() != p_list.size()) {
905  piranha_throw(std::invalid_argument,"the list of momenta contains duplicate entries");
906  }
907  if (std::unordered_set<std::string>(q_list.begin(),q_list.end()).size() != q_list.size()) {
908  piranha_throw(std::invalid_argument,"the list of coordinates contains duplicate entries");
909  }
910  const auto size = new_p.size();
911  for (decltype(new_p.size()) i = 0u; i < size; ++i) {
912  for (decltype(new_p.size()) j = 0u; j < size; ++j) {
913  // NOTE: no need for actually doing computations when i == j.
914  if (i != j && !math::is_zero(math::pbracket(*new_p[i],*new_p[j],p_list,q_list))) {
915  return false;
916  }
917  if (i != j && !math::is_zero(math::pbracket(*new_q[i],*new_q[j],p_list,q_list))) {
918  return false;
919  }
920  // Poisson bracket needs to be zero for i != j, one for i == j.
921  // NOTE: cast from bool to int is always 0 or 1.
922  if (math::pbracket(*new_q[i],*new_p[j],p_list,q_list) != p_type(static_cast<int>(i == j))) {
923  return false;
924  }
925  }
926  }
927  return true;
928 }
