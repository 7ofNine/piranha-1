70 // Generic binomial implementation using the falling factorial. U must be an integer
71 // type, T can be anything that supports basic arithmetics. k must be non-negative.
72 template <typename T, typename U>
73 inline T generic_binomial(const T &x, const U &k)
74 {
75  const U zero(0), one(1);
76  if (generic_binomial_check_k(k, zero)) {
77  piranha_throw(std::invalid_argument, "negative k value in binomial coefficient");
78  }
79  // Zero at bottom results always in 1.
80  if (k == zero) {
81  return T(1);
82  }
83  T tmp(x), retval = x / T(k);
84  --tmp;
85  for (auto i = static_cast<U>(k - one); i >= one; --i, --tmp) {
86  retval *= tmp;
87  retval /= T(i);
88  }
89  return retval;
90 }

92 // Compute gamma(a)/(gamma(b) * gamma(c)), assuming a, b and c are not negative ints.
93 template <typename T>
94 inline T compute_3_gamma(const T &a, const T &b, const T &c)
95 {
96  // Here we should never enter with negative ints.
97  piranha_assert(a >= T(0) || std::trunc(a) != a);
98  piranha_assert(b >= T(0) || std::trunc(b) != b);
99  piranha_assert(c >= T(0) || std::trunc(c) != c);
100  const T pi = boost::math::constants::pi<T>();
101  T tmp0(0), tmp1(1);
102  if (a < T(0)) {
103  tmp0 -= std::lgamma(T(1) - a);
104  tmp1 *= pi / std::sin(a * pi);
105  } else {
106  tmp0 += std::lgamma(a);
107  }
108  if (b < T(0)) {
109  tmp0 += std::lgamma(T(1) - b);
110  tmp1 *= std::sin(b * pi) / pi;
111  } else {
112  tmp0 -= std::lgamma(b);
113  }
114  if (c < T(0)) {
115  tmp0 += std::lgamma(T(1) - c);
116  tmp1 *= std::sin(c * pi) / pi;
117  } else {
118  tmp0 -= std::lgamma(c);
119  }
120  return std::exp(tmp0) * tmp1;
121 }

123 // Implementation of the generalised binomial coefficient for floating-point types.
124 template <typename T>
125 inline T fp_binomial(const T &x, const T &y)
126 {
127  static_assert(std::is_floating_point<T>::value, "Invalid type for fp_binomial.");
128  if (unlikely(!std::isfinite(x) || !std::isfinite(y))) {
129  piranha_throw(std::invalid_argument,
130  "cannot compute binomial coefficient with non-finite floating-point argument(s)");
131  }
132  const bool neg_int_x = std::trunc(x + T(1)) == (x + T(1)) && (x + T(1)) <= T(0),
133  neg_int_y = std::trunc(y + T(1)) == (y + T(1)) && (y + T(1)) <= T(0),
134  neg_int_x_y = std::trunc(x - y + T(1)) == (x - y + T(1)) && (x - y + T(1)) <= T(0);
135  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
136  switch (mask) {
137  case 0u:
138  // Case 0 is the non-special one, use the default implementation.
139  return compute_3_gamma(x + T(1), y + T(1), x - y + T(1));
140  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
141  case 2u:
142  case 4u:
143  // These are finite numerators with infinite denominators.
144  return T(0.);
145  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
146  case 3u: {
147  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
148  // formula to make them finite.
149  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
150  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
151  // here.
152  const auto phase = std::pow(T(-1), x + T(1)) / std::pow(T(-1), y + T(1));
153  return compute_3_gamma(-y, -x, x - y + T(1)) * phase;
154  }
155  case 5u: {
156  const auto phase = std::pow(T(-1), x - y + T(1)) / std::pow(T(-1), x + T(1));
157  return compute_3_gamma(-(x - y), y + T(1), -x) * phase;
158  }
159  }
160  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
161  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
162  return T(0);
163 }
