86 template <typename T, typename U>
87 class lambdified
88 {
89  static_assert(std::is_same<T,typename std::decay<T>::type>::value,"Invalid type.");
90  static_assert(std::is_same<U,typename std::decay<U>::type>::value,"Invalid type.");
91  static_assert(detail::math_lambdified_reqs<T,U>::value,"Invalid types.");
92  using eval_type = decltype(math::evaluate(std::declval<const T &>(),
93  std::declval<const std::unordered_map<std::string,U> &>()));
94  // Constructor implementation.
95  void construct(const std::vector<std::string> &names)
96  {
97  // Check if there are duplicates.
98  auto names_copy(names);
99  std::sort(names_copy.begin(),names_copy.end());
100  if (std::unique(names_copy.begin(),names_copy.end()) != names_copy.end()) {
101  piranha_throw(std::invalid_argument,"the list of evaluation symbols contains duplicates");
102  }
103  // Fill in the eval dict and the vector of pointers.
104  for (const auto &s: names) {
105  const auto ret = m_eval_dict.emplace(std::make_pair(s,U{}));
106  piranha_assert(ret.second);
107  m_ptrs.push_back(std::addressof(ret.first->second));
108  }
109  }
110  // Reconstruct the vector of pointers after a copy or move operation.
111  void reconstruct_ptrs()
112  {
113  piranha_assert(m_ptrs.empty());
114  piranha_assert(m_names.size() == m_eval_dict.size());
115  for (const auto &s: m_names) {
116  piranha_assert(m_eval_dict.count(s) == 1u);
117  m_ptrs.push_back(std::addressof(m_eval_dict.find(s)->second));
118  }
119  }
120  public:

141  {
142  construct(names);
143  }

180  m_eval_dict(std::move(other.m_eval_dict))
181  {
182  // NOTE: it looks like we cannot be sure the moved-in pointers are still valid.
183  // Let's just make sure.
184  reconstruct_ptrs();
185  }

209  eval_type operator()(const std::vector<U> &values)
210  {
211  if (unlikely(values.size() != m_eval_dict.size())) {
212  piranha_throw(std::invalid_argument,"the size of the vector of evaluation values does not "
213  "match the size of the symbol list used during construction");
214  }
215  piranha_assert(values.size() == m_names.size());
216  for (decltype(values.size()) i = 0u; i < values.size(); ++i) {
217  auto ptr = m_ptrs[static_cast<decltype(m_ptrs.size())>(i)];
218  piranha_assert(ptr == std::addressof(m_eval_dict.find(
219  m_names[static_cast<decltype(m_names.size())>(i)])->second));
220  *ptr = values[i];
221  }
222  return math::evaluate(m_x,m_eval_dict);
223  }
