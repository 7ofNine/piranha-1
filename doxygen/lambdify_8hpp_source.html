88 template <typename T, typename U>
89 class lambdified
90 {
91  static_assert(std::is_same<T,typename std::decay<T>::type>::value,"Invalid type.");
92  static_assert(std::is_same<U,typename std::decay<U>::type>::value,"Invalid type.");
93  static_assert(detail::math_lambdified_reqs<T,U>::value,"Invalid types.");
94  // Constructor implementation.
95  void construct()
96  {
97  // Check if there are duplicates.
98  auto names_copy(m_names);
99  std::sort(names_copy.begin(),names_copy.end());
100  if (std::unique(names_copy.begin(),names_copy.end()) != names_copy.end()) {
101  piranha_throw(std::invalid_argument,"the list of evaluation symbols contains duplicates");
102  }
103  // Fill in the eval dict and the vector of pointers, and make sure
104  // that m_extra_map does not contain anything that is already in m_names.
105  for (const auto &s: m_names) {
106  if (m_extra_map.find(s) != m_extra_map.end()) {
107  piranha_throw(std::invalid_argument,"the extra symbols map contains symbol '"
108  + s + "', which is already in the symbol list used for the construction "
109  "of the lambdified object");
110  }
111  const auto ret = m_eval_dict.emplace(std::make_pair(s,U{}));
112  piranha_assert(ret.second);
113  m_ptrs.push_back(std::addressof(ret.first->second));
114  }
115  // Fill in the extra symbols.
116  for (const auto &p: m_extra_map) {
117  const auto ret = m_eval_dict.emplace(std::make_pair(p.first,U{}));
118  piranha_assert(ret.second);
119  m_ptrs.push_back(std::addressof(ret.first->second));
120  }
121  }
122  // Reconstruct the vector of pointers following copy or move construction.
123  void reconstruct_ptrs()
124  {
125  // m_ptrs must be empty because we assume this method is called only as part of
126  // the copy/move ctors.
127  piranha_assert(m_ptrs.empty());
128  piranha_assert(m_eval_dict.size() >= m_extra_map.size() &&
129  m_names.size() == m_eval_dict.size() - m_extra_map.size());
130  // Take care first of the positional argument.
131  for (const auto &s: m_names) {
132  piranha_assert(m_eval_dict.count(s) == 1u);
133  m_ptrs.push_back(std::addressof(m_eval_dict.find(s)->second));
134  }
135  // The extra symbols.
136  for (const auto &p: m_extra_map) {
137  piranha_assert(m_eval_dict.count(p.first) == 1u);
138  m_ptrs.push_back(std::addressof(m_eval_dict.find(p.first)->second));
139  }
140  // Make sure the sizes are consistent.
141  piranha_assert(m_ptrs.size() == static_cast<decltype(m_ptrs.size())>(m_names.size()) +
142  m_extra_map.size());
143  }

267  eval_type operator()(const std::vector<U> &values)
268  {
269  if (unlikely(values.size() != m_eval_dict.size() - m_extra_map.size())) {
270  piranha_throw(std::invalid_argument,"the size of the vector of evaluation values does not "
271  "match the size of the symbol list used during construction");
272  }
273  piranha_assert(values.size() == m_names.size());
274  decltype(values.size()) i = 0u;
275  for (; i < values.size(); ++i) {
276  auto ptr = m_ptrs[static_cast<decltype(m_ptrs.size())>(i)];
277  piranha_assert(ptr == std::addressof(m_eval_dict.find(
278  m_names[static_cast<decltype(m_names.size())>(i)])->second));
279  *ptr = values[i];
280  }
281  // NOTE: here it is crucial that the iteration order on m_extra_map is the same
282  // as it was during the construction of the m_ptrs vector, otherwise we are associating
283  // values to the wrong symbols. Luckily, it seems like iterating on a const unordered_map
284  // multiple times yields the same order:
285  // http://stackoverflow.com/questions/18301302/is-forauto-i-unordered-map-guaranteed-to-have-the-same-order-every-time
286  // https://groups.google.com/a/isocpp.org/forum/#!topic/std-discussion/kHYFUhsauhU
287  for (const auto &p: m_extra_map) {
288  piranha_assert(i < m_ptrs.size());
289  auto ptr = m_ptrs[static_cast<decltype(m_ptrs.size())>(i)];
290  piranha_assert(ptr == std::addressof(m_eval_dict.find(p.first)->second));
291  *ptr = p.second(values);
292  ++i;
293  }
294  return math::evaluate(m_x,m_eval_dict);
295  }
