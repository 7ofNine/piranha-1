230  template <typename T>
231  typename std::enable_if<std::is_floating_point<T>::value,T>::type convert_to_impl() const
232  {
233  if (is_nan()) {
234  if (std::numeric_limits<T>::has_quiet_NaN) {
235  return std::numeric_limits<T>::quiet_NaN();
236  } else {
237  piranha_throw(std::overflow_error,"cannot convert NaN to floating-point type");
238  }
239  }
240  if (is_inf()) {
241  piranha_assert(sign() != 0);
242  if (std::numeric_limits<T>::has_infinity && sign() > 0) {
243  return std::numeric_limits<T>::infinity();
244  } else if (std::numeric_limits<T>::has_infinity && sign() < 0) {
245  return std::copysign(std::numeric_limits<T>::infinity(),std::numeric_limits<T>::lowest());
246  } else {
247  piranha_throw(std::overflow_error,"cannot convert infinity to floating-point type");
248  }
249  }
250  if (std::is_same<T,long double>::value) {
251  return static_cast<T>(::mpfr_get_ld(m_value,default_rnd));
252  }
253  if (std::is_same<T,double>::value) {
254  return static_cast<T>(::mpfr_get_d(m_value,default_rnd));
255  }
256  return static_cast<T>(::mpfr_get_flt(m_value,default_rnd));
257  }

260  template <typename T>
261  typename std::enable_if<detail::is_mp_rational<T>::value,T>::type convert_to_impl() const
262  {
263  if (is_nan()) {
264  piranha_throw(std::overflow_error,"cannot convert NaN to rational");
265  }
266  if (is_inf()) {
267  piranha_throw(std::overflow_error,"cannot convert infinity to rational");
268  }
269  if (sign() == 0) {
270  return T{};
271  }
272  // Get string representation.
273  ::mpfr_exp_t exp(0);
274  char *cptr = ::mpfr_get_str(nullptr,&exp,10,0,m_value,default_rnd);
275  if (unlikely(!cptr)) {
276  piranha_throw(std::overflow_error,"error in conversion of real to rational: the call to the MPFR function failed");
277  }
278  smart_mpfr_str str_ptr(cptr,::mpfr_free_str);
279  // Transform into fraction.
280  std::size_t digits = 0u;
281  for (; *cptr != '\0'; ++cptr) {
282  if (detail::is_digit(*cptr)) {
283  ++digits;
284  }
285  }
286  if (!digits) {
287  piranha_throw(std::overflow_error,"error in conversion of real to rational: invalid number of digits");
288  }
289  // NOTE: here the only exception that can be thrown is when raising to a power
290  // that cannot be represented by unsigned long.
291  try {
292  T retval(str_ptr.get());
293  // NOTE: possible optimizations here include going through direct GMP routines.
294  retval *= T(1,10).pow(digits);
295  retval *= T(10).pow(exp);
296  return retval;
297  } catch (...) {
298  piranha_throw(std::overflow_error,"error in conversion of real to rational: exponent is too large");
299  }
300  }

1954 // Compute gamma(a)/(gamma(b) * gamma(c)), assuming a, b and c are not negative ints. The logarithm
1955 // of the gamma function is used internally.
1956 inline real real_compute_3_gamma(const real &a, const real &b, const real &c, const ::mpfr_prec_t &prec)
1957 {
1958  // Here we should never enter with negative ints.
1959  piranha_assert(a.sign() >= 0 || a.truncated() != a);
1960  piranha_assert(b.sign() >= 0 || b.truncated() != b);
1961  piranha_assert(c.sign() >= 0 || c.truncated() != c);
1962  const real pi = real{0,prec}.pi();
1963  real tmp0(0), tmp1(1);
1964  if (a.sign() < 0) {
1965  tmp0 -= (1 - a).lgamma();
1966  tmp1 *= pi / (a * pi).sin();
1967  } else {
1968  tmp0 += a.lgamma();
1969  }
1970  if (b.sign() < 0) {
1971  tmp0 += (1 - b).lgamma();
1972  tmp1 *= (b * pi).sin() / pi;
1973  } else {
1974  tmp0 -= b.lgamma();
1975  }
1976  if (c.sign() < 0) {
1977  tmp0 += (1 - c).lgamma();
1978  tmp1 *= (c * pi).sin() / pi;
1979  } else {
1980  tmp0 -= c.lgamma();
1981  }
1982  return tmp0.exp() * tmp1;
1983 }

2002 inline real real::binomial(const real &y) const
2003 {
2004  if (unlikely(is_nan() || is_inf() || y.is_nan() || y.is_inf())) {
2005  piranha_throw(std::invalid_argument,"cannot compute binomial coefficient with non-finite real argument(s)");
2006  }
2007  // Work with the max precision.
2008  const ::mpfr_prec_t max_prec = std::max< ::mpfr_prec_t>(get_prec(),y.get_prec());
2009  const bool neg_int_x = truncated() == (*this) && sign() < 0,
2010  neg_int_y = y.truncated() == y && y.sign() < 0,
2011  neg_int_x_y = ((*this) - y).truncated() == ((*this) - y) && ((*this) - y).sign() < 0;
2012  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
2013  switch (mask) {
2014  case 0u:
2015  // Case 0 is the non-special one, use the default implementation.
2016  return detail::real_compute_3_gamma((*this) + 1,y + 1,(*this) - y + 1,max_prec);
2017  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
2018  case 2u:
2019  case 4u:
2020  // These are finite numerators with infinite denominators.
2021  return real{0,max_prec};
2022  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
2023  case 3u:
2024  {
2025  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
2026  // formula to make them finite.
2027  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
2028  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
2029  // here.
2030  const auto phase = math::pow(-1,(*this) + 1) / math::pow(-1,y + 1);
2031  return detail::real_compute_3_gamma(-y,-(*this),(*this) - y + 1,max_prec) * phase;
2032  }
2033  case 5u:
2034  {
2035  const auto phase = math::pow(-1,(*this) - y + 1) / math::pow(-1,(*this) + 1);
2036  return detail::real_compute_3_gamma(-((*this) - y),y + 1,-(*this),max_prec) * phase;
2037  }
2038  }
2039  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
2040  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
2041  return real{0,max_prec};
2042 }
