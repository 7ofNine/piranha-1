231  template <typename T>
232  typename std::enable_if<std::is_floating_point<T>::value,T>::type convert_to_impl() const
233  {
234  if (is_nan()) {
235  if (std::numeric_limits<T>::has_quiet_NaN) {
236  return std::numeric_limits<T>::quiet_NaN();
237  } else {
238  piranha_throw(std::overflow_error,"cannot convert NaN to floating-point type");
239  }
240  }
241  if (is_inf()) {
242  piranha_assert(sign() != 0);
243  if (std::numeric_limits<T>::has_infinity && sign() > 0) {
244  return std::numeric_limits<T>::infinity();
245  } else if (std::numeric_limits<T>::has_infinity && sign() < 0) {
246  return std::copysign(std::numeric_limits<T>::infinity(),std::numeric_limits<T>::lowest());
247  } else {
248  piranha_throw(std::overflow_error,"cannot convert infinity to floating-point type");
249  }
250  }
251  if (std::is_same<T,long double>::value) {
252  return static_cast<T>(::mpfr_get_ld(m_value,default_rnd));
253  }
254  if (std::is_same<T,double>::value) {
255  return static_cast<T>(::mpfr_get_d(m_value,default_rnd));
256  }
257  return static_cast<T>(::mpfr_get_flt(m_value,default_rnd));
258  }

261  template <typename T>
262  typename std::enable_if<detail::is_mp_rational<T>::value,T>::type convert_to_impl() const
263  {
264  if (is_nan()) {
265  piranha_throw(std::overflow_error,"cannot convert NaN to rational");
266  }
267  if (is_inf()) {
268  piranha_throw(std::overflow_error,"cannot convert infinity to rational");
269  }
270  if (sign() == 0) {
271  return T{};
272  }
273  // Get string representation.
274  ::mpfr_exp_t exp(0);
275  char *cptr = ::mpfr_get_str(nullptr,&exp,10,0,m_value,default_rnd);
276  if (unlikely(!cptr)) {
277  piranha_throw(std::overflow_error,"error in conversion of real to rational: the call to the MPFR function failed");
278  }
279  smart_mpfr_str str_ptr(cptr,::mpfr_free_str);
280  // Transform into fraction.
281  std::size_t digits = 0u;
282  for (; *cptr != '\0'; ++cptr) {
283  if (detail::is_digit(*cptr)) {
284  ++digits;
285  }
286  }
287  if (!digits) {
288  piranha_throw(std::overflow_error,"error in conversion of real to rational: invalid number of digits");
289  }
290  // NOTE: here the only exception that can be thrown is when raising to a power
291  // that cannot be represented by unsigned long.
292  try {
293  T retval(str_ptr.get());
294  // NOTE: possible optimizations here include going through direct GMP routines.
295  retval *= T(1,10).pow(digits);
296  retval *= T(10).pow(exp);
297  return retval;
298  } catch (...) {
299  piranha_throw(std::overflow_error,"error in conversion of real to rational: exponent is too large");
300  }
301  }

1956 // Compute gamma(a)/(gamma(b) * gamma(c)), assuming a, b and c are not negative ints. The logarithm
1957 // of the gamma function is used internally.
1958 inline real real_compute_3_gamma(const real &a, const real &b, const real &c, const ::mpfr_prec_t &prec)
1959 {
1960  // Here we should never enter with negative ints.
1961  piranha_assert(a.sign() >= 0 || a.truncated() != a);
1962  piranha_assert(b.sign() >= 0 || b.truncated() != b);
1963  piranha_assert(c.sign() >= 0 || c.truncated() != c);
1964  const real pi = real{0,prec}.pi();
1965  real tmp0(0), tmp1(1);
1966  if (a.sign() < 0) {
1967  tmp0 -= (1 - a).lgamma();
1968  tmp1 *= pi / (a * pi).sin();
1969  } else {
1970  tmp0 += a.lgamma();
1971  }
1972  if (b.sign() < 0) {
1973  tmp0 += (1 - b).lgamma();
1974  tmp1 *= (b * pi).sin() / pi;
1975  } else {
1976  tmp0 -= b.lgamma();
1977  }
1978  if (c.sign() < 0) {
1979  tmp0 += (1 - c).lgamma();
1980  tmp1 *= (c * pi).sin() / pi;
1981  } else {
1982  tmp0 -= c.lgamma();
1983  }
1984  return tmp0.exp() * tmp1;
1985 }

2004 inline real real::binomial(const real &y) const
2005 {
2006  if (unlikely(is_nan() || is_inf() || y.is_nan() || y.is_inf())) {
2007  piranha_throw(std::invalid_argument,"cannot compute binomial coefficient with non-finite real argument(s)");
2008  }
2009  // Work with the max precision.
2010  const ::mpfr_prec_t max_prec = std::max< ::mpfr_prec_t>(get_prec(),y.get_prec());
2011  const bool neg_int_x = truncated() == (*this) && sign() < 0,
2012  neg_int_y = y.truncated() == y && y.sign() < 0,
2013  neg_int_x_y = ((*this) - y).truncated() == ((*this) - y) && ((*this) - y).sign() < 0;
2014  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
2015  switch (mask) {
2016  case 0u:
2017  // Case 0 is the non-special one, use the default implementation.
2018  return detail::real_compute_3_gamma((*this) + 1,y + 1,(*this) - y + 1,max_prec);
2019  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
2020  case 2u:
2021  case 4u:
2022  // These are finite numerators with infinite denominators.
2023  return real{0,max_prec};
2024  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
2025  case 3u:
2026  {
2027  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
2028  // formula to make them finite.
2029  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
2030  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
2031  // here.
2032  const auto phase = math::pow(-1,(*this) + 1) / math::pow(-1,y + 1);
2033  return detail::real_compute_3_gamma(-y,-(*this),(*this) - y + 1,max_prec) * phase;
2034  }
2035  case 5u:
2036  {
2037  const auto phase = math::pow(-1,(*this) - y + 1) / math::pow(-1,(*this) + 1);
2038  return detail::real_compute_3_gamma(-((*this) - y),y + 1,-(*this),max_prec) * phase;
2039  }
2040  }
2041  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
2042  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
2043  return real{0,max_prec};
2044 }
