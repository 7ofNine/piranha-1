235  template <typename T>
236  typename std::enable_if<std::is_floating_point<T>::value, T>::type convert_to_impl() const
237  {
238  if (is_nan()) {
239  if (std::numeric_limits<T>::has_quiet_NaN) {
240  return std::numeric_limits<T>::quiet_NaN();
241  } else {
242  piranha_throw(std::overflow_error, "cannot convert NaN to floating-point type");
243  }
244  }
245  if (is_inf()) {
246  piranha_assert(sign() != 0);
247  if (std::numeric_limits<T>::has_infinity && sign() > 0) {
248  return std::numeric_limits<T>::infinity();
249  } else if (std::numeric_limits<T>::has_infinity && sign() < 0) {
250  return std::copysign(std::numeric_limits<T>::infinity(), std::numeric_limits<T>::lowest());
251  } else {
252  piranha_throw(std::overflow_error, "cannot convert infinity to floating-point type");
253  }
254  }
255  if (std::is_same<T, long double>::value) {
256  return static_cast<T>(::mpfr_get_ld(m_value, default_rnd));
257  }
258  if (std::is_same<T, double>::value) {
259  return static_cast<T>(::mpfr_get_d(m_value, default_rnd));
260  }
261  return static_cast<T>(::mpfr_get_flt(m_value, default_rnd));
262  }

265  template <typename T>
266  typename std::enable_if<detail::is_mp_rational<T>::value, T>::type convert_to_impl() const
267  {
268  if (is_nan()) {
269  piranha_throw(std::overflow_error, "cannot convert NaN to rational");
270  }
271  if (is_inf()) {
272  piranha_throw(std::overflow_error, "cannot convert infinity to rational");
273  }
274  if (sign() == 0) {
275  return T{};
276  }
277  // Get string representation.
278  ::mpfr_exp_t exp(0);
279  char *cptr = ::mpfr_get_str(nullptr, &exp, 10, 0, m_value, default_rnd);
280  if (unlikely(!cptr)) {
281  piranha_throw(std::overflow_error,
282  "error in conversion of real to rational: the call to the MPFR function failed");
283  }
284  smart_mpfr_str str_ptr(cptr, ::mpfr_free_str);
285  // Transform into fraction.
286  std::size_t digits = 0u;
287  for (; *cptr != '\0'; ++cptr) {
288  if (detail::is_digit(*cptr)) {
289  ++digits;
290  }
291  }
292  if (!digits) {
293  piranha_throw(std::overflow_error, "error in conversion of real to rational: invalid number of digits");
294  }
295  // NOTE: here the only exception that can be thrown is when raising to a power
296  // that cannot be represented by unsigned long.
297  try {
298  T retval(str_ptr.get());
299  // NOTE: possible optimizations here include going through direct GMP routines.
300  retval *= T(1, 10).pow(digits);
301  retval *= T(10).pow(exp);
302  return retval;
303  } catch (...) {
304  piranha_throw(std::overflow_error, "error in conversion of real to rational: exponent is too large");
305  }
306  }

2210 inline real real::binomial(const real &y) const
2211 {
2212  if (unlikely(is_nan() || is_inf() || y.is_nan() || y.is_inf())) {
2213  piranha_throw(std::invalid_argument, "cannot compute binomial coefficient with non-finite real argument(s)");
2214  }
2215  // Work with the max precision.
2216  const ::mpfr_prec_t max_prec = std::max<::mpfr_prec_t>(get_prec(), y.get_prec());
2217  const bool neg_int_x = truncated() == (*this) && sign() < 0, neg_int_y = y.truncated() == y && y.sign() < 0,
2218  neg_int_x_y = ((*this) - y).truncated() == ((*this) - y) && ((*this) - y).sign() < 0;
2219  const unsigned mask = unsigned(neg_int_x) + (unsigned(neg_int_y) << 1u) + (unsigned(neg_int_x_y) << 2u);
2220  switch (mask) {
2221  case 0u:
2222  // Case 0 is the non-special one, use the default implementation.
2223  return detail::real_compute_3_gamma((*this) + 1, y + 1, (*this) - y + 1, max_prec);
2224  // NOTE: case 1 is not possible: x < 0, y > 0 implies x - y < 0 always.
2225  case 2u:
2226  case 4u:
2227  // These are finite numerators with infinite denominators.
2228  return real{0, max_prec};
2229  // NOTE: case 6 is not possible: x > 0, y < 0 implies x - y > 0 always.
2230  case 3u: {
2231  // 3 and 5 are the cases with 1 inf in num and 1 inf in den. Use the transformation
2232  // formula to make them finite.
2233  // NOTE: the phase here is really just a sign, but it seems tricky to compute this exactly
2234  // due to potential rounding errors. We are attempting to err on the safe side by using pow()
2235  // here.
2236  const auto phase = math::pow(-1, (*this) + 1) / math::pow(-1, y + 1);
2237  return detail::real_compute_3_gamma(-y, -(*this), (*this) - y + 1, max_prec) * phase;
2238  }
2239  case 5u: {
2240  const auto phase = math::pow(-1, (*this) - y + 1) / math::pow(-1, (*this) + 1);
2241  return detail::real_compute_3_gamma(-((*this) - y), y + 1, -(*this), max_prec) * phase;
2242  }
2243  }
2244  // Case 7 returns zero -> from inf / (inf * inf) it becomes a / (b * inf) after the transform.
2245  // NOTE: put it here so the compiler does not complain about missing return statement in the switch block.
2246  return real{0, max_prec};
