266  // Canonical term: the first nonzero element is positive and all the gcd of all elements is 1 or -1.
267  // NOTE: this also includes the check for all zero elements, as gcd(0,0,...,0) = 0.
268  static bool term_is_canonical(const p_type &p)
269  {
270  bool first_nonzero_found = false;
271  value_type cd(0);
272  for (const auto &n : p.v) {
273  if (!first_nonzero_found && !math::is_zero(n)) {
274  if (n < 0) {
275  return false;
276  }
277  first_nonzero_found = true;
278  }
279  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
280  math::gcd3(cd, cd, n);
281  }
282  if (cd != 1 && cd != -1) {
283  return false;
284  }
285  return true;
286  }
287  // Range check on a term - meaningful only if T is a C++ integral type.
288  template <typename U = T, typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
289  static bool term_range_check(const p_type &p)
290  {
291  return std::all_of(p.v.begin(), p.v.end(), [](const value_type &x) {
292  return x >= -detail::safe_abs_sint<value_type>::value && x <= detail::safe_abs_sint<value_type>::value;
293  });
294  }
295  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value, int>::type = 0>
296  static bool term_range_check(const p_type &)
297  {
298  return true;
299  }
300  bool destruction_checks() const
301  {
302  const auto it_f = m_container.end();
303  auto it = m_container.begin();
304  const typename v_type::size_type v_size = (it == it_f) ? 0u : it->v.size();
305  for (; it != it_f; ++it) {
306  // Check: the exponent must be greater than zero.
307  if (it->e <= 0) {
308  return false;
309  }
310  // Check: range.
311  if (!term_range_check(*it)) {
312  return false;
313  }
314  // Check: canonical.
315  if (!term_is_canonical(*it)) {
316  return false;
317  }
318  // Check: all vectors have the same size.
319  if (it->v.size() != v_size) {
320  return false;
321  }
322  }
323  return true;
324  }
325  // Insertion machinery.
326  template <typename Term>
327  void insertion_impl(Term &&term)
328  {
329  // Handle the case of a table with no buckets.
330  if (unlikely(!m_container.bucket_count())) {
331  m_container._increase_size();
332  }
333  // Try to locate the term.
334  auto bucket_idx = m_container._bucket(term);
335  const auto it = m_container._find(term, bucket_idx);
336  if (it == m_container.end()) {
337  // New term.
338  if (unlikely(m_container.size() == std::numeric_limits<size_type>::max())) {
339  piranha_throw(std::overflow_error, "maximum number of elements reached");
340  }
341  // Term is new. Handle the case in which we need to rehash because of load factor.
342  if (unlikely(static_cast<double>(m_container.size() + size_type(1u))
343  / static_cast<double>(m_container.bucket_count())
344  > m_container.max_load_factor())) {
345  m_container._increase_size();
346  // We need a new bucket index in case of a rehash.
347  bucket_idx = m_container._bucket(term);
348  }
349  // Actually perform the insertion and finish by updating the size.
350  m_container._unique_insert(std::forward<Term>(term), bucket_idx);
351  m_container._update_size(m_container.size() + size_type(1u));
352  } else {
353  // Existing term - update the exponent.
354  update_exponent(it->e, term.e);
355  }
356  }
357  template <typename U = T, typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
358  static void update_exponent(value_type &a, const value_type &b)
359  {
360  // Check the range when the exponent is an integral type.
361  piranha_assert(a > 0);
362  piranha_assert(b > 0);
363  // NOTE: this is safe as we require b to be a positive value.
364  if (unlikely(a > std::numeric_limits<value_type>::max() - b)) {
365  piranha_throw(std::invalid_argument, "overflow in the computation of the exponent of a divisor term");
366  }
367  a = static_cast<value_type>(a + b);
368  }
369  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value, int>::type = 0>
370  static void update_exponent(value_type &a, const value_type &b)
371  {
372  // For integer exponents, just do normal addition.
373  a += b;
374  }

483  template <typename It, typename Exponent, insert_enabler<It, Exponent> = 0>
484  void insert(It begin, It end, const Exponent &e)
485  {
486  p_type term;
487  // Assign the exponent.
488  term.e = safe_cast<value_type>(e);
489  if (unlikely(term.e <= 0)) {
490  piranha_throw(std::invalid_argument, "a term of a divisor must have a positive exponent");
491  }
492  // Build the vector to be inserted.
493  for (; begin != end; ++begin) {
494  term.v.push_back(safe_cast<value_type>(*begin));
495  }
496  // Range check.
497  if (unlikely(!term_range_check(term))) {
498  piranha_throw(std::invalid_argument, "an element in a term of a divisor is outside the allowed range");
499  }
500  // Check that the term is canonical.
501  if (unlikely(!term_is_canonical(term))) {
502  piranha_throw(std::invalid_argument, "term not in canonical form");
503  }
504  // Perform the insertion.
505  insertion_impl(std::move(term));
506  }

565  std::size_t hash() const
566  {
567  // NOTE: here we are using a simple sum to compute the hash of a vector of vectors. Indeed,
568  // we need some add/multiply like operation in order to make sure that two divisors
569  // with the same elements stored in different order hash to the same thing. This seems
570  // to be ok, according to this,
571  // http://stackoverflow.com/questions/18021643/hashing-a-set-of-integers-in-an-order-independent-way,
572  // as the hash of each subvector should be rather good (it comes from boost hash combine).
573  // XOR is also commutative and might be used (see link above).
574  // If problem arises, we can consider muliplying the hash of each subvector by a prime
575  // (similar to Kronecker substitution).
576  std::size_t retval = 0u;
577  p_type_hasher hasher;
578  const auto it_f = m_container.end();
579  for (auto it = m_container.begin(); it != it_f; ++it) {
580  retval = static_cast<std::size_t>(retval + hasher(*it));
581  }
582  return retval;
583  }

676  void print(std::ostream &os, const symbol_set &args) const
677  {
678  // Don't print anything if there are no terms.
679  if (m_container.empty()) {
680  return;
681  }
682  if (unlikely(m_container.begin()->v.size() != args.size())) {
683  piranha_throw(std::invalid_argument, "invalid size of arguments set");
684  }
685  const auto it_f = m_container.end();
686  bool first_term = true;
687  os << "1/[";
688  for (auto it = m_container.begin(); it != it_f; ++it) {
689  // If this is not the first term, print a leading '*' operator.
690  if (first_term) {
691  first_term = false;
692  } else {
693  os << '*';
694  }
695  bool printed_something = false;
696  os << '(';
697  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i) {
698  // If the aij is zero, don't print anything.
699  if (math::is_zero(it->v[i])) {
700  continue;
701  }
702  // A positive aij, in case previous output exists, must be preceded
703  // by a "+" sign.
704  if (it->v[i] > 0 && printed_something) {
705  os << '+';
706  }
707  // Print the aij, unless it's "-1": in that case, just print the minus sign.
708  if (it->v[i] == -1) {
709  os << '-';
710  } else if (it->v[i] != 1) {
711  os << detail::prepare_for_print(it->v[i]) << '*';
712  }
713  // Finally, print name of variable.
714  os << args[i].get_name();
715  printed_something = true;
716  }
717  os << ')';
718  // Print the exponent, if different from one.
719  if (it->e != 1) {
720  os << "**" << detail::prepare_for_print(it->e);
721  }
722  }
723  os << ']';
724  }

800  template <typename U>
801  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
802  {
803  // Just return 1 if the container is empty.
804  eval_type<U> retval(1);
805  if (m_container.empty()) {
806  return retval;
807  }
808  // NOTE: the positions map must have the same number of elements as this, and it must
809  // be made of consecutive positions [0,size - 1].
810  if (unlikely(pmap.size() != m_container.begin()->v.size()
811  || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
812  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
813  }
814  // Args must be compatible with this.
815  if (unlikely(args.size() != m_container.begin()->v.size())) {
816  piranha_throw(std::invalid_argument, "invalid size of arguments set");
817  }
818  const auto it_f = m_container.end();

864  {

873  const divisor &large = (t1.m_key.size() >= t2.m_key.size()) ? t1.m_key : t2.m_key;
