156  // Canonical term: the first nonzero element is positive and all the gcd of all elements is 1 or -1.
157  // NOTE: this also includes the check for all zero elements, as gcd(0,0,...,0) = 0.
158  static bool term_is_canonical(const p_type &p)
159  {
160  bool first_nonzero_found = false;
161  value_type cd(0);
162  for (const auto &n : p.v) {
163  if (!first_nonzero_found && !math::is_zero(n)) {
164  if (n < 0) {
165  return false;
166  }
167  first_nonzero_found = true;
168  }
169  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
170  math::gcd3(cd, cd, n);
171  }
172  if (cd != 1 && cd != -1) {
173  return false;
174  }
175  return true;
176  }

190  bool destruction_checks() const
191  {
192  const auto it_f = m_container.end();
193  auto it = m_container.begin();
194  const typename v_type::size_type v_size = (it == it_f) ? 0u : it->v.size();
195  for (; it != it_f; ++it) {
196  // Check: the exponent must be greater than zero.
197  if (it->e <= 0) {
198  return false;
199  }
200  // Check: range.
201  if (!term_range_check(*it)) {
202  return false;
203  }
204  // Check: canonical.
205  if (!term_is_canonical(*it)) {
206  return false;
207  }
208  // Check: all vectors have the same size.
209  if (it->v.size() != v_size) {
210  return false;
211  }
212  }
213  return true;

215  // Insertion machinery.
216  template <typename Term>
217  void insertion_impl(Term &&term)
218  {
219  // Handle the case of a table with no buckets.
220  if (unlikely(!m_container.bucket_count())) {
221  m_container._increase_size();
222  }
223  // Try to locate the term.
224  auto bucket_idx = m_container._bucket(term);
225  const auto it = m_container._find(term, bucket_idx);
226  if (it == m_container.end()) {
227  // New term.
228  if (unlikely(m_container.size() == std::numeric_limits<size_type>::max())) {
229  piranha_throw(std::overflow_error, "maximum number of elements reached");
230  }
231  // Term is new. Handle the case in which we need to rehash because of load factor.
232  if (unlikely(static_cast<double>(m_container.size() + size_type(1u))
233  / static_cast<double>(m_container.bucket_count())
234  > m_container.max_load_factor())) {
235  m_container._increase_size();
236  // We need a new bucket index in case of a rehash.
237  bucket_idx = m_container._bucket(term);
238  }
239  // Actually perform the insertion and finish by updating the size.
240  m_container._unique_insert(std::forward<Term>(term), bucket_idx);
241  m_container._update_size(m_container.size() + size_type(1u));
242  } else {
243  // Existing term - update the exponent.
244  update_exponent(it->e, term.e);
245  }
246  }
247  template <typename U = T, typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
248  static void update_exponent(value_type &a, const value_type &b)
249  {
250  // Check the range when the exponent is an integral type.
251  piranha_assert(a > 0);
252  piranha_assert(b > 0);
253  // NOTE: this is safe as we require b to be a positive value.
254  if (unlikely(a > std::numeric_limits<value_type>::max() - b)) {
255  piranha_throw(std::invalid_argument, "overflow in the computation of the exponent of a divisor term");
256  }
257  a = static_cast<value_type>(a + b);
258  }
259  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value, int>::type = 0>
260  static void update_exponent(value_type &a, const value_type &b)
261  {
262  // For integer exponents, just do normal addition.
263  a += b;
264  }

279  template <class Archive>
280  void load(Archive &ar, unsigned int)
281  {
282  divisor new_d;
283  try {
284  // NOTE: here we could throw either because the archive is garbage (Boost.Serialization throwing
285  // in this case) or because the loaded terms are invalid. In the second case we must make sure
286  // to destroy the content of new_d before exiting, otherwise the dtor of the divisor will
287  // hit assertion failures in debug mode. We need the second clear() in the catch block
288  // because we could be in the situation that the archive contains invaild terms and it is garbage
289  // after a while - in thie case invalid terms coud be living in new_d and the first clear() is never
290  // hit.
291  ar &new_d.m_container;
292  // Run the destruction checks, if they fail throw.
293  if (unlikely(!new_d.destruction_checks())) {
294  new_d.clear();
295  piranha_throw(std::invalid_argument, "error during the deserialization of a divisor, the loaded data "
296  "is invalid");
297  }
298  } catch (...) {
299  new_d.clear();
300  throw;
301  }
302  // Move in.
303  *this = std::move(new_d);
304  }

407  template <typename It, typename Exponent, insert_enabler<It, Exponent> = 0>
408  void insert(It begin, It end, const Exponent &e)
409  {
410  p_type term;
411  // Assign the exponent.
412  term.e = safe_cast<value_type>(e);
413  if (unlikely(term.e <= 0)) {
414  piranha_throw(std::invalid_argument, "a term of a divisor must have a positive exponent");
415  }
416  // Build the vector to be inserted.
417  for (; begin != end; ++begin) {
418  term.v.push_back(safe_cast<value_type>(*begin));
419  }
420  // Range check.
421  if (unlikely(!term_range_check(term))) {
422  piranha_throw(std::invalid_argument, "an element in a term of a divisor is outside the allowed range");
423  }
424  // Check that the term is canonical.
425  if (unlikely(!term_is_canonical(term))) {
426  piranha_throw(std::invalid_argument, "term not in canonical form");
427  }
428  // Perform the insertion.
429  insertion_impl(std::move(term));
430  }

489  std::size_t hash() const
490  {
491  // NOTE: here we are using a simple sum to compute the hash of a vector of vectors. Indeed,
492  // we need some add/multiply like operation in order to make sure that two divisors
493  // with the same elements stored in different order hash to the same thing. This seems
494  // to be ok, according to this,
495  // http://stackoverflow.com/questions/18021643/hashing-a-set-of-integers-in-an-order-independent-way,
496  // as the hash of each subvector should be rather good (it comes from boost hash combine).
497  // XOR is also commutative and might be used (see link above).
498  // If problem arises, we can consider muliplying the hash of each subvector by a prime
499  // (similar to Kronecker substitution).
500  std::size_t retval = 0u;
501  p_type_hasher hasher;
502  const auto it_f = m_container.end();
503  for (auto it = m_container.begin(); it != it_f; ++it) {
504  retval = static_cast<std::size_t>(retval + hasher(*it));
505  }
506  return retval;
507  }

602  void print(std::ostream &os, const symbol_set &args) const
603  {
604  // Don't print anything if there are no terms.
605  if (m_container.empty()) {
606  return;
607  }
608  if (unlikely(m_container.begin()->v.size() != args.size())) {
609  piranha_throw(std::invalid_argument, "invalid size of arguments set");
610  }
611  const auto it_f = m_container.end();
612  bool first_term = true;
613  os << "1/[";
614  for (auto it = m_container.begin(); it != it_f; ++it) {
615  // If this is not the first term, print a leading '*' operator.
616  if (first_term) {
617  first_term = false;
618  } else {
619  os << '*';
620  }
621  bool printed_something = false;
622  os << '(';
623  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i) {
624  // If the aij is zero, don't print anything.
625  if (math::is_zero(it->v[i])) {
626  continue;
627  }
628  // A positive aij, in case previous output exists, must be preceded
629  // by a "+" sign.
630  if (it->v[i] > 0 && printed_something) {
631  os << '+';
632  }
633  // Print the aij, unless it's "-1": in that case, just print the minus sign.
634  if (it->v[i] == -1) {
635  os << '-';
636  } else if (it->v[i] != 1) {
637  os << detail::prepare_for_print(it->v[i]) << '*';
638  }
639  // Finally, print name of variable.
640  os << args[i].get_name();
641  printed_something = true;
642  }
643  os << ')';
644  // Print the exponent, if different from one.
645  if (it->e != 1) {
646  os << "**" << detail::prepare_for_print(it->e);
647  }
648  }
649  os << ']';
650  }

725  template <typename U>
726  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
727  {
728  // Just return 1 if the container is empty.
729  eval_type<U> retval(1);
730  if (m_container.empty()) {
731  return retval;
732  }
733  // NOTE: the positions map must have the same number of elements as this, and it must
734  // be made of consecutive positions [0,size - 1].
735  if (unlikely(pmap.size() != m_container.begin()->v.size()
736  || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
737  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
738  }
739  // Args must be compatible with this.
740  if (unlikely(args.size() != m_container.begin()->v.size())) {
741  piranha_throw(std::invalid_argument, "invalid size of arguments set");
742  }
743  const auto it_f = m_container.end();
744  for (auto it = m_container.begin(); it != it_f; ++it) {
745  eval_sum_type<U> tmp(0);
746  auto map_it = pmap.begin();
747  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i, ++map_it) {
748  piranha_assert(map_it != pmap.end() && map_it->first == i);
749  // NOTE: might use multadd here eventually for performance.
750  tmp += it->v[i] * map_it->second;
751  }
752  piranha_assert(map_it == pmap.end());
753  // NOTE: consider rewriting this in terms of multiplications as a performance
754  // improvement - the eval_type deduction should be changed accordingly.
755  retval /= math::pow(tmp, it->e);
756  }
757  return retval;
758  }

786  template <typename Cf, multiply_enabler<Cf> = 0>
787  static void multiply(std::array<term<Cf, divisor>, multiply_arity> &res, const term<Cf, divisor> &t1,
788  const term<Cf, divisor> &t2, const symbol_set &args)
789  {
790  term<Cf, divisor> &t = res[0u];
791  if (unlikely(!t1.m_key.is_compatible(args) || !t2.m_key.is_compatible(args))) {
792  piranha_throw(std::invalid_argument, "invalid size of arguments set");
793  }
794  // Coefficient.
795  detail::cf_mult_impl(t.m_cf, t1.m_cf, t2.m_cf);
796  // Now deal with the key.
797  // Establish the largest and smallest divisor.
798  const divisor &large = (t1.m_key.size() >= t2.m_key.size()) ? t1.m_key : t2.m_key;
799  const divisor &small = (t1.m_key.size() < t2.m_key.size()) ? t1.m_key : t2.m_key;
800  // Assign the large to the result.
801  t.m_key = large;
802  // Run the multiplication.
803  const auto it_f = small.m_container.end();
804  for (auto it = small.m_container.begin(); it != it_f; ++it) {
805  t.m_key.insertion_impl(*it);
806  }
807  }
