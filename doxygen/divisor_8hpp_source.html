266  // Canonical term: the first nonzero element is positive and all the gcd of all elements is 1 or -1.
267  // NOTE: this also includes the check for all zero elements, as gcd(0,0,...,0) = 0.
268  static bool term_is_canonical(const p_type &p)
269  {
270  bool first_nonzero_found = false;
271  value_type cd(0);
272  for (const auto &n : p.v) {
273  if (!first_nonzero_found && !math::is_zero(n)) {
274  if (n < 0) {
275  return false;
276  }
277  first_nonzero_found = true;
278  }
279  // NOTE: gcd(0,n) == n (or +-n, in our case) for all n, zero included.
280  math::gcd3(cd, cd, n);
281  }
282  if (cd != 1 && cd != -1) {
283  return false;
284  }
285  return true;
286  }
287  // Range check on a term - meaningful only if T is a C++ integral type.
288  template <typename U = T, typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
289  static bool term_range_check(const p_type &p)
290  {
291  return std::all_of(p.v.begin(), p.v.end(), [](const value_type &x) {
292  return x >= -detail::safe_abs_sint<value_type>::value && x <= detail::safe_abs_sint<value_type>::value;
293  });
294  }
295  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value, int>::type = 0>
296  static bool term_range_check(const p_type &)
297  {
298  return true;

300  bool destruction_checks() const
301  {
302  const auto it_f = m_container.end();
303  auto it = m_container.begin();
304  const typename v_type::size_type v_size = (it == it_f) ? 0u : it->v.size();
305  for (; it != it_f; ++it) {
306  // Check: the exponent must be greater than zero.
307  if (it->e <= 0) {
308  return false;
309  }
310  // Check: range.
311  if (!term_range_check(*it)) {
312  return false;
313  }
314  // Check: canonical.
315  if (!term_is_canonical(*it)) {
316  return false;
317  }
318  // Check: all vectors have the same size.
319  if (it->v.size() != v_size) {
320  return false;
321  }
322  }
323  return true;
324  }
325  // Insertion machinery.
326  template <typename Term>
327  void insertion_impl(Term &&term)
328  {
329  // Handle the case of a table with no buckets.
330  if (unlikely(!m_container.bucket_count())) {
331  m_container._increase_size();
332  }
333  // Try to locate the term.
334  auto bucket_idx = m_container._bucket(term);
335  const auto it = m_container._find(term, bucket_idx);
336  if (it == m_container.end()) {
337  // New term.
338  if (unlikely(m_container.size() == std::numeric_limits<size_type>::max())) {
339  piranha_throw(std::overflow_error, "maximum number of elements reached");
340  }
341  // Term is new. Handle the case in which we need to rehash because of load factor.
342  if (unlikely(static_cast<double>(m_container.size() + size_type(1u))
343  / static_cast<double>(m_container.bucket_count())
344  > m_container.max_load_factor())) {
345  m_container._increase_size();
346  // We need a new bucket index in case of a rehash.
347  bucket_idx = m_container._bucket(term);
348  }
349  // Actually perform the insertion and finish by updating the size.
350  m_container._unique_insert(std::forward<Term>(term), bucket_idx);
351  m_container._update_size(m_container.size() + size_type(1u));
352  } else {
353  // Existing term - update the exponent.
354  update_exponent(it->e, term.e);
355  }
356  }
357  template <typename U = T, typename std::enable_if<std::is_integral<U>::value, int>::type = 0>
358  static void update_exponent(value_type &a, const value_type &b)
359  {
360  // Check the range when the exponent is an integral type.
361  piranha_assert(a > 0);
362  piranha_assert(b > 0);
363  // NOTE: this is safe as we require b to be a positive value.
364  if (unlikely(a > std::numeric_limits<value_type>::max() - b)) {
365  piranha_throw(std::invalid_argument, "overflow in the computation of the exponent of a divisor term");
366  }
367  a = static_cast<value_type>(a + b);
368  }
369  template <typename U = T, typename std::enable_if<!std::is_integral<U>::value, int>::type = 0>
370  static void update_exponent(value_type &a, const value_type &b)
371  {
372  // For integer exponents, just do normal addition.
373  a += b;
374  }

495  template <typename It, typename Exponent, insert_enabler<It, Exponent> = 0>
496  void insert(It begin, It end, const Exponent &e)
497  {
498  p_type term;
499  // Assign the exponent.
500  term.e = safe_cast<value_type>(e);
501  if (unlikely(term.e <= 0)) {
502  piranha_throw(std::invalid_argument, "a term of a divisor must have a positive exponent");
503  }
504  // Build the vector to be inserted.
505  for (; begin != end; ++begin) {
506  term.v.push_back(safe_cast<value_type>(*begin));
507  }
508  // Range check.
509  if (unlikely(!term_range_check(term))) {
510  piranha_throw(std::invalid_argument, "an element in a term of a divisor is outside the allowed range");
511  }
512  // Check that the term is canonical.
513  if (unlikely(!term_is_canonical(term))) {
514  piranha_throw(std::invalid_argument, "term not in canonical form");
515  }
516  // Perform the insertion.
517  insertion_impl(std::move(term));
518  }

577  std::size_t hash() const
578  {
579  // NOTE: here we are using a simple sum to compute the hash of a vector of vectors. Indeed,
580  // we need some add/multiply like operation in order to make sure that two divisors
581  // with the same elements stored in different order hash to the same thing. This seems
582  // to be ok, according to this,
583  // http://stackoverflow.com/questions/18021643/hashing-a-set-of-integers-in-an-order-independent-way,
584  // as the hash of each subvector should be rather good (it comes from boost hash combine).
585  // XOR is also commutative and might be used (see link above).
586  // If problem arises, we can consider muliplying the hash of each subvector by a prime
587  // (similar to Kronecker substitution).
588  std::size_t retval = 0u;
589  p_type_hasher hasher;
590  const auto it_f = m_container.end();
591  for (auto it = m_container.begin(); it != it_f; ++it) {
592  retval = static_cast<std::size_t>(retval + hasher(*it));
593  }
594  return retval;
595  }

688  void print(std::ostream &os, const symbol_set &args) const
689  {
690  // Don't print anything if there are no terms.
691  if (m_container.empty()) {
692  return;
693  }
694  if (unlikely(m_container.begin()->v.size() != args.size())) {
695  piranha_throw(std::invalid_argument, "invalid size of arguments set");
696  }
697  const auto it_f = m_container.end();
698  bool first_term = true;
699  os << "1/[";
700  for (auto it = m_container.begin(); it != it_f; ++it) {
701  // If this is not the first term, print a leading '*' operator.
702  if (first_term) {
703  first_term = false;
704  } else {
705  os << '*';
706  }
707  bool printed_something = false;
708  os << '(';
709  for (typename v_type::size_type i = 0u; i < it->v.size(); ++i) {
710  // If the aij is zero, don't print anything.
711  if (math::is_zero(it->v[i])) {
712  continue;
713  }
714  // A positive aij, in case previous output exists, must be preceded
715  // by a "+" sign.
716  if (it->v[i] > 0 && printed_something) {
717  os << '+';
718  }
719  // Print the aij, unless it's "-1": in that case, just print the minus sign.
720  if (it->v[i] == -1) {
721  os << '-';
722  } else if (it->v[i] != 1) {
723  os << detail::prepare_for_print(it->v[i]) << '*';
724  }
725  // Finally, print name of variable.
726  os << args[i].get_name();
727  printed_something = true;
728  }
729  os << ')';
730  // Print the exponent, if different from one.
731  if (it->e != 1) {
732  os << "**" << detail::prepare_for_print(it->e);
733  }
734  }
735  os << ']';
736  }

812  template <typename U>
813  eval_type<U> evaluate(const symbol_set::positions_map<U> &pmap, const symbol_set &args) const
814  {
815  // Just return 1 if the container is empty.
816  eval_type<U> retval(1);
817  if (m_container.empty()) {
818  return retval;
819  }
820  // NOTE: the positions map must have the same number of elements as this, and it must
821  // be made of consecutive positions [0,size - 1].
822  if (unlikely(pmap.size() != m_container.begin()->v.size()
823  || (pmap.size() && pmap.back().first != pmap.size() - 1u))) {
824  piranha_throw(std::invalid_argument, "invalid positions map for evaluation");
825  }

837  tmp += it->v[i] * map_it->second;
