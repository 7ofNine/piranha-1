112 // Apply a functor to a single-coefficient series. This is in the spirit of the old apply_cf_functor method,
113 // and it is used in some of the math:: overrides below.
114 template <typename Functor, typename RetT, typename T>
115 inline RetT apply_cf_functor(const T &s)
116 {
117  using term_type = typename RetT::term_type;
118  using cf_type = typename term_type::cf_type;
119  using key_type = typename term_type::key_type;
120  if (!s.is_single_coefficient()) {
121  piranha_throw(std::invalid_argument,std::string("cannot compute ") + Functor::name + ", series is not single-coefficient");
122  }
123  RetT retval;
124  Functor f;
125  if (s.empty()) {
126  retval.insert(term_type(f(cf_type(0)),key_type(symbol_set{})));
127  } else {
128  retval.insert(term_type(f(s._container().begin()->m_cf),key_type(symbol_set{})));
129  }
130  return retval;
131 }

523 class series_operators
524 {
525  // A couple of handy aliases.
526  // NOTE: we need both bso_type and common_type because bso gives access to the ::value
527  // member to decide on the implementation, common_type is handy because it's shorter
528  // than using ::type on the bso.
529  template <typename T, typename U, int N>
530  using bso_type = detail::binary_series_op_return_type<typename std::decay<T>::type,
531  typename std::decay<U>::type,N>;
532  template <typename T, typename U, int N>
533  using series_common_type = typename bso_type<T,U,N>::type;
534  // Main implementation function for add/sub: all the possible cases end up here eventually, after
535  // any necessary conversion.
536  template <bool Sign, typename T, typename U>
537  static series_common_type<T,U,0> binary_add_impl(T &&x, U &&y)
538  {
539  // This is the same as T and U.
540  using ret_type = series_common_type<T,U,0>;
541  static_assert(std::is_same<typename std::decay<T>::type,ret_type>::value,"Invalid type.");
542  static_assert(std::is_same<typename std::decay<U>::type,ret_type>::value,"Invalid type.");
543  // Init the return value from the first operand.
544  // NOTE: this is always possible to do, a series is always copy/move-constructible.
545  ret_type retval(std::forward<T>(x));
546  // NOTE: here we cover the corner case in which x and y are the same object.
547  // This could be problematic in the algorithm below if retval was move-cted from
548  // x, since in such a case y will also be erased. This will happen if one does
549  // std::move(a) + std::move(a)
550  if (unlikely(&x == &y)) {
551  retval.template merge_terms<Sign>(retval);
552  return retval;
553  }
554  // NOTE: we do not use x any more, as it might have been moved.
555  if (likely(retval.m_symbol_set == y.m_symbol_set)) {
556  retval.template merge_terms<Sign>(std::forward<U>(y));
557  } else {
558  // Let's fix the args of the first series, if needed.
559  auto merge = retval.m_symbol_set.merge(y.m_symbol_set);
560  if (merge != retval.m_symbol_set) {
561  // This is a move assignment, always possible.
562  retval = retval.merge_arguments(merge);
563  }
564  // Fix the args of the second series.
565  if (merge != y.m_symbol_set) {
566  retval.template merge_terms<Sign>(y.merge_arguments(merge));
567  } else {
568  retval.template merge_terms<Sign>(std::forward<U>(y));
569  }
570  }
571  return retval;
572  }
573  // NOTE: this case has no special algorithmic requirements, the base requirements for cf and series types
574  // already cover everything (including the needeed series constructors and cf arithmetic operators).
575  template <typename T, typename U, typename std::enable_if<bso_type<T,U,0>::value == 0u,int>::type = 0>
576  static series_common_type<T,U,0> dispatch_binary_add(T &&x, U &&y)
577  {
578  return binary_add_impl<true>(std::forward<T>(x),std::forward<U>(y));
579  }
580  // NOTE: this covers two cases:
581  // - equal recursion, first series wins,
582  // - first higher recursion, coefficient of first series wins,
583  // In both cases we need to construct a T from y.
584  template <typename T, typename U, typename std::enable_if<(bso_type<T,U,0>::value == 1u || bso_type<T,U,0>::value == 4u) &&
585  // In order for the implementation to work, we need to be able to build T from U.
586  std::is_constructible<typename std::decay<T>::type,const typename std::decay<U>::type &>::value,
587  int>::type = 0>
588  static series_common_type<T,U,0> dispatch_binary_add(T &&x, U &&y)
589  {
590  typename std::decay<T>::type y1(std::forward<U>(y));
591  return dispatch_binary_add(std::forward<T>(x),std::move(y1));
592  }
593  // Symmetric of 1.
594  template <typename T, typename U, typename std::enable_if<bso_type<T,U,0>::value == 2u,int>::type = 0>
595  static auto dispatch_binary_add(T &&x, U &&y) -> decltype(dispatch_binary_add(std::forward<U>(y),std::forward<T>(x)))
596  {
597  return dispatch_binary_add(std::forward<U>(y),std::forward<T>(x));
598  }
599  // Two cases:
600  // - equal series recursion, 3rd coefficient type generated,

1276  if (unlikely(!m_container.bucket_count())) {

2178  pow_ret_type<T,U> pow(const T &x) const
2179  {
2180  // NOTE: there are 3 types involved here:
2181  // - Derived,
2182  // - the return type (which is Derived or a rebound type from Derived),
2183  // - the type of Derived * Derived.
2184  using ret_type = pow_ret_type<T,U>;
2185  using r_term_type = typename ret_type::term_type;
2186  using r_cf_type = typename r_term_type::cf_type;
2187  using cf_type = typename term_type::cf_type;
2188  using key_type = typename term_type::key_type;
2189  using m_type = pow_m_type<Derived>;
2190  using m_term_type = typename m_type::term_type;
2191  using m_cf_type = typename m_term_type::cf_type;
2192  using m_key_type = typename m_term_type::key_type;
2193  // Handle the case of single coefficient series.
2194  if (is_single_coefficient()) {
2195  ret_type retval;
2196  if (empty()) {
2197  // An empty series is equal to zero.
2198  retval.insert(r_term_type(math::pow(cf_type(0),x),key_type(symbol_set{})));
2199  } else {
2200  retval.insert(r_term_type(math::pow(m_container.begin()->m_cf,x),key_type(symbol_set{})));
2201  }
2202  return retval;
2203  }
2204  // Handle the case of zero exponent.
2205  if (math::is_zero(x)) {
2206  ret_type retval;
2207  retval.insert(r_term_type(r_cf_type(1),key_type(symbol_set{})));
2208  return retval;
2209  }
2210  // Exponentiation by repeated multiplications.
2211  integer n;
2212  try {
2213  n = safe_cast<integer>(x);
2214  } catch (const std::invalid_argument &) {
2215  piranha_throw(std::invalid_argument,"invalid argument for series exponentiation: non-integral value");
2216  }
2217  if (n.sign() < 0) {
2218  piranha_throw(std::invalid_argument,"invalid argument for series exponentiation: negative integral value");
2219  }
2220  // Lock the cache for the rest of the method.
2221  std::lock_guard<std::mutex> lock(s_pow_mutex);
2222  auto &v = get_pow_cache()[*static_cast<Derived const *>(this)];
2223  using s_type = decltype(v.size());
2224  // Init the vector, if needed.
2225  if (!v.size()) {
2226  m_type tmp;
2227  tmp.insert(m_term_type(m_cf_type(1),m_key_type(symbol_set{})));
2228  v.push_back(std::move(tmp));
2229  }
2230  // Fill in the missing powers.
2231  while (v.size() <= n) {
2232  // NOTE: for series it seems like it is better to run the dumb algorithm instead of, e.g.,
2233  // exponentiation by squaring - the growth in number of terms seems to be slower.
2234  v.push_back(v.back() * (*static_cast<Derived const *>(this)));
2235  }
2236  return ret_type(v[static_cast<s_type>(n)]);
2237  }
