112 // Apply a functor to a single-coefficient series. This is in the spirit of the old apply_cf_functor method,
113 // and it is used in some of the math:: overrides below.
114 template <typename Functor, typename RetT, typename T>
115 inline RetT apply_cf_functor(const T &s)
116 {
117  using term_type = typename RetT::term_type;
118  using cf_type = typename term_type::cf_type;
119  using key_type = typename term_type::key_type;
120  if (!s.is_single_coefficient()) {
121  piranha_throw(std::invalid_argument,
122  std::string("cannot compute ") + Functor::name + ", series is not single-coefficient");
123  }
124  RetT retval;
125  Functor f;
126  if (s.empty()) {
127  retval.insert(term_type(f(cf_type(0)), key_type(symbol_set{})));
128  } else {
129  retval.insert(term_type(f(s._container().begin()->m_cf), key_type(symbol_set{})));
130  }
131  return retval;
132 }
133 }

360 // Case 0:
361 // a. both are series with same recursion index,
362 // b. the coefficients are the same and the op results in the same coefficient,
363 // c. the two series types are equal.
364 template <typename S1, typename S2, int N>
365 struct binary_series_op_return_type<S1, S2, N,
366  typename std::enable_if<
367  // NOTE: in all these checks, every access to S1 and S2 should not incur in hard
368  // errors:
369  // - the series recursion index works on all types safely,
370  // - the various _t aliases are SFINAE friendly - if S1 or S2 are not series
371  // with the appropriate characteristics,
372  // or the coefficients are not oppable, etc., there should be a soft error.
373  /*a*/ series_recursion_index<S1>::value == series_recursion_index<S2>::value
374  && series_recursion_index<S1>::value != 0u &&
375  /*b*/ std::is_same<bso_cf_op_t<S1, S2, N>, bso_cf_t<S1>>::value
376  && std::is_same<bso_cf_op_t<S1, S2, N>, bso_cf_t<S2>>::value &&
377  /*c*/ std::is_same<S1, S2>::value>::type> {
378  using type = S1;
379  static const unsigned value = 0u;
380 };

383 // a. both are series with same recursion index,
384 // b. the coefficients are not the same and the op results in the first coefficient.

390  /*b*/ std::is_same<bso_cf_op_t<S1, S2, N>, bso_cf_t<S1>>::value

541 class series_operators
542 {
543  // A couple of handy aliases.
544  // NOTE: we need both bso_type and common_type because bso gives access to the ::value
545  // member to decide on the implementation, common_type is handy because it's shorter
546  // than using ::type on the bso.
547  template <typename T, typename U, int N>
548  using bso_type
549  = detail::binary_series_op_return_type<typename std::decay<T>::type, typename std::decay<U>::type, N>;
550  template <typename T, typename U, int N>
551  using series_common_type = typename bso_type<T, U, N>::type;
552  // Main implementation function for add/sub: all the possible cases end up here eventually, after
553  // any necessary conversion.
554  template <bool Sign, typename T, typename U>
555  static series_common_type<T, U, 0> binary_add_impl(T &&x, U &&y)
556  {
557  // This is the same as T and U.
558  using ret_type = series_common_type<T, U, 0>;
559  static_assert(std::is_same<typename std::decay<T>::type, ret_type>::value, "Invalid type.");
560  static_assert(std::is_same<typename std::decay<U>::type, ret_type>::value, "Invalid type.");
561  // Init the return value from the first operand.
562  // NOTE: this is always possible to do, a series is always copy/move-constructible.
563  ret_type retval(std::forward<T>(x));
564  // NOTE: here we cover the corner case in which x and y are the same object.
565  // This could be problematic in the algorithm below if retval was move-cted from
566  // x, since in such a case y will also be erased. This will happen if one does
567  // std::move(a) + std::move(a)
568  if (unlikely(&x == &y)) {
569  retval.template merge_terms<Sign>(retval);
570  return retval;
571  }
572  // NOTE: we do not use x any more, as it might have been moved.
573  if (likely(retval.m_symbol_set == y.m_symbol_set)) {
574  retval.template merge_terms<Sign>(std::forward<U>(y));
575  } else {
576  // Let's fix the args of the first series, if needed.
577  auto merge = retval.m_symbol_set.merge(y.m_symbol_set);
578  if (merge != retval.m_symbol_set) {
579  // This is a move assignment, always possible.
580  retval = retval.merge_arguments(merge);
581  }
582  // Fix the args of the second series.
583  if (merge != y.m_symbol_set) {
584  retval.template merge_terms<Sign>(y.merge_arguments(merge));
585  } else {
586  retval.template merge_terms<Sign>(std::forward<U>(y));
587  }
588  }
589  return retval;
590  }
591  // NOTE: this case has no special algorithmic requirements, the base requirements for cf and series types
592  // already cover everything (including the needeed series constructors and cf arithmetic operators).
593  template <typename T, typename U, typename std::enable_if<bso_type<T, U, 0>::value == 0u, int>::type = 0>
594  static series_common_type<T, U, 0> dispatch_binary_add(T &&x, U &&y)
595  {
596  return binary_add_impl<true>(std::forward<T>(x), std::forward<U>(y));
597  }
598  // NOTE: this covers two cases:
599  // - equal recursion, first series wins,
600  // - first higher recursion, coefficient of first series wins,
601  // In both cases we need to construct a T from y.
602  template <
603  typename T, typename U,
604  typename std::enable_if<(bso_type<T, U, 0>::value == 1u || bso_type<T, U, 0>::value == 4u) &&
605  // In order for the implementation to work, we need to be able to build T from U.
606  std::is_constructible<typename std::decay<T>::type,
607  const typename std::decay<U>::type &>::value,
608  int>::type
609  = 0>
610  static series_common_type<T, U, 0> dispatch_binary_add(T &&x, U &&y)
611  {
612  typename std::decay<T>::type y1(std::forward<U>(y));
613  return dispatch_binary_add(std::forward<T>(x), std::move(y1));
614  }
615  // Symmetric of 1.
616  template <typename T, typename U, typename std::enable_if<bso_type<T, U, 0>::value == 2u, int>::type = 0>
617  static auto dispatch_binary_add(T &&x, U &&y)
618  -> decltype(dispatch_binary_add(std::forward<U>(y), std::forward<T>(x)))
619  {
620  return dispatch_binary_add(std::forward<U>(y), std::forward<T>(x));
621  }
622  // Two cases:
623  // - equal series recursion, 3rd coefficient type generated,
624  // - first higher recursion, coefficient result is different from first series.
625  // In both cases we need to promote both operands to a 3rd type.
