143  // Canonicalisation.
144  static std::pair<p_type, p_type> canonicalise_impl(const p_type &n, const p_type &d)
145  {
146  // First let's check for negative exponents.
147  detail::poly_expo_checker(n);
148  detail::poly_expo_checker(d);
149  // Handle a zero divisor.
150  if (unlikely(math::is_zero(d))) {
151  piranha_throw(zero_division_error, "null denominator in rational function");
152  }
153  // If the numerator is null, return {0,1}.
154  if (math::is_zero(n)) {
155  return std::make_pair(p_type{}, p_type{1});
156  }
157  // NOTE: maybe these checks should go directly into the poly GCD routine. Keep it
158  // in mind for the future.
159  // NOTE: it would make sense here to deal with the single coefficient denominator case as well.
160  // This should give good performance when one is using rational_function as a rational polynomial,
161  // no need to go through a costly canonicalisation.
162  // If the denominator is unitary, just return the input values.
163  if (math::is_unitary(d)) {
164  return std::make_pair(n, d);
165  }
166  // Handle single-coefficient polys.
167  if (n.is_single_coefficient() && d.is_single_coefficient()) {
168  piranha_assert(n.size() == 1u && d.size() == 1u);
169  // Use a rational to construct the canonical form.
170  rational tmp{n._container().begin()->m_cf, d._container().begin()->m_cf};
171  return std::make_pair(p_type{tmp.num()}, p_type{tmp.den()});
172  }
173  // Compute the GCD and create the return values.
174  auto tup_res = p_type::gcd(n, d, true);
175  auto retval = std::make_pair(std::move(std::get<1u>(tup_res)), std::move(std::get<2u>(tup_res)));
176  // Check if we need to adjust the sign of the leading term
177  // of the denominator.
178  // NOTE: here we ask for the lterm into something that potentially might be zero
179  // if truncation is active. Keep this in mind.
180  if (detail::poly_lterm(retval.second)->m_cf.sign() < 0) {
181  math::negate(retval.first);
182  math::negate(retval.second);
183  }
184  return retval;
185  }
186  // Utility function to convert a q_type to a pair (p_type,integer), representing the numerator
187  // and denominator of a rational function.
188  static std::pair<p_type, integer> q_to_p_type(const q_type &q)
189  {
190  // Init the numerator.
191  p_type ret_p;
192  ret_p._container().rehash(q._container().bucket_count());
193  ret_p.set_symbol_set(q.get_symbol_set());
194  // Compute the least common multiplier.
195  integer lcm{1};
196  // The GCD.
197  integer g;
198  for (const auto &t : q._container()) {
199  math::gcd3(g, lcm, t.m_cf.den());
200  math::mul3(lcm, lcm, t.m_cf.den());
201  integer::_divexact(lcm, lcm, g);
202  }
203  // All these computations involve only positive numbers,
204  // the GCD must always be positive.
205  piranha_assert(lcm.sign() == 1);
206  // Fill in the numerator.
207  for (const auto &t : q._container()) {
208  using term_type = typename p_type::term_type;
209  // NOTE: possibility of unique insertion here.
210  // NOTE: possibility of exact division.
211  ret_p.insert(term_type{lcm / t.m_cf.den() * t.m_cf.num(), t.m_key});
212  }
213  return std::make_pair(std::move(ret_p), std::move(lcm));
214  }

267  // Addition.
268  static rational_function dispatch_binary_add(const rational_function &a, const rational_function &b)
269  {
270  // Detect unitary denominators.
271  const bool uda = math::is_unitary(a.den()), udb = math::is_unitary(b.den());
272  rational_function retval;
273  if (uda && udb) {
274  // With unitary denominators, just add the numerators and set den to 1.
275  retval.m_num = a.m_num + b.m_num;
276  retval.m_den = 1;
277  } else if (uda) {
278  // Only a is unitary.
279  retval.m_num = a.m_num * b.m_den + b.m_num;
280  retval.m_den = b.m_den;
281  } else if (udb) {
282  // Only b is unitary.
283  retval.m_num = a.m_num + b.m_num * a.m_den;
284  retval.m_den = a.m_den;
285  } else {
286  // The general case.
287  retval.m_num = a.m_num * b.m_den + a.m_den * b.m_num;
288  retval.m_den = a.m_den * b.m_den;
289  // NOTE: if the canonicalisation fails for some reason, it is not a problem
290  // as retval is a local variable that will just be destroyed. The destructor
291  // does not run any check.
292  retval.canonicalise();
293  }
294  return retval;
295  }

767  void print_tex(std::ostream &os) const
768  {
769  if (math::is_zero(*this)) {
770  // Special case for zero.
771  os << "0";
772  } else if (math::is_unitary(m_den)) {
773  // If the denominator is 1, print just the numerator.

775  } else {

777  // NOTE: here we are relying on the fact that negation of poly proceeds

779  piranha_assert(m_num.size() >= 1u);
780  piranha_assert(m_den.size() >= 1u);

782  const bool negate_den = m_den._container().begin()->m_cf.sign() < 0;
783  if (negate_num != negate_den) {

785  // then we need to prepend the minus sign.
786  os << '-';
787  }

790  (-m_num).print_tex(os);
791  } else {
792  m_num.print_tex(os);
793  }
794  os << "}{";
795  if (negate_den) {
796  (-m_den).print_tex(os);
797  } else {

799  }

1117  template <typename T, pow_enabler<T> = 0>
1118  rational_function pow(const T &n) const
1119  {
1120  // NOTE: here we are renouncing the pow() optimisation
1121  // implemented for the polynomials. Consider bringing
1122  // it back if it becomes important.
1123  const integer n_int{n}, un_int = n_int.abs();
1124  rational_function retval;
1125  {
1126  // Lock the cache.
1127  std::lock_guard<std::mutex> lock(s_pow_mutex);
1128  auto &v = get_pow_cache()[*this];
1129  using s_type = decltype(v.size());
1130  // Init the vector, if needed.
1131  if (!v.size()) {
1132  v.push_back(rational_function{1});
1133  }
1134  // Fill in the missing powers.
1135  while (v.size() <= un_int) {
1136  // NOTE: avoid canonicalisation by setting directly
1137  // the num/den.
1138  // NOTE: this will have to be replaced by explicit untruncated
1139  // multiplication.
1140  rational_function tmp;
1141  tmp.m_num = v.back().m_num * m_num;
1142  tmp.m_den = v.back().m_den * m_den;
1143  v.push_back(std::move(tmp));
1144  }
1145  retval = v[static_cast<s_type>(un_int)];
1146  }
1147  // We need to fix retval in case of negative powers.
1148  if (n_int.sign() < 0) {
1149  if (unlikely(math::is_zero(retval.m_num))) {
1150  piranha_throw(zero_division_error, "zero denominator in rational_function exponentiation");
1151  }
1152  // Swap num/den.
1153  std::swap(retval.m_num, retval.m_den);
1154  // The only canonicalisation we need is checking the sign of the new
1155  // denominator.
1156  if (detail::poly_lterm(retval.m_den)->m_cf.sign() < 0) {
1157  math::negate(retval.m_num);
1158  math::negate(retval.m_den);
1159  }
1160  }
1161  return retval;
1162  }
