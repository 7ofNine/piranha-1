90 class symbol_set
91 {
92  bool check() const
93  {
94  // Check for sorted range.
95  if (!std::is_sorted(begin(),end())) {
96  return false;
97  }
98  if (size() < 2u) {
99  return true;
100  }
101  // Check for duplicates.
102  for (size_type i = 0u; i < size() - 1u; ++i) {
103  if (m_values[i] == m_values[i + 1u]) {
104  return false;
105  }
106  }
107  return true;
108  }
109  // Enabler for ctor from iterator.
110  template <typename Iterator, typename Symbol>
111  using it_ctor_enabler = typename std::enable_if<is_input_iterator<Iterator>::value &&
112  std::is_constructible<Symbol,decltype(*(std::declval<const Iterator &>()))>::value,int>::type;
113  // Serialization support.
114  friend class boost::serialization::access;
115  template <class Archive>
116  void save(Archive &ar, unsigned int) const
117  {
118  const auto size = m_values.size();
119  ar & size;
120  for (const auto &s: m_values) {
121  // Store the names.
122  ar & s.get_name();
123  }
124  }
125  template <class Archive>
126  void load(Archive &ar, unsigned int)
127  {
128  // A new, empty set.
129  symbol_set ss;
130  // First recover the size.
131  std::vector<symbol>::size_type size;
132  ar & size;
133  // Restore using strings.
134  // NOTE: this is the slowest and safest way of doing it, even in face
135  // of bogus input from the archive (e.g., out-of-order symbols, duplicate names,
136  // etc.). This can certainly be optimised in the future.
137  std::string name;
138  for (decltype(size) i = 0u; i < size; ++i) {
139  ar & name;
140  ss.add(symbol(name));
141  }
142  // Move in the new set.
143  *this = std::move(ss);
144  }

425  void add(const symbol &s)
426  {
427  // Copy it to provide strong exception safety.
428  std::vector<symbol> new_values;
429  new_values.reserve(size() + size_type(1u));
430  std::copy(begin(),end(),std::back_inserter(new_values));
431  const auto it = std::lower_bound(new_values.begin(),new_values.end(),s);
432  if (unlikely(it != new_values.end() && *it == s)) {
433  piranha_throw(std::invalid_argument,"symbol already present in this set");
434  }
435  new_values.insert(it,s);
436  // Move in the new args vector.
437  m_values = std::move(new_values);
438  piranha_assert(check());
439  }

559  size_type index_of(const symbol &s) const
560  {
561  // Need to check for potential overflows here.
562  using diff_type = std::iterator_traits<const_iterator>::difference_type;
563  using udiff_type = std::make_unsigned<diff_type>::type;
564  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
565  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
566  piranha_throw(std::overflow_error,"potential overflow in the computaion of the "
567  "index of a symbol");
568  }
569  const auto it = std::lower_bound(m_values.begin(),m_values.end(),s);
570  if (it != m_values.end() && *it == s) {
571  auto retval = it - m_values.begin();
572  piranha_assert(retval >= diff_type(0));
573  return static_cast<size_type>(retval);
574  }
575  return m_values.size();
576  }

603 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
604 {
605  size_type ia = 0u, ib = 0u;
606  while (true) {
607  if (ia == a.size() || ib == b.size()) {
608  break;
609  }
610  if (a[ia] == b[ib]) {
611  m_values.push_back(ia);
612  ++ia;
613  ++ib;
614  } else if (a[ia] < b[ib]) {
615  ++ia;
616  } else {
617  ++ib;
618  }
619  }
620 }

633 template <typename T>
634 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol,T> &map)
635 {
636  for (const auto &p: map) {
637  const auto idx = a.index_of(p.first);
638  if (idx != a.size()) {
639  m_pairs.push_back(std::make_pair(idx,p.second));
640  }
641  }
642  std::stable_sort(m_pairs.begin(),m_pairs.end(),[](const value_type &p1, const value_type &p2) {
643  return p1.first < p2.first;
644  });
645  // Check that there are no duplicate positions.
646  piranha_assert(std::unique(m_pairs.begin(),m_pairs.end(),[](const value_type &p1, const value_type &p2) {
647  return p1.first == p2.first;
648  }) == m_pairs.end());
649 }
