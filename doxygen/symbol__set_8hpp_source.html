86  if (!std::is_sorted(begin(), end())) {
87  return false;

386  void add(const symbol &s)
387  {
388  // Copy it to provide strong exception safety.
389  std::vector<symbol> new_values;
390  new_values.reserve(size() + size_type(1u));
391  std::copy(begin(), end(), std::back_inserter(new_values));
392  const auto it = std::lower_bound(new_values.begin(), new_values.end(), s);
393  if (unlikely(it != new_values.end() && *it == s)) {
394  piranha_throw(std::invalid_argument, "symbol already present in this set");
395  }
396  new_values.insert(it, s);
397  // Move in the new args vector.
398  m_values = std::move(new_values);
399  piranha_assert(check());
400  }

519  size_type index_of(const symbol &s) const
520  {
521  // Need to check for potential overflows here.
522  using diff_type = std::iterator_traits<const_iterator>::difference_type;
523  using udiff_type = std::make_unsigned<diff_type>::type;
524  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
525  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
526  piranha_throw(std::overflow_error, "potential overflow in the computaion of the "
527  "index of a symbol");
528  }
529  const auto it = std::lower_bound(m_values.begin(), m_values.end(), s);
530  if (it != m_values.end() && *it == s) {
531  auto retval = it - m_values.begin();
532  piranha_assert(retval >= diff_type(0));
533  return static_cast<size_type>(retval);
534  }
535  return m_values.size();
536  }

565 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
566 {
567  size_type ia = 0u, ib = 0u;
568  while (true) {
569  if (ia == a.size() || ib == b.size()) {
570  break;
571  }
572  if (a[ia] == b[ib]) {
573  m_values.push_back(ia);
574  ++ia;
575  ++ib;
576  } else if (a[ia] < b[ib]) {
577  ++ia;
578  } else {
579  ++ib;
580  }
581  }
582 }

595 template <typename T>
596 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol, T> &map)
597 {
598  for (const auto &p : map) {
599  const auto idx = a.index_of(p.first);
600  if (idx != a.size()) {
601  m_pairs.push_back(std::make_pair(idx, p.second));
602  }
603  }
604  std::stable_sort(m_pairs.begin(), m_pairs.end(),
605  [](const value_type &p1, const value_type &p2) { return p1.first < p2.first; });
606  // Check that there are no duplicate positions.
607  piranha_assert(std::unique(m_pairs.begin(), m_pairs.end(),
608  [](const value_type &p1, const value_type &p2) { return p1.first == p2.first; })
609  == m_pairs.end());
610 }
