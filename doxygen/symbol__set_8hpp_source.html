86 class symbol_set
87 {
88  bool check() const
89  {
90  // Check for sorted range.
91  if (!std::is_sorted(begin(), end())) {
92  return false;
93  }
94  if (size() < 2u) {
95  return true;
96  }
97  // Check for duplicates.
98  for (size_type i = 0u; i < size() - 1u; ++i) {
99  if (m_values[i] == m_values[i + 1u]) {
100  return false;
101  }
102  }
103  return true;
104  }
105  // Enabler for ctor from iterator.
106  template <typename Iterator, typename Symbol>
107  using it_ctor_enabler =
108  typename std::enable_if<is_input_iterator<Iterator>::value
109  && std::is_constructible<Symbol,
110  decltype(*(std::declval<const Iterator &>()))>::value,
111  int>::type;
112  // Serialization support.
113  friend class boost::serialization::access;
114  template <class Archive>
115  void save(Archive &ar, unsigned int) const
116  {
117  const auto size = m_values.size();
118  ar &size;
119  for (const auto &s : m_values) {
120  // Store the names.
121  ar &s.get_name();
122  }
123  }
124  template <class Archive>
125  void load(Archive &ar, unsigned int)
126  {
127  // A new, empty set.
128  symbol_set ss;
129  // First recover the size.
130  std::vector<symbol>::size_type size;
131  ar &size;
132  // Restore using strings.
133  // NOTE: this is the slowest and safest way of doing it, even in face
134  // of bogus input from the archive (e.g., out-of-order symbols, duplicate names,
135  // etc.). This can certainly be optimised in the future.
136  std::string name;
137  for (decltype(size) i = 0u; i < size; ++i) {
138  ar &name;
139  ss.add(symbol(name));
140  }
141  // Move in the new set.
142  *this = std::move(ss);
143  }

424  void add(const symbol &s)
425  {
426  // Copy it to provide strong exception safety.
427  std::vector<symbol> new_values;
428  new_values.reserve(size() + size_type(1u));
429  std::copy(begin(), end(), std::back_inserter(new_values));
430  const auto it = std::lower_bound(new_values.begin(), new_values.end(), s);
431  if (unlikely(it != new_values.end() && *it == s)) {
432  piranha_throw(std::invalid_argument, "symbol already present in this set");
433  }
434  new_values.insert(it, s);
435  // Move in the new args vector.
436  m_values = std::move(new_values);
437  piranha_assert(check());
438  }

557  size_type index_of(const symbol &s) const
558  {
559  // Need to check for potential overflows here.
560  using diff_type = std::iterator_traits<const_iterator>::difference_type;
561  using udiff_type = std::make_unsigned<diff_type>::type;
562  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
563  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
564  piranha_throw(std::overflow_error, "potential overflow in the computaion of the "
565  "index of a symbol");
566  }
567  const auto it = std::lower_bound(m_values.begin(), m_values.end(), s);
568  if (it != m_values.end() && *it == s) {
569  auto retval = it - m_values.begin();
570  piranha_assert(retval >= diff_type(0));
571  return static_cast<size_type>(retval);
572  }
573  return m_values.size();
574  }

603 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
604 {
605  size_type ia = 0u, ib = 0u;
606  while (true) {
607  if (ia == a.size() || ib == b.size()) {
608  break;
609  }
610  if (a[ia] == b[ib]) {
611  m_values.push_back(ia);
612  ++ia;
613  ++ib;
614  } else if (a[ia] < b[ib]) {
615  ++ia;
616  } else {
617  ++ib;
618  }
619  }

633 template <typename T>
634 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol, T> &map)
635 {
636  for (const auto &p : map) {
637  const auto idx = a.index_of(p.first);
638  if (idx != a.size()) {
639  m_pairs.push_back(std::make_pair(idx, p.second));
640  }
641  }
642  std::stable_sort(m_pairs.begin(), m_pairs.end(),
643  [](const value_type &p1, const value_type &p2) { return p1.first < p2.first; });
644  // Check that there are no duplicate positions.
645  piranha_assert(std::unique(m_pairs.begin(), m_pairs.end(),
646  [](const value_type &p1, const value_type &p2) { return p1.first == p2.first; })
