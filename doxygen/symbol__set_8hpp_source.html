88 class symbol_set
89 {
90  bool check() const
91  {
92  // Check for sorted range.
93  if (!std::is_sorted(begin(),end())) {
94  return false;
95  }
96  if (size() < 2u) {
97  return true;
98  }
99  // Check for duplicates.
100  for (size_type i = 0u; i < size() - 1u; ++i) {
101  if (m_values[i] == m_values[i + 1u]) {
102  return false;
103  }
104  }
105  return true;
106  }
107  // Enabler for ctor from iterator.
108  template <typename Iterator, typename Symbol>
109  using it_ctor_enabler = typename std::enable_if<is_input_iterator<Iterator>::value &&
110  std::is_constructible<Symbol,decltype(*(std::declval<const Iterator &>()))>::value,int>::type;
111  // Serialization support.
112  friend class boost::serialization::access;
113  template <class Archive>
114  void save(Archive &ar, unsigned int) const
115  {
116  const auto size = m_values.size();
117  ar & size;
118  for (const auto &s: m_values) {
119  // Store the names.
120  ar & s.get_name();
121  }
122  }
123  template <class Archive>
124  void load(Archive &ar, unsigned int)
125  {
126  // A new, empty set.
127  symbol_set ss;
128  // First recover the size.
129  std::vector<symbol>::size_type size;
130  ar & size;
131  // Restore using strings.
132  // NOTE: this is the slowest and safest way of doing it, even in face
133  // of bogus input from the archive (e.g., out-of-order symbols, duplicate names,
134  // etc.). This can certainly be optimised in the future.
135  std::string name;
136  for (decltype(size) i = 0u; i < size; ++i) {
137  ar & name;
138  ss.add(symbol(name));
139  }
140  // Move in the new set.
141  *this = std::move(ss);
142  }

421  void add(const symbol &s)
422  {
423  // Copy it to provide strong exception safety.
424  std::vector<symbol> new_values;
425  new_values.reserve(size() + size_type(1u));
426  std::copy(begin(),end(),std::back_inserter(new_values));
427  const auto it = std::lower_bound(new_values.begin(),new_values.end(),s);
428  if (unlikely(it != new_values.end() && *it == s)) {
429  piranha_throw(std::invalid_argument,"symbol already present in this set");
430  }
431  new_values.insert(it,s);
432  // Move in the new args vector.
433  m_values = std::move(new_values);
434  piranha_assert(check());
435  }

555  size_type index_of(const symbol &s) const
556  {
557  // Need to check for potential overflows here.
558  using diff_type = std::iterator_traits<const_iterator>::difference_type;
559  using udiff_type = std::make_unsigned<diff_type>::type;
560  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
561  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
562  piranha_throw(std::overflow_error,"potential overflow in the computaion of the "
563  "index of a symbol");
564  }
565  const auto it = std::lower_bound(m_values.begin(),m_values.end(),s);
566  if (it != m_values.end() && *it == s) {
567  auto retval = it - m_values.begin();
568  piranha_assert(retval >= diff_type(0));
569  return static_cast<size_type>(retval);
570  }
571  return m_values.size();
572  }

599 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
600 {
601  size_type ia = 0u, ib = 0u;
602  while (true) {
603  if (ia == a.size() || ib == b.size()) {
604  break;
605  }
606  if (a[ia] == b[ib]) {
607  m_values.push_back(ia);
608  ++ia;
609  ++ib;
610  } else if (a[ia] < b[ib]) {
611  ++ia;
612  } else {
613  ++ib;
614  }
615  }

630 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol,T> &map)
631 {

634  if (idx != a.size()) {
