376  void add(const symbol &s)
377  {
378  // Copy it to provide strong exception safety.
379  std::vector<symbol> new_values;
380  new_values.reserve(size() + size_type(1u));
381  std::copy(begin(), end(), std::back_inserter(new_values));
382  const auto it = std::lower_bound(new_values.begin(), new_values.end(), s);
383  if (unlikely(it != new_values.end() && *it == s)) {
384  piranha_throw(std::invalid_argument, "symbol already present in this set");
385  }
386  new_values.insert(it, s);
387  // Move in the new args vector.
388  m_values = std::move(new_values);
389  piranha_assert(check());
390  }

509  size_type index_of(const symbol &s) const
510  {
511  // Need to check for potential overflows here.
512  using diff_type = std::iterator_traits<const_iterator>::difference_type;
513  using udiff_type = std::make_unsigned<diff_type>::type;
514  // This is not an exact check, it is conservative by 1 unit in order to simplify the logic.
515  if (unlikely(m_values.size() > static_cast<udiff_type>(std::numeric_limits<diff_type>::max()))) {
516  piranha_throw(std::overflow_error, "potential overflow in the computaion of the "
517  "index of a symbol");
518  }
519  const auto it = std::lower_bound(m_values.begin(), m_values.end(), s);
520  if (it != m_values.end() && *it == s) {
521  auto retval = it - m_values.begin();
522  piranha_assert(retval >= diff_type(0));
523  return static_cast<size_type>(retval);
524  }
525  return m_values.size();
526  }

555 inline symbol_set::positions::positions(const symbol_set &a, const symbol_set &b)
556 {
557  size_type ia = 0u, ib = 0u;
558  while (true) {
559  if (ia == a.size() || ib == b.size()) {
560  break;
561  }
562  if (a[ia] == b[ib]) {
563  m_values.push_back(ia);
564  ++ia;
565  ++ib;
566  } else if (a[ia] < b[ib]) {
567  ++ia;
568  } else {
569  ++ib;
570  }
571  }

585 template <typename T>
586 inline symbol_set::positions_map<T>::positions_map(const symbol_set &a, const std::unordered_map<symbol, T> &map)
587 {
588  for (const auto &p : map) {
589  const auto idx = a.index_of(p.first);
590  if (idx != a.size()) {
591  m_pairs.emplace_back(idx, p.second);
592  }
593  }
594  std::stable_sort(m_pairs.begin(), m_pairs.end(),
595  [](const value_type &p1, const value_type &p2) { return p1.first < p2.first; });
596  // Check that there are no duplicate positions.
597  piranha_assert(std::unique(m_pairs.begin(), m_pairs.end(),
598  [](const value_type &p1, const value_type &p2) { return p1.first == p2.first; })
599  == m_pairs.end());
600 }
