617 // Wrapper for std stream classes for use in msgpack. The reason for this wrapper is that msgpack expects
618 // streams with a write(const char *, std::size_t) method, but in std streams the second param is std::streamsize
619 // (which is a signed int). Hence we wrap the write method to do the safe conversion from size_t to streamsize.
620 template <typename Stream>
621 class msgpack_stream_wrapper : public Stream
622 {
623 public:
624  // Inherit ctors.
625  using Stream::Stream;
626  auto write(const typename Stream::char_type *p, std::size_t count)
627  -> decltype(std::declval<Stream &>().write(p, std::streamsize(0)))
628  {
629  return static_cast<Stream *>(this)->write(p, safe_cast<std::streamsize>(count));
630  }
631 };
632

634 using msgpack_stream_write_t
635  = decltype(std::declval<T &>().write(std::declval<const char *>(), std::declval<std::size_t>()));
636 }

744 template <typename Stream>
745 struct msgpack_pack_impl<Stream, long double, msgpack_ld_enabler<Stream>> {
747 
759  void operator()(msgpack::packer<Stream> &packer, const long double &x, msgpack_format f) const
760  {
761  if (f == msgpack_format::binary) {
762  packer.pack_bin(sizeof(long double));
763  packer.pack_bin_body(reinterpret_cast<const char *>(&x), sizeof(long double));
764  } else {
765  if (std::isnan(x)) {
766  if (std::signbit(x)) {
767  msgpack_pack(packer, std::string("-nan"), f);
768  } else {
769  msgpack_pack(packer, std::string("+nan"), f);
770  }
771  } else if (std::isinf(x)) {
772  if (std::signbit(x)) {
773  msgpack_pack(packer, std::string("-inf"), f);
774  } else {
775  msgpack_pack(packer, std::string("+inf"), f);
776  }
777  } else {
778  std::ostringstream oss;
779  // Make sure we are using the C locale.
780  oss.imbue(std::locale::classic());
781  // Use scientific format.
782  oss << std::scientific;
783  // http://stackoverflow.com/questions/554063/how-do-i-print-a-double-value-with-full-precision-using-cout
784  // NOTE: this does not mean that the *exact* value of the long double is printed, just that the
785  // value is recovered exactly if reloaded on the same machine. This is a compromise, as the exact
786  // printing of the value in string form would take up hundreds of digits. On the other hand, there is a
787  // similar situation also for float and double, as there is not guarantee that they conform to IEEE. In
788  // the end it seems the only practical approach is to consider all floating-point types as approximate
789  // values, subject to various platform/architecture vagaries.
790  oss.precision(std::numeric_limits<long double>::max_digits10);
791  oss << x;
792  msgpack_pack(packer, oss.str(), f);
793  }
794  }
795  }
796 };

1007  } else {

1009  "quiet NaNs");
1010  }

1013  x = std::copysign(lim::quiet_NaN(), -1.l);
1014  } else {
1015  piranha_throw(std::invalid_argument, "cannot deserialize a NaN if the platform does not support "
1016  "quiet NaNs");
1017  }
1018  } else if (tmp == "+inf") {
1019  if (lim::has_infinity) {
1020  x = lim::infinity();
1021  } else {
1022  piranha_throw(std::invalid_argument, "infinities are not supported by the platform");
1023  }
1024  } else if (tmp == "-inf") {
1025  if (lim::has_infinity) {
1026  x = std::copysign(lim::infinity(), -1.l);
1027  } else {
1028  piranha_throw(std::invalid_argument, "infinities are not supported by the platform");
1029  }
1030  } else {
1031  std::istringstream iss;
1032  iss.imbue(std::locale::classic());
1033  // NOTE: is seems like the std::scientific format flag has an effect on input
1034  // streams as well. See the example here:
1035  // http://en.cppreference.com/w/cpp/io/manip/fixed
1036  iss >> std::scientific;
1037  iss.str(tmp);
1038  iss >> x;
1039  if (unlikely(iss.fail())) {
1040  piranha_throw(std::invalid_argument, "failed to parse the string '" + tmp + "' as a long double");

1250 // NOTE: no need for ifdefs guards here, as the compression-specific stuff is hidden in the CompressionFilter type.
1251 template <typename CompressionFilter, typename T>
1252 inline void save_file_boost_compress_impl(const T &x, std::ofstream &ofile, data_format f)
1253 {
1254  piranha_assert(f == data_format::boost_binary || f == data_format::boost_portable);
1255  // NOTE: there seem to be 2 choices here: stream or streambuf. The first does some formatting, while the second
1256  // one is "raw" but does not provide the stream interface (which is used, e.g., by msgpack). Since we always
1257  // open files in binary mode (as suggested by the Boost serialization library), this should not matter in the end.
1258  // Some resources:
1259  // http://stackoverflow.com/questions/1753469/how-to-hook-up-boost-serialization-iostreams-to-serialize-gzip-an-object-to
1260  // https://code.google.com/p/cloudobserver/wiki/TutorialsBoostIOstreams
1261  // http://stackoverflow.com/questions/8116541/what-exactly-is-streambuf-how-do-i-use-it
1262  // http://www.boost.org/doc/libs/1_46_1/libs/serialization/doc/special.html
1263  boost::iostreams::filtering_ostream out;
1264  out.push(CompressionFilter{});
1265  out.push(ofile);
1266  if (f == data_format::boost_binary) {
1267  boost::archive::binary_oarchive oa(out);
1268  boost_save(oa, x);
1269  } else {
1270  boost::archive::text_oarchive oa(out);
1271  boost_save(oa, x);
1272  }
1273 }

1508 // Utility to deduce compression and data format from a filename.
1509 inline std::pair<compression, data_format> get_cdf_from_filename(std::string filename)
1510 {
1511  const auto orig_fname = filename;
1512  compression c = compression::none;
1513  if (boost::ends_with(filename, ".bz2")) {
1514  c = compression::bzip2;
1515  filename.erase(filename.end() - 4, filename.end());
1516  } else if (boost::ends_with(filename, ".gz")) {
1517  c = compression::gzip;
1518  filename.erase(filename.end() - 3, filename.end());
1519  } else if (boost::ends_with(filename, ".zip")) {
1520  c = compression::zlib;
1521  filename.erase(filename.end() - 4, filename.end());
1522  }
1523  data_format f;
1524  if (boost::ends_with(filename, ".boostb")) {
1525  f = data_format::boost_binary;
1526  } else if (boost::ends_with(filename, ".boostp")) {
1527  f = data_format::boost_portable;
1528  } else if (boost::ends_with(filename, ".mpackb")) {
1529  f = data_format::msgpack_binary;
1530  } else if (boost::ends_with(filename, ".mpackp")) {
1531  f = data_format::msgpack_portable;
1532  } else {
1533  piranha_throw(std::invalid_argument,
1534  "unable to deduce the data format from the filename '" + orig_fname
1535  + "'. The filename must end with one of ['.boostb','.boostp','.mpackb','.mpackp'], "
1536  "optionally followed by one of ['.bz2','gz','zip'].");
1537  }
1538  return std::make_pair(c, f);
1539 }
