81 template <typename T, typename Derived, typename S = std::integral_constant<std::size_t, 0u>>

84  // NOTE: the general idea about requirements here is that these are the bare minimum
85  // to make a simple array_key suitable as key in a piranha::series. There are additional
86  // requirements which are optional and checked in the member functions.
87  PIRANHA_TT_CHECK(std::is_constructible, T, int);
88  PIRANHA_TT_CHECK(is_less_than_comparable, T);

101 private:
102  // Enabler for constructor from init list.
103  template <typename U>
104  using init_list_enabler =
105  typename std::enable_if<std::is_constructible<container_type, std::initializer_list<U>>::value, int>::type;
106  // Enabler for generic ctor.
107  template <typename U>
108  using generic_ctor_enabler = typename std::enable_if<has_safe_cast<value_type, U>::value, int>::type;
109  // Enabler for addition and subtraction.
110  // NOTE: here it is not difficult to make this an int enabler, like we usually do elsewhere,
111  // but alas the intel compiler chokes on this :/.
112  template <typename U>
113  using add_enabler = decltype(std::declval<U const &>().add(std::declval<U &>(), std::declval<U const &>()));
114  template <typename U>
115  using sub_enabler = decltype(std::declval<U const &>().sub(std::declval<U &>(), std::declval<U const &>()));

186  template <typename U, typename Derived2, typename S2, generic_ctor_enabler<U> = 0>
187  explicit array_key(const array_key<U, Derived2, S2> &other, const symbol_set &args)
188  {
189  if (unlikely(other.size() != args.size())) {
190  piranha_throw(std::invalid_argument, "inconsistent sizes in generic array_key constructor");
191  }
192  // NOTE: here the requirement is that value_type is move-assignable, which is already
193  // fulfilled by is_container_element.
194  std::transform(other.begin(), other.end(), std::back_inserter(m_container),
195  [](const U &x) { return safe_cast<value_type>(x); });
196  }

354  void trim_identify(symbol_set &candidates, const symbol_set &args) const
355  {
356  if (unlikely(m_container.size() != args.size())) {
357  piranha_throw(std::invalid_argument, "invalid arguments set for trim_identify()");
358  }
359  for (size_type i = 0u; i < m_container.size(); ++i) {
360  if (!math::is_zero(m_container[i]) && std::binary_search(candidates.begin(), candidates.end(),
361  args[static_cast<symbol_set::size_type>(i)])) {
362  candidates.remove(args[static_cast<symbol_set::size_type>(i)]);
363  }
364  }
365  }

379  Derived trim(const symbol_set &trim_args, const symbol_set &orig_args) const
380  {
381  if (unlikely(m_container.size() != orig_args.size())) {
382  piranha_throw(std::invalid_argument, "invalid arguments set for trim()");
383  }
384  Derived retval;
385  for (size_type i = 0u; i < m_container.size(); ++i) {
386  if (!std::binary_search(trim_args.begin(), trim_args.end(),
387  orig_args[static_cast<symbol_set::size_type>(i)])) {
388  retval.push_back(m_container[i]);
389  }
390  }
391  return retval;
392  }
