360  void trim_identify(symbol_set &candidates, const symbol_set &args) const
361  {
362  if (unlikely(m_container.size() != args.size())) {
363  piranha_throw(std::invalid_argument,"invalid arguments set for trim_identify()");
364  }
365  for (size_type i = 0u; i < m_container.size(); ++i) {
366  if (!math::is_zero(m_container[i]) && std::binary_search(candidates.begin(),candidates.end(),
367  args[static_cast<symbol_set::size_type>(i)]))
368  {
369  candidates.remove(args[static_cast<symbol_set::size_type>(i)]);
370  }
371  }

386  Derived trim(const symbol_set &trim_args, const symbol_set &orig_args) const
387  {
388  if (unlikely(m_container.size() != orig_args.size())) {
389  piranha_throw(std::invalid_argument,"invalid arguments set for trim()");
390  }
391  Derived retval;
392  for (size_type i = 0u; i < m_container.size(); ++i) {
393  if (!std::binary_search(trim_args.begin(),trim_args.end(),
394  orig_args[static_cast<symbol_set::size_type>(i)]))
395  {
396  retval.push_back(m_container[i]);

398  }
