358  void trim_identify(symbol_set &candidates, const symbol_set &args) const
359  {
360  if (unlikely(m_container.size() != args.size())) {
361  piranha_throw(std::invalid_argument,"invalid arguments set for trim_identify()");
362  }
363  for (size_type i = 0u; i < m_container.size(); ++i) {
364  if (!math::is_zero(m_container[i]) && std::binary_search(candidates.begin(),candidates.end(),
365  args[static_cast<symbol_set::size_type>(i)]))
366  {
367  candidates.remove(args[static_cast<symbol_set::size_type>(i)]);
368  }
369  }
370  }

384  Derived trim(const symbol_set &trim_args, const symbol_set &orig_args) const
385  {
386  if (unlikely(m_container.size() != orig_args.size())) {
387  piranha_throw(std::invalid_argument,"invalid arguments set for trim()");
388  }
389  Derived retval;
390  for (size_type i = 0u; i < m_container.size(); ++i) {
391  if (!std::binary_search(trim_args.begin(),trim_args.end(),
392  orig_args[static_cast<symbol_set::size_type>(i)]))
393  {
394  retval.push_back(m_container[i]);
395  }
396  }
397  return retval;
398  }
